22use super::{CorruptedError, InternalError, SqliteFullDatabase};

26/// Opens the database using the given [`Config`].
27///
28/// Note that this doesn't return a [`SqliteFullDatabase`], but rather a [`DatabaseOpen`].
29pub fn open(config: Config) -> Result<DatabaseOpen, InternalError> {
30    let flags = rusqlite::OpenFlags::SQLITE_OPEN_READ_WRITE |
31        rusqlite::OpenFlags::SQLITE_OPEN_CREATE |
32        // The "no mutex" option opens SQLite in "multi-threaded" mode, meaning that it can safely
33        // be used from multiple threads as long as we don't access the connection from multiple
34        // threads *at the same time*. Since we put the connection behind a `Mutex`, and that the
35        // underlying library implements `!Sync` for `Connection` as a safety measure anyway, it
36        // is safe to enable this option.
37        // See https://www.sqlite.org/threadsafe.html
38        rusqlite::OpenFlags::SQLITE_OPEN_NO_MUTEX;
39
40    let database = match config.ty {
41        ConfigTy::Disk { path, .. } => rusqlite::Connection::open_with_flags(path, flags),
42        ConfigTy::Memory => rusqlite::Connection::open_in_memory_with_flags(flags),
43    }
44    .map_err(InternalError)?;
45
46    // The underlying SQLite wrapper maintains a cache of prepared statements. We set it to a
47    // value superior to the number of different queries we make.
48    database.set_prepared_statement_cache_capacity(64);
49
50    // Configure the database connection.
51    database
52        .execute_batch(
53            r#"
54-- See https://sqlite.org/pragma.html and https://www.sqlite.org/wal.html
55PRAGMA journal_mode = WAL;
56PRAGMA synchronous = NORMAL;
57PRAGMA locking_mode = EXCLUSIVE;
58PRAGMA encoding = 'UTF-8';
59PRAGMA trusted_schema = false;
60PRAGMA foreign_keys = ON;
61            "#,
62        )
63        .map_err(InternalError)?;
64
65    // `PRAGMA` queries can't be parametrized, and thus we have to use `format!`.
66    database
67        .execute(
68            &format!(
69                "PRAGMA cache_size = {}",
70                0i64.saturating_sub_unsigned(
71                    u64::try_from((config.cache_size.saturating_sub(1) / 1024).saturating_add(1))
72                        .unwrap_or(u64::MAX),
73                )
74            ),
75            (),
76        )
77        .map_err(InternalError)?;
78
79    // `PRAGMA` queries can't be parametrized, and thus we have to use `format!`.
80    if let ConfigTy::Disk {
81        memory_map_size, ..
82    } = config.ty
83    {
84        database
85            .execute_batch(&format!("PRAGMA mmap_size = {}", memory_map_size))
86            .map_err(InternalError)?;
87    }
88
89    // Each SQLite database contains a "user version" whose value can be used by the API user
90    // (that's us!) however they want. Its value defaults to 0 for new database. We use it to
91    // store the schema version.
92    let user_version = database
93        .prepare_cached("PRAGMA user_version")
94        .map_err(InternalError)?
95        .query_row((), |row| row.get::<_, i64>(0))
96        .map_err(InternalError)?;
97
98    // Migrations.
99    if user_version <= 0 {
100        database
101            .execute_batch(
102                r#"
103-- `auto_vacuum` can switched between `NONE` and non-`NONE` on newly-created database.
104PRAGMA auto_vacuum = INCREMENTAL;
105
106/*
107Contains all the "global" values in the database.
108A value must be present either in `value_blob` or `value_number` depending on the type of data.
109
110Keys in that table:
111
112 - `best` (blob): Hash of the best block.
113
114 - `finalized` (number): Height of the finalized block, as a 64bits big endian number.
115
116*/
117CREATE TABLE meta(
118    key STRING NOT NULL PRIMARY KEY,
119    value_blob BLOB,
120    value_number INTEGER,
121    -- Either `value_blob` or `value_number` must be NULL but not both.
122    CHECK((value_blob IS NULL OR value_number IS NULL) AND (value_blob IS NOT NULL OR value_number IS NOT NULL))
123);
124
125/*
126List of all trie nodes of all blocks whose trie is stored in the database.
127*/
128CREATE TABLE trie_node(
129    hash BLOB NOT NULL PRIMARY KEY,
130    partial_key BLOB NOT NULL    -- Each byte is a nibble, in other words all bytes are <16
131);
132
133/*
134Storage associated to a trie node.
135For each entry in `trie_node` there exists either 0 or 1 entry in `trie_node_storage` indicating
136the storage value associated to this node.
137*/
138CREATE TABLE trie_node_storage(
139    node_hash BLOB NOT NULL PRIMARY KEY,
140    value BLOB,
141    trie_root_ref BLOB,
142    trie_entry_version INTEGER NOT NULL,
143    FOREIGN KEY (node_hash) REFERENCES trie_node(hash) ON UPDATE CASCADE ON DELETE CASCADE
144    CHECK((value IS NULL) != (trie_root_ref IS NULL))
145);
146CREATE INDEX trie_node_storage_by_trie_root_ref ON trie_node_storage(trie_root_ref);
147
148/*
149Parent-child relationship between trie nodes.
150*/
151CREATE TABLE trie_node_child(
152    hash BLOB NOT NULL,
153    child_num BLOB NOT NULL,   -- Always contains one single byte. We use `BLOB` instead of `INTEGER` because SQLite stupidly doesn't provide any way of converting between integers and blobs
154    child_hash BLOB NOT NULL,
155    PRIMARY KEY (hash, child_num),
156    FOREIGN KEY (hash) REFERENCES trie_node(hash) ON UPDATE CASCADE ON DELETE CASCADE
157    CHECK(LENGTH(child_num) == 1 AND HEX(child_num) < '10')
158);
159CREATE INDEX trie_node_child_by_hash ON trie_node_child(hash);
160CREATE INDEX trie_node_child_by_child_hash ON trie_node_child(child_hash);
161
162/*
163List of all known blocks, indexed by their hash or number.
164*/
165CREATE TABLE blocks(
166    hash BLOB NOT NULL PRIMARY KEY,
167    parent_hash BLOB,  -- NULL only for the genesis block
168    state_trie_root_hash BLOB,  -- NULL if and only if the trie is empty or if the trie storage has been pruned from the database
169    number INTEGER NOT NULL,
170    header BLOB NOT NULL,
171    justification BLOB,
172    is_best_chain BOOLEAN NOT NULL,
173    UNIQUE(number, hash)
174);
175CREATE INDEX blocks_by_number ON blocks(number);
176CREATE INDEX blocks_by_parent ON blocks(parent_hash);
177CREATE INDEX blocks_by_state_trie_root_hash ON blocks(state_trie_root_hash);
178CREATE INDEX blocks_by_best ON blocks(number, is_best_chain);
179
180/*
181Each block has a body made from 0+ extrinsics (in practice, there's always at least one extrinsic,
182but the database supports 0). This table contains these extrinsics.
183The `idx` field contains the index between `0` and `num_extrinsics - 1`. The values in `idx` must
184be contiguous for each block.
185*/
186CREATE TABLE blocks_body(
187    hash BLOB NOT NULL,
188    idx INTEGER NOT NULL,
189    extrinsic BLOB NOT NULL,
190    UNIQUE(hash, idx),
191    CHECK(length(hash) == 32),
192    FOREIGN KEY (hash) REFERENCES blocks(hash) ON UPDATE CASCADE ON DELETE CASCADE
193);
194CREATE INDEX blocks_body_by_block ON blocks_body(hash);
195
196PRAGMA user_version = 1;
197
198        "#,
199            )
200            .map_err(InternalError)?
201    }
202
203    let is_empty = database
204        .prepare_cached("SELECT COUNT(*) FROM meta WHERE key = ?")
205        .map_err(InternalError)?
206        .query_row(("best",), |row| row.get::<_, i64>(0))
207        .map_err(InternalError)?
208        == 0;
209
210    Ok(if !is_empty {
211        DatabaseOpen::Open(SqliteFullDatabase {
212            database: parking_lot::Mutex::new(database),
213            block_number_bytes: config.block_number_bytes, // TODO: consider storing this value in the DB and check it when opening
214        })
215    } else {
216        DatabaseOpen::Empty(DatabaseEmpty {
217            database,
218            block_number_bytes: config.block_number_bytes,
219        })
220    })
221}

223/// Configuration for the database.
224#[derive(Debug)]
225pub struct Config<'a> {
226    /// Type of database.
227    pub ty: ConfigTy<'a>,
228
229    /// Number of bytes used to encode the block number.
230    pub block_number_bytes: usize,
231
232    /// Maximum allowed size, in bytes, of the SQLite cache.
233    pub cache_size: usize,
234}
235
236/// Type of database.
237#[derive(Debug)]
238pub enum ConfigTy<'a> {
239    /// Store the database on disk.
240    Disk {
241        /// Path to the directory containing the database.
242        path: &'a Path,
243        /// Maximum allowed amount of memory, in bytes, that SQLite will reserve to memory-map
244        /// files.
245        memory_map_size: usize,
246    },
247    /// Store the database in memory. The database is discarded on destruction.
248    Memory,
249}
250
251/// Either existing database or database prototype.
252pub enum DatabaseOpen {
253    /// A database already existed and has now been opened.
254    Open(SqliteFullDatabase),
255
256    /// Either a database has just been created, or there existed a database but it is empty.
257    ///
258    /// > **Note**: The situation where a database existed but is empty can happen if you have
259    /// >           previously called [`open`] then dropped the [`DatabaseOpen`] object without
260    /// >           filling the newly-created database with data.
261    Empty(DatabaseEmpty),
262}
263
264/// An open database. Holds file descriptors.
265pub struct DatabaseEmpty {
266    /// See the similar field in [`SqliteFullDatabase`].
267    database: rusqlite::Connection,
268
269    /// See the similar field in [`SqliteFullDatabase`].
270    block_number_bytes: usize,
271}
272
273impl DatabaseEmpty {
274    /// Inserts the given finalized block in the database prototype in order to turn it into
275    /// an actual database.
276    // TODO: can a database not be empty?
277    pub fn initialize<'a>(
278        self,
279        finalized_block_header: &[u8],
280        finalized_block_body: impl ExactSizeIterator<Item = &'a [u8]>,
281        finalized_block_justification: Option<Vec<u8>>,
282    ) -> Result<SqliteFullDatabase, CorruptedError> {
283        let database = SqliteFullDatabase {
284            database: parking_lot::Mutex::new(self.database),
285            block_number_bytes: self.block_number_bytes,
286        };
287
288        database.reset(
289            finalized_block_header,
290            finalized_block_body,
291            finalized_block_justification,
292        )?;
293
294        Ok(database)
295    }
