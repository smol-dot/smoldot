18//! Given as input a partial base trie and a diff, calculates the new root of the trie.
19//!
20//! # Implementation
21//!
22//! The algorithm for calculating the trie root is roughly:
23//!
24//! ```notrust
25//! TrieRoot = ClosestDescendantMerkleValue(∅, False) || EmptyTrieRootHash
26//!
27//! ClosestDescendantMerkleValue(Key, ForceRecalculate) =
28//!     If !ForceRecalculate AND !DiffContainsEntryWithPrefix(Key)
29//!         MaybeMerkleValue := BaseTrieClosestDescendantMerkleValue(Key)
30//!         If MaybeMerkleValue
31//!             return MaybeMerkleValue
32//!
33//!     Btcd := BaseTrieClosestDescendant(Key)
34//!     MaybeNodeKey := LongestCommonDenominator(Btcd, ...DiffInsertsNodesWithPrefix(Key))
35//!     MaybeChildren := Array(ClosestDescendantMerkleValue(Concat(MaybeNodeKey, 0), MaybeNodeKey != Btcd), ..., ClosestDescendantMerkleValue(Concat(MaybeNodeKey, 15), MaybeNodeKey != Btcd))
36//!     MaybeStorageValue := DiffInserts(MaybeNodeKey) || (BaseTrieStorageValue(MaybeNodeKey) - DiffErases(MaybeNodeKey))
37//!
38//!     If MaybeStorageValue = ∅ AND NumChildren(MaybeChildren) == 0
39//!         ∅
40//!     ElseIf NumChildren(MaybeChildren) == 1 AND BaseTrieStorageValue(MaybeNodeKey) != ∅ AND MaybeStorageValue = ∅
41//!         # Because the partial key of the child has changed, we have to recalculate it.
42//!         ClosestDescendantMerkleValue(ChildThatExists(MaybeChildren), True)
43//!     ElseIf NumChildren(MaybeChildren) == 1 AND BaseTrieStorageValue(MaybeNodeKey) = ∅ AND MaybeStorageValue = ∅
44//!         ChildThatExists(MaybeChildren)
45//!     Else
46//!         Encode(MaybeStorageValue, MaybeChildren)
47//! ```
48//!
49//! The public API functions are thus `BaseTrieClosestDescendant`,
50//! `BaseTrieClosestDescendantMerkleValue`, and `BaseTrieStorageValue`.
51//!
52//! The algorithm above is recursive. In order to implement it, we instead maintain a stack of
53//! nodes that are currently being calculated.
54//!

66/// Configuration for [`trie_root_calculator`].
67pub struct Config {
68    /// Diff that is being applied on top of the base trie.
69    pub diff: TrieDiff,
70
71    /// Version to use for the storage values written by [`Config::diff`].
72    pub diff_trie_entries_version: TrieEntryVersion,
73
74    /// Depth level of the deepest node whose value needs to be calculated. The root node has a
75    /// depth level of 1, its children a depth level of 2, etc.
76    ///
77    /// Used as a hint to pre-allocate a container. Getting the value wrong has no ill
78    /// consequence except a very small performance hit.
79    pub max_trie_recalculation_depth_hint: usize,
80}
81
82/// Starts a new calculation.
83pub fn trie_root_calculator(config: Config) -> InProgress {
84    Box::new(Inner {
85        stack: Vec::with_capacity(config.max_trie_recalculation_depth_hint),
86        diff: config.diff,
87        diff_trie_entries_version: config.diff_trie_entries_version,
88    })
89    .next()
90}

92/// Trie root calculation in progress.
93pub enum InProgress {
94    /// Calculation has successfully finished.
95    Finished {
96        /// Calculated trie root hash.
97        trie_root_hash: [u8; 32],
98    },
99    /// See [`ClosestDescendant`].
100    ClosestDescendant(ClosestDescendant),
101    /// See [`StorageValue`].
102    StorageValue(StorageValue),
103    /// See [`ClosestDescendantMerkleValue`].
104    ClosestDescendantMerkleValue(ClosestDescendantMerkleValue),
105    /// See [`TrieNodeInsertUpdateEvent`].
106    // TODO: consider guaranteeing an ordering in the events?
107    TrieNodeInsertUpdateEvent(TrieNodeInsertUpdateEvent),
108    /// See [`TrieNodeRemoveEvent`].
109    TrieNodeRemoveEvent(TrieNodeRemoveEvent),
110}

112/// In order to continue the calculation, must find in the base trie the closest descendant
113/// (including branch node) to a certain key in the trie. This can be equal to the requested key
114/// itself.
115pub struct ClosestDescendant {
116    inner: Box<Inner>,
117    /// Extra nibbles to append at the end of what `self.key()` returns.
118    key_extra_nibbles: Vec<Nibble>,
119    /// `Some` if the diff inserts a value that is equal or a descendant of `self.key()`. If
120    /// `Some`, contains the least common denominator shared amongst all these insertions.
121    diff_inserts_lcd: Option<Vec<Nibble>>,
122}
123
124impl ClosestDescendant {
125    /// Returns an iterator of slices, which, when joined together, form the full key of the trie
126    /// node whose closest descendant must be fetched.
127    pub fn key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
128        self.inner
129            .current_node_full_key()
130            .map(either::Left)
131            .chain(iter::once(either::Right(&self.key_extra_nibbles)))
132    }
133
134    /// Returns the same value as [`ClosestDescendant::key`] but as a `Vec`.
135    pub fn key_as_vec(&self) -> Vec<Nibble> {
136        self.key().fold(Vec::new(), |mut a, b| {
137            a.extend_from_slice(b.as_ref());
138            a
139        })
140    }
141
142    /// Indicates the key of the closest descendant to the key indicated by
143    /// [`ClosestDescendant::key`] and resume the calculation.
144    pub fn inject(
145        mut self,
146        closest_descendant: Option<impl Iterator<Item = Nibble>>,
147    ) -> InProgress {
148        // We are after a call to `BaseTrieClosestDescendant`.
149        debug_assert!(
150            self.inner
151                .stack
152                .last()
153                .map_or(true, |n| n.children.len() != 16)
154        );
155
156        // Length of the key of the currently-iterated node.
157        let iter_key_len = self
158            .inner
159            .current_node_full_key()
160            .fold(0, |acc, k| acc + k.as_ref().len());
161
162        // Find the value of `MaybeNode`.
163        let (maybe_node_partial_key, children_partial_key_changed) =
164            match (self.diff_inserts_lcd, closest_descendant) {
165                (Some(inserted_key), Some(base_trie_key)) => {
166                    // `children_partial_key_changed` is `true` if
167                    // `diff_inserts_lcd < closest_descendant`.
168                    let mut inserted_key_iter = inserted_key.iter().copied().skip(iter_key_len);
169                    let mut base_trie_key = base_trie_key.skip(iter_key_len);
170                    let mut maybe_node_pk =
171                        Vec::with_capacity(inserted_key.len() - iter_key_len + 8);
172                    let mut children_pk_changed = false;
173                    loop {
174                        match (inserted_key_iter.next(), base_trie_key.next()) {
175                            (Some(inib), Some(bnib)) if inib == bnib => maybe_node_pk.push(inib),
176                            (Some(_), Some(_)) | (None, Some(_)) => {
177                                children_pk_changed = true;
178                                break;
179                            }
180                            (Some(_), None) | (None, None) => break,
181                        }
182                    }
183                    (maybe_node_pk, children_pk_changed)
184                }
185                (Some(inserted_key), None) => (
186                    inserted_key
187                        .into_iter()
188                        .skip(iter_key_len)
189                        .collect::<Vec<_>>(),
190                    true,
191                ),
192                (None, Some(base_trie_key)) => (base_trie_key.skip(iter_key_len).collect(), false),
193                (None, None) => {
194                    // If neither the base trie nor the diff contain any descendant, then skip ahead.
195                    return if let Some(parent_node) = self.inner.stack.last_mut() {
196                        // If the element has a parent, indicate that the current iterated node
197                        // doesn't exist and continue the algorithm.
198                        debug_assert_ne!(parent_node.children.len(), 16);
199                        parent_node.children.push(None);
200                        self.inner.next()
201                    } else {
202                        // If the element doesn't have a parent, then the trie is completely empty.
203                        InProgress::Finished {
204                            trie_root_hash: trie::EMPTY_BLAKE2_TRIE_MERKLE_VALUE,
205                        }
206                    };
207                }
208            };
209
210        // Push `MaybeNode` onto the stack and continue the algorithm.
211        self.inner.stack.push(InProgressNode {
212            partial_key: maybe_node_partial_key,
213            children_partial_key_changed,
214            children: arrayvec::ArrayVec::new(),
215        });
216        self.inner.next()
217    }
218}

220/// In order to continue, must fetch the storage value of the given key in the base trie.
221pub struct StorageValue(Box<Inner>);
222
223impl StorageValue {
224    /// Returns an iterator of slices, which, when joined together, form the full key of the trie
225    /// node whose storage value must be fetched.
226    pub fn key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
227        self.0.current_node_full_key()
228    }
229
230    /// Returns the same value as [`ClosestDescendant::key`] but as a `Vec`.
231    pub fn key_as_vec(&self) -> Vec<Nibble> {
232        self.key().fold(Vec::new(), |mut a, b| {
233            a.extend_from_slice(b.as_ref());
234            a
235        })
236    }
237
238    /// Indicate the storage value and trie entry version of the trie node indicated by
239    /// [`StorageValue::key`] and resume the calculation.
240    pub fn inject_value(
241        mut self,
242        base_trie_storage_value: Option<(&[u8], TrieEntryVersion)>,
243    ) -> InProgress {
244        // We have finished obtaining `BaseTrieStorageValue` and we are at the last step of
245        // `ClosestDescendantMerkleValue` in the algorithm shown at the top.
246
247        // Adjust the storage value to take the diff into account.
248        // In other words, we calculate `MaybeStorageValue` from `BaseTrieStorageValue`.
249        let maybe_storage_value = if self.key().fold(0, |a, b| a + b.as_ref().len()) % 2 == 0 {
250            // TODO: could be optimized?
251            let key_nibbles = self.key().fold(Vec::new(), |mut a, b| {
252                a.extend_from_slice(b.as_ref());
253                a
254            });
255            debug_assert_eq!(key_nibbles.len() % 2, 0);
256            let key_as_u8 =
257                trie::nibbles_to_bytes_suffix_extend(key_nibbles.into_iter()).collect::<Vec<_>>();
258            if let Some((value, _)) = self.0.diff.diff_get(&key_as_u8) {
259                value.map(|v| (v, self.0.diff_trie_entries_version))
260            } else {
261                base_trie_storage_value
262            }
263        } else {
264            base_trie_storage_value
265        };
266
267        // Remove the element that is being calculated from the stack, as we finish the
268        // calculation down below.
269        let calculated_elem = self.0.stack.pop().unwrap_or_else(|| panic!());
270        debug_assert_eq!(calculated_elem.children.len(), 16);
271
272        let node_num_children = calculated_elem
273            .children
274            .iter()
275            .filter(|c| c.is_some())
276            .count();
277
278        let parent_node = self.0.stack.last_mut();
279        debug_assert!(parent_node.map_or(true, |p| p.children.len() != 16));
280
281        match (
282            base_trie_storage_value,
283            maybe_storage_value,
284            node_num_children,
285            self.0.stack.last_mut(),
286        ) {
287            (_, None, 0, Some(_parent_node)) => {
288                // Trie node no longer exists after the diff has been applied.
289                // This path is only reached if the trie node has a parent, as otherwise the trie
290                // node is the trie root and thus necessarily exists.
291                InProgress::TrieNodeRemoveEvent(TrieNodeRemoveEvent {
292                    inner: self.0,
293                    calculated_elem,
294                    ty: TrieNodeRemoveEventTy::NoChildrenLeft,
295                })
296            }
297
298            (_, None, 1, _parent_node) if !calculated_elem.children_partial_key_changed => {
299                // Trie node doesn't exists after the diff has been applied because it has exactly
300                // one child.
301                // Since `children_partial_key_changed` is `true`, we know that the node existed
302                // and generate a `TrieNodeRemoveEvent`.
303                // Unfortunately, the child Merkle value is wrong as its partial key has changed
304                // and has to be recalculated.
305
306                // To handle this situation, we back jump to `ClosestDescendant` but this time
307                // make sure to skip over `calculated_elem`.
308                // This isn't done here but in `TrieNodeRemoveEvent::resume`.
309                InProgress::TrieNodeRemoveEvent(TrieNodeRemoveEvent {
310                    inner: self.0,
311                    calculated_elem,
312                    ty: TrieNodeRemoveEventTy::ReplacedWithSingleChild,
313                })
314            }
315
316            (_, None, 1, _parent_node) => {
317                // Same as the previous block, except that `children_partial_key_changed` is
318                // `false`. Therefore the node didn't exist in the base trie, and no event shall
319                // be generated. Instead, we immediately call `resume` in order to continue
320                // execution without notifying the user.
321                TrieNodeRemoveEvent {
322                    inner: self.0,
323                    calculated_elem,
324                    ty: TrieNodeRemoveEventTy::ReplacedWithSingleChild,
325                }
326                .resume()
327            }
328
329            (Some(_), None, 0, None) => {
330                // Root node of the trie was deleted and trie is now empty.
331                InProgress::TrieNodeRemoveEvent(TrieNodeRemoveEvent {
332                    inner: self.0,
333                    calculated_elem,
334                    ty: TrieNodeRemoveEventTy::NoChildrenLeft,
335                })
336            }
337
338            (None, None, 0, None) => {
339                // Trie is empty.
340                // This case is handled separately in order to not generate
341                // a `TrieNodeInsertUpdateEvent` for a node that doesn't actually exist.
342                InProgress::Finished {
343                    trie_root_hash: trie::EMPTY_BLAKE2_TRIE_MERKLE_VALUE,
344                }
345            }
346
347            (_, _, _, parent_node) => {
348                // Trie node still exists or was created. Calculate its Merkle value.
349
350                // Due to some borrow checker troubles, we need to calculate the storage value
351                // hash ahead of time if relevant.
352                let storage_value_hash =
353                    if let Some((value, TrieEntryVersion::V1)) = maybe_storage_value {
354                        if value.len() >= 33 {
355                            Some(blake2_rfc::blake2b::blake2b(32, &[], value))
356                        } else {
357                            None
358                        }
359                    } else {
360                        None
361                    };
362                let merkle_value = trie::trie_node::calculate_merkle_value(
363                    trie::trie_node::Decoded {
364                        children: array::from_fn(|n| calculated_elem.children[n].as_ref()),
365                        partial_key: calculated_elem.partial_key.iter().copied(),
366                        storage_value: match (maybe_storage_value, storage_value_hash.as_ref()) {
367                            (_, Some(storage_value_hash)) => trie::trie_node::StorageValue::Hashed(
368                                <&[u8; 32]>::try_from(storage_value_hash.as_bytes())
369                                    .unwrap_or_else(|_| panic!()),
370                            ),
371                            (Some((value, _)), None) => {
372                                trie::trie_node::StorageValue::Unhashed(value)
373                            }
374                            (None, _) => trie::trie_node::StorageValue::None,
375                        },
376                    },
377                    trie::HashFunction::Blake2,
378                    parent_node.is_none(),
379                )
380                .unwrap_or_else(|_| panic!());
381
382                // The stack is updated in `TrieNodeInsertUpdateEvent::resume`.
383                InProgress::TrieNodeInsertUpdateEvent(TrieNodeInsertUpdateEvent {
384                    inner: self.0,
385                    children: calculated_elem.children,
386                    inserted_elem_partial_key: calculated_elem.partial_key,
387                    merkle_value,
388                })
389            }
390        }
391    }
392}

394/// In order to continue, must fetch the Merkle value of the closest descendant to the given key
395/// in the base trie.
396///
397/// It is possible to continue the calculation even if the Merkle value is unknown, in which case
398/// the calculation will walk down the trie in order to calculate the Merkle value manually.
399pub struct ClosestDescendantMerkleValue {
400    inner: Box<Inner>,
401}
402
403impl ClosestDescendantMerkleValue {
404    /// Returns an iterator of slices, which, when joined together, form the full key of the trie
405    /// node whose closest descendant Merkle value must be fetched.
406    pub fn key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
407        // A `ClosestDescendantMerkleValue` is created directly in response to a `ClosestAncestor` without
408        // updating the `Inner`.
409        self.inner.current_node_full_key()
410    }
411
412    /// Returns the same value as [`ClosestDescendantMerkleValue::key`] but as a `Vec`.
413    pub fn key_as_vec(&self) -> Vec<Nibble> {
414        self.key().fold(Vec::new(), |mut a, b| {
415            a.extend_from_slice(b.as_ref());
416            a
417        })
418    }
419
420    /// Indicate that the closest descendant or its Merkle value is unknown and resume the
421    /// calculation.
422    ///
423    /// This function be used if you are unaware of the Merkle value. The algorithm will perform
424    /// the calculation of this Merkle value manually, which takes more time.
425    pub fn resume_unknown(self) -> InProgress {
426        // The element currently being iterated was `Btcd`, and is now switched to being
427        // `MaybeNodeKey`. Because a `ClosestDescendantMerkleValue` is only ever created if the diff doesn't
428        // contain any entry that descends the currently iterated node, we know for sure that
429        // `MaybeNodeKey` is equal to `Btcd`.
430        InProgress::ClosestDescendant(ClosestDescendant {
431            inner: self.inner,
432            key_extra_nibbles: Vec::new(),
433            diff_inserts_lcd: None,
434        })
435    }
436
437    /// Indicate the Merkle value of closest descendant of the trie node indicated by
438    /// [`ClosestDescendantMerkleValue::key`] and resume the calculation.
439    pub fn inject_merkle_value(mut self, merkle_value: &[u8]) -> InProgress {
440        // We are after a call to `BaseTrieClosestDescendantMerkleValue` in the algorithm shown
441        // at the top.
442
443        // Check with a debug_assert! that this is a valid Merkle value. While this algorithm
444        // doesn't actually care about the content, providing a wrong value clearly indicates a
445        // bug somewhere in the API user's code.
446        debug_assert!(merkle_value.len() == 32 || trie::trie_node::decode(merkle_value).is_ok());
447
448        if let Some(parent_node) = self.inner.stack.last_mut() {
449            // If the element has a parent, add the Merkle value to its children and resume the
450            // algorithm.
451            debug_assert_ne!(parent_node.children.len(), 16);
452            parent_node
453                .children
454                .push(Some(trie::trie_node::MerkleValueOutput::from_bytes(
455                    AsRef::as_ref(&merkle_value),
456                )));
457            self.inner.next()
458        } else {
459            // If the element doesn't have a parent, then the Merkle value is the root of trie!
460            // This should only ever happen if the diff is empty.
461            debug_assert_eq!(self.inner.diff.diff_range_ordered::<Vec<u8>>(..).count(), 0);
462
463            // The trie root hash is always 32 bytes. If not, it indicates a bug in the API user's
464            // code.
465            let trie_root_hash = <[u8; 32]>::try_from(AsRef::as_ref(&merkle_value))
466                .unwrap_or_else(|_| panic!("invalid node value provided"));
467            InProgress::Finished { trie_root_hash }
468        }
469    }
470}

472/// Event indicating that a trie node has been inserted or updated.
473///
474/// The API user has nothing to do, but they can use this event to update their local version of
475/// the trie.
476pub struct TrieNodeInsertUpdateEvent {
477    inner: Box<Inner>,
478
479    /// Partial key of the node that was inserted or updated.
480    inserted_elem_partial_key: Vec<Nibble>,
481
482    /// Merkle values of the children of the node that was inserted or updated.
483    children: arrayvec::ArrayVec<Option<trie::trie_node::MerkleValueOutput>, 16>,
484
485    /// Merkle value of the node that was inserted or updated.
486    merkle_value: trie::trie_node::MerkleValueOutput,
487}
488
489impl TrieNodeInsertUpdateEvent {
490    /// Returns the key of the trie node that was inserted or updated.
491    pub fn key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
492        self.inner
493            .current_node_full_key()
494            .map(either::Left)
495            .chain(iter::once(either::Right(&self.inserted_elem_partial_key)))
496    }
497
498    /// Returns the same value as [`TrieNodeInsertUpdateEvent::key`] but as a `Vec`.
499    pub fn key_as_vec(&self) -> Vec<Nibble> {
500        self.key().fold(Vec::new(), |mut a, b| {
501            a.extend_from_slice(b.as_ref());
502            a
503        })
504    }
505
506    /// Returns the new Merkle value of the trie node that was inserted or updated.
507    pub fn merkle_value(&self) -> &[u8] {
508        self.merkle_value.as_ref()
509    }
510
511    /// Returns the Merkle values of the children of the trie node that was inserted or updated.
512    ///
513    /// The iterator always yields exactly 16 elements. The yielded element is `None` if there is
514    /// no child at this position.
515    pub fn children_merkle_values(&self) -> impl ExactSizeIterator<Item = Option<&[u8]>> {
516        self.children
517            .iter()
518            .map(|child| child.as_ref().map(|merkle_value| merkle_value.as_ref()))
519    }
520
521    /// Returns the partial key of the trie node that was inserted or updated.
522    pub fn partial_key(&self) -> &[Nibble] {
523        &self.inserted_elem_partial_key
524    }
525
526    /// Resume the computation.
527    pub fn resume(mut self) -> InProgress {
528        if let Some(parent_node) = self.inner.stack.last_mut() {
529            parent_node.children.push(Some(self.merkle_value));
530            self.inner.next()
531        } else {
532            // No more node in the stack means that this was the root node. The calculated
533            // Merkle value is the trie root hash.
534            InProgress::Finished {
535                // Guaranteed to never panic for the root node.
536                trie_root_hash: self.merkle_value.try_into().unwrap_or_else(|_| panic!()),
537            }
538        }
539    }
540}

542/// Event indicating that a trie node has been destroyed.
543///
544/// The API user has nothing to do, but they can use this event to update their local version of
545/// the trie.
546pub struct TrieNodeRemoveEvent {
547    inner: Box<Inner>,
548
549    /// Element that was removed from the stack and that we know no longer exists.
550    calculated_elem: InProgressNode,
551
552    /// Why the node was removed.
553    ty: TrieNodeRemoveEventTy,
554}
555
556enum TrieNodeRemoveEventTy {
557    NoChildrenLeft,
558    ReplacedWithSingleChild,
559}
560
561impl TrieNodeRemoveEvent {
562    /// Returns the key of the trie node that was removed.
563    pub fn key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
564        self.inner
565            .current_node_full_key()
566            .map(either::Left)
567            .chain(iter::once(either::Right(&self.calculated_elem.partial_key)))
568    }
569
570    /// Returns the same value as [`TrieNodeRemoveEvent::key`] but as a `Vec`.
571    pub fn key_as_vec(&self) -> Vec<Nibble> {
572        self.key().fold(Vec::new(), |mut a, b| {
573            a.extend_from_slice(b.as_ref());
574            a
575        })
576    }
577
578    /// Resume the computation.
579    pub fn resume(mut self) -> InProgress {
580        match self.ty {
581            TrieNodeRemoveEventTy::NoChildrenLeft => {
582                if let Some(parent_node) = self.inner.stack.last_mut() {
583                    parent_node.children.push(None);
584                    self.inner.next()
585                } else {
586                    InProgress::Finished {
587                        trie_root_hash: trie::EMPTY_BLAKE2_TRIE_MERKLE_VALUE,
588                    }
589                }
590            }
591            TrieNodeRemoveEventTy::ReplacedWithSingleChild => {
592                let child_index = self
593                    .calculated_elem
594                    .children
595                    .iter()
596                    .position(|c| c.is_some())
597                    .unwrap_or_else(|| panic!());
598
599                let mut partial_key = self.calculated_elem.partial_key;
600                partial_key.push(
601                    Nibble::try_from(u8::try_from(child_index).unwrap_or_else(|_| panic!()))
602                        .unwrap_or_else(|_| panic!()),
603                );
604
605                // Now calculate `LongestCommonDenominator(DiffInsertsNodesWithPrefix)`.
606                let (_, diff_inserts_lcd) = diff_inserts_least_common_denominator(
607                    &self.inner.diff,
608                    self.inner
609                        .current_node_full_key()
610                        .map(either::Left)
611                        .chain(iter::once(either::Right(&partial_key))),
612                );
613
614                InProgress::ClosestDescendant(ClosestDescendant {
615                    inner: self.inner,
616                    key_extra_nibbles: partial_key,
617                    diff_inserts_lcd,
618                })
619            }
620        }
621    }
622}
623
624struct Inner {
625    /// Stack of node entries whose value is currently being calculated. Every time
626    /// `BaseTrieClosestDescendant` returns we push an element here.
627    ///
628    /// The node currently being iterated is either `Btcd` or `MaybeNodeValue` depending on where
629    /// we are in the algorithm.
630    ///
631    /// If the entry at the top of the stack has `children.len() == 16`, then the node currently
632    /// being iterated is the one at the top of the stack, otherwise it's the child of the one at
633    /// the top of the stack whose entry is past the end of `children`.
634    stack: Vec<InProgressNode>,
635
636    /// Same value as [`Config::diff`].
637    diff: TrieDiff,
638
639    /// Same value as [`Config::diff_trie_entries_version`].
640    diff_trie_entries_version: TrieEntryVersion,
641}
642
643#[derive(Debug)]
644struct InProgressNode {
645    /// Partial key of the node currently being calculated.
646    partial_key: Vec<Nibble>,
647
648    /// If `true`, the partial keys of the children of this node have changed compared to their
649    /// values in the base trie.
650    children_partial_key_changed: bool,
651
652    /// Merkle values of the children of the node. Filled up to 16 elements, then the storage
653    /// value is requested. Each element is `Some` or `None` depending on whether a child exists.
654    children: arrayvec::ArrayVec<Option<trie::trie_node::MerkleValueOutput>, 16>,
655}
656
657impl Inner {
658    /// Analyzes the content of the [`Inner`] and progresses the algorithm.
659    fn next(self: Box<Self>) -> InProgress {
660        if self.stack.last().map_or(false, |n| n.children.len() == 16) {
661            // Finished obtaining `MaybeChildren` and jumping to obtaining `MaybeStorageValue`.
662            return InProgress::StorageValue(StorageValue(self));
663        }
664
665        // In the paths below, we want to obtain `MaybeChildren` and thus we're going to return
666        // a `ClosestDescendant`.
667        // Now calculate `ForceCalculate` and
668        // `LongestCommonDenominator(DiffInsertsNodesWithPrefix)`.
669        let (force_recalculate, diff_inserts_lcd) =
670            diff_inserts_least_common_denominator(&self.diff, self.current_node_full_key());
671
672        // If the diff doesn't contain any descendant, jump to calling `BaseTrieMerkleValue`.
673        if !force_recalculate
674            && diff_inserts_lcd.is_none()
675            && self
676                .stack
677                .last()
678                .map_or(true, |n| !n.children_partial_key_changed)
679        {
680            return InProgress::ClosestDescendantMerkleValue(ClosestDescendantMerkleValue {
681                inner: self,
682            });
683        }
684
685        InProgress::ClosestDescendant(ClosestDescendant {
686            inner: self,
687            key_extra_nibbles: Vec::new(),
688            diff_inserts_lcd,
689        })
690    }
691
692    /// Iterator of arrays which, when joined together, form the full key of the node currently
693    /// being iterated.
694    fn current_node_full_key(&self) -> impl Iterator<Item = impl AsRef<[Nibble]>> {
695        self.stack.iter().flat_map(move |node| {
696            let maybe_child_nibble_u8 =
697                u8::try_from(node.children.len()).unwrap_or_else(|_| unreachable!());
698            let child_nibble = Nibble::try_from(maybe_child_nibble_u8).ok();
699
700            iter::once(either::Right(&node.partial_key))
701                .chain(child_nibble.map(|n| either::Left([n])))
702        })
703    }
704}
705
706/// Given a diff and a prefix, returns:
707///
708/// - Whether the diff contains any entry starts with this prefix.
709/// - The least common denominator of all the insertions that start with this prefix, or `None` if
710///   there is no insertion.
711///
712fn diff_inserts_least_common_denominator(
713    diff: &TrieDiff,
714    prefix: impl Iterator<Item = impl AsRef<[Nibble]>>,
715) -> (bool, Option<Vec<Nibble>>) {
716    let mut force_recalculate = false;
717    let mut diff_inserts_lcd = None::<Vec<Nibble>>;
718
719    // TODO: could be optimized?
720    let prefix_nibbles = prefix.fold(Vec::new(), |mut a, b| {
721        a.extend_from_slice(b.as_ref());
722        a
723    });
724    let prefix =
725        trie::nibbles_to_bytes_suffix_extend(prefix_nibbles.iter().copied()).collect::<Vec<_>>();
726    for (key, inserts_entry) in
727        diff.diff_range_ordered::<[u8]>((ops::Bound::Included(&prefix[..]), ops::Bound::Unbounded))
728    {
729        let key = trie::bytes_to_nibbles(key.iter().copied()).collect::<Vec<_>>();
730        if !key.starts_with(&prefix_nibbles) {
731            break;
732        }
733
734        force_recalculate = true;
735
736        if inserts_entry {
737            diff_inserts_lcd = match diff_inserts_lcd.take() {
738                Some(mut v) => {
739                    let lcd = key.iter().zip(v.iter()).take_while(|(a, b)| a == b).count();
740                    debug_assert!(lcd >= prefix_nibbles.len());
741                    debug_assert_eq!(&key[..lcd], &v[..lcd]);
742                    v.truncate(lcd);
743                    Some(v)
744                }
745                None => Some(key),
746            };
747        }
748    }
749
750    (force_recalculate, diff_inserts_lcd)
751}
