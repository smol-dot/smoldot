18//! Parachains and parathreads syncing.
19//!
20//! A parachain is a blockchain whose best and finalized blocks are determined by looking at the
21//! storage of a different chain called the relay chain.
22//!
23//! In order to obtain the current best block of a parachain, you must obtain the parachains
24//! persisted validation data from the relay chain. This is done by calling the
25//! `ParachainHost_persisted_validation_data` runtime function. The runtime function returns
26//! an opaque set of bytes called the "head data" whose meaning depends on the parachain. Most of
27//! the time, it is a block hash.
28//!
29//! In order to obtain the current finalized block of a parachain, do the same but on the current
30//! finalized block of the relay chain.
31//!
32//! See the [`persisted_validation_data_parameters`] to obtain the input to pass to the runtime
33//! function. The first parameter is a `para_id` found in the chain specification of the
34//! parachain of parathread.
35
36/// Produces the input to pass to the `ParachainHost_persisted_validation_data` runtime call.
37pub fn persisted_validation_data_parameters(
38    para_id: u32,
39    assumption: OccupiedCoreAssumption,
40) -> impl Iterator<Item = impl AsRef<[u8]>> + Clone {
41    [
42        either::Left(para_id.to_le_bytes()),
43        either::Right(assumption.scale_encoded()),
44    ]
45    .into_iter()
46}
47
48/// Name of the runtime function to call in order to obtain the parachain heads.
49pub const PERSISTED_VALIDATION_FUNCTION_NAME: &str = "ParachainHost_persisted_validation_data";
50
51/// An assumption being made about the state of an occupied core.
52// TODO: what does that mean?
53#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
54pub enum OccupiedCoreAssumption {
55    /// The candidate occupying the core was made available and included to free the core.
56    Included,
57    /// The candidate occupying the core timed out and freed the core without advancing the para.
58    TimedOut,
59    /// The core was not occupied to begin with.
60    Free,
61}
62
63impl OccupiedCoreAssumption {
64    /// Returns the SCALE encoding of this type.
65    pub fn scale_encoded(&self) -> impl AsRef<[u8]> + Clone + use<> {
66        match self {
67            OccupiedCoreAssumption::Included => [0],
68            OccupiedCoreAssumption::TimedOut => [1],
69            OccupiedCoreAssumption::Free => [2],
70        }
71    }
72}
73
74/// Attempt to decode the return value of the `ParachainHost_persisted_validation_data` runtime
75/// call.
76pub fn decode_persisted_validation_data_return_value(
77    scale_encoded: &'_ [u8],
78    block_number_bytes: usize,
79) -> Result<Option<PersistedValidationDataRef<'_>>, Error> {
80    let res: Result<_, nom::Err<nom::error::Error<_>>> = nom::Parser::parse(
81        &mut nom::combinator::all_consuming(crate::util::nom_option_decode(
82            persisted_validation_data(block_number_bytes),
83        )),
84        scale_encoded,
85    );
86    match res {
87        Ok((_, data)) => Ok(data),
88        Err(nom::Err::Error(err) | nom::Err::Failure(err)) => Err(Error(err.code)),
89        Err(_) => unreachable!(),
90    }
91}

93/// Error that can happen during the decoding.
94#[derive(Debug, derive_more::Display, derive_more::Error)]
95#[display("Error decoding persisted validation data")]
96// TODO: nom doesn't implement the Error trait; remove not(source) at some point
97pub struct Error(#[error(not(source))] nom::error::ErrorKind);

99/// Decoded persisted validation data.
100// TODO: document and explain
101#[derive(Debug, Copy, Clone, PartialEq, Eq)]
102pub struct PersistedValidationDataRef<'a> {
103    /// Opaque data representing the best block (or similar concept) of the parachain/parathread.
104    ///
105    /// The meaning of this data depends on the chain, but for chains built on top of Cumulus
106    /// (i.e. the vast majority of chains) this consists in a block header.
107    pub parent_head: &'a [u8],
108    pub relay_parent_number: u64,
109    pub relay_parent_storage_root: &'a [u8; 32],
110
111    /// Maximum legal size of a POV block, in bytes.
112    pub max_pov_size: u32,
113}
114
115/// `Nom` combinator that parses a [`PersistedValidationDataRef`].
116fn persisted_validation_data<'a, E: nom::error::ParseError<&'a [u8]>>(
117    block_number_bytes: usize,
118) -> impl nom::Parser<&'a [u8], Output = PersistedValidationDataRef<'a>, Error = E> {
119    nom::combinator::map(
120        (
121            crate::util::nom_bytes_decode,
122            crate::util::nom_varsize_number_decode_u64(block_number_bytes),
123            nom::bytes::streaming::take(32u32),
124            nom::number::streaming::le_u32,
125        ),
126        |(parent_head, relay_parent_number, relay_parent_storage_root, max_pov_size)| {
127            PersistedValidationDataRef {
128                parent_head,
129                relay_parent_number,
130                relay_parent_storage_root: <&[u8; 32]>::try_from(relay_parent_storage_root)
131                    .unwrap(),
132                max_pov_size,
133            }
134        },
135    )
136}
