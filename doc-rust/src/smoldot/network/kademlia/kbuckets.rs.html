18//! K-buckets are a collection used to store a partial view of the list of nodes in a
19//! peer-to-peer network.
20//!
21//! # How it works
22//!
23//! The k-buckets consist in 256 so-called "buckets", each containing up to `ENTRIES_PER_BUCKET`
24//! elements. The value for `ENTRIES_PER_BUCKET` is configurable as a parameter of the [`KBuckets`]
25//! struct, but is typically equal to 20. Therefore, the k-buckets cannot contain more than `256 *
26//! ENTRIES_PER_BUCKET` elements in total.
27//!
28//! The API of the [`KBuckets`] struct is similar to the one of a `key => value` map. In addition
29//! to its value, each element also contains a [`PeerState`] indicating whether this node is
30//! connected to the local node.
31//!
32//! In order to insert an element, its key is first hashed using the SHA-256 algorithm, then this
33//! hash is compared with the hash of a `local_key` that was passed at initialization. The
34//! position of the highest non-matching bit in this comparison determines in which of the 256
35//! buckets the element will be inserted. It is forbidden to insert the `local_key` itself.
36//!
37//! Within a bucket, each new element is inserted one after the other, except that the elements
38//! whose [`PeerState`] is [`PeerState::Connected`] are always earlier in the list compared to the
39//! ones whose state is [`PeerState::Disconnected`]. If the state of an element is updated using
40//! [`OccupiedEntry::set_state`], the elements are re-ordered accordingly.
41//!
42//! Each bucket can only contain `ENTRIES_PER_BUCKET` elements. If a bucket is full and contains
43//! only elements in the [`PeerState::Connected`] state, then no new element can be added. If a
44//! bucket is full and contains at least one [`PeerState::Disconnected`] elements, then the last
45//! element in the bucket will expire after a certain time after which it can be replaced with a
46//! new one.
47//!
48//! # Properties
49//!
50//! While this data structure is generic, the `local_key` passed at initialization is typically
51//! the network identity of the local node, and the keys being inserted are typically the network
52//! identities of the other nodes of the peer-to-peer network.
53//!
54//! Assuming that all the network identities that exist are distributed uniformly, the k-buckets
55//! will hold more network identities that are close to the local node's network identity, and
56//! fewer network identities that are far away. In other words, the k-buckets store the neighbors
57//! of the local node, and a few far-away nodes.
58//!
59//! Since all the nodes of the network do the same, one can find all the node whose identity is
60//! closest to a certain key `K` by doing the following:
61//!
62//! - Find in our local k-buckets the node `N` closest to `K`.
63//! - Ask `N` to look into its own k-buckets what is the node closest to `K`.
64//! - If `N` has a node `N2` where `distance(K, N2) < distance(K, N)`, then repeat the previous
65//! step but with `N2`.
66//! - If no closer node is found, we know that `N` is the node closest to `K` in the network.
67//!

73/// K-buckets, as popularized by the Kademlia algorithm, and defined by the libp2p specification.
74pub struct KBuckets<K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
75    /// Key of the "local" node, that holds the buckets.
76    local_key: (K, Key),
77    /// List of buckets, ordered by increasing distance. In other words, the first elements of
78    /// this field are the ones that are the closest to [`KBuckets::local_key`].
79    buckets: Vec<Bucket<K, V, TNow, ENTRIES_PER_BUCKET>>,
80    /// Duration after which the last entry of each bucket will expired if it is disconnected.
81    pending_timeout: Duration,
82}
83
84impl<K, V, TNow, const ENTRIES_PER_BUCKET: usize> KBuckets<K, V, TNow, ENTRIES_PER_BUCKET>
85where
86    K: Clone + PartialEq + AsRef<[u8]>,
87    TNow: Clone + Add<Duration, Output = TNow> + Ord,
88{
89    pub fn new(local_key: K, pending_timeout: Duration) -> Self {
90        let local_key_hashed = Key::new(local_key.as_ref());
91
92        KBuckets {
93            local_key: (local_key, local_key_hashed),
94            buckets: (0..256)
95                .map(|_| Bucket {
96                    entries: arrayvec::ArrayVec::new(),
97                    num_connected_entries: 0,
98                    pending_entry: None,
99                })
100                .collect(),
101            pending_timeout,
102        }
103    }
104
105    /// Returns the local key that was passed to [`KBuckets::new`].
106    pub fn local_key(&self) -> &K {
107        &self.local_key.0
108    }
109
110    /// Returns the value corresponding to the given key. Returns `None` if the key can't be found.
111    pub fn get(&self, key: &K) -> Option<&V> {
112        let key_hashed = Key::new(key.as_ref());
113        let distance = match distance_log2(&self.local_key.1, &key_hashed) {
114            Some(d) => d,
115            None => return None,
116        };
117
118        self.buckets[usize::from(distance)].get(key)
119    }
120
121    /// Returns the value corresponding to the given key. Returns `None` if the key can't be found.
122    pub fn get_mut(&mut self, key: &K) -> Option<&mut V> {
123        let key_hashed = Key::new(key.as_ref());
124        let distance = match distance_log2(&self.local_key.1, &key_hashed) {
125            Some(d) => d,
126            None => return None,
127        };
128
129        self.buckets[usize::from(distance)].get_mut(key)
130    }
131
132    /// Inserts or updates an entry in the buckets.
133    pub fn entry<'a>(&'a mut self, key: &'a K) -> Entry<'a, K, V, TNow, ENTRIES_PER_BUCKET>
134    where
135        K: Clone,
136    {
137        let key_hashed = Key::new(key.as_ref());
138        let distance = match distance_log2(&self.local_key.1, &key_hashed) {
139            Some(d) => d,
140            None => return Entry::LocalKey,
141        };
142
143        if self.buckets[usize::from(distance)].get_mut(key).is_some() {
144            return Entry::Occupied(OccupiedEntry {
145                inner: self,
146                key,
147                distance,
148            });
149        }
150
151        Entry::Vacant(VacantEntry {
152            inner: self,
153            key,
154            distance,
155        })
156    }
157
158    /// Returns the list of entries in the k-buckets, ordered by increasing distance with the
159    /// target.
160    pub fn closest_entries(&self, target: &K) -> impl Iterator<Item = (&K, &V)> {
161        // TODO: this is extremely unoptimized
162        let target_hashed = Key::new(target.as_ref());
163        let mut list = self.iter_ordered().collect::<Vec<_>>();
164        list.sort_by_key(|(key, _)| {
165            let key_hashed = Key::new(key.as_ref());
166            distance_log2(&key_hashed, &target_hashed).map_or(0, |d| u16::from(d) + 1)
167        });
168        list.into_iter()
169    }
170}

172impl<K, V, TNow, const ENTRIES_PER_BUCKET: usize> KBuckets<K, V, TNow, ENTRIES_PER_BUCKET> {
173    /// Iterates over all the peers in the k-buckets.
174    ///
175    /// The buckets are iterated one by one from closest to furthest away, and within each bucket
176    /// elements are ordered by descending time since connectivity.
177    pub fn iter_ordered(&self) -> impl Iterator<Item = (&K, &V)> {
178        self.buckets
179            .iter()
180            .flat_map(|b| b.entries.iter().map(|(k, v)| (k, v)))
181    }
182
183    /// Iterates over all the peers in the k-buckets.
184    ///
185    /// The buckets are iterated one by one from closest to furthest away, and within each bucket
186    /// elements are ordered by descending time since connectivity.
187    pub fn iter_mut_ordered(&mut self) -> impl Iterator<Item = (&K, &mut V)> {
188        self.buckets
189            .iter_mut()
190            .flat_map(|b| b.entries.iter_mut().map(|(k, v)| (&*k, v)))
191    }
192}

205pub enum Entry<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
206    /// Requested key is the same as local key. The local key is never present in the k-buckets.
207    LocalKey,
208    Vacant(VacantEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>),
209    Occupied(OccupiedEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>),
210}
211
212impl<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize> Entry<'a, K, V, TNow, ENTRIES_PER_BUCKET>
213where
214    K: Clone + PartialEq + AsRef<[u8]>,
215    TNow: Clone + Add<Duration, Output = TNow> + Ord,
216{
217    /// If `self` is [`Entry::Occupied`], returns the inner [`OccupiedEntry`]. Otherwise returns
218    /// `None`.
219    pub fn into_occupied(self) -> Option<OccupiedEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>> {
220        match self {
221            Entry::LocalKey | Entry::Vacant(_) => None,
222            Entry::Occupied(e) => Some(e),
223        }
224    }
225
226    pub fn or_insert(
227        self,
228        value: V,
229        now: &TNow,
230        state: PeerState,
231    ) -> Result<OccupiedEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>, OrInsertError> {
232        match self {
233            Entry::LocalKey => Err(OrInsertError::LocalKey),
234            Entry::Vacant(v) => match v.insert(value, now, state) {
235                Ok(InsertResult { entry, .. }) => Ok(entry),
236                Err(InsertError::Full) => Err(OrInsertError::Full),
237            },
238            Entry::Occupied(e) => Ok(e),
239        }
240    }
241}

252pub struct VacantEntry<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
253    inner: &'a mut KBuckets<K, V, TNow, ENTRIES_PER_BUCKET>,
254    key: &'a K,
255    distance: u8,
256}
257
258/// See [`VacantEntry::insert`].
259pub struct InsertResult<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
260    /// Entry that was just inserted.
261    pub entry: OccupiedEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>,
262    /// Entry that had to be removed in order to make space for the newly-inserted element, if
263    /// any. This removed entry was always in a disconnected state.
264    pub removed_entry: Option<(K, V)>,
265}
266
267impl<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize>
268    VacantEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>
269where
270    K: Clone + PartialEq + AsRef<[u8]>,
271    TNow: Clone + Add<Duration, Output = TNow> + Ord,
272{
273    /// Inserts the entry in the vacant slot. Returns an error if the k-buckets are full.
274    pub fn insert(
275        self,
276        value: V,
277        now: &TNow,
278        state: PeerState,
279    ) -> Result<InsertResult<'a, K, V, TNow, ENTRIES_PER_BUCKET>, InsertError> {
280        let bucket = &mut self.inner.buckets[usize::from(self.distance)];
281
282        let removed_entry = match state {
283            PeerState::Connected if bucket.num_connected_entries < ENTRIES_PER_BUCKET => {
284                let mut previous_entry = None;
285
286                if bucket.entries.is_full() {
287                    previous_entry = bucket.entries.pop();
288                    debug_assert!(previous_entry.is_some());
289                    bucket.pending_entry = Some(now.clone() + self.inner.pending_timeout);
290                }
291
292                bucket
293                    .entries
294                    .insert(bucket.num_connected_entries, (self.key.clone(), value));
295                bucket.num_connected_entries += 1;
296
297                if bucket.num_connected_entries == ENTRIES_PER_BUCKET {
298                    bucket.pending_entry = None;
299                }
300
301                previous_entry
302            }
303            PeerState::Connected => {
304                debug_assert!(bucket.entries.is_full());
305                debug_assert_eq!(bucket.num_connected_entries, ENTRIES_PER_BUCKET);
306                debug_assert!(bucket.pending_entry.is_none());
307                return Err(InsertError::Full);
308            }
309            PeerState::Disconnected if bucket.entries.is_full() => {
310                if bucket.num_connected_entries == ENTRIES_PER_BUCKET {
311                    return Err(InsertError::Full);
312                }
313
314                if *bucket.pending_entry.as_ref().unwrap() > *now {
315                    return Err(InsertError::Full);
316                }
317
318                let previous_entry = bucket.entries.pop();
319                bucket.entries.push((self.key.clone(), value));
320                bucket.pending_entry = Some(now.clone() + self.inner.pending_timeout);
321                previous_entry
322            }
323            PeerState::Disconnected => {
324                debug_assert!(!bucket.entries.is_full());
325                debug_assert!(bucket.pending_entry.is_none());
326                bucket.entries.push((self.key.clone(), value));
327
328                if bucket.entries.is_full() {
329                    bucket.pending_entry = Some(now.clone() + self.inner.pending_timeout);
330                }
331
332                None
333            }
334        };
335
336        Ok(InsertResult {
337            entry: OccupiedEntry {
338                inner: self.inner,
339                key: self.key,
340                distance: self.distance,
341            },
342            removed_entry,
343        })
344    }
345}

354pub struct OccupiedEntry<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
355    inner: &'a mut KBuckets<K, V, TNow, ENTRIES_PER_BUCKET>,
356    key: &'a K,
357    distance: u8,
358}
359
360impl<'a, K, V, TNow, const ENTRIES_PER_BUCKET: usize>
361    OccupiedEntry<'a, K, V, TNow, ENTRIES_PER_BUCKET>
362where
363    K: Clone + PartialEq + AsRef<[u8]>,
364    TNow: Clone + Add<Duration, Output = TNow> + Ord,
365{
366    /// Updates the state of this entry.
367    pub fn set_state(&mut self, now: &TNow, state: PeerState) {
368        let bucket = &mut self.inner.buckets[usize::from(self.distance)];
369        let position = bucket
370            .entries
371            .iter()
372            .position(|(k, _)| *k == *self.key)
373            .unwrap();
374
375        match state {
376            PeerState::Connected if position >= bucket.num_connected_entries => {
377                debug_assert!(bucket.num_connected_entries < ENTRIES_PER_BUCKET);
378                let entry = bucket.entries.remove(position);
379                bucket.entries.insert(bucket.num_connected_entries, entry);
380                bucket.num_connected_entries += 1;
381
382                // If the peer we switch from disconnected to connected was the last one, reset
383                // the expiration.
384                if position == bucket.entries.capacity() - 1 {
385                    debug_assert!(bucket.pending_entry.is_some());
386                    bucket.pending_entry = Some(now.clone() + self.inner.pending_timeout);
387                }
388            }
389
390            PeerState::Disconnected if position < bucket.num_connected_entries => {
391                let entry = bucket.entries.remove(position);
392                bucket.num_connected_entries -= 1;
393                bucket.entries.insert(bucket.num_connected_entries, entry);
394
395                // If the peer we switch from connected to disconnected is now the last one, start
396                // the expiration.
397                if bucket.num_connected_entries == bucket.entries.capacity() - 1 {
398                    debug_assert!(bucket.pending_entry.is_none());
399                    bucket.pending_entry = Some(now.clone() + self.inner.pending_timeout);
400                }
401            }
402
403            _ => {}
404        }
405    }
406
407    pub fn get_mut(&mut self) -> &mut V {
408        self.inner.buckets[usize::from(self.distance)]
409            .get_mut(self.key)
410            .unwrap()
411    }
412}

419struct Bucket<K, V, TNow, const ENTRIES_PER_BUCKET: usize> {
420    /// List of entries in the bucket. Ordered by decreasing importance. The first entries in
421    /// the list are the ones we've been connected to for the longest time, while the last entries
422    /// are the ones we've disconnected from for a long time.
423    entries: arrayvec::ArrayVec<(K, V), ENTRIES_PER_BUCKET>,
424    /// Number of entries in the [`Bucket::entries`] that are in the [`PeerState::Connected`]
425    /// state.
426    num_connected_entries: usize,
427    /// If `Some`, the last entry in [`Bucket::entries`] is going to get kicked out after `TNow`.
428    /// Always `None` if [`Bucket::entries`] isn't full or contains only connected entries.
429    pending_entry: Option<TNow>,
430}
431
432impl<K, V, TNow, const ENTRIES_PER_BUCKET: usize> Bucket<K, V, TNow, ENTRIES_PER_BUCKET>
433where
434    K: PartialEq,
435{
436    fn get(&self, key: &K) -> Option<&V> {
437        if let Some((_, value)) = self.entries.iter().find(|e| e.0 == *key) {
438            return Some(value);
439        }
440
441        // TODO: check expiration?
442
443        None
444    }
445
446    fn get_mut(&mut self, key: &K) -> Option<&mut V> {
447        if let Some((_, value)) = self.entries.iter_mut().find(|e| e.0 == *key) {
448            return Some(value);
449        }
450
451        // TODO: check expiration?
452
453        None
454    }
455}

472        Self { digest: hash }
473    }

476/// Returns the `log2` distance between two keys. Returns `None` if the distance is zero.
477fn distance_log2(a: &Key, b: &Key) -> Option<u8> {
478    for n in 0..32 {
479        let a = a.digest[n];
480        let b = b.digest[n];
481        let xor_leading_zeroes = (a ^ b).leading_zeros();
482        if xor_leading_zeroes == 8 {
483            continue;
484        }
485
486        let xor_distance = u32::try_from((31 - n) * 8).unwrap() + (8 - xor_leading_zeroes);
487        debug_assert!(xor_distance > 0);
488        debug_assert!(xor_distance <= 256);
489        return Some(u8::try_from(xor_distance - 1).unwrap());
490    }
491
492    None
493}
