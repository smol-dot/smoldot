25/// Prototype for a Merkle proof whose building is in progress.
26pub struct ProofBuilder {
27    /// Contains a subset of the trie. Each node is associated with its node value if it is known,
28    /// or `None` if it isn't known.
29    ///
30    /// The `TrieStructure` data structure is very explicit in its usage. Nodes such as branch
31    /// nodes are not implicitly but explicitly created. However, the trie structure is normally
32    /// not meant to contain branch nodes without any children. In order to bypass this
33    /// restriction, we pretend that nodes have a storage value when necessary even if this is
34    /// not the case.
35    trie_structure: trie_structure::TrieStructure<Option<Node>>,
36
37    /// List of keys of the nodes in [`ProofBuilder::trie_structure`] whose user data is `None`.
38    missing_node_values: hashbrown::HashSet<Vec<Nibble>, fnv::FnvBuildHasher>,
39}
40
41#[derive(Debug, Clone)]
42struct Node {
43    /// Value passed as `node_value` by the API user.
44    node_value: Vec<u8>,
45    /// Node containing the storage value associated with this node.
46    storage_value_node: Option<Vec<u8>>,
47}
48
49impl ProofBuilder {
50    /// Initializes a new empty proof builder.
51    ///
52    /// This is equivalent to calling [`ProofBuilder::with_nodes_capacity`] with a value of 0.
53    pub fn new() -> Self {
54        Self::with_nodes_capacity(0)
55    }
56
57    /// Initializes a new empty proof builder.
58    ///
59    /// Memory is allocated to store `capacity` nodes, in other words the number of nodes added
60    /// using [`ProofBuilder::set_node_value`].
61    pub fn with_nodes_capacity(capacity: usize) -> Self {
62        ProofBuilder {
63            trie_structure: trie_structure::TrieStructure::with_capacity(capacity),
64            missing_node_values: hashbrown::HashSet::with_capacity_and_hasher(
65                capacity,
66                Default::default(),
67            ),
68        }
69    }
70
71    /// Inserts the node value of a given trie node into the builder.
72    ///
73    /// Overwrites any previously-set value for this key.
74    ///
75    /// The `node_value` is decoded in order for the proof builder to determine the hierarchy of
76    /// the trie and know which node values are missing. If the `node_value` is invalid, this
77    /// function panics. In order to avoid difficult-to-debug corner cases, this function also
78    /// panics if it is no possible for `node_value` to be found at the given `key` due to a
79    /// mismatch.
80    ///
81    /// If the `node_value` contains a storage value as a hash, then a `unhashed_storage_value`
82    /// can optionally be provided in order to provide the unhashed version of this value.
83    ///
84    /// The validity of the `node_value` in the context of the other node values that have been
85    /// stored in this proof builder isn't verified. For example, a node value can indicate no
86    /// children while the node value of a child has been added, or a node value can indicate a
87    /// child with a specific hash, while the child in question has a different hash. This will
88    /// lead to an invalid proof being generated.
89    ///
90    /// # Panic
91    ///
92    /// Panics if `node_value` is not a valid node value.
93    /// Panics if the partial key found in `node_value` doesn't match the last elements of `key`.
94    /// Panics in case `node_value` indicates no storage value, but `unhashed_storage_value`
95    /// is `Some`.
96    ///
97    pub fn set_node_value(
98        &mut self,
99        key: &[Nibble],
100        node_value: &[u8],
101        unhashed_storage_value: Option<&[u8]>,
102    ) {
103        // The first thing to do is decode the node value, in order to detect invalid node values
104        // first things first.
105        let decoded_node_value = match trie_node::decode(node_value) {
106            Ok(d) => d,
107            Err(err) => panic!("failed to decode node value: {err:?}; value: {node_value:?}"),
108        };
109
110        // Check consistency between `node_value` and `unhashed_storage_value` and determine
111        // whether a separate storage node should be included in the proof.
112        let storage_value_node = match (&decoded_node_value.storage_value, &unhashed_storage_value)
113        {
114            (trie_node::StorageValue::Unhashed(in_node_value), Some(user_provided)) => {
115                assert_eq!(in_node_value, user_provided);
116                None
117            }
118            (trie_node::StorageValue::Hashed(_), Some(value)) => Some(value.to_vec()),
119            (trie_node::StorageValue::None, Some(_)) => panic!(),
120            (_, None) => None,
121        };
122
123        // Value that is going to be inserted in the trie.
124        let trie_structure_value = Node {
125            node_value: node_value.to_owned(),
126            storage_value_node,
127        };
128
129        match self.trie_structure.node(key.iter().copied()) {
130            trie_structure::Entry::Occupied(mut entry) => {
131                let _was_in = self.missing_node_values.remove(key);
132                debug_assert_eq!(_was_in, entry.user_data().is_none());
133                *entry.user_data() = Some(trie_structure_value);
134            }
135            trie_structure::Entry::Vacant(entry) => {
136                // We insert a storage value for the given node, even though the node might be
137                // a branch node. This is necessary in order for the trie structure to store our
138                // node, as otherwise the node possibly wouldn't exist.
139                match entry.insert_storage_value() {
140                    trie_structure::PrepareInsert::One(insert) => {
141                        insert.insert(Some(trie_structure_value));
142                    }
143                    trie_structure::PrepareInsert::Two(insert) => {
144                        let _was_inserted = self
145                            .missing_node_values
146                            .insert(insert.branch_node_key().collect());
147                        debug_assert!(_was_inserted);
148                        insert.insert(Some(trie_structure_value), None);
149                    }
150                }
151            }
152        }
153
154        // We must also make sure that the parent of the node is in the proof. Insert the parent
155        // in the trie structure as well.
156        // This shouldn't be done if the node is the root node of the trie.
157        let partial_key_len = decoded_node_value.partial_key.len();
158        assert!(
159            key.len() >= partial_key_len,
160            "mismatch between node value partial key and provided key"
161        );
162        assert!(
163            itertools::equal(
164                key[(key.len() - partial_key_len)..].iter().copied(),
165                decoded_node_value.partial_key.clone()
166            ),
167            "mismatch between node value partial key and provided key"
168        );
169        if key.len() != partial_key_len {
170            let parent_key = &key[..(key.len() - partial_key_len - 1)];
171            match self.trie_structure.node(parent_key.iter().copied()) {
172                trie_structure::Entry::Occupied(_) => {
173                    // The parent is already in the structure. Nothing to do.
174                }
175                trie_structure::Entry::Vacant(entry) => match entry.insert_storage_value() {
176                    trie_structure::PrepareInsert::One(insert) => {
177                        let _was_inserted = self.missing_node_values.insert(parent_key.to_owned());
178                        debug_assert!(_was_inserted);
179                        insert.insert(None);
180                    }
181                    trie_structure::PrepareInsert::Two(insert) => {
182                        let _was_inserted = self.missing_node_values.insert(parent_key.to_owned());
183                        debug_assert!(_was_inserted);
184
185                        let _was_inserted = self
186                            .missing_node_values
187                            .insert(insert.branch_node_key().collect());
188                        debug_assert!(_was_inserted);
189
190                        insert.insert(None, None);
191                    }
192                },
193            }
194        }
195    }
196
197    /// Returns a list of keys for which the node value must be known in order to be able to build
198    /// the proof.
199    ///
200    /// For each entry returned by this iterator, [`ProofBuilder::set_node_value`] must be called.
201    ///
202    /// This function has a complexity of `O(1)` and thus can be called repeatedly.
203    pub fn missing_node_values(&self) -> impl Iterator<Item = &[Nibble]> {
204        self.missing_node_values.iter().map(|v| &v[..])
205    }
206
207    /// Returns the hash of the trie root node.
208    ///
209    /// This function returns `None` if the proof is empty or if the trie root node is missing
210    /// from the proof, in which case [`ProofBuilder::missing_node_values`] will return it.
211    ///
212    /// In other words, if the proof is not empty and if [`ProofBuilder::missing_node_values`]
213    /// returns an empty iterator, then you are guaranteed that this function returns `Some`.
214    ///
215    /// This function has a complexity of `O(1)`.
216    pub fn trie_root_hash(&self) -> Option<[u8; 32]> {
217        let node_value = &self.trie_structure.root_user_data()?.as_ref()?.node_value;
218        Some(blake2_hash(node_value))
219    }
220
221    /// Modifies the node values that have been inserted in the proof builder in order to make the
222    /// proof coherent, if necessary.
223    ///
224    /// If all the node values that have been added through [`ProofBuilder::set_node_value`] come
225    /// from a valid trie, then the proof is already coherent and there is no need to call this
226    /// function. This function is necessary only in situations where the node values have been
227    /// manually modified.
228    ///
229    /// Calling this function when the node values aren't coherent will modify the hash of the trie
230    /// root that is found in the proof. Most of the time, the verification of a trie proof checks
231    /// whether the hash of the trie root in the proof matches an expected value. When that is the
232    /// case, then calling this function would produce a proof that is no longer accepted.
233    ///
234    /// This function only updates the hashes of storage values and children. Storage values
235    /// themselves are always left untouched.
236    /// This function also updates the presence of children. If a node has been added using
237    /// [`ProofBuilder::set_node_value`] but its parent indicates a lack of child, then calling
238    /// this function will modify the parent in order to indicate the presence of a child.
239    /// However, this function never removes children. If a node indicates that it has a child, but
240    /// the child has not been added using [`ProofBuilder::set_node_value`], it simply indicates
241    /// that the child in question is (intentionally) not part of the proof, and the proof is still
242    /// valid.
243    ///
244    /// This function works even if [`ProofBuilder::missing_node_values`] returns a non-empty
245    /// iterator, but will not be able to update everything. You are encouraged to call this
246    /// function only after having added all missing node values;
247    pub fn make_coherent(&mut self) {
248        // The implementation of this function iterates over the nodes of the trie in a specific
249        // order: we start with the deepest child possible of the root node, then we jump from each
250        // node to the deepest child possible of its next sibling. If a node is the last sibling,
251        // jump to its parent.
252        // This order of iteration guarantees that we traverse all the nodes of the trie, and that
253        // we don't traverse a parent before having traversed of all its children.
254        // When traversing a node, we calculate the hash of its children and update the node value
255        // of that node.
256
257        // Node currently being iterated.
258        let mut iter: trie_structure::NodeAccess<_> = {
259            let mut node = match self.trie_structure.root_node() {
260                Some(c) => c,
261                None => {
262                    // Trie is empty.
263                    return;
264                }
265            };
266
267            loop {
268                match node.into_first_child() {
269                    Ok(c) => node = c,
270                    Err(c) => break c,
271                }
272            }
273        };
274
275        loop {
276            // Due to borrowing issues, we calculate ahead of time the values of the children of
277            // the node.
278            let children_values: [Option<Option<arrayvec::ArrayVec<u8, 32>>>; 16] =
279                array::from_fn(|nibble| {
280                    let nibble = nibble::Nibble::try_from(u8::try_from(nibble).unwrap()).unwrap();
281
282                    if let Some(child_node_info) = iter.child_user_data(nibble) {
283                        if let Some(child_node_info) = child_node_info {
284                            // Node values of length < 32 are inlined.
285                            if child_node_info.node_value.len() < 32 {
286                                Some(Some(child_node_info.node_value.iter().copied().collect()))
287                            } else {
288                                Some(Some(blake2_hash(&child_node_info.node_value).into()))
289                            }
290                        } else {
291                            // Missing node value. Don't update anything.
292                            None
293                        }
294                    } else {
295                        Some(None)
296                    }
297                });
298
299            // We only update the current node if its node value has been inserted by the user.
300            if let Some(node_info) = iter.user_data().as_mut() {
301                // We already make sure that node values are valid when inserting them. As such,
302                // it is ok to `unwrap()` here.
303                let mut decoded_node_value = trie_node::decode(&node_info.node_value).unwrap();
304
305                // Update the hash of the storage value contained in `decoded_node_value`.
306                // This is done in a slightly weird way due to borrowing issues.
307                let storage_value_hash = node_info
308                    .storage_value_node
309                    .as_ref()
310                    .map(|v| blake2_hash(v));
311                debug_assert_eq!(
312                    storage_value_hash.is_some(),
313                    matches!(
314                        decoded_node_value.storage_value,
315                        trie_node::StorageValue::Hashed(_)
316                    )
317                );
318                if let Some(storage_value_hash) = storage_value_hash.as_ref() {
319                    decoded_node_value.storage_value =
320                        trie_node::StorageValue::Hashed(storage_value_hash);
321                }
322
323                // Update the children.
324                for (nibble, child_value) in children_values.iter().enumerate() {
325                    if let Some(child_value) = child_value {
326                        decoded_node_value.children[nibble] = child_value.as_deref();
327                    }
328                    // As documented, children are never removed. If `child_value` is `None`, we
329                    // intentionally keep the existing value in `decoded_node_value.children`.
330                }
331
332                // Re-encode the node value after its updates, and store it.
333                // `encode` can return an error only if there's no children and no storage value.
334                // Because we are guaranteed that the node was valid when we decoded it, and that
335                // we only ever add a storage value or add children, we are sure that encoding
336                // can't reach this situation.
337                let updated_node_value =
338                    trie_node::encode(decoded_node_value)
339                        .unwrap()
340                        .fold(Vec::new(), |mut a, b| {
341                            a.extend_from_slice(b.as_ref());
342                            a
343                        });
344                node_info.node_value = updated_node_value;
345            }
346
347            // Jump to the next node in the order of iteration described at the top of this
348            // function.
349            match iter.into_next_sibling() {
350                Err(n) => match n.into_parent() {
351                    Some(p) => iter = p,
352                    None => break, // Finished.
353                },
354                Ok(mut node) => {
355                    iter = loop {
356                        match node.into_first_child() {
357                            Ok(c) => node = c,
358                            Err(c) => break c,
359                        }
360                    };
361                }
362            }
363        }
364    }

366    /// Builds the Merkle proof.
367    ///
368    /// This function returns an iterator of buffers. The actual Merkle proof consists in the
369    /// concatenation of all the buffers.
370    ///
371    /// This function will succeed if no entry at all has been inserted in the [`ProofBuilder`].
372    ///
373    /// This function will succeed even if [`ProofBuilder::missing_node_values`] returns a
374    /// non-zero number of elements. However, the proof produced will then be invalid.
375    pub fn build(mut self) -> impl Iterator<Item = impl AsRef<[u8]> + Clone> {
376        // Index of the root node in the trie, if any.
377        let root_node_index = self.trie_structure.root_node().map(|n| n.node_index());
378
379        // Collect the entries in the proof into a `HashSet` in order to de-duplicate them.
380        // TODO: we need to collect the indices into a Vec due to the API of trie_structure not allowing non-mutable access to nodes
381        let entries = self
382            .trie_structure
383            .iter_unordered()
384            .collect::<Vec<_>>()
385            .into_iter()
386            .flat_map(move |node_index| {
387                let Some(trie_structure_value) = self
388                    .trie_structure
389                    .node_by_index(node_index)
390                    .unwrap()
391                    .user_data()
392                    .take()
393                // Ignore nodes whose value is missing.
394                else {
395                    return either::Right(iter::empty());
396                };
397
398                // Nodes of length < 32 should have been inlined within their parent or ancestor.
399                // We thus skip them, unless they're the root node.
400                if root_node_index != Some(node_index) && trie_structure_value.node_value.len() < 32
401                {
402                    debug_assert!(trie_structure_value.storage_value_node.is_none());
403                    return either::Right(iter::empty());
404                }
405
406                // For each node, there are either one or two things to output: the node value,
407                // and the storage value.
408                either::Left(
409                    iter::once(trie_structure_value.node_value)
410                        .chain(trie_structure_value.storage_value_node),
411                )
412            })
413            .collect::<hashbrown::HashSet<_, fnv::FnvBuildHasher>>();
414
415        // The first bytes of the proof contain the number of entries in the proof.
416        let num_entries_encoded = crate::util::encode_scale_compact_usize(entries.len());
417
418        // Add the size of each entry before each entry.
419        let entries = entries.into_iter().flat_map(|entry| {
420            let len = crate::util::encode_scale_compact_usize(entry.len());
421            [either::Left(len), either::Right(entry)].into_iter()
422        });
423
424        iter::once(either::Left(num_entries_encoded)).chain(entries.into_iter().map(either::Right))
425    }
426
427    /// Similar to [`ProofBuilder::build`], but returns a `Vec`.
428    ///
429    /// This is a convenience wrapper around [`ProofBuilder::build`].
430    pub fn build_to_vec(self) -> Vec<u8> {
431        self.build().fold(Vec::new(), |mut a, b| {
432            a.extend_from_slice(b.as_ref());
433            a
434        })
435    }
436}

444fn blake2_hash(data: &[u8]) -> [u8; 32] {
445    <[u8; 32]>::try_from(blake2_rfc::blake2b::blake2b(32, &[], data).as_bytes()).unwrap()
446}

522        // This test builds a proof of a randomly-generated trie, then fixes it, then checks
523        // whether the decoder considers the proof as valid.
524
525        // We repeat the test many times due to its random factor.
526        for _ in 0..1500 {
527            // Build a trie with entries with randomly generated keys.
528            let mut trie = trie_structure::TrieStructure::new();
529            // Generate at least one node, as otherwise the test panics. Empty proofs are however
530            // valid.
531            for _ in 0..Uniform::new_inclusive(1, 32).sample(&mut rand::thread_rng()) {
532                let mut key = Vec::new();
533                for _ in 0..Uniform::new_inclusive(0, 12).sample(&mut rand::thread_rng()) {
534                    key.push(
535                        nibble::Nibble::try_from(
536                            Uniform::new_inclusive(0, 15).sample(&mut rand::thread_rng()),
537                        )
538                        .unwrap(),
539                    );
540                }
541
542                match trie.node(key.into_iter()) {
543                    trie_structure::Entry::Vacant(e) => {
544                        e.insert_storage_value().insert((), ());
545                    }
546                    trie_structure::Entry::Occupied(trie_structure::NodeAccess::Branch(e)) => {
547                        e.insert_storage_value();
548                    }
549                    trie_structure::Entry::Occupied(trie_structure::NodeAccess::Storage(_)) => {}
550                }
551            }
552
553            // Put the content of the trie into the proof builder.
554            let mut proof_builder = super::ProofBuilder::new();
555            for node_index in trie.iter_unordered().collect::<Vec<_>>() {
556                let key = trie
557                    .node_full_key_by_index(node_index)
558                    .unwrap()
559                    .collect::<Vec<_>>();
560
561                // This randomly-generated storage might end up not being used, but that's not
562                // problematic.
563                let mut storage_value = Vec::new();
564                for _ in 0..Uniform::new_inclusive(0, 64).sample(&mut rand::thread_rng()) {
565                    storage_value
566                        .push(Uniform::new_inclusive(0, 255).sample(&mut rand::thread_rng()));
567                }
568
569                let node_value = trie_node::encode_to_vec(trie_node::Decoded {
570                    children: array::from_fn(|nibble| {
571                        let nibble =
572                            nibble::Nibble::try_from(u8::try_from(nibble).unwrap()).unwrap();
573                        if trie
574                            .node_by_index(node_index)
575                            .unwrap()
576                            .child_user_data(nibble)
577                            .is_some()
578                        {
579                            Some(&[][..])
580                        } else {
581                            None
582                        }
583                    }),
584                    partial_key: trie
585                        .node_by_index(node_index)
586                        .unwrap()
587                        .partial_key()
588                        .collect::<Vec<_>>()
589                        .into_iter(),
590                    storage_value: if trie.node_by_index(node_index).unwrap().has_storage_value() {
591                        trie_node::StorageValue::Unhashed(&storage_value)
592                    } else {
593                        trie_node::StorageValue::None
594                    },
595                })
596                .unwrap();
597
598                proof_builder.set_node_value(&key, &node_value, None);
599            }
600
601            // Generate the proof.
602            assert!(proof_builder.missing_node_values().next().is_none());
603            proof_builder.make_coherent();
604            let trie_root_hash = proof_builder.trie_root_hash().unwrap();
605            let proof = proof_builder.build_to_vec();
606
607            // Verify the correctness of the proof.
608            let proof =
609                proof_decode::decode_and_verify_proof(proof_decode::Config { proof }).unwrap();
610            assert!(
611                proof
612                    .closest_descendant_merkle_value(&trie_root_hash, iter::empty())
613                    .is_ok()
614            );
615        }
616    }
