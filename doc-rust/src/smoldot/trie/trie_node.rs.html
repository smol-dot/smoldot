22/// Encodes the components of a node value into the node value itself.
23///
24/// This function returns an iterator of buffers. The actual node value is the concatenation of
25/// these buffers put together.
26///
27/// > **Note**: The returned iterator might contain a reference to the storage value and children
28/// >           values in the [`Decoded`]. By returning an iterator of buffers, we avoid copying
29/// >           these storage value and children values.
30///
31/// This encoding is independent of the trie version.
32pub fn encode<'a>(
33    decoded: Decoded<
34        'a,
35        impl ExactSizeIterator<Item = nibble::Nibble> + Clone,
36        impl AsRef<[u8]> + Clone + 'a,
37    >,
38) -> Result<impl Iterator<Item = impl AsRef<[u8]> + Clone> + Clone, EncodeError> {
39    // The return value is composed of three parts:
40    // - Before the storage value.
41    // - The storage value (which can be empty).
42    // - The children nodes.
43
44    // Contains the encoding before the storage value.
45    let mut before_storage_value: Vec<u8> = Vec::with_capacity(decoded.partial_key.len() / 2 + 32);
46
47    let has_children = decoded.children.iter().any(Option::is_some);
48
49    // We first push the node header.
50    // See https://spec.polkadot.network/#defn-node-header
51    {
52        let (first_byte_msb, pk_len_first_byte_bits): (u8, _) =
53            match (has_children, decoded.storage_value) {
54                (false, StorageValue::Unhashed(_)) => (0b01, 6),
55                (true, StorageValue::None) => (0b10, 6),
56                (true, StorageValue::Unhashed(_)) => (0b11, 6),
57                (false, StorageValue::Hashed(_)) => (0b001, 5),
58                (true, StorageValue::Hashed(_)) => (0b0001, 4),
59                (false, StorageValue::None) => {
60                    if decoded.partial_key.len() != 0 {
61                        return Err(EncodeError::PartialKeyButNoChildrenNoStorageValue);
62                    } else {
63                        (0, 6)
64                    }
65                }
66            };
67
68        let max_representable_in_first_byte = (1 << pk_len_first_byte_bits) - 1;
69        let first_byte = (first_byte_msb << pk_len_first_byte_bits)
70            | u8::try_from(cmp::min(
71                decoded.partial_key.len(),
72                max_representable_in_first_byte,
73            ))
74            .unwrap();
75        before_storage_value.push(first_byte);
76
77        // Note that if the partial key length is exactly equal to `pk_len_first_byte_bits`, we
78        // need to push a `0` afterwards in order to avoid an ambiguity. Similarly, if
79        // `remain_pk_len` is at any point equal to 255, we must push an additional `0`
80        // afterwards.
81        let mut remain_pk_len = decoded
82            .partial_key
83            .len()
84            .checked_sub(max_representable_in_first_byte);
85        while let Some(pk_len_inner) = remain_pk_len {
86            before_storage_value.push(u8::try_from(cmp::min(pk_len_inner, 255)).unwrap());
87            remain_pk_len = pk_len_inner.checked_sub(255);
88        }
89    }
90
91    // We then push the partial key.
92    before_storage_value.extend(nibble::nibbles_to_bytes_prefix_extend(
93        decoded.partial_key.clone(),
94    ));
95
96    // After the partial key, the node value optionally contains a bitfield of child nodes.
97    if has_children {
98        before_storage_value.extend_from_slice(&decoded.children_bitmap().to_le_bytes());
99    }
100
101    // Then, the storage value.
102    let storage_value = match decoded.storage_value {
103        StorageValue::Hashed(hash) => &hash[..],
104        StorageValue::None => &[][..],
105        StorageValue::Unhashed(storage_value) => {
106            before_storage_value.extend_from_slice(
107                crate::util::encode_scale_compact_usize(storage_value.len()).as_ref(),
108            );
109            storage_value
110        }
111    };
112
113    // Finally, the children node values.
114    let children_nodes = decoded
115        .children
116        .into_iter()
117        .flatten()
118        .flat_map(|child_value| {
119            let size = crate::util::encode_scale_compact_usize(child_value.as_ref().len());
120            [either::Left(size), either::Right(child_value)].into_iter()
121        });
122
123    // The return value is the combination of these components.
124    Ok(iter::once(either::Left(before_storage_value))
125        .chain(iter::once(either::Right(storage_value)))
126        .map(either::Left)
127        .chain(children_nodes.map(either::Right)))
128}

130/// Error potentially returned by [`encode`].
131#[derive(Debug, derive_more::Display, derive_more::Error, Clone)]
132pub enum EncodeError {
133    /// Nodes that have no children nor storage value are invalid unless they are the root node.
134    PartialKeyButNoChildrenNoStorageValue,
135}

137/// Encodes the components of a node value into the node value itself.
138///
139/// This is a convenient wrapper around [`encode`]. See the documentation of [`encode`] for more
140/// details.
141pub fn encode_to_vec(
142    decoded: Decoded<
143        '_,
144        impl ExactSizeIterator<Item = nibble::Nibble> + Clone,
145        impl AsRef<[u8]> + Clone,
146    >,
147) -> Result<Vec<u8>, EncodeError> {
148    let capacity = decoded.partial_key.len() / 2
149        + match decoded.storage_value {
150            StorageValue::Hashed(_) => 32,
151            StorageValue::None => 0,
152            StorageValue::Unhashed(v) => v.len(),
153        }
154        + 16 * 32
155        + 32; // The last `+ 32` is an arbitrary margin, for length prefixes and the header.
156
157    let result = encode(decoded)?.fold(Vec::with_capacity(capacity), |mut a, b| {
158        a.extend_from_slice(b.as_ref());
159        a
160    });
161
162    // Check that `capacity` was calculated correctly and that no re-allocation has happened.
163    debug_assert_eq!(result.capacity(), capacity);
164
165    Ok(result)
166}

168/// Calculates the Merkle value of the given node.
169///
170/// `is_root_node` must be `true` if the encoded node is the root node of the trie.
171///
172/// This is similar to [`encode`], except that the encoding is then optionally hashed.
173///
174/// Hashing is performed if the encoded value is 32 bytes or more, or if `is_root_node` is `true`.
175/// This is the reason why `is_root_node` must be provided.
176pub fn calculate_merkle_value(
177    decoded: Decoded<
178        '_,
179        impl ExactSizeIterator<Item = nibble::Nibble> + Clone,
180        impl AsRef<[u8]> + Clone,
181    >,
182    hash_function: HashFunction,
183    is_root_node: bool,
184) -> Result<MerkleValueOutput, EncodeError> {
185    /// The Merkle value of a node is defined as either the hash of the node value, or the node value
186    /// itself if it is shorted than 32 bytes (or if we are the root).
187    ///
188    /// This struct serves as a helper to handle these situations. Rather than putting intermediary
189    /// values in buffers then hashing the node value as a whole, we push the elements of the node
190    /// value to this struct which automatically switches to hashing if the value exceeds 32 bytes.
191    enum HashOrInline {
192        Inline(arrayvec::ArrayVec<u8, 31>),
193        Blake2Hasher(blake2_rfc::blake2b::Blake2b),
194        Keccak256Hasher(sha3::Keccak256),
195    }
196
197    let mut merkle_value_sink = match (is_root_node, hash_function) {
198        (true, HashFunction::Blake2) => {
199            HashOrInline::Blake2Hasher(blake2_rfc::blake2b::Blake2b::new(32))
200        }
201        (true, HashFunction::Keccak256) => {
202            HashOrInline::Keccak256Hasher(<sha3::Keccak256 as sha3::Digest>::new())
203        }
204        (false, _) => HashOrInline::Inline(arrayvec::ArrayVec::new()),
205    };
206
207    for buffer in encode(decoded)? {
208        let buffer = buffer.as_ref();
209        match &mut merkle_value_sink {
210            HashOrInline::Inline(curr) => {
211                if curr.try_extend_from_slice(buffer).is_ok() {
212                    continue;
213                }
214
215                match hash_function {
216                    HashFunction::Blake2 => {
217                        let mut hasher = blake2_rfc::blake2b::Blake2b::new(32);
218                        hasher.update(curr);
219                        hasher.update(buffer);
220                        merkle_value_sink = HashOrInline::Blake2Hasher(hasher);
221                    }
222                    HashFunction::Keccak256 => {
223                        let mut hasher = <sha3::Keccak256 as sha3::Digest>::new();
224                        sha3::Digest::update(&mut hasher, curr);
225                        sha3::Digest::update(&mut hasher, buffer);
226                        merkle_value_sink = HashOrInline::Keccak256Hasher(hasher);
227                    }
228                }
229            }
230            HashOrInline::Blake2Hasher(hasher) => {
231                hasher.update(buffer);
232            }
233            HashOrInline::Keccak256Hasher(hasher) => {
234                sha3::Digest::update(hasher, buffer);
235            }
236        }
237    }
238
239    Ok(MerkleValueOutput {
240        inner: match merkle_value_sink {
241            HashOrInline::Inline(b) => MerkleValueOutputInner::Inline(b),
242            HashOrInline::Blake2Hasher(h) => MerkleValueOutputInner::Blake2Hasher(h.finalize()),
243            HashOrInline::Keccak256Hasher(h) => {
244                MerkleValueOutputInner::Keccak256Hasher(sha3::Digest::finalize(h).into())
245            }
246        },
247    })
248}

250/// Output of the calculation.
251#[derive(Clone)]
252pub struct MerkleValueOutput {
253    inner: MerkleValueOutputInner,
254}

256#[derive(Clone)]
257enum MerkleValueOutputInner {
258    Inline(arrayvec::ArrayVec<u8, 31>),
259    Blake2Hasher(blake2_rfc::blake2b::Blake2bResult),
260    Keccak256Hasher([u8; 32]),
261    Bytes(arrayvec::ArrayVec<u8, 32>),
262}

264impl MerkleValueOutput {
265    /// Builds a [`MerkleValueOutput`] from a slice of bytes.
266    ///
267    /// # Panic
268    ///
269    /// Panics if `bytes.len() > 32`.
270    ///
271    pub fn from_bytes(bytes: &[u8]) -> MerkleValueOutput {
272        assert!(bytes.len() <= 32);
273        MerkleValueOutput {
274            inner: MerkleValueOutputInner::Bytes({
275                let mut v = arrayvec::ArrayVec::new();
276                v.try_extend_from_slice(bytes).unwrap();
277                v
278            }),
279        }
280    }
281}

283impl AsRef<[u8]> for MerkleValueOutput {
284    fn as_ref(&self) -> &[u8] {
285        match &self.inner {
286            MerkleValueOutputInner::Inline(a) => a.as_slice(),
287            MerkleValueOutputInner::Blake2Hasher(a) => a.as_bytes(),
288            MerkleValueOutputInner::Keccak256Hasher(a) => &a[..],
289            MerkleValueOutputInner::Bytes(a) => a.as_slice(),
290        }
291    }

314/// Decodes a node value found in a proof into its components.
315///
316/// This can decode nodes no matter their version or hash algorithm.
317pub fn decode(
318    mut node_value: &'_ [u8],
319) -> Result<Decoded<'_, DecodedPartialKey<'_>, &'_ [u8]>, Error> {
320    if node_value.is_empty() {
321        return Err(Error::Empty);
322    }
323
324    // See https://spec.polkadot.network/#defn-node-header
325    let (has_children, storage_value_hashed, pk_len_first_byte_bits) = match node_value[0] >> 6 {
326        0b00 => {
327            if (node_value[0] >> 5) == 0b001 {
328                (false, Some(true), 5)
329            } else if (node_value[0] >> 4) == 0b0001 {
330                (true, Some(true), 4)
331            } else if node_value[0] == 0 {
332                (false, None, 6)
333            } else {
334                return Err(Error::InvalidHeaderBits);
335            }
336        }
337        0b10 => (true, None, 6),
338        0b01 => (false, Some(false), 6),
339        0b11 => (true, Some(false), 6),
340        _ => unreachable!(),
341    };
342
343    // Length of the partial key, in nibbles.
344    let pk_len = {
345        let mut accumulator = usize::from(node_value[0] & ((1 << pk_len_first_byte_bits) - 1));
346        node_value = &node_value[1..];
347        let mut continue_iter = accumulator == ((1 << pk_len_first_byte_bits) - 1);
348        while continue_iter {
349            if node_value.is_empty() {
350                return Err(Error::PartialKeyLenTooShort);
351            }
352            continue_iter = node_value[0] == 255;
353            accumulator = accumulator
354                .checked_add(usize::from(node_value[0]))
355                .ok_or(Error::PartialKeyLenOverflow)?;
356            node_value = &node_value[1..];
357        }
358        accumulator
359    };
360
361    // No children and no storage value can only indicate the root of an empty trie, in which case
362    // a non-empty partial key is invalid.
363    if pk_len != 0 && !has_children && storage_value_hashed.is_none() {
364        return Err(Error::EmptyTrieWithPartialKey);
365    }
366
367    // Iterator to the partial key found in the node value of `proof_iter`.
368    let partial_key = {
369        // Length of the partial key, in bytes.
370        let pk_len_bytes = if pk_len == 0 {
371            0
372        } else {
373            1 + ((pk_len - 1) / 2)
374        };
375        if node_value.len() < pk_len_bytes {
376            return Err(Error::PartialKeyTooShort);
377        }
378
379        let pk = &node_value[..pk_len_bytes];
380        node_value = &node_value[pk_len_bytes..];
381
382        if (pk_len % 2) == 1 && (pk[0] & 0xf0) != 0 {
383            return Err(Error::InvalidPartialKeyPadding);
384        }
385
386        pk
387    };
388
389    // After the partial key, the node value optionally contains a bitfield of child nodes.
390    let children_bitmap = if has_children {
391        if node_value.len() < 2 {
392            return Err(Error::ChildrenBitmapTooShort);
393        }
394        let val = u16::from_le_bytes(<[u8; 2]>::try_from(&node_value[..2]).unwrap());
395        if val == 0 {
396            return Err(Error::ZeroChildrenBitmap);
397        }
398        node_value = &node_value[2..];
399        val
400    } else {
401        0
402    };
403
404    // Now at the value that interests us.
405    let storage_value = match storage_value_hashed {
406        Some(false) => {
407            let (node_value_update, len) = crate::util::nom_scale_compact_usize(node_value)
408                .map_err(|_: nom::Err<nom::error::Error<&[u8]>>| Error::StorageValueLenDecode)?;
409            node_value = node_value_update;
410            if node_value.len() < len {
411                return Err(Error::StorageValueTooShort);
412            }
413            let storage_value = &node_value[..len];
414            node_value = &node_value[len..];
415            StorageValue::Unhashed(storage_value)
416        }
417        Some(true) => {
418            if node_value.len() < 32 {
419                return Err(Error::StorageValueTooShort);
420            }
421            let storage_value_hash = <&[u8; 32]>::try_from(&node_value[..32]).unwrap();
422            node_value = &node_value[32..];
423            StorageValue::Hashed(storage_value_hash)
424        }
425        None => StorageValue::None,
426    };
427
428    let mut children = [None; 16];
429    for (n, child) in children.iter_mut().enumerate() {
430        if children_bitmap & (1 << n) == 0 {
431            continue;
432        }
433
434        // Find the Merkle value of that child in `node_value`.
435        let (node_value_update, len) = crate::util::nom_scale_compact_usize(node_value)
436            .map_err(|_: nom::Err<nom::error::Error<&[u8]>>| Error::ChildLenDecode)?;
437        if len > 32 {
438            return Err(Error::ChildTooLarge);
439        }
440        node_value = node_value_update;
441        if node_value.len() < len {
442            return Err(Error::ChildrenTooShort);
443        }
444
445        *child = Some(&node_value[..len]);
446        node_value = &node_value[len..];
447    }
448
449    if !node_value.is_empty() {
450        return Err(Error::TooLong);
451    }
452
453    Ok(Decoded {
454        partial_key: if (pk_len % 2) == 1 {
455            DecodedPartialKey::from_bytes_skip_first(partial_key)
456        } else {
457            DecodedPartialKey::from_bytes(partial_key)
458        },
459        children,
460        storage_value,
461    })
462}

464/// Decoded node value. Returned by [`decode`] or passed as parameter to [`encode`].
465#[derive(Debug, Clone)]
466pub struct Decoded<'a, I, C> {
467    /// Iterator to the nibbles of the partial key of the node.
468    pub partial_key: I,
469
470    /// All 16 possible children. `Some` if a child is present, and `None` otherwise. The `&[u8]`
471    /// can be:
472    ///
473    /// - Of length 32, in which case the slice is the hash of the node value of the child (also
474    ///   known as the Merkle value).
475    /// - Empty when decoding a compact trie proof.
476    /// - Of length inferior to 32, in which case the slice is directly the node value.
477    ///
478    pub children: [Option<C>; 16],
479
480    /// Storage value of this node.
481    pub storage_value: StorageValue<'a>,
482}
483
484impl<'a, I, C> Decoded<'a, I, C> {
485    /// Returns a bits map of the children that are present, as found in the node value.
486    pub fn children_bitmap(&self) -> u16 {
487        let mut out = 0u16;
488        for n in 0..16 {
489            if self.children[n].is_none() {
490                continue;
491            }
492            out |= 1 << n;
493        }
494        out
495    }
496}

498/// See [`Decoded::storage_value`].
499#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
500pub enum StorageValue<'a> {
501    /// Storage value of the item is present in the node value.
502    Unhashed(&'a [u8]),
503    /// Hash of the storage value of the item is present in the node value.
504    Hashed(&'a [u8; 32]),
505    /// Item doesn't have any storage value.
506    None,
507}

509/// Iterator to the nibbles of the partial key. See [`Decoded::partial_key`].
510#[derive(Clone)]
511pub struct DecodedPartialKey<'a> {
512    inner: nibble::BytesToNibbles<iter::Copied<slice::Iter<'a, u8>>>,
513    skip_first: bool,
514}
515
516impl<'a> DecodedPartialKey<'a> {
517    /// Returns a [`DecodedPartialKey`] iterator that produces the nibbles encoded as the given
518    /// bytes. Each byte is turned into two nibbles.
519    ///
520    /// > **Note**: This function is a convenient wrapper around [`nibble::bytes_to_nibbles`].
521    pub fn from_bytes(bytes: &'a [u8]) -> Self {
522        DecodedPartialKey {
523            inner: nibble::bytes_to_nibbles(bytes.iter().copied()),
524            skip_first: false,
525        }
526    }
527
528    /// Equivalent to [`DecodedPartialKey::from_bytes`], but skips the first nibble.
529    ///
530    /// This is useful for situations where the partial key contains a `0` prefix that exists for
531    /// alignment but doesn't actually represent a nibble.
532    ///
533    /// > **Note**: This is equivalent to `from_bytes(bytes).skip(1)`. The possibility to skip the
534    /// >           first nibble is built into this code due to how frequent it is necessary.
535    pub fn from_bytes_skip_first(bytes: &'a [u8]) -> Self {
536        DecodedPartialKey {
537            inner: nibble::bytes_to_nibbles(bytes.iter().copied()),
538            skip_first: true,
539        }
540    }
541}
542
543impl<'a> Iterator for DecodedPartialKey<'a> {
544    type Item = nibble::Nibble;
545
546    fn next(&mut self) -> Option<nibble::Nibble> {
547        loop {
548            let nibble = self.inner.next()?;
549            if self.skip_first {
550                debug_assert_eq!(u8::from(nibble), 0);
551                self.skip_first = false;
552                continue;
553            }
554            break Some(nibble);
555        }
556    }
557
558    fn size_hint(&self) -> (usize, Option<usize>) {
559        let mut len = self.inner.len();
560        if self.skip_first {
561            len -= 1;
562        }
563        (len, Some(len))
564    }
565}

569impl<'a> fmt::Debug for DecodedPartialKey<'a> {
570    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
571        const HEX_TABLE: &[u8] = b"0123456789abcdef";
572        write!(f, "0x")?;
573        for nibble in self.clone() {
574            let chr = HEX_TABLE[usize::from(u8::from(nibble))];
575            write!(f, "{}", char::from(chr))?;
576        }
577        Ok(())
578    }

581/// Possible error returned by [`decode`].
582#[derive(Debug, Clone, derive_more::Display, derive_more::Error)]
583pub enum Error {
584    /// Node value is empty.
585    Empty,
586    /// Bits in the header have an invalid format.
587    InvalidHeaderBits,
588    /// Node value ends while parsing partial key length.
589    PartialKeyLenTooShort,
590    /// Length of partial key is too large to be reasonable.
591    PartialKeyLenOverflow,
592    /// Node value ends within partial key.
593    PartialKeyTooShort,
594    /// If partial key is of uneven length, then it must be padded with `0`.
595    InvalidPartialKeyPadding,
596    /// End of data within the children bitmap.
597    ChildrenBitmapTooShort,
598    /// The children bitmap is equal to 0 despite the header indicating the presence of children.
599    ZeroChildrenBitmap,
600    /// Error while decoding length of child.
601    ChildLenDecode,
602    /// Node value ends within a child value.
603    ChildrenTooShort,
604    /// Child value is superior to 32 bytes.
605    ChildTooLarge,
606    /// Error while decoding length of storage value.
607    StorageValueLenDecode,
608    /// Node value ends within the storage value.
609    StorageValueTooShort,
610    /// Node value is longer than expected.
611    TooLong,
612    /// Node value indicates that it is the root of an empty trie but contains a non-empty partial
613    /// key.
614    EmptyTrieWithPartialKey,
615}
