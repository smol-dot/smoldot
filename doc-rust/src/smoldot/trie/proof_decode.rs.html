18//! Decodes and verifies a trie proof.
19//!
20//! A trie proof is a proof that a certain key in the trie has a certain storage value (or lacks
21//! a storage value). The proof can be verified by knowing only the Merkle value of the root node.
22//!
23//! # Details
24//!
25//! > **Note**: For reminder, the Merkle value of a node is the hash of its node value, or the
26//! >           node value directly if its length is smaller than 32 bytes.
27//!
28//! A trie proof consists in a list of node values of nodes in the trie. For the proof to be valid,
29//! the hash of one of these node values must match the expected trie root node value. Since a
30//! node value contains the Merkle values of the children of the node, it is possible to iterate
31//! down the hierarchy of nodes until the one closest to the desired key is found.
32//!
33//! # Usage
34//!
35//! This modules provides the [`decode_and_verify_proof`] function that decodes a proof and
36//! verifies whether it is correct.
37//!
38//! Once decoded, one can examine the content of the proof, in other words the list of storage
39//! items and values.

48/// Configuration to pass to [`decode_and_verify_proof`].
49pub struct Config<I> {
50    /// List of node values of nodes found in the trie. At least one entry corresponding to the
51    /// root node of the trie must be present in order for the verification to succeed.
52    pub proof: I,
53}

55/// Verifies whether a proof is correct and returns an object that allows examining its content.
56///
57/// The proof is then stored within the [`DecodedTrieProof`].
58///
59/// Due to the generic nature of this function, the proof can be either a `Vec<u8>` or a `&[u8]`.
60///
61/// Returns an error if the proof is invalid, or if the proof contains entries that are
62/// disconnected from the root node of the trie.
63pub fn decode_and_verify_proof<T>(config: Config<T>) -> Result<DecodedTrieProof<T>, Error>
64where
65    T: AsRef<[u8]>,
66{
67    // Call `as_ref()` once at the beginning in order to guarantee stability of the memory
68    // location.
69    let proof_as_ref = config.proof.as_ref();
70
71    struct InProgressEntry<'a> {
72        index_in_proof: usize,
73        range_in_proof: ops::Range<usize>,
74        decode_result: Result<
75            trie_node::Decoded<'a, trie_node::DecodedPartialKey<'a>, &'a [u8]>,
76            trie_node::Error,
77        >,
78    }
79
80    // A Merkle proof is a SCALE-encoded `Vec<Vec<u8>>`.
81    //
82    // This `Vec` contains two types of items: trie node values, and standalone storage items. In
83    // both cases, we will later need a hashed version of them. Create a list of hashes, one per
84    // entry in `proof`.
85    //
86    // This hashmap uses a FNV hasher, theoretically vulnerable to HashDos attacks. While it is
87    // possible for an attacker to craft a proof that leads to all entries being in the same
88    // bucket, this proof is going to be invalid (unless the blake2 hash function is broken, which
89    // we assume it isn't). So while an attacker can slightly increase the time that this function
90    // takes, it is always cause this function to return an error and is actually likely to make
91    // the function actually take less time than if it was a legitimate proof.
92    let entries_by_merkle_value = {
93        // TODO: don't use a Vec?
94        let (_, decoded_proof) = nom::Parser::parse(
95            &mut nom::combinator::all_consuming(nom::combinator::flat_map(
96                crate::util::nom_scale_compact_usize,
97                |num_elems| {
98                    nom::multi::many_m_n(num_elems, num_elems, crate::util::nom_bytes_decode)
99                },
100            )),
101            config.proof.as_ref(),
102        )
103        .map_err(|_: nom::Err<nom::error::Error<&[u8]>>| Error::InvalidFormat)?;
104
105        let entries_by_merkle_value = decoded_proof
106            .iter()
107            .copied()
108            .enumerate()
109            .map(
110                |(proof_entry_num, proof_entry)| -> ([u8; 32], InProgressEntry) {
111                    // The merkle value of a trie node is normally either its hash or the node
112                    // itself if its length is < 32. In the context of a proof, however, nodes
113                    // whose length is < 32 aren't supposed to be their own entry. For this reason,
114                    // we only hash each entry.
115                    let hash = *<&[u8; 32]>::try_from(
116                        blake2_rfc::blake2b::blake2b(32, &[], proof_entry).as_bytes(),
117                    )
118                    .unwrap();
119
120                    let proof_entry_offset = if proof_entry.is_empty() {
121                        0
122                    } else {
123                        proof_entry.as_ptr() as usize - proof_as_ref.as_ptr() as usize
124                    };
125
126                    (
127                        hash,
128                        InProgressEntry {
129                            index_in_proof: proof_entry_num,
130                            range_in_proof: proof_entry_offset
131                                ..(proof_entry_offset + proof_entry.len()),
132                            decode_result: trie_node::decode(proof_entry),
133                        },
134                    )
135                },
136            )
137            .collect::<hashbrown::HashMap<_, _, fnv::FnvBuildHasher>>();
138
139        // Using a hashmap has the consequence that if multiple proof entries were identical, only
140        // one would be tracked. This allows us to make sure that the proof doesn't contain
141        // multiple identical entries.
142        if entries_by_merkle_value.len() != decoded_proof.len() {
143            return Err(Error::DuplicateProofEntry);
144        }
145
146        entries_by_merkle_value
147    };
148
149    // Start by iterating over each element of the proof, and keep track of elements that are
150    // decodable but aren't mentioned in any other element. This gives us the tries roots.
151    let trie_roots = {
152        let mut maybe_trie_roots = entries_by_merkle_value
153            .keys()
154            .collect::<hashbrown::HashSet<_, fnv::FnvBuildHasher>>();
155        for (hash, InProgressEntry { decode_result, .. }) in entries_by_merkle_value.iter() {
156            let Ok(decoded) = decode_result else {
157                maybe_trie_roots.remove(hash);
158                continue;
159            };
160            for child in decoded.children.into_iter().flatten() {
161                if let Ok(child) = &<[u8; 32]>::try_from(child) {
162                    maybe_trie_roots.remove(child);
163                }
164            }
165        }
166        maybe_trie_roots
167    };
168
169    // The implementation below iterates down the tree of nodes represented by this proof, keeping
170    // note of the traversed elements.
171
172    // Keep track of all the entries found in the proof.
173    let mut entries: Vec<Entry> = Vec::with_capacity(entries_by_merkle_value.len());
174
175    let mut trie_roots_with_entries =
176        hashbrown::HashMap::with_capacity_and_hasher(trie_roots.len(), Default::default());
177
178    // Keep track of the proof entries that haven't been visited when traversing.
179    let mut unvisited_proof_entries =
180        (0..entries_by_merkle_value.len()).collect::<hashbrown::HashSet<_, fnv::FnvBuildHasher>>();
181
182    // We repeat this operation for every trie root.
183    for trie_root_hash in trie_roots {
184        struct StackEntry<'a> {
185            range_in_proof: ops::Range<usize>,
186            index_in_entries: usize,
187            num_visited_children: u8,
188            children_node_values: [Option<&'a [u8]>; 16],
189        }
190
191        // Keep track of the number of entries before this trie root.
192        // This allows us to truncate `entries` to this value in case of decoding failure.
193        let num_entries_before_current_trie_root = entries.len();
194
195        // Keep track of the indices of the proof entries that are visited when traversing this trie.
196        let mut visited_proof_entries_during_trie =
197            Vec::with_capacity(entries_by_merkle_value.len());
198
199        // TODO: configurable capacity?
200        let mut visited_entries_stack: Vec<StackEntry> = Vec::with_capacity(24);
201
202        loop {
203            // Find which node to visit next.
204            // This is the next child of the node at the top of the stack, or if the node at
205            // the top of the stack doesn't have any child, we pop it and continue iterating.
206            // If the stack is empty, we are necessarily at the first iteration.
207            let (visited_node_entry_range, visited_node_decoded) =
208                match visited_entries_stack.last_mut() {
209                    None => {
210                        // Stack is empty.
211                        // Because we immediately `break` after popping the last element, the stack
212                        // can only ever be empty at the very start.
213                        let InProgressEntry {
214                            index_in_proof: root_position,
215                            range_in_proof: root_range,
216                            decode_result,
217                        } = entries_by_merkle_value.get(&trie_root_hash[..]).unwrap();
218                        visited_proof_entries_during_trie.push(*root_position);
219                        // If the node can't be decoded, we ignore the entire trie and jump
220                        // to the next one.
221                        let Ok(decoded) = decode_result.clone() else {
222                            debug_assert_eq!(num_entries_before_current_trie_root, entries.len());
223                            break;
224                        };
225                        (root_range.clone(), decoded)
226                    }
227                    Some(StackEntry {
228                        num_visited_children: stack_top_visited_children,
229                        ..
230                    }) if *stack_top_visited_children == 16 => {
231                        // We have visited all the children of the top of the stack. Pop the node from
232                        // the stack.
233                        let Some(StackEntry {
234                            index_in_entries: stack_top_index_in_entries,
235                            ..
236                        }) = visited_entries_stack.pop()
237                        else {
238                            unreachable!()
239                        };
240
241                        // Update the value of `child_entries_follow_up`
242                        // and `children_present_in_proof_bitmap` of the parent.
243                        if let Some(&StackEntry {
244                            index_in_entries: parent_index_in_entries,
245                            num_visited_children: parent_children_visited,
246                            ..
247                        }) = visited_entries_stack.last()
248                        {
249                            entries[parent_index_in_entries].child_entries_follow_up +=
250                                entries[stack_top_index_in_entries].child_entries_follow_up + 1;
251                            entries[parent_index_in_entries].children_present_in_proof_bitmap |=
252                                1 << (parent_children_visited - 1);
253                        }
254
255                        // If we popped the last node of the stack, we have finished the iteration.
256                        if visited_entries_stack.is_empty() {
257                            trie_roots_with_entries
258                                .insert(*trie_root_hash, num_entries_before_current_trie_root);
259                            // Remove the visited entries from `unvisited_proof_entries`.
260                            // Note that it is questionable what to do if the same entry is visited
261                            // multiple times. In case where multiple storage branches are identical,
262                            // the sender of the proof should de-duplicate the identical nodes. For
263                            // this reason, it could be legitimate for the same proof entry to be
264                            // visited multiple times.
265                            for entry_num in visited_proof_entries_during_trie {
266                                unvisited_proof_entries.remove(&entry_num);
267                            }
268                            break;
269                        } else {
270                            continue;
271                        }
272                    }
273                    Some(StackEntry {
274                        range_in_proof: stack_top_proof_range,
275                        num_visited_children: stack_top_visited_children,
276                        children_node_values: stack_top_children,
277                        ..
278                    }) => {
279                        // Find the next child of the top of the stack.
280                        let stack_top_entry = &proof_as_ref[stack_top_proof_range.clone()];
281
282                        // Find the index of the next child (that we are about to visit).
283                        let next_child_to_visit = stack_top_children
284                            .iter()
285                            .skip(usize::from(*stack_top_visited_children))
286                            .position(|c| c.is_some())
287                            .map(|idx| u8::try_from(idx).unwrap() + *stack_top_visited_children)
288                            .unwrap_or(16);
289
290                        // `continue` if all children have been visited. The next iteration will
291                        // pop the stack entry.
292                        if next_child_to_visit == 16 {
293                            *stack_top_visited_children = 16;
294                            continue;
295                        }
296                        *stack_top_visited_children = next_child_to_visit + 1;
297
298                        // The value of the child node is either directly inlined (if less
299                        // than 32 bytes) or is a hash.
300                        let child_node_value =
301                            stack_top_children[usize::from(next_child_to_visit)].unwrap();
302                        debug_assert!(child_node_value.len() <= 32); // Guaranteed by decoding API.
303                        if child_node_value.len() < 32 {
304                            let offset = stack_top_proof_range.start
305                                + if !child_node_value.is_empty() {
306                                    child_node_value.as_ptr() as usize
307                                        - stack_top_entry.as_ptr() as usize
308                                } else {
309                                    0
310                                };
311                            debug_assert!(offset == 0 || offset >= stack_top_proof_range.start);
312                            debug_assert!(
313                                offset <= (stack_top_proof_range.start + stack_top_entry.len())
314                            );
315
316                            let child_range_in_proof = offset..(offset + child_node_value.len());
317
318                            // Decodes the child.
319                            // If the node can't be decoded, we ignore the entire trie and jump
320                            // to the next one.
321                            let Ok(child_decoded) =
322                                trie_node::decode(&proof_as_ref[child_range_in_proof.clone()])
323                            else {
324                                entries.truncate(num_entries_before_current_trie_root);
325                                break;
326                            };
327
328                            (child_range_in_proof, child_decoded)
329                        } else if let Some(&InProgressEntry {
330                            index_in_proof: child_position,
331                            range_in_proof: ref child_entry_range,
332                            ref decode_result,
333                        }) = entries_by_merkle_value.get(child_node_value)
334                        {
335                            // If the node value of the child is less than 32 bytes long, it should
336                            // have been inlined instead of given separately.
337                            if child_entry_range.end - child_entry_range.start < 32 {
338                                entries.truncate(num_entries_before_current_trie_root);
339                                break;
340                            }
341
342                            visited_proof_entries_during_trie.push(child_position);
343
344                            // If the node can't be decoded, we ignore the entire trie and jump
345                            // to the next one.
346                            let Ok(decoded) = decode_result.clone() else {
347                                entries.truncate(num_entries_before_current_trie_root);
348                                break;
349                            };
350                            (child_entry_range.clone(), decoded)
351                        } else {
352                            // Child is a hash that was not found in the proof. Simply continue
353                            // iterating, in order to try to find the follow-up child.
354                            continue;
355                        }
356                    }
357                };
358
359            // All nodes must either have a child or a storage value or be the root.
360            if visited_node_decoded.children_bitmap() == 0
361                && matches!(
362                    visited_node_decoded.storage_value,
363                    trie_node::StorageValue::None
364                )
365                && !visited_entries_stack.is_empty()
366            {
367                entries.truncate(num_entries_before_current_trie_root);
368                break;
369            }
370
371            // Nodes with no storage value and one children are forbidden.
372            if visited_node_decoded
373                .children
374                .iter()
375                .filter(|c| c.is_some())
376                .count()
377                == 1
378                && matches!(
379                    visited_node_decoded.storage_value,
380                    trie_node::StorageValue::None
381                )
382            {
383                entries.truncate(num_entries_before_current_trie_root);
384                break;
385            }
386
387            // Add an entry for this node in the final list of entries.
388            entries.push(Entry {
389                parent_entry_index: visited_entries_stack.last().map(|entry| {
390                    (
391                        entry.index_in_entries,
392                        nibble::Nibble::try_from(entry.num_visited_children - 1).unwrap(),
393                    )
394                }),
395                range_in_proof: visited_node_entry_range.clone(),
396                storage_value_in_proof: match visited_node_decoded.storage_value {
397                    trie_node::StorageValue::None => None,
398                    trie_node::StorageValue::Hashed(value_hash) => {
399                        if let Some(&InProgressEntry {
400                            index_in_proof: value_position,
401                            range_in_proof: ref value_entry_range,
402                            ..
403                        }) = entries_by_merkle_value.get(&value_hash[..])
404                        {
405                            visited_proof_entries_during_trie.push(value_position);
406                            Some(value_entry_range.clone())
407                        } else {
408                            None
409                        }
410                    }
411                    trie_node::StorageValue::Unhashed(v) => {
412                        let offset = if !v.is_empty() {
413                            v.as_ptr() as usize - proof_as_ref.as_ptr() as usize
414                        } else {
415                            0
416                        };
417                        debug_assert!(offset == 0 || offset >= visited_node_entry_range.start);
418                        debug_assert!(offset <= visited_node_entry_range.end);
419                        Some(offset..offset + v.len())
420                    }
421                },
422                child_entries_follow_up: 0,          // Filled later.
423                children_present_in_proof_bitmap: 0, // Filled later.
424            });
425
426            // Add the visited node to the stack. The next iteration will either go to its first
427            // child, or pop the node from the stack.
428            visited_entries_stack.push(StackEntry {
429                range_in_proof: visited_node_entry_range,
430                index_in_entries: entries.len() - 1,
431                num_visited_children: 0,
432                children_node_values: visited_node_decoded.children,
433            });
434        }
435    }
436
437    // The entire reason why we track the unvisited proof entries is to return this error if
438    // necessary.
439    if !unvisited_proof_entries.is_empty() {
440        return Err(Error::UnusedProofEntry);
441    }
442
443    drop(entries_by_merkle_value);
444    Ok(DecodedTrieProof {
445        proof: config.proof,
446        entries,
447        trie_roots: trie_roots_with_entries,
448    })
449}

455

461    /// Given that hashes are verified to actually match their values, there is no risk of

465}

504                            if self.0.is_empty() {
505                                write!(f, "∅")?
506                            }
507                            for byte in self.0 {
508                                write!(f, "{:02x}", *byte)?
509                            }
510                            Ok(())
511                        }
512                    }
513
514                    struct DummyNibbles<'a, T: AsRef<[u8]>>(EntryKeyIter<'a, T>);
515                    impl<'a, T: AsRef<[u8]>> fmt::Debug for DummyNibbles<'a, T> {
516                        fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
517                            let mut any_written = false;
518                            for nibble in self.0.clone() {
519                                any_written = true;
520                                write!(f, "{:x}", nibble)?
521                            }
522                            if !any_written {
523                                write!(f, "∅")?
524                            }
525                            Ok(())
526                        }
527                    }

529                    (
530                        (DummyHash(trie_root_hash), DummyNibbles(key)),
531                        (
532                            entry.trie_node_info.children,
533                            entry.trie_node_info.storage_value,
534                        ),
535                    )
536                },
537            ))
538            .finish()
539    }

550impl<T: AsRef<[u8]>> DecodedTrieProof<T> {
551    /// Returns a list of all elements of the proof, ordered by key in lexicographic order.
552    ///
553    /// This function is a convenient wrapper around [`DecodedTrieProof::iter_ordered`] that
554    /// converts the keys into arrays of bytes. If a key can't be represented as an array of
555    /// bytes, then it is filtered out. Assuming that the trie has only ever been used in the
556    /// context of the runtime, then this cannot happen. See the section below for an
557    /// explanation.
558    ///
559    /// The iterator might include branch nodes. It is not possible for this function to
560    /// differentiate between value-less nodes that are present in the proof only because they are
561    /// branch nodes, and value-less nodes that are present in the proof because the fact that they
562    /// have no value is important for the proof.
563    ///
564    /// # Detailed explanation
565    ///
566    /// The trie consists of nodes, each with a key and a value. The keys consist of an array of
567    /// "nibbles", which are 4 bits each.
568    ///
569    /// When the runtime writes a value in the trie, it passes a key as an array a bytes. In order
570    /// to know where to write this value, this array of bytes is converted into an array of
571    /// nibbles by turning each byte into two nibbles.
572    ///
573    /// Due to the fact that the host-runtime interface only ever uses arrays of bytes, it is not
574    /// possible for the runtime to store a value or read a value in the trie at a key that
575    /// consists in an uneven number of nibbles, as an uneven number of nibbles cannot be
576    /// converted to an array of bytes.
577    ///
578    /// In other words, if a trie has only ever been used in the context of a runtime, then it is
579    /// guaranteed to not contain any storage value at key that consists in an uneven number of
580    /// nibbles.
581    ///
582    /// The trie format itself, however, technically doesn't forbid storing reading and writing
583    /// values at keys that consist in an uneven number of nibbles. For this reason, a proof
584    /// containing a value at a key that consists in an uneven number of nibbles is considered as
585    /// valid according to [`decode_and_verify_proof`].
586    ///
587    // TODO: paragraph below not true anymore
588    /// However, given that [`decode_and_verify_proof`] verifies the trie proof against the state
589    /// trie root hash, we are also guaranteed that this proof reflects the actual trie. If the
590    /// actual trie can't contain any storage value at a key that consists in an uneven number of
591    /// nibbles, then the proof is also guaranteed to not contain any storage value at a key that
592    /// consists in an uneven number of nibbles. Importantly, this is only true if we are sure that
593    /// the block is valid, in other words that it has indeed been created using a runtime. Blocks
594    /// that are invalid might have been created through a fake trie.
595    ///
596    /// As a conclusion, if this proof is made against a trie that has only ever been used in the
597    /// context of a runtime, and that the block using this trie is guaranteed to be valid, then
598    /// this function will work as intended and return the entire content of the proof.
599    ///
600    // TODO: ordering between trie roots unspecified
601    // TODO: consider not returning a Vec
602    pub fn iter_runtime_context_ordered(
603        &self,
604    ) -> impl Iterator<Item = (EntryKey<'_, Vec<u8>>, StorageValue)> {
605        self.iter_ordered().filter_map(
606            |(
607                EntryKey {
608                    trie_root_hash,
609                    key,
610                },
611                entry,
612            )| {
613                let value = entry.trie_node_info.storage_value;
614
615                if key.clone().count() % 2 != 0 {
616                    return None;
617                }
618
619                let key = nibble::nibbles_to_bytes_suffix_extend(key).collect();
620                Some((
621                    EntryKey {
622                        trie_root_hash,
623                        key,
624                    },
625                    value,
626                ))
627            },
628        )
629    }

631    /// Returns a list of all elements of the proof, ordered by key in lexicographic order.
632    ///
633    /// The iterator includes branch nodes.
634    // TODO: ordering between trie roots unspecified
635    pub fn iter_ordered(
636        &self,
637    ) -> impl Iterator<Item = (EntryKey<'_, EntryKeyIter<'_, T>>, ProofEntry<'_, T>)> {
638        let proof = self.proof.as_ref();
639
640        self.trie_roots
641            .iter()
642            .flat_map(|(trie_root_hash, &trie_root_entry_index)| {
643                self.entries
644                    .iter()
645                    .enumerate()
646                    .skip(trie_root_entry_index)
647                    .take(self.entries[trie_root_entry_index].child_entries_follow_up + 1)
648                    .map(|(entry_index, entry)| {
649                        let key = EntryKey {
650                            trie_root_hash,
651                            key: EntryKeyIter::new(self, entry_index),
652                        };
653
654                        let Ok(entry_index_decoded) =
655                            trie_node::decode(&proof[entry.range_in_proof.clone()])
656                        else {
657                            // Proof has been checked to be entirely decodable.
658                            unreachable!()
659                        };
660
661                        let entry = ProofEntry {
662                            merkle_value: if let Some((parent_index, parent_nibble)) =
663                                self.entries[entry_index].parent_entry_index
664                            {
665                                let Ok(parent_decoded) = trie_node::decode(
666                                    &proof[self.entries[parent_index].range_in_proof.clone()],
667                                ) else {
668                                    // Proof has been checked to be entirely decodable.
669                                    unreachable!()
670                                };
671                                parent_decoded.children[usize::from(parent_nibble)]
672                                    .as_ref()
673                                    .unwrap()
674                            } else {
675                                trie_root_hash
676                            },
677                            node_value: &self.proof.as_ref()[entry.range_in_proof.clone()],
678                            partial_key_nibbles: entry_index_decoded.partial_key,
679                            unhashed_storage_value: entry
680                                .storage_value_in_proof
681                                .as_ref()
682                                .map(|range| &proof[range.clone()]),
683                            trie_node_info: TrieNodeInfo {
684                                children: Children {
685                                    children: {
686                                        let mut children = core::array::from_fn(|_| Child::NoChild);
687                                        let mut i = entry_index + 1;
688                                        for child_num in 0..16 {
689                                            let Some(child_merkle_value) =
690                                                entry_index_decoded.children[child_num]
691                                            else {
692                                                continue;
693                                            };
694                                            if entry.children_present_in_proof_bitmap
695                                                & (1 << child_num)
696                                                != 0
697                                            {
698                                                children[child_num] = Child::InProof {
699                                                    child_key: EntryKeyIter::new(self, i),
700                                                    merkle_value: child_merkle_value,
701                                                };
702                                                i += self.entries[i].child_entries_follow_up;
703                                                i += 1;
704                                            } else {
705                                                children[child_num] = Child::AbsentFromProof {
706                                                    merkle_value: child_merkle_value,
707                                                };
708                                            }
709                                        }
710                                        children
711                                    },
712                                },
713                                storage_value: match (
714                                    entry_index_decoded.storage_value,
715                                    &entry.storage_value_in_proof,
716                                ) {
717                                    (trie_node::StorageValue::Unhashed(value), _) => {
718                                        StorageValue::Known {
719                                            value,
720                                            inline: true,
721                                        }
722                                    }
723                                    (trie_node::StorageValue::Hashed(_), Some(value_range)) => {
724                                        StorageValue::Known {
725                                            value: &proof[value_range.clone()],
726                                            inline: false,
727                                        }
728                                    }
729                                    (trie_node::StorageValue::Hashed(hash), None) => {
730                                        StorageValue::HashKnownValueMissing(hash)
731                                    }
732                                    (trie_node::StorageValue::None, _v) => {
733                                        debug_assert!(_v.is_none());
734                                        StorageValue::None
735                                    }
736                                },
737                            },
738                        };
739
740                        (key, entry)
741                    })
742            })
743    }

745    /// Returns the key of the closest ancestor to the given key that can be found in the proof.
746    /// If `key` is in the proof, returns `key`.
747    pub fn closest_ancestor_in_proof<'a>(
748        &'a self,
749        trie_root_merkle_value: &[u8; 32],
750        mut key: impl Iterator<Item = nibble::Nibble>,
751    ) -> Result<Option<EntryKeyIter<'a, T>>, IncompleteProofError> {
752        let proof = self.proof.as_ref();
753
754        // If the proof doesn't contain any entry for the requested trie, then we have no
755        // information about the node whatsoever.
756        // This check is necessary because we assume below that a lack of ancestor means that the
757        // key is outside of the trie.
758        let Some(&(mut iter_entry)) = self.trie_roots.get(trie_root_merkle_value) else {
759            return Err(IncompleteProofError());
760        };
761
762        loop {
763            let Ok(iter_entry_decoded) =
764                trie_node::decode(&proof[self.entries[iter_entry].range_in_proof.clone()])
765            else {
766                unreachable!()
767            };
768
769            let mut iter_entry_partial_key_iter = iter_entry_decoded.partial_key;
770            loop {
771                match (key.next(), iter_entry_partial_key_iter.next()) {
772                    (Some(a), Some(b)) if a == b => {}
773                    (_, Some(_)) => {
774                        // Mismatch in partial key. Closest ancestor is the parent entry.
775                        let Some((parent_entry, _)) = self.entries[iter_entry].parent_entry_index
776                        else {
777                            // Key is completely outside of the trie.
778                            return Ok(None);
779                        };
780                        return Ok(Some(EntryKeyIter::new(self, parent_entry)));
781                    }
782                    (Some(child_num), None) => {
783                        if let Some(_) = iter_entry_decoded.children[usize::from(child_num)] {
784                            // Key points to child. Update `iter_entry` and continue.
785                            let children_present_in_proof_bitmap =
786                                self.entries[iter_entry].children_present_in_proof_bitmap;
787
788                            // If the child isn't present in the proof, then the proof is
789                            // incomplete.
790                            if children_present_in_proof_bitmap & (1 << u8::from(child_num)) == 0 {
791                                return Err(IncompleteProofError());
792                            }
793
794                            for c in 0..u8::from(child_num) {
795                                if children_present_in_proof_bitmap & (1 << c) != 0 {
796                                    iter_entry += 1;
797                                    iter_entry += self.entries[iter_entry].child_entries_follow_up;
798                                }
799                            }
800                            iter_entry += 1;
801                        } else {
802                            // Key points to non-existing child. Closest ancestor is `iter_entry`.
803                            return Ok(Some(EntryKeyIter::new(self, iter_entry)));
804                        }
805                        break;
806                    }
807                    (None, None) => {
808                        // Exact match. Closest ancestor is `iter_entry`.
809                        return Ok(Some(EntryKeyIter::new(self, iter_entry)));
810                    }
811                }
812            }
813        }
814    }

816    /// Returns information about a trie node.
817    ///
818    /// Returns an error if the proof doesn't contain enough information about this trie node.
819    ///
820    /// This function will return `Ok` even if there is no node in the trie for `key`, in which
821    /// case the returned [`TrieNodeInfo`] will indicate no storage value and no children.
822    pub fn trie_node_info(
823        &self,
824        trie_root_merkle_value: &[u8; 32],
825        mut key: impl Iterator<Item = nibble::Nibble>,
826    ) -> Result<TrieNodeInfo<'_, T>, IncompleteProofError> {
827        let proof = self.proof.as_ref();
828
829        // Find the starting point of the requested trie.
830        let Some((mut iter_entry_merkle_value, mut iter_entry)) = self
831            .trie_roots
832            .get_key_value(trie_root_merkle_value)
833            .map(|(k, v)| (&k[..], *v))
834        else {
835            return Err(IncompleteProofError());
836        };
837
838        loop {
839            let Ok(iter_entry_decoded) =
840                trie_node::decode(&proof[self.entries[iter_entry].range_in_proof.clone()])
841            else {
842                // Proof has been checked to be entirely decodable.
843                unreachable!()
844            };
845
846            let mut iter_entry_partial_key_iter = iter_entry_decoded.partial_key;
847            loop {
848                match (key.next(), iter_entry_partial_key_iter.next()) {
849                    (Some(a), Some(b)) if a == b => {}
850                    (Some(_), Some(_)) => {
851                        // Mismatch in partial key. No node with the requested key in the trie.
852                        return Ok(TrieNodeInfo {
853                            storage_value: StorageValue::None,
854                            children: Children {
855                                children: core::array::from_fn(|_| Child::NoChild),
856                            },
857                        });
858                    }
859                    (None, Some(a)) => {
860                        // Input key is a subslice of `iter_entry`'s key.
861                        // One has one descendant that is `iter_entry`.
862                        let mut children = core::array::from_fn(|_| Child::NoChild);
863                        children[usize::from(a)] = Child::InProof {
864                            child_key: EntryKeyIter::new(self, iter_entry),
865                            merkle_value: iter_entry_merkle_value,
866                        };
867                        return Ok(TrieNodeInfo {
868                            storage_value: StorageValue::None,
869                            children: Children { children },
870                        });
871                    }
872                    (Some(child_num), None) => {
873                        if let Some(child) = iter_entry_decoded.children[usize::from(child_num)] {
874                            // Key points in the direction of a child.
875                            let children_present_in_proof_bitmap =
876                                self.entries[iter_entry].children_present_in_proof_bitmap;
877
878                            // If the child isn't present in the proof, then the proof is
879                            // incomplete.
880                            if children_present_in_proof_bitmap & (1 << u8::from(child_num)) == 0 {
881                                return Err(IncompleteProofError());
882                            }
883
884                            // Child is present in the proof. Update `iter_entry` and continue.
885                            iter_entry_merkle_value = child;
886                            for c in 0..u8::from(child_num) {
887                                if children_present_in_proof_bitmap & (1 << c) != 0 {
888                                    iter_entry += 1;
889                                    iter_entry += self.entries[iter_entry].child_entries_follow_up;
890                                }
891                            }
892                            iter_entry += 1;
893                            break;
894                        } else {
895                            // Key points to non-existing child.
896                            return Ok(TrieNodeInfo {
897                                storage_value: StorageValue::None,
898                                children: Children {
899                                    children: core::array::from_fn(|_| Child::NoChild),
900                                },
901                            });
902                        }
903                    }
904                    (None, None) => {
905                        // Exact match. Trie node is `iter_entry`.
906                        return Ok(TrieNodeInfo {
907                            storage_value: match (
908                                iter_entry_decoded.storage_value,
909                                &self.entries[iter_entry].storage_value_in_proof,
910                            ) {
911                                (trie_node::StorageValue::Unhashed(value), _) => {
912                                    StorageValue::Known {
913                                        value,
914                                        inline: true,
915                                    }
916                                }
917                                (trie_node::StorageValue::Hashed(_), Some(value_range)) => {
918                                    StorageValue::Known {
919                                        value: &proof[value_range.clone()],
920                                        inline: false,
921                                    }
922                                }
923                                (trie_node::StorageValue::Hashed(hash), None) => {
924                                    StorageValue::HashKnownValueMissing(hash)
925                                }
926                                (trie_node::StorageValue::None, _v) => {
927                                    debug_assert!(_v.is_none());
928                                    StorageValue::None
929                                }
930                            },
931                            children: Children {
932                                children: {
933                                    let mut children = core::array::from_fn(|_| Child::NoChild);
934                                    let mut i = iter_entry + 1;
935                                    for child_num in 0..16 {
936                                        let Some(child_merkle_value) =
937                                            iter_entry_decoded.children[child_num]
938                                        else {
939                                            continue;
940                                        };
941                                        if self.entries[iter_entry].children_present_in_proof_bitmap
942                                            & (1 << child_num)
943                                            != 0
944                                        {
945                                            children[child_num] = Child::InProof {
946                                                child_key: EntryKeyIter::new(self, i),
947                                                merkle_value: child_merkle_value,
948                                            };
949                                            i += self.entries[i].child_entries_follow_up;
950                                            i += 1;
951                                        } else {
952                                            children[child_num] = Child::AbsentFromProof {
953                                                merkle_value: child_merkle_value,
954                                            };
955                                        }
956                                    }
957                                    children
958                                },
959                            },
960                        });
961                    }
962                }
963            }
964        }
965    }

1000    /// Find in the proof the trie node that follows `key_before` in lexicographic order.
1001    ///
1002    /// If `or_equal` is `true`, then `key_before` is returned if it is equal to a node in the
1003    /// trie. If `false`, then only keys that are strictly superior are returned.
1004    ///
1005    /// The returned value must always start with `prefix`. Note that the value of `prefix` is
1006    /// important as it can be the difference between `Err(IncompleteProofError)` and `Ok(None)`.
1007    ///
1008    /// If `branch_nodes` is `false`, then trie nodes that don't have a storage value are skipped.
1009    ///
1010    /// Returns an error if the proof doesn't contain enough information to determine the next key.
1011    /// Returns `Ok(None)` if the proof indicates that there is no next key (within the given
1012    /// prefix).
1013    pub fn next_key(
1014        &self,
1015        trie_root_merkle_value: &[u8; 32],
1016        key_before: impl Iterator<Item = nibble::Nibble>,
1017        mut or_equal: bool,
1018        prefix: impl Iterator<Item = nibble::Nibble>,
1019        branch_nodes: bool,
1020    ) -> Result<Option<EntryKeyIter<'_, T>>, IncompleteProofError> {
1021        let proof = self.proof.as_ref();
1022
1023        // The implementation below might continue iterating `prefix` even after it has returned
1024        // `None`, thus we have to fuse it.
1025        let mut prefix = prefix.fuse();
1026
1027        // The implementation below might continue iterating `key_before` even after it has
1028        // returned `None`, thus we have to fuse it.
1029        // Furthermore, `key_before` might be modified in the implementation below. When that
1030        // happens, de do this by setting it to `either::Right`.
1031        let mut key_before = either::Left(key_before.fuse());
1032
1033        // Find the starting point of the requested trie.
1034        let Some(&(mut iter_entry)) = self.trie_roots.get(trie_root_merkle_value) else {
1035            return Err(IncompleteProofError());
1036        };
1037
1038        // If `true`, then it was determined that `key_before` was after the last child of
1039        // `iter_entry`. The next iteration must find `iter_entry`'s next sibling.
1040        let mut iterating_up: bool = false;
1041
1042        // Indicates the depth of ancestors of `iter_entry` that match `prefix`.
1043        // For example, if the value is 2, then `iter_entry`'s parent and grandparent match
1044        //`prefix`, but `iter_entry`'s grandparent parent does not.
1045        let mut prefix_match_iter_entry_ancestor_depth = 0;
1046
1047        loop {
1048            let Ok(iter_entry_decoded) =
1049                trie_node::decode(&proof[self.entries[iter_entry].range_in_proof.clone()])
1050            else {
1051                // Proof has been checked to be entirely decodable.
1052                unreachable!()
1053            };
1054
1055            if iterating_up {
1056                debug_assert!(matches!(key_before, either::Right(_)));
1057                debug_assert!(or_equal);
1058
1059                // It was determined that `key_before` was after the last child of `iter_entry`.
1060                // The next iteration must find `iter_entry`'s next sibling.
1061
1062                if prefix_match_iter_entry_ancestor_depth == 0 {
1063                    // `prefix` only matches `iter_entry` and not its parent and
1064                    // thus wouldn't match `iter_entry`'s sibling or uncle.
1065                    // Therefore, the next node is `None`.
1066                    return Ok(None);
1067                }
1068
1069                // Go to `iter_entry`'s next sibling, if any.
1070                let Some((parent_entry, parent_to_child_nibble)) =
1071                    self.entries[iter_entry].parent_entry_index
1072                else {
1073                    // `iter_entry` is the root of the trie. `key_before` is therefore
1074                    // after the last entry of the trie.
1075                    return Ok(None);
1076                };
1077                let Ok(parent_entry_decoded) =
1078                    trie_node::decode(&proof[self.entries[parent_entry].range_in_proof.clone()])
1079                else {
1080                    // Proof has been checked to be entirely decodable.
1081                    unreachable!()
1082                };
1083
1084                let Some(child_num) = parent_entry_decoded
1085                    .children
1086                    .iter()
1087                    .skip(usize::from(parent_to_child_nibble) + 1)
1088                    .position(|c| c.is_some())
1089                    .map(|n| n + usize::from(parent_to_child_nibble) + 1)
1090                else {
1091                    // No child found. Continue iterating up.
1092                    iterating_up = true;
1093                    iter_entry = parent_entry;
1094                    prefix_match_iter_entry_ancestor_depth -= 1;
1095                    continue;
1096                };
1097
1098                // Found a next sibling.
1099
1100                let children_present_in_proof_bitmap =
1101                    self.entries[parent_entry].children_present_in_proof_bitmap;
1102
1103                // If the child isn't present in the proof, then the proof is
1104                // incomplete. While in some situations we could prove that the child
1105                // is necessarily the next key, there is no way to know its full key.
1106                if children_present_in_proof_bitmap & (1 << child_num) == 0 {
1107                    return Err(IncompleteProofError());
1108                }
1109
1110                // `iter_entry` is still pointing to the child at index `parent_to_child_nibble`.
1111                // Since we're jumping to its next sibling, all we have to do is skip over it
1112                // and its descedants.
1113                iter_entry += self.entries[iter_entry].child_entries_follow_up;
1114                iter_entry += 1;
1115                iterating_up = false;
1116                continue; // Continue in order to refresh `iter_entry_decoded`.
1117            }
1118
1119            let mut iter_entry_partial_key_iter = iter_entry_decoded.partial_key;
1120            loop {
1121                match (
1122                    key_before.next(),
1123                    prefix.next(),
1124                    iter_entry_partial_key_iter.next(),
1125                ) {
1126                    (Some(k), Some(p), Some(pk)) if k == p && p == pk => {
1127                        // Continue descending down the tree.
1128                    }
1129                    (None, Some(p), Some(pk)) if p == pk => {
1130                        // Continue descending down the tree.
1131                    }
1132                    (Some(k), None, Some(pk)) if k == pk => {
1133                        // Continue descending down the tree.
1134                    }
1135                    (None, None, Some(_)) => {
1136                        // Exact match. Due to the `branch_nodes` setting, we can't just
1137                        // return `iter_entry` and instead continue iterating.
1138                        or_equal = true;
1139                    }
1140                    (Some(k), None, Some(pk)) if k < pk => {
1141                        // `key_before` points to somewhere between `iter_entry`'s parent
1142                        // and `iter_entry`. Due to the `branch_nodes` setting, we can't just
1143                        // return `iter_entry` and instead continue iterating.
1144                        key_before = either::Right(iter::empty().fuse());
1145                        or_equal = true;
1146                    }
1147                    (Some(k), Some(p), _) if k > p => {
1148                        // `key_before` is strictly superior to `prefix`. The next key is
1149                        // thus necessarily `None`.
1150                        // Note that this is not a situation that is expected to be common, as
1151                        // doing such a call is pointless.
1152                        return Ok(None);
1153                    }
1154                    (Some(k), Some(p), Some(pk)) if k < p && p == pk => {
1155                        // The key and prefix diverge.
1156                        // We know that there isn't any key in the trie between `key_before`
1157                        // and `prefix`.
1158                        // Starting next iteration, the next node matching the prefix
1159                        // will be the result.
1160                        key_before = either::Right(iter::empty().fuse());
1161                        or_equal = true;
1162                    }
1163                    (_, Some(p), Some(pk)) => {
1164                        debug_assert!(p != pk); // Other situations covered by other match blocks.
1165
1166                        // Mismatch between prefix and partial key. No matter the value of
1167                        // `key_before` There is no node in the trie that starts with `prefix`.
1168                        return Ok(None);
1169                    }
1170                    (Some(k), None, Some(pk)) if k < pk => {
1171                        // `key_before` points to somewhere between `iter_entry`'s parent
1172                        // and `iter_entry`. We know that `iter_entry` necessarily matches
1173                        // the prefix. The next key is necessarily `iter_entry`.
1174                        return Ok(Some(EntryKeyIter::new(self, iter_entry)));
1175                    }
1176                    (Some(k), None, Some(pk)) => {
1177                        debug_assert!(k > pk); // Checked above.
1178
1179                        // `key_before` points to somewhere between the last child of `iter_entry`
1180                        // and its next sibling.
1181                        // The next key is thus the next sibling of `iter_entry`.
1182                        iterating_up = true;
1183                        key_before = either::Right(iter::empty().fuse());
1184                        or_equal = true;
1185                        break;
1186                    }
1187                    (None, None, None)
1188                        if or_equal
1189                            && (branch_nodes
1190                                || !matches!(
1191                                    iter_entry_decoded.storage_value,
1192                                    trie_node::StorageValue::None,
1193                                )) =>
1194                    {
1195                        // Exact match. Next key is `iter_entry`.
1196                        return Ok(Some(EntryKeyIter::new(self, iter_entry)));
1197                    }
1198                    (key_before_nibble, prefix_nibble, None) => {
1199                        // The moment when we have finished traversing a node and go to the
1200                        // next one is the most complicated situation.
1201
1202                        // We know for sure that `iter_entry` can't be returned, as it is covered
1203                        // by the other match variants above. Therefore, `key_before_nibble` equal
1204                        // to `None` is the same as if it is equal to `0`.
1205                        let key_before_nibble = match key_before_nibble {
1206                            Some(n) => u8::from(n),
1207                            None => {
1208                                or_equal = true;
1209                                0
1210                            }
1211                        };
1212
1213                        // Try to find the first child that is after `key_before`.
1214                        if let Some(child_num) = iter_entry_decoded
1215                            .children
1216                            .iter()
1217                            .skip(usize::from(key_before_nibble))
1218                            .position(|c| c.is_some())
1219                            .map(|n| n + usize::from(key_before_nibble))
1220                        {
1221                            // Found a child. Make sure that it matches the prefix nibble.
1222                            if prefix_nibble.map_or(false, |p| child_num != usize::from(p)) {
1223                                // Child doesn't match prefix. No next key.
1224                                return Ok(None);
1225                            }
1226
1227                            // Continue iterating down through the child that has been found.
1228
1229                            let children_present_in_proof_bitmap =
1230                                self.entries[iter_entry].children_present_in_proof_bitmap;
1231
1232                            // If the child isn't present in the proof, then the proof is
1233                            // incomplete. While in some situations we could prove that the child
1234                            // is necessarily the next key, there is no way to know its full key.
1235                            if children_present_in_proof_bitmap & (1 << child_num) == 0 {
1236                                return Err(IncompleteProofError());
1237                            }
1238
1239                            for c in 0..child_num {
1240                                if children_present_in_proof_bitmap & (1 << c) != 0 {
1241                                    iter_entry += 1;
1242                                    iter_entry += self.entries[iter_entry].child_entries_follow_up;
1243                                }
1244                            }
1245
1246                            iter_entry += 1;
1247                            prefix_match_iter_entry_ancestor_depth += 1;
1248                            if usize::from(key_before_nibble) != child_num {
1249                                key_before = either::Right(iter::empty().fuse());
1250                                or_equal = true;
1251                            }
1252                            break;
1253                        } else {
1254                            // Childless branch nodes are forbidden. This is checked when
1255                            // decoding the proof.
1256                            debug_assert!(!matches!(
1257                                iter_entry_decoded.storage_value,
1258                                trie_node::StorageValue::None,
1259                            ));
1260
1261                            // `key_before` is after the last child of `iter_entry`. The next
1262                            // node is thus a sibling or uncle of `iter_entry`.
1263                            iterating_up = true;
1264                            key_before = either::Right(iter::empty().fuse());
1265                            or_equal = true;
1266                            break;
1267                        }
1268                    }
1269                }
1270            }
1271        }
1272    }
1273
1274    /// Find in the proof the closest trie node that descends from `key` and returns its Merkle
1275    /// value.
1276    ///
1277    /// Returns an error if the proof doesn't contain enough information to determine the Merkle
1278    /// value.
1279    /// Returns `Ok(None)` if the proof indicates that there is no descendant.
1280    pub fn closest_descendant_merkle_value(
1281        &self,
1282        trie_root_merkle_value: &[u8; 32],
1283        mut key: impl Iterator<Item = nibble::Nibble>,
1284    ) -> Result<Option<&[u8]>, IncompleteProofError> {
1285        let proof = self.proof.as_ref();
1286
1287        // Find the starting point of the requested trie.
1288        let Some((mut iter_entry_merkle_value, mut iter_entry)) = self
1289            .trie_roots
1290            .get_key_value(trie_root_merkle_value)
1291            .map(|(k, v)| (&k[..], *v))
1292        else {
1293            return Err(IncompleteProofError());
1294        };
1295
1296        loop {
1297            let Ok(iter_entry_decoded) =
1298                trie_node::decode(&proof[self.entries[iter_entry].range_in_proof.clone()])
1299            else {
1300                // Proof has been checked to be entirely decodable.
1301                unreachable!()
1302            };
1303
1304            let mut iter_entry_partial_key_iter = iter_entry_decoded.partial_key;
1305            loop {
1306                match (key.next(), iter_entry_partial_key_iter.next()) {
1307                    (Some(a), Some(b)) if a == b => {}
1308                    (Some(_), Some(_)) => {
1309                        // Mismatch in partial key. No descendant.
1310                        return Ok(None);
1311                    }
1312                    (None, Some(_)) => {
1313                        // Input key is a subslice of `iter_entry`'s key.
1314                        // Descendant is thus `iter_entry`.
1315                        return Ok(Some(iter_entry_merkle_value));
1316                    }
1317                    (Some(child_num), None) => {
1318                        if let Some(child) = iter_entry_decoded.children[usize::from(child_num)] {
1319                            // Key points in the direction of a child.
1320                            let children_present_in_proof_bitmap =
1321                                self.entries[iter_entry].children_present_in_proof_bitmap;
1322
1323                            // If the child isn't present in the proof, then the proof is
1324                            // incomplete, unless the input key doesn't have any nibble anymore,
1325                            // in which case we know that the closest descendant is this child,
1326                            // even if it's not present.
1327                            if children_present_in_proof_bitmap & (1 << u8::from(child_num)) == 0 {
1328                                if key.next().is_none() {
1329                                    return Ok(Some(child));
1330                                } else {
1331                                    return Err(IncompleteProofError());
1332                                }
1333                            }
1334
1335                            // Child is present in the proof. Update `iter_entry` and continue.
1336                            iter_entry_merkle_value = child;
1337                            for c in 0..u8::from(child_num) {
1338                                if children_present_in_proof_bitmap & (1 << c) != 0 {
1339                                    iter_entry += 1;
1340                                    iter_entry += self.entries[iter_entry].child_entries_follow_up;
1341                                }
1342                            }
1343                            iter_entry += 1;
1344                            break;
1345                        } else {
1346                            // Key points to non-existing child. We know that there's no
1347                            // descendant.
1348                            return Ok(None);
1349                        }
1350                    }
1351                    (None, None) => {
1352                        // Exact match. Closest descendant is `iter_entry`.
1353                        return Ok(Some(iter_entry_merkle_value));
1354                    }
1355                }
1356            }
1357        }
1358    }
1359}

1423pub struct EntryKeyIter<'a, T> {
1424    proof: &'a DecodedTrieProof<T>,
1425    target_entry: usize,
1426    /// [`EntryKeyIter::entry_key_iterator`] is the `target_entry_depth_remaining` nth ancestor
1427    /// of [`EntryKeyIter::target_entry`].
1428    target_entry_depth_remaining: usize,
1429    /// `None` if iteration is finished.
1430    entry_key_iterator: Option<trie_node::DecodedPartialKey<'a>>,
1431}
1432
1433impl<'a, T: AsRef<[u8]>> EntryKeyIter<'a, T> {
1434    fn new(proof: &'a DecodedTrieProof<T>, target_entry: usize) -> Self {
1435        // Find the number of nodes between `target_entry` and the trie root.
1436        let mut entry_iter = target_entry;
1437        let mut target_entry_depth_remaining = 0;
1438        loop {
1439            if let Some((parent, _)) = proof.entries[entry_iter].parent_entry_index {
1440                target_entry_depth_remaining += 1;
1441                entry_iter = parent;
1442            } else {
1443                break;
1444            }
1445        }
1446
1447        let Ok(decoded_entry) = trie_node::decode(
1448            &proof.proof.as_ref()[proof.entries[entry_iter].range_in_proof.clone()],
1449        ) else {
1450            unreachable!()
1451        };
1452
1453        EntryKeyIter {
1454            proof,
1455            target_entry,
1456            target_entry_depth_remaining,
1457            entry_key_iterator: Some(decoded_entry.partial_key),
1458        }
1459    }
1460}
1461
1462impl<'a, T: AsRef<[u8]>> Iterator for EntryKeyIter<'a, T> {
1463    type Item = nibble::Nibble;
1464
1465    fn next(&mut self) -> Option<Self::Item> {
1466        // `entry_key_iterator` is `None` only if the iteration is finished, as a way to fuse
1467        // the iterator.
1468        let Some(entry_key_iterator) = &mut self.entry_key_iterator else {
1469            return None;
1470        };
1471
1472        // Still yielding from `entry_key_iterator`.
1473        if let Some(nibble) = entry_key_iterator.next() {
1474            return Some(nibble);
1475        }
1476
1477        // `entry_key_iterator` has finished iterating. Update the local state for the next node
1478        // in the hierarchy.
1479        if self.target_entry_depth_remaining == 0 {
1480            // Iteration finished.
1481            self.entry_key_iterator = None;
1482            return None;
1483        }
1484        self.target_entry_depth_remaining -= 1;
1485
1486        // Find the `target_entry_depth_remaining`th ancestor of `target_entry`.
1487        let mut entry_iter = self.target_entry;
1488        for _ in 0..self.target_entry_depth_remaining {
1489            let Some((parent, _)) = self.proof.entries[entry_iter].parent_entry_index else {
1490                unreachable!()
1491            };
1492            entry_iter = parent;
1493        }
1494
1495        // Store the partial key of `entry_iter` in `entry_key_iterator`, so that it starts being
1496        // yielded at the next iteration.
1497        self.entry_key_iterator = Some(
1498            trie_node::decode(
1499                &self.proof.proof.as_ref()[self.proof.entries[entry_iter].range_in_proof.clone()],
1500            )
1501            .unwrap_or_else(|_| unreachable!())
1502            .partial_key,
1503        );
1504
1505        // Yield the "parent-child nibble" of `entry_iter`.
1506        Some(
1507            self.proof.entries[entry_iter]
1508                .parent_entry_index
1509                .unwrap_or_else(|| unreachable!())
1510                .1,
1511        )
1512    }
1513
1514    fn size_hint(&self) -> (usize, Option<usize>) {
1515        // We know we're going to yield `self.target_entry_depth_remaining` "parent-child nibbles".
1516        // We add to that the size hint of `entry_key_iterator`.
1517        let entry_key_iterator = self
1518            .entry_key_iterator
1519            .as_ref()
1520            .map_or(0, |i| i.size_hint().0);
1521        (entry_key_iterator + self.target_entry_depth_remaining, None)
1522    }
1523}
