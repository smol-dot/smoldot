24/// Public key of a node's identity.
25///
26/// Libp2p specifies multiple different possible algorithms, but only Ed25519 support is
27/// mandatory.
28#[derive(Clone, Debug, PartialEq, Eq)]
29pub enum PublicKey {
30    /// An Ed25519 public key.
31    Ed25519([u8; 32]),
32}
33
34impl PublicKey {
35    /// Encode the public key into a Protobuf structure for exchange with other nodes.
36    ///
37    /// As indicated in the libp2p specification, the encoding is done deterministically despite
38    /// the fact that the Protobuf format isn't deterministic.
39    ///
40    /// See <https://github.com/libp2p/specs/blob/master/peer-ids/peer-ids.md#keys>.
41    pub fn to_protobuf_encoding(&self) -> Vec<u8> {
42        match self {
43            PublicKey::Ed25519(key) => {
44                const CAPACITY: usize = 32 + 4;
45                let mut out = Vec::with_capacity(CAPACITY);
46                for slice in protobuf::enum_tag_encode(1, 1) {
47                    out.extend_from_slice(slice.as_ref());
48                }
49                for slice in protobuf::bytes_tag_encode(2, key) {
50                    out.extend_from_slice(slice.as_ref());
51                }
52                debug_assert_eq!(out.len(), CAPACITY);
53                out
54            }
55        }
56    }
57
58    /// Decode a public key from a Protobuf structure, e.g. read from storage or received from
59    /// another node.
60    pub fn from_protobuf_encoding(bytes: &[u8]) -> Result<PublicKey, FromProtobufEncodingError> {
61        struct ErrorWrapper(FromProtobufEncodingError);
62        impl<'a> nom::error::ParseError<&'a [u8]> for ErrorWrapper {
63            fn from_error_kind(_: &'a [u8], _: nom::error::ErrorKind) -> Self {
64                ErrorWrapper(FromProtobufEncodingError::ProtobufDecodeError)
65            }
66            fn append(_: &'a [u8], _: nom::error::ErrorKind, other: Self) -> Self {
67                other
68            }
69        }
70        impl<'a> nom::error::FromExternalError<&'a [u8], FromProtobufEncodingError> for ErrorWrapper {
71            fn from_external_error(
72                _: &'a [u8],
73                _: nom::error::ErrorKind,
74                e: FromProtobufEncodingError,
75            ) -> Self {
76                ErrorWrapper(e)
77            }
78        }
79
80        // As indicated in the libp2p specification, the public key must be encoded
81        // deterministically, and thus the fields are decoded deterministically in a precise order.
82        let mut parser =
83            nom::combinator::all_consuming::<_, ErrorWrapper, _>(nom::combinator::complete((
84                nom::sequence::preceded(
85                    nom::combinator::peek(nom::combinator::verify(
86                        protobuf::tag_decode,
87                        |(field_num, _)| *field_num == 1,
88                    )),
89                    nom::combinator::map_res(protobuf::enum_tag_decode, |val| match val {
90                        0..=3 => Ok(val),
91                        _ => Err(FromProtobufEncodingError::UnknownAlgorithm),
92                    }),
93                ),
94                nom::sequence::preceded(
95                    nom::combinator::peek(nom::combinator::verify(
96                        protobuf::tag_decode,
97                        |(field_num, _)| *field_num == 2,
98                    )),
99                    nom::combinator::map_res(protobuf::bytes_tag_decode, |d| {
100                        <[u8; 32]>::try_from(d)
101                            .map_err(|_| FromProtobufEncodingError::BadEd25519Key)
102                    }),
103                ),
104            )));
105
106        match nom::Finish::finish(nom::Parser::parse(&mut parser, bytes)) {
107            Ok((_, (1, key))) => Ok(PublicKey::Ed25519(key)),
108            Ok((_, (_, _))) => Err(FromProtobufEncodingError::UnsupportedAlgorithm),
109            Err(err) => Err(err.0),
110        }
111    }
112
113    /// Convert the [`PublicKey`] into the corresponding [`PeerId`].
114    pub fn into_peer_id(self) -> PeerId {
115        self.into()
116    }
117
118    /// Verifies whether the given signature is valid for the given message using `self` as the
119    /// public key.
120    pub fn verify(&self, message: &[u8], signature: &[u8]) -> Result<(), SignatureVerifyFailed> {
121        let PublicKey::Ed25519(public_key) = self;
122        let public_key = ed25519_zebra::VerificationKey::try_from(*public_key)
123            .map_err(|_| SignatureVerifyFailed())?;
124        let signature =
125            ed25519_zebra::Signature::try_from(signature).map_err(|_| SignatureVerifyFailed())?;
126        public_key
127            .verify(&signature, message)
128            .map_err(|_| SignatureVerifyFailed())?;
129        Ok(())
130    }
131}

133/// Error potentially returned by [`PublicKey::from_protobuf_encoding`].
134#[derive(Debug, derive_more::Display, derive_more::Error)]
135pub enum FromProtobufEncodingError {
136    /// Error decoding the Protobuf message.
137    ProtobufDecodeError,
138    /// Public key algorithm unknown.
139    UnknownAlgorithm,
140    /// Ed25519 key doesn't have a correct length.
141    BadEd25519Key,
142    /// Algorithms other than Ed25519 aren't supported.
143    UnsupportedAlgorithm,
144}

146/// Call to [`PublicKey::verify`] has failed. No reason is provided for security reasons.
147#[derive(Debug, derive_more::Display, derive_more::Error)]
148pub struct SignatureVerifyFailed();

150/// Public keys with byte-lengths smaller than `MAX_INLINE_KEY_LENGTH` will be
151/// automatically used as the peer id using an identity multihash.
152const MAX_INLINE_KEY_LENGTH: usize = 42;
153
154/// Identifier of a node of the network.
155///
156/// The data is a multihash of the public key of the peer.
157#[derive(Clone, Eq)]
158pub struct PeerId {
159    /// Always contains a valid multihash.
160    multihash: Vec<u8>,
161}
162
163impl PeerId {
164    /// Builds the [`PeerId`] corresponding to a public key.
165    pub fn from_public_key(key: &PublicKey) -> PeerId {
166        let key_enc = key.to_protobuf_encoding();
167
168        let out = if key_enc.len() <= MAX_INLINE_KEY_LENGTH {
169            multihash::Multihash::identity(&key_enc).into_bytes()
170        } else {
171            let mut out = vec![0; 34];
172            out[0] = 0x12;
173            out[1] = 0x32;
174
175            let mut hasher = <sha2::Sha256 as sha2::Digest>::new();
176            sha2::Digest::update(&mut hasher, &key_enc);
177            sha2::Digest::finalize_into(
178                hasher,
179                sha2::digest::generic_array::GenericArray::from_mut_slice(&mut out[2..]),
180            );
181
182            out
183        };
184
185        PeerId { multihash: out }
186    }
187
188    /// Checks whether `data` is a valid [`PeerId`].
189    ///
190    /// In case of error, returns the bytes passed as parameter in addition to the error.
191    pub fn from_bytes(data: Vec<u8>) -> Result<PeerId, (FromBytesError, Vec<u8>)> {
192        let result = match multihash::Multihash::from_bytes(&data) {
193            Ok(hash) => {
194                // For a PeerId to be valid, it must use either the "identity" multihash code (0x0)
195                // or the "sha256" multihash code (0x12).
196                if hash.hash_algorithm_code() == 0 {
197                    if let Err(err) = PublicKey::from_protobuf_encoding(hash.data()) {
198                        Err(FromBytesError::InvalidPublicKey(err))
199                    } else {
200                        Ok(())
201                    }
202                } else if hash.hash_algorithm_code() == 0x12 {
203                    Ok(())
204                } else {
205                    Err(FromBytesError::InvalidMultihashAlgorithm)
206                }
207            }
208            Err((err, _)) => Err(FromBytesError::DecodeError(err)),
209        };
210
211        match result {
212            Ok(()) => Ok(PeerId { multihash: data }),
213            Err(err) => Err((err, data)),
214        }
215    }
216
217    /// Returns a raw bytes representation of this `PeerId`.
218    pub fn into_bytes(self) -> Vec<u8> {
219        self.multihash
220    }
221
222    /// Returns a raw bytes representation of this `PeerId`.
223    pub fn as_bytes(&self) -> &[u8] {
224        &self.multihash
225    }
226
227    /// Returns a base-58 encoded string of this `PeerId`.
228    pub fn to_base58(&self) -> String {
229        bs58::encode(self.as_bytes()).into_string()
230    }
231}

303impl FromStr for PeerId {
304    type Err = ParseError;
305
306    fn from_str(s: &str) -> Result<Self, Self::Err> {
307        let bytes = bs58::decode(s)
308            .into_vec()
309            .map_err(Bs58DecodeError)
310            .map_err(ParseError::Bs58)?;
311        PeerId::from_bytes(bytes).map_err(|(err, _)| ParseError::NotPeerId(err))
312    }
313}

315/// Error when turning bytes into a [`PeerId`].
316#[derive(Debug, derive_more::Display, derive_more::Error)]
317pub enum FromBytesError {
318    /// Failed to decode bytes into a multihash.
319    DecodeError(multihash::FromBytesError),
320    /// The algorithm used in the multihash isn't identity or SHA-256.
321    InvalidMultihashAlgorithm,
322    /// Multihash uses the identity algorithm, but the data isn't a valid public key.
323    #[display("Failed to decode public key protobuf: {_0}")]
324    InvalidPublicKey(FromProtobufEncodingError),
325}
326
327/// Error when parsing a string to a [`PeerId`].
328#[derive(Debug, derive_more::Display, derive_more::Error)]
329pub enum ParseError {
330    /// Error decoding the Base58 encoding.
331    #[display("Base58 decoding error: {_0}")]
332    Bs58(Bs58DecodeError),
333    /// Decoded bytes aren't a valid [`PeerId`].
334    #[display("{_0}")]
335    NotPeerId(FromBytesError),
336}
337
338/// Error when decoding Base58 encoding.
339#[derive(Debug, derive_more::Display, derive_more::Error, derive_more::From)]
340#[display("{_0}")]
341// TODO: bs58 doesn't implement the Error trait; remove not(source) at some point
342pub struct Bs58DecodeError(#[error(not(source))] bs58::decode::Error);
