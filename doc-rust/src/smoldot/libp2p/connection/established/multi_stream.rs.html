36/// State machine of a fully-established connection where substreams are handled externally.
37pub struct MultiStream<TNow, TSubId, TSubUd> {
38    /// Events that should be yielded from [`MultiStream::pull_event`].
39    pending_events: VecDeque<Event<TSubUd>>,
40
41    /// List of all open substreams, both inbound and outbound.
42    ///
43    /// There are two substreams namespaces: "out substreams", used for API purposes when it comes
44    /// to notifications and requests, and "in substreams", used for API purposes when it comes to
45    /// raw data sent/received on a substream. When the user for example resets an "in substream",
46    /// the "out substream" remains valid.
47    in_substreams: hashbrown::HashMap<TSubId, Substream<TNow, TSubUd>, util::SipHasherBuild>,
48
49    out_in_substreams_map: hashbrown::HashMap<u32, TSubId, fnv::FnvBuildHasher>,
50
51    next_out_substream_id: u32,
52
53    /// List of outgoing substreams that aren't opened yet.
54    ///
55    /// Every time an outgoing substream is opened, an item is pulled from this list.
56    ///
57    /// Does not include the ping substream.
58    desired_out_substreams: VecDeque<Substream<TNow, TSubUd>>,
59
60    /// Substream used for outgoing pings.
61    ///
62    /// Initially contains `None` as the substream for pings isn't opened yet.
63    ///
64    /// Because of the API of [`substream::Substream`] concerning pings, there is no need to
65    /// handle situations where the substream fails to negotiate, as this is handled by making
66    /// outgoing pings error. This substream is therefore constant.
67    ping_substream: Option<TSubId>,
68    /// When to start the next ping attempt.
69    next_ping: TNow,
70    /// Source of randomness to generate ping payloads.
71    ///
72    /// Note that we use ChaCha20 because the rest of the code base also uses ChaCha20. This avoids
73    /// unnecessary code being included in the binary and reduces the binary size.
74    ping_payload_randomness: rand_chacha::ChaCha20Rng,
75
76    /// See [`Config::max_inbound_substreams`].
77    // TODO: not enforced at the moment
78    _max_inbound_substreams: usize,
79    /// See [`Config::max_protocol_name_len`].
80    max_protocol_name_len: usize,
81    /// See [`Config::ping_protocol`].
82    ping_protocol: String,
83    /// See [`Config::ping_interval`].
84    ping_interval: Duration,
85    /// See [`Config::ping_timeout`].
86    ping_timeout: Duration,
87}
88
89struct Substream<TNow, TSubUd> {
90    id: u32,
91    /// Opaque data decided by the user. `None` if the substream doesn't exist on the API layer
92    /// yet.
93    user_data: Option<TSubUd>,
94    /// Underlying state machine for the substream. Always `Some` while the substream is alive,
95    /// and `None` if it has been reset.
96    inner: Option<substream::Substream<TNow>>,
97    /// State of the message frames.
98    framing: webrtc_framing::WebRtcFraming,
99}
100
101const MAX_PENDING_EVENTS: usize = 4;
102
103impl<TNow, TSubId, TSubUd> MultiStream<TNow, TSubId, TSubUd>
104where
105    TNow: Clone + Add<Duration, Output = TNow> + Sub<TNow, Output = Duration> + Ord,
106    TSubId: Clone + PartialEq + Eq + Hash,
107{
108    /// Creates a new connection from the given configuration.
109    pub fn webrtc(config: Config<TNow>) -> MultiStream<TNow, TSubId, TSubUd> {
110        let mut randomness = rand_chacha::ChaCha20Rng::from_seed(config.randomness_seed);
111
112        MultiStream {
113            pending_events: {
114                // Note that the capacity is higher than `MAX_PENDING_EVENTS` because resetting
115                // substreams can unconditionally queue an event, and the API doesn't give the
116                // possibility to not reset a substream (as that would introduce too much
117                // complexity). For this reason, we reserve enough for the events that can happen
118                // by reading/writing substreams plus events that can happen by resetting
119                // substreams.
120                let cap = MAX_PENDING_EVENTS + config.substreams_capacity;
121                VecDeque::with_capacity(cap)
122            },
123            in_substreams: hashbrown::HashMap::with_capacity_and_hasher(
124                config.substreams_capacity,
125                util::SipHasherBuild::new({
126                    let mut seed = [0; 16];
127                    randomness.fill_bytes(&mut seed);
128                    seed
129                }),
130            ),
131            out_in_substreams_map: hashbrown::HashMap::with_capacity_and_hasher(
132                config.substreams_capacity,
133                Default::default(),
134            ),
135            next_out_substream_id: 0,
136            desired_out_substreams: VecDeque::with_capacity(config.substreams_capacity),
137            ping_substream: None,
138            next_ping: config.first_out_ping,
139            ping_payload_randomness: randomness,
140            _max_inbound_substreams: config.max_inbound_substreams,
141            max_protocol_name_len: config.max_protocol_name_len,
142            ping_protocol: config.ping_protocol,
143            ping_interval: config.ping_interval,
144            ping_timeout: config.ping_timeout,
145        }
146    }

148    /// Removes an event from the queue of events and returns it.
149    ///
150    /// This method should be called after [`MultiStream::substream_read_write`] or
151    /// [`MultiStream::reset_substream`] is called.
152    pub fn pull_event(&mut self) -> Option<Event<TSubUd>> {
153        self.pending_events.pop_front()
154    }

163    /// Returns the number of new outbound substreams that the state machine would like to see
164    /// opened.
165    ///
166    /// This value doesn't change automatically over time but only after a call to
167    /// [`MultiStream::substream_read_write`], [`MultiStream::add_substream`],
168    /// [`MultiStream::reset_substream`], [`MultiStream::add_request`], or
169    /// [`MultiStream::open_notifications_substream`].
170    ///
171    /// Note that the user is expected to track the number of substreams that are currently being
172    /// opened. For example, if this function returns 2 and there are already 2 substreams
173    /// currently being opened, then there is no need to open any additional one.
174    pub fn desired_outbound_substreams(&self) -> u32 {
175        u32::try_from(self.desired_out_substreams.len())
176            .unwrap_or(u32::MAX)
177            .saturating_add(if self.ping_substream.is_none() { 1 } else { 0 })
178    }

180    /// Notifies the state machine that a new substream has been opened.
181    ///
182    /// `outbound` indicates whether the substream has been opened by the remote (`false`) or
183    /// locally (`true`).
184    ///
185    /// If `outbound` is `true`, then the value returned by
186    /// [`MultiStream::desired_outbound_substreams`] will decrease by one.
187    ///
188    /// # Panic
189    ///
190    /// Panics if there already exists a substream with an identical identifier.
191    ///
192    pub fn add_substream(&mut self, id: TSubId, outbound: bool) {
193        let substream = if !outbound {
194            let out_substream_id = self.next_out_substream_id;
195            self.next_out_substream_id += 1;
196
197            Substream {
198                id: out_substream_id,
199                inner: Some(substream::Substream::ingoing(self.max_protocol_name_len)),
200                user_data: None,
201                framing: webrtc_framing::WebRtcFraming::new(),
202            }
203        } else if self.ping_substream.is_none() {
204            let out_substream_id = self.next_out_substream_id;
205            self.next_out_substream_id += 1;
206
207            self.ping_substream = Some(id.clone());
208
209            Substream {
210                id: out_substream_id,
211                inner: Some(substream::Substream::ping_out(self.ping_protocol.clone())),
212                user_data: None,
213                framing: webrtc_framing::WebRtcFraming::new(),
214            }
215        } else if let Some(desired) = self.desired_out_substreams.pop_front() {
216            desired
217        } else {
218            // TODO: reset the new substream
219            todo!()
220        };
221
222        let _prev_val = self.out_in_substreams_map.insert(substream.id, id.clone());
223        debug_assert!(_prev_val.is_none());
224
225        let previous_value = self.in_substreams.insert(id, substream);
226        if previous_value.is_some() {
227            // There is already a substream with that identifier. This is forbidden by the API of
228            // this function.
229            panic!()
230        }
231    }

233    /// Immediately destroys the substream with the given identifier.
234    ///
235    /// The given identifier is now considered invalid by the state machine.
236    ///
237    /// # Panic
238    ///
239    /// Panics if there is no substream with that identifier.
240    ///
241    pub fn reset_substream(&mut self, substream_id: &TSubId) {
242        let mut substream = self.in_substreams.remove(substream_id).unwrap();
243        let _was_in = self.out_in_substreams_map.remove(&substream.id);
244        debug_assert!(_was_in.is_none());
245
246        if Some(substream_id) == self.ping_substream.as_ref() {
247            self.ping_substream = None;
248        }
249
250        let maybe_event = substream.inner.unwrap().reset();
251        if let Some(event) = maybe_event {
252            Self::on_substream_event(
253                &mut self.pending_events,
254                substream.id,
255                &mut substream.user_data,
256                event,
257            );
258        }
259    }

261    /// Reads/writes data on the substream.
262    ///
263    /// If the method returns [`SubstreamFate::Reset`], then the substream is now considered dead
264    /// according to the state machine and its identifier is now invalid. If the reading or
265    /// writing side of the substream was still open, then the user should reset that substream.
266    ///
267    /// This method will refuse to accept data if too many events are already queued. Use
268    /// [`MultiStream::pull_event`] to empty the queue of events between calls to this method.
269    ///
270    /// In the case of a WebRTC connection, the [`ReadWrite::incoming_buffer`] and
271    /// [`ReadWrite::write_bytes_queueable`] must always be `Some`.
272    ///
273    /// # Panic
274    ///
275    /// Panics if there is no substream with that identifier.
276    /// Panics if this is a WebRTC connection, and the reading or writing side is closed.
277    ///
278    #[must_use]
279    pub fn substream_read_write(
280        &mut self,
281        substream_id: &TSubId,
282        read_write: &mut ReadWrite<TNow>,
283    ) -> SubstreamFate {
284        let substream = self.in_substreams.get_mut(substream_id).unwrap();
285
286        // In WebRTC, the reading and writing side is never closed.
287        assert!(
288            read_write.expected_incoming_bytes.is_some()
289                && read_write.write_bytes_queueable.is_some()
290        );
291
292        // Reading/writing the ping substream is used to queue new outgoing pings.
293        if Some(substream_id) == self.ping_substream.as_ref() {
294            if read_write.now >= self.next_ping {
295                let mut payload = [0u8; 32];
296                self.ping_payload_randomness.fill_bytes(&mut payload);
297                substream.inner.as_mut().unwrap().queue_ping(
298                    &payload,
299                    read_write.now.clone(),
300                    self.ping_timeout,
301                );
302                self.next_ping = read_write.now.clone() + self.ping_interval;
303            }
304
305            read_write.wake_up_after(&self.next_ping);
306        }
307
308        // Don't process any more data before events are pulled.
309        if self.pending_events.len() >= MAX_PENDING_EVENTS {
310            return SubstreamFate::Continue;
311        }
312
313        // Now process the substream.
314        let event = match substream.framing.read_write(read_write) {
315            Ok(mut framing) => {
316                let (substream_update, event) =
317                    substream.inner.take().unwrap().read_write(&mut framing);
318                substream.inner = substream_update;
319                event
320            }
321            Err(_) => substream.inner.take().unwrap().reset(),
322        };
323
324        if let Some(event) = event {
325            read_write.wake_up_asap();
326            Self::on_substream_event(
327                &mut self.pending_events,
328                substream.id,
329                &mut substream.user_data,
330                event,
331            )
332        }
333
334        // The substream is `None` if it needs to be reset.
335        if substream.inner.is_none() {
336            if Some(substream_id) == self.ping_substream.as_ref() {
337                self.ping_substream = None;
338            }
339            self.out_in_substreams_map.remove(&substream.id);
340            self.in_substreams.remove(substream_id);
341            SubstreamFate::Reset
342        } else {
343            SubstreamFate::Continue
344        }
345    }

347    /// Turns an event from the [`substream`] module into an [`Event`] and adds it to the queue.
348    fn on_substream_event(
349        pending_events: &mut VecDeque<Event<TSubUd>>,
350        substream_id: u32,
351        substream_user_data: &mut Option<TSubUd>,
352        event: substream::Event,
353    ) {
354        pending_events.push_back(match event {
355            substream::Event::InboundError {
356                error,
357                was_accepted: false,
358            } => Event::InboundError(error),
359            substream::Event::InboundError {
360                was_accepted: true, ..
361            } => Event::InboundAcceptedCancel {
362                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
363                user_data: substream_user_data.take().unwrap(),
364            },
365            substream::Event::InboundNegotiated(protocol_name) => Event::InboundNegotiated {
366                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
367                protocol_name,
368            },
369            substream::Event::InboundNegotiatedCancel => Event::InboundAcceptedCancel {
370                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
371                user_data: substream_user_data.take().unwrap(),
372            },
373            substream::Event::RequestIn { request } => Event::RequestIn {
374                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
375                request,
376            },
377            substream::Event::Response { response } => Event::Response {
378                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
379                response,
380                user_data: substream_user_data.take().unwrap(),
381            },
382            substream::Event::NotificationsInOpen { handshake } => Event::NotificationsInOpen {
383                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
384                handshake,
385            },
386            substream::Event::NotificationsInOpenCancel => Event::NotificationsInOpenCancel {
387                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
388            },
389            substream::Event::NotificationIn { notification } => Event::NotificationIn {
390                notification,
391                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
392            },
393            substream::Event::NotificationsInClose { outcome } => Event::NotificationsInClose {
394                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
395                outcome,
396                user_data: substream_user_data.take().unwrap(),
397            },
398            substream::Event::NotificationsOutResult { result } => Event::NotificationsOutResult {
399                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
400                result: match result {
401                    Ok(r) => Ok(r),
402                    Err(err) => Err((err, substream_user_data.take().unwrap())),
403                },
404            },
405            substream::Event::NotificationsOutCloseDemanded => {
406                Event::NotificationsOutCloseDemanded {
407                    id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
408                }
409            }
410            substream::Event::NotificationsOutReset => Event::NotificationsOutReset {
411                id: SubstreamId(SubstreamIdInner::MultiStream(substream_id)),
412                user_data: substream_user_data.take().unwrap(),
413            },
414            substream::Event::PingOutSuccess { ping_time } => Event::PingOutSuccess { ping_time },
415            substream::Event::PingOutError { .. } => {
416                // Because ping events are automatically generated by the external API without any
417                // guarantee, it is safe to merge multiple failed pings into one.
418                Event::PingOutFailed
419            }
420        });
421    }

423    /// Sends a request to the remote.
424    ///
425    /// This method only inserts the request into the connection object. The request will later
426    /// be sent out through [`MultiStream::substream_read_write`].
427    ///
428    /// Assuming that the remote is using the same implementation, an [`Event::RequestIn`] will
429    /// be generated on its side.
430    ///
431    /// If `request` is `None`, then no request is sent to the remote at all. If `request` is
432    /// `Some`, then a (potentially-empty) request is sent. If `Some(&[])` is provided, a
433    /// length-prefix containing a 0 is sent to the remote.
434    ///
435    /// After the remote has sent back a response, an [`Event::Response`] event will be generated
436    /// locally. The `user_data` parameter will be passed back.
437    ///
438    /// The timeout is the time between the moment the substream is opened and the moment the
439    /// response is sent back. If the emitter doesn't send the request or if the receiver doesn't
440    /// answer during this time window, the request is considered failed.
441    pub fn add_request(
442        &mut self,
443        protocol_name: String,
444        request: Option<Vec<u8>>,
445        timeout: TNow,
446        max_response_size: usize,
447        user_data: TSubUd,
448    ) -> SubstreamId {
449        let substream_id = self.next_out_substream_id;
450        self.next_out_substream_id += 1;
451
452        self.desired_out_substreams.push_back(Substream {
453            id: substream_id,
454            inner: Some(substream::Substream::request_out(
455                protocol_name,
456                timeout,
457                request,
458                max_response_size,
459            )),
460            user_data: Some(user_data),
461            framing: webrtc_framing::WebRtcFraming::new(),
462        });
463
464        // TODO: ? do this? substream.reserve_window(128 * 1024 * 1024 + 128); // TODO: proper max size
465
466        SubstreamId(SubstreamIdInner::MultiStream(substream_id))
467    }

528    /// Call after an [`Event::InboundNegotiated`] has been emitted in order to accept the protocol
529    /// name and indicate the type of the protocol.
530    ///
531    /// # Panic
532    ///
533    /// Panics if the substream is not in the correct state.
534    ///
535    pub fn accept_inbound(&mut self, substream_id: SubstreamId, ty: InboundTy, user_data: TSubUd) {
536        let substream_id = match substream_id.0 {
537            SubstreamIdInner::MultiStream(id) => id,
538            _ => panic!(),
539        };
540
541        let inner_substream_id = self.out_in_substreams_map.get(&substream_id).unwrap();
542
543        let substream = self.in_substreams.get_mut(inner_substream_id).unwrap();
544        substream.inner.as_mut().unwrap().accept_inbound(ty);
545        debug_assert!(substream.user_data.is_none());
546        substream.user_data = Some(user_data);
547    }
