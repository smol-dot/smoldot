18//! State machine handling a single TCP or WebSocket libp2p connection.
19//!
20//! # About resources allocation and back-pressure
21//!
22//! In order to avoid DoS attacks, it is important, in networking code, to make sure that the
23//! amount of memory allocated directly or indirectly by a connection stays bounded.
24//!
25//! The situations in the [`SingleStream`] that lead to an increase in memory consumption are:
26//!
27//! 1- On incoming or outgoing substreams.
28//! 2- When sending a request or receiving a response in a request-response protocol.
29//! 3- When sending a notification.
30//! 4- When receiving a request and sending back a response.
31//! 5- When receiving a notification.
32//! // TODO: 6- on Yamux ping frames
33//!
34//! In order to solve 1-, there exists a maximum number of simultaneous substreams allowed by the
35//! protocol, thereby guaranteeing that the memory consumption doesn't exceed a certain bound.
36//! Since receiving a request and a response is a one-time process that occupies an entire
37//! substream, allocations referenced by points 2- and 4- are also bounded thanks to this limit.
38//! Request-response protocols enforce a limit to the size of the request and response, again
39//! guaranteeing a bound on the memory consumption.
40//!
41//! In order to solve 3-, always use [`SingleStream::notification_substream_queued_bytes`] in order
42//! to check the current amount of buffered data before calling
43//! [`SingleStream::write_notification_unbounded`]. See the documentation of
44//! [`SingleStream::write_notification_unbounded`] for more details.
45//!
46//! In order to solve 5-, // TODO: .
47//!

70/// State machine of a fully-established connection.
71pub struct SingleStream<TNow, TSubUd> {
72    /// Encryption layer applied directly on top of the incoming data and outgoing data.
73    encryption: noise::Noise,
74
75    /// Extra fields. Segregated in order to solve borrowing questions.
76    inner: Box<Inner<TNow, TSubUd>>,
77}
78
79/// Extra fields. Segregated in order to solve borrowing questions.
80struct Inner<TNow, TSubUd> {
81    /// State of the various substreams of the connection.
82    /// Consists in a collection of substreams, each of which holding a [`substream::Substream`]
83    /// object, or `None` if the substream has been reset.
84    yamux: yamux::Yamux<TNow, Option<(substream::Substream<TNow>, Option<TSubUd>)>>,
85
86    /// Substream in [`Inner::yamux`] used for outgoing pings.
87    ///
88    /// Because of the API of [`substream::Substream`] concerning pings, there is no need to
89    /// handle situations where the substream fails to negotiate, as this is handled by making
90    /// outgoing pings error. This substream is therefore constant.
91    ///
92    /// It is possible, however, that the remote resets the ping substream. In other words, this
93    /// substream might not be found in [`Inner::yamux`]. When that happens, all outgoing pings
94    /// are immediately considered as failed.
95    outgoing_pings: yamux::SubstreamId,
96    /// When to start the next ping attempt.
97    next_ping: TNow,
98    /// Source of randomness to generate ping payloads.
99    ///
100    /// Note that we use ChaCha20 because the rest of the code base also uses ChaCha20. This avoids
101    /// unnecessary code being included in the binary and reduces the binary size.
102    ping_payload_randomness: rand_chacha::ChaCha20Rng,
103
104    /// See [`Config::max_inbound_substreams`].
105    max_inbound_substreams: usize,
106    /// See [`Config::max_protocol_name_len`].
107    max_protocol_name_len: usize,
108    /// See [`Config::ping_interval`].
109    ping_interval: Duration,
110    /// See [`Config::ping_timeout`].
111    ping_timeout: Duration,
112}
113
114impl<TNow, TSubUd> SingleStream<TNow, TSubUd>
115where
116    TNow: Clone + Add<Duration, Output = TNow> + Sub<TNow, Output = Duration> + Ord,
117{
118    /// Reads data coming from the socket, updates the internal state machine, and writes data
119    /// destined to the socket through the [`ReadWrite`].
120    ///
121    /// In order to avoid unnecessary memory allocations, only one [`Event`] is returned at a time.
122    /// Consequently, this method returns as soon as an event is available, even if the buffers
123    /// haven't finished being read. Call this method in a loop until the number of bytes read and
124    /// written are both 0, and the returned [`Event`] is `None`.
125    ///
126    /// If an error is returned, the socket should be entirely shut down.
127    // TODO: consider exposing an API more similar to the one of substream::Substream::read_write?
128    pub fn read_write(
129        mut self,
130        read_write: &mut ReadWrite<TNow>,
131    ) -> Result<(SingleStream<TNow, TSubUd>, Option<Event<TSubUd>>), Error> {
132        // Start any outgoing ping if necessary.
133        if read_write.now >= self.inner.next_ping {
134            self.inner.next_ping = read_write.now.clone() + self.inner.ping_interval;
135
136            // It might be that the remote has reset the ping substream, in which case the out ping
137            // substream no longer exists and we immediately consider the ping as failed.
138            if self.inner.yamux.has_substream(self.inner.outgoing_pings) {
139                let mut payload = [0u8; 32];
140                self.inner.ping_payload_randomness.fill_bytes(&mut payload);
141                self.inner.yamux[self.inner.outgoing_pings]
142                    .as_mut()
143                    .unwrap()
144                    .0
145                    .queue_ping(&payload, read_write.now.clone(), self.inner.ping_timeout);
146                self.inner
147                    .yamux
148                    .mark_substream_write_ready(self.inner.outgoing_pings);
149            } else {
150                return Ok((self, Some(Event::PingOutFailed)));
151            }
152        }
153        read_write.wake_up_after(&self.inner.next_ping);
154
155        // If we have both sent and received a GoAway frame, that means that no new substream
156        // can be opened. If in addition to this there is no substream in the connection,
157        // then we can safely close it as a normal termination.
158        // Note that, because we have no guarantee that the remote has received our GoAway
159        // frame yet, it is possible to receive requests for new substreams even after having
160        // sent the GoAway. Because we close the writing side, it is not possible to indicate
161        // to the remote that these new substreams are denied. However, this is not a problem
162        // as the remote interprets our GoAway frame as an automatic refusal of all its pending
163        // substream requests.
164        // TODO: review w.r.t. https://github.com/smol-dot/smoldot/issues/1121
165        if (self.inner.yamux.len()
166            == if self.inner.yamux.has_substream(self.inner.outgoing_pings) {
167                1
168            } else {
169                0
170            })
171            && self.inner.yamux.goaway_sent()
172            && self.inner.yamux.received_goaway().is_some()
173        {
174            read_write.close_write();
175        }
176
177        // Note that we treat the reading side being closed the same way as no data being
178        // received. The fact that the remote has closed their writing side is no different
179        // than them leaving their writing side open but no longer send any data at all.
180        // The remote is free to close their writing side at any point if it judges that it
181        // will no longer need to send anymore data.
182        // Note, however, that in principle the remote should have sent a GoAway frame prior
183        // to closing their writing side. But this is not something we check or really care
184        // about.
185
186        // Pass the `read_write` through the Noise state machine.
187        let mut decrypted_read_write = self
188            .encryption
189            .read_write(read_write)
190            .map_err(Error::Noise)?;
191
192        // Pass the Noise decrypted stream through the Yamux state machine.
193        let yamux_rw_outcome = self
194            .inner
195            .yamux
196            .read_write(&mut decrypted_read_write)
197            .map_err(Error::Yamux)?;
198
199        match yamux_rw_outcome {
200            yamux::ReadWriteOutcome::Idle { yamux } => {
201                self.inner.yamux = yamux;
202
203                // Nothing happened, and thus there is nothing more to do.
204                drop(decrypted_read_write);
205                return Ok((self, None));
206            }
207            yamux::ReadWriteOutcome::IncomingSubstream { mut yamux } => {
208                debug_assert!(!yamux.goaway_queued_or_sent());
209
210                // Receive a request from the remote for a new incoming substream.
211                // These requests are automatically accepted unless the total limit to the
212                // number of substreams has been reached.
213                // Note that `num_inbound()` counts substreams that have been closed but not
214                // yet removed from the state machine. This can affect the actual limit in a
215                // subtle way. At the time of writing of this comment the limit should be
216                // properly enforced, however it is not considered problematic if it weren't.
217                if yamux.num_inbound() >= self.inner.max_inbound_substreams {
218                    // Can only error if there's no incoming substream, which we know for sure
219                    // is the case here.
220                    yamux
221                        .reject_pending_substream()
222                        .unwrap_or_else(|_| panic!());
223                } else {
224                    // Can only error if there's no incoming substream, which we know for sure
225                    // is the case here.
226                    yamux
227                        .accept_pending_substream(Some((
228                            substream::Substream::ingoing(self.inner.max_protocol_name_len),
229                            None,
230                        )))
231                        .unwrap_or_else(|_| panic!());
232                }
233
234                self.inner.yamux = yamux;
235
236                drop(decrypted_read_write);
237                return Ok((self, None));
238            }
239            yamux::ReadWriteOutcome::ProcessSubstream {
240                mut substream_read_write,
241            } => {
242                // The Yamux state machine needs to process a substream.
243
244                // Temporarily extract the substream's fields to put them back later.
245                let (state_machine, mut substream_user_data) =
246                    substream_read_write.user_data_mut().take().unwrap();
247                let (state_machine_update, event) =
248                    state_machine.read_write(substream_read_write.read_write());
249
250                let event_to_yield = event.map(|ev| {
251                    Self::pass_through_substream_event(
252                        substream_read_write.substream_id(),
253                        &mut substream_user_data,
254                        ev,
255                    )
256                });
257
258                match state_machine_update {
259                    Some(s) => {
260                        *substream_read_write.user_data_mut() = Some((s, substream_user_data));
261                        self.inner.yamux = substream_read_write.finish();
262                    }
263                    None => {
264                        self.inner.yamux = substream_read_write.reset();
265                    }
266                }
267
268                if let Some(event_to_yield) = event_to_yield {
269                    drop(decrypted_read_write);
270                    return Ok((self, Some(event_to_yield)));
271                }
272            }
273            yamux::ReadWriteOutcome::StreamReset { yamux, .. } => {
274                self.inner.yamux = yamux;
275                decrypted_read_write.wake_up_asap();
276            }
277            yamux::ReadWriteOutcome::GoAway { yamux, .. } => {
278                self.inner.yamux = yamux;
279                drop(decrypted_read_write);
280                return Ok((self, Some(Event::NewOutboundSubstreamsForbidden)));
281            }
282            yamux::ReadWriteOutcome::PingResponse { .. } => {
283                // Can only happen if we send out Yamux pings, which we never do.
284                unreachable!()
285            }
286        }
287
288        drop(decrypted_read_write);
289
290        // Substreams that have been closed or reset aren't immediately removed the yamux state
291        // machine. They must be removed manually, which is what is done here.
292        // TODO: could be optimized by doing it only through a Yamux event? this is the case for StreamReset but not for graceful streams closures
293        let dead_substream_ids = self
294            .inner
295            .yamux
296            .dead_substreams()
297            .map(|(id, death_ty, _)| (id, death_ty))
298            .collect::<Vec<_>>();
299        for (dead_substream_id, death_ty) in dead_substream_ids {
300            match death_ty {
301                yamux::DeadSubstreamTy::Reset => {
302                    // If the substream was reset by the remote, then the substream state
303                    // machine will still be `Some`.
304                    if let Some((state_machine, mut user_data)) =
305                        self.inner.yamux.remove_dead_substream(dead_substream_id)
306                    {
307                        // TODO: consider changing this `state_machine.reset()` function to be a state transition of the substream state machine (that doesn't take ownership), to simplify the implementation of both the substream state machine and this code
308                        if let Some(event) = state_machine.reset() {
309                            return Ok((
310                                self,
311                                Some(Self::pass_through_substream_event(
312                                    dead_substream_id,
313                                    &mut user_data,
314                                    event,
315                                )),
316                            ));
317                        }
318                    };
319
320                    // Removing a dead substream might lead to Yamux being able to process more
321                    // incoming data. As such, we loop again.
322                    read_write.wake_up_asap();
323                }
324                yamux::DeadSubstreamTy::ClosedGracefully => {
325                    self.inner.yamux.remove_dead_substream(dead_substream_id);
326                }
327            }
328        }
329
330        Ok((self, None))
331    }

333    /// Turns an event from the [`substream`] module into an [`Event`].
334    fn pass_through_substream_event(
335        substream_id: yamux::SubstreamId,
336        substream_user_data: &mut Option<TSubUd>,
337        event: substream::Event,
338    ) -> Event<TSubUd> {
339        match event {
340            substream::Event::InboundError {
341                error,
342                was_accepted: false,
343            } => Event::InboundError(error),
344            substream::Event::InboundError {
345                was_accepted: true, ..
346            } => Event::InboundAcceptedCancel {
347                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
348                user_data: substream_user_data.take().unwrap(),
349                // TODO: notify of the error?
350            },
351            substream::Event::InboundNegotiated(protocol_name) => Event::InboundNegotiated {
352                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
353                protocol_name,
354            },
355            substream::Event::InboundNegotiatedCancel => Event::InboundNegotiatedCancel {
356                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
357            },
358            substream::Event::RequestIn { request } => Event::RequestIn {
359                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
360                request,
361            },
362            substream::Event::Response { response } => Event::Response {
363                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
364                response,
365                user_data: substream_user_data.take().unwrap(),
366            },
367            substream::Event::NotificationsInOpen { handshake } => Event::NotificationsInOpen {
368                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
369                handshake,
370            },
371            substream::Event::NotificationsInOpenCancel => Event::NotificationsInOpenCancel {
372                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
373            },
374            substream::Event::NotificationIn { notification } => Event::NotificationIn {
375                notification,
376                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
377            },
378            substream::Event::NotificationsInClose { outcome } => Event::NotificationsInClose {
379                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
380                outcome,
381                user_data: substream_user_data.take().unwrap(),
382            },
383            substream::Event::NotificationsOutResult { result } => Event::NotificationsOutResult {
384                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
385                result: match result {
386                    Ok(r) => Ok(r),
387                    Err(err) => Err((err, substream_user_data.take().unwrap())),
388                },
389            },
390            substream::Event::NotificationsOutCloseDemanded => {
391                Event::NotificationsOutCloseDemanded {
392                    id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
393                }
394            }
395            substream::Event::NotificationsOutReset => Event::NotificationsOutReset {
396                id: SubstreamId(SubstreamIdInner::SingleStream(substream_id)),
397                user_data: substream_user_data.take().unwrap(),
398            },
399            substream::Event::PingOutSuccess { ping_time } => Event::PingOutSuccess { ping_time },
400            substream::Event::PingOutError { .. } => {
401                // Because ping events are automatically generated by the external API without any
402                // guarantee, it is safe to merge multiple failed pings into one.
403                Event::PingOutFailed
404            }
405        }
406    }

408    /// Close the incoming substreams, automatically denying any new substream request from the
409    /// remote.
410    ///
411    /// Note that this does not prevent incoming-substreams-related events
412    /// (such as [`Event::RequestIn`]) from being generated, as it is possible that the remote has
413    /// already opened a substream but has no sent all the necessary handshake messages yet.
414    ///
415    /// # Panic
416    ///
417    /// Panic if this function has been called before. It is illegal to call
418    /// [`SingleStream::deny_new_incoming_substreams`] more than one on the same connections.
419    ///
420    pub fn deny_new_incoming_substreams(&mut self) {
421        // TODO: arbitrary yamux error code
422        self.inner
423            .yamux
424            .send_goaway(yamux::GoAwayErrorCode::NormalTermination)
425            .unwrap()
426    }

435    /// Sends a request to the remote.
436    ///
437    /// This method only inserts the request into the connection object. Use
438    /// [`SingleStream::read_write`] in order to actually send out the request.
439    ///
440    /// Assuming that the remote is using the same implementation, an [`Event::RequestIn`] will
441    /// be generated on its side.
442    ///
443    /// If `request` is `None`, then no request is sent to the remote at all. If `request` is
444    /// `Some`, then a (potentially-empty) request is sent. If `Some(&[])` is provided, a
445    /// length-prefix containing a 0 is sent to the remote.
446    ///
447    /// After the remote has sent back a response, an [`Event::Response`] event will be generated
448    /// locally. The `user_data` parameter will be passed back.
449    ///
450    /// The timeout is the time between the moment the substream is opened and the moment the
451    /// response is sent back. If the emitter doesn't send the request or if the receiver doesn't
452    /// answer during this time window, the request is considered failed.
453    ///
454    /// # Panic
455    ///
456    /// Panics if a [`Event::NewOutboundSubstreamsForbidden`] event has been generated in the past.
457    ///
458    pub fn add_request(
459        &mut self,
460        protocol_name: String,
461        request: Option<Vec<u8>>,
462        timeout: TNow,
463        max_response_size: usize,
464        user_data: TSubUd,
465    ) -> SubstreamId {
466        let substream_id = self
467            .inner
468            .yamux
469            .open_substream(Some((
470                substream::Substream::request_out(
471                    protocol_name,
472                    timeout,
473                    request,
474                    max_response_size,
475                ),
476                Some(user_data),
477            )))
478            .unwrap(); // TODO: consider not panicking
479
480        // TODO: we add some bytes due to the length prefix, this is a bit hacky as we should ask this information from the substream
481        self.inner.yamux.add_remote_window_saturating(
482            substream_id,
483            u64::try_from(max_response_size)
484                .unwrap_or(u64::MAX)
485                .saturating_add(64)
486                .saturating_sub(yamux::NEW_SUBSTREAMS_FRAME_SIZE),
487        );
488
489        SubstreamId(SubstreamIdInner::SingleStream(substream_id))
490    }

533    /// Call after an [`Event::InboundNegotiated`] has been emitted in order to accept the protocol
534    /// name and indicate the type of the protocol.
535    ///
536    /// # Panic
537    ///
538    /// Panics if the substream is not in the correct state.
539    ///
540    pub fn accept_inbound(&mut self, substream_id: SubstreamId, ty: InboundTy, user_data: TSubUd) {
541        let substream_id = match substream_id.0 {
542            SubstreamIdInner::SingleStream(id) => id,
543            _ => panic!(),
544        };
545
546        let (substream, ud) = self.inner.yamux[substream_id].as_mut().unwrap();
547        substream.accept_inbound(ty);
548        debug_assert!(ud.is_none());
549        *ud = Some(user_data);
550        self.inner.yamux.mark_substream_write_ready(substream_id);
551    }

561        let substream_id = match substream_id.0 {

601    /// # Panic
602    ///
603    /// Panics if the substream id is not valid or the substream is of the wrong type.
604    ///
605    pub fn reject_in_notifications_substream(&mut self, substream_id: SubstreamId) {
606        let substream_id = match substream_id.0 {
607            SubstreamIdInner::SingleStream(id) => id,
608            _ => panic!(),
609        };
610
611        self.inner.yamux[substream_id]
612            .as_mut()
613            .unwrap()
614            .0
615            .reject_in_notifications_substream();
616        self.inner.yamux.mark_substream_write_ready(substream_id);
617    }

619    /// Queues a notification to be written out on the given substream.
620    ///
621    /// # About back-pressure
622    ///
623    /// This method unconditionally queues up data. You must be aware that the remote, however,
624    /// can decide to delay indefinitely the sending of that data, which can potentially lead to
625    /// an unbounded increase in memory.
626    ///
627    /// As such, you are encouraged to call this method only if the amount of queued data (as
628    /// determined by calling [`SingleStream::notification_substream_queued_bytes`]) is below a
629    /// certain threshold. If above, the notification should be silently discarded.
630    ///
631    /// # Panic
632    ///
633    /// Panics if the [`SubstreamId`] doesn't correspond to a notifications substream, or if the
634    /// notifications substream isn't in the appropriate state.
635    ///
636    pub fn write_notification_unbounded(
637        &mut self,
638        substream_id: SubstreamId,
639        notification: Vec<u8>,
640    ) {
641        let substream_id = match substream_id.0 {
642            SubstreamIdInner::SingleStream(id) => id,
643            _ => panic!(),
644        };
645
646        self.inner.yamux[substream_id]
647            .as_mut()
648            .unwrap()
649            .0
650            .write_notification_unbounded(notification);
651        self.inner.yamux.mark_substream_write_ready(substream_id);
652    }
653
654    /// Returns the number of bytes waiting to be sent out on that substream.
655    ///
656    /// See the documentation of [`SingleStream::write_notification_unbounded`] for context.
657    ///
658    /// # Panic
659    ///
660    /// Panics if the [`SubstreamId`] doesn't correspond to a notifications substream, or if the
661    /// notifications substream isn't in the appropriate state.
662    ///
663    pub fn notification_substream_queued_bytes(&self, substream_id: SubstreamId) -> usize {
664        let substream_id = match substream_id.0 {
665            SubstreamIdInner::SingleStream(id) => id,
666            _ => panic!(),
667        };
668
669        // Note that this doesn't take into account data that the Yamux or Noise state machines
670        // have extracted from the substream but hasn't sent out yet, because the objective of this
671        // function is to provide a hint about when to stop sending more data, and the size of the
672        // data that Noise and Yamux have extracted is always bounded anyway. It's not worth the
673        // effort of reporting a 100% accurate information when a 100% accurate information isn't
674        // needed.
675        self.inner.yamux[substream_id]
676            .as_ref()
677            .unwrap()
678            .0
679            .notification_substream_queued_bytes()
680    }

682    /// Closes a notifications substream opened after a successful
683    /// [`Event::NotificationsOutResult`].
684    ///
685    /// This can be done even when in the negotiation phase, in other words before the remote has
686    /// accepted/refused the substream.
687    ///
688    /// # Panic
689    ///
690    /// Panics if the [`SubstreamId`] doesn't correspond to a notifications substream, or if the
691    /// notifications substream isn't in the appropriate state.
692    ///
693    pub fn close_out_notifications_substream(&mut self, substream_id: SubstreamId) {
694        let substream_id = match substream_id.0 {
695            SubstreamIdInner::SingleStream(id) => id,
696            _ => panic!(),
697        };
698
699        if !self.inner.yamux.has_substream(substream_id) {
700            panic!()
701        }
702
703        self.inner.yamux[substream_id]
704            .as_mut()
705            .unwrap()
706            .0
707            .close_out_notifications_substream();
708        self.inner.yamux.mark_substream_write_ready(substream_id);
709    }

711    /// Closes a notifications substream that was accepted using
712    /// [`SingleStream::accept_in_notifications_substream`].
713    ///
714    /// # Panic
715    ///
716    /// Panics if the [`SubstreamId`] doesn't correspond to a notifications substream, or if the
717    /// notifications substream isn't in the appropriate state.
718    ///
719    pub fn close_in_notifications_substream(&mut self, substream_id: SubstreamId, timeout: TNow) {
720        let substream_id = match substream_id.0 {
721            SubstreamIdInner::SingleStream(id) => id,
722            _ => panic!(),
723        };
724
725        if !self.inner.yamux.has_substream(substream_id) {
726            panic!()
727        }
728
729        self.inner.yamux[substream_id]
730            .as_mut()
731            .unwrap()
732            .0
733            .close_in_notifications_substream(timeout);
734        self.inner.yamux.mark_substream_write_ready(substream_id);
735    }

737    /// Responds to an incoming request. Must be called in response to a [`Event::RequestIn`].
738    ///
739    /// Passing an `Err` corresponds, on the other side, to a
740    /// [`substream::RequestError::SubstreamClosed`].
741    ///
742    /// Returns an error if the [`SubstreamId`] is invalid.
743    pub fn respond_in_request(
744        &mut self,
745        substream_id: SubstreamId,
746        response: Result<Vec<u8>, ()>,
747    ) -> Result<(), RespondInRequestError> {
748        let substream_id = match substream_id.0 {
749            SubstreamIdInner::SingleStream(id) => id,
750            _ => return Err(RespondInRequestError::SubstreamClosed),
751        };
752
753        if !self.inner.yamux.has_substream(substream_id) {
754            return Err(RespondInRequestError::SubstreamClosed);
755        }
756
757        self.inner.yamux[substream_id]
758            .as_mut()
759            .unwrap()
760            .0
761            .respond_in_request(response)?;
762        self.inner.yamux.mark_substream_write_ready(substream_id);
763        Ok(())
764    }

812/// Error during a connection. The connection should be shut down.
813#[derive(Debug, derive_more::Display, derive_more::Error)]
814pub enum Error {
815    /// Error in the noise cipher. Data has most likely been corrupted.
816    #[display("Noise error: {_0}")]
817    Noise(noise::CipherError),
818    /// Error while encoding noise data.
819    #[display("{_0}")]
820    NoiseEncrypt(noise::EncryptError),
821    /// Error in the Yamux multiplexing protocol.
822    #[display("Yamux error: {_0}")]
823    Yamux(yamux::Error),
824}
825
826/// Successfully negotiated connection. Ready to be turned into a [`SingleStream`].
827pub struct ConnectionPrototype {
828    encryption: noise::Noise,
829}
830
831impl ConnectionPrototype {
832    /// Builds a new [`ConnectionPrototype`] of a connection using the Noise and Yamux protocols.
833    pub(crate) fn from_noise_yamux(encryption: noise::Noise) -> Self {
834        ConnectionPrototype { encryption }
835    }
836
837    /// Extracts the Noise state machine from this prototype.
838    pub fn into_noise_state_machine(self) -> noise::Noise {
839        self.encryption
840    }
841
842    /// Turns this prototype into an actual connection.
843    pub fn into_connection<TNow, TSubUd>(self, config: Config<TNow>) -> SingleStream<TNow, TSubUd>
844    where
845        TNow: Clone + Add<Duration, Output = TNow> + Sub<TNow, Output = Duration> + Ord,
846    {
847        let mut randomness = rand_chacha::ChaCha20Rng::from_seed(config.randomness_seed);
848
849        let mut yamux = yamux::Yamux::new(yamux::Config {
850            is_initiator: self.encryption.is_initiator(),
851            capacity: config.substreams_capacity,
852            randomness_seed: {
853                let mut seed = [0; 32];
854                randomness.fill_bytes(&mut seed);
855                seed
856            },
857            max_out_data_frame_size: NonZero::<u32>::new(8192).unwrap(), // TODO: make configurable?
858            max_simultaneous_queued_pongs: NonZero::<usize>::new(4).unwrap(),
859            max_simultaneous_rst_substreams: NonZero::<usize>::new(1024).unwrap(),
860        });
861
862        let outgoing_pings = yamux
863            .open_substream(Some((
864                substream::Substream::ping_out(config.ping_protocol.clone()),
865                None,
866            )))
867            // Can only panic if a `GoAway` has been received, or if there are too many substreams
868            // already open, which we know for sure can't happen here
869            .unwrap_or_else(|_| panic!());
870
871        SingleStream {
872            encryption: self.encryption,
873            inner: Box::new(Inner {
874                yamux,
875                outgoing_pings,
876                next_ping: config.first_out_ping,
877                ping_payload_randomness: randomness,
878                max_inbound_substreams: config.max_inbound_substreams,
879                max_protocol_name_len: config.max_protocol_name_len,
880                ping_interval: config.ping_interval,
881                ping_timeout: config.ping_timeout,
882            }),
883        }
884    }
885}

890    }
