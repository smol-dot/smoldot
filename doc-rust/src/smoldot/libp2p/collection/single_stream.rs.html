44/// State machine dedicated to a single single-stream connection.
45pub struct SingleStreamConnectionTask<TNow> {
46    /// State machine of the underlying connection.
47    connection: SingleStreamConnectionTaskInner<TNow>,
48
49    /// Buffer of messages destined to the coordinator.
50    ///
51    /// Never goes above a few elements.
52    pending_messages: VecDeque<ConnectionToCoordinatorInner>,
53}
54
55enum SingleStreamConnectionTaskInner<TNow> {
56    /// Connection is still in its handshake phase.
57    Handshake {
58        handshake: single_stream_handshake::HealthyHandshake,
59
60        /// Seed that will be used to initialize randomness when building the
61        /// [`established::SingleStream`].
62        /// This seed is computed during the handshake in order to avoid having to access a shared
63        /// state when the handshake is over. While it seems a bit dangerous to leave a randomness
64        /// seed in plain memory, the randomness isn't used for anything critical or related to
65        /// cryptography, but only for example to avoid hash collision attacks.
66        randomness_seed: [u8; 32],
67
68        /// When the handshake phase times out.
69        timeout: TNow,
70
71        /// See [`super::Config::max_inbound_substreams`].
72        max_inbound_substreams: usize,
73
74        /// See [`Config::substreams_capacity`].
75        substreams_capacity: usize,
76
77        max_protocol_name_len: usize,
78
79        /// See [`super::Config::ping_protocol`].
80        ping_protocol: Arc<str>,
81    },
82
83    /// Connection has been fully established.
84    Established {
85        established: established::SingleStream<TNow, Option<SubstreamId>>,
86
87        /// Because outgoing substream ids are assigned by the coordinator, we maintain a mapping
88        /// of the "outer ids" to "inner ids".
89        outbound_substreams_map:
90            hashbrown::HashMap<SubstreamId, established::SubstreamId, fnv::FnvBuildHasher>,
91
92        /// After a [`ConnectionToCoordinatorInner::NotificationsInOpenCancel`] or a
93        /// [`ConnectionToCoordinatorInner::NotificationsInClose`] is emitted, an
94        /// entry is added to this list. If the coordinator accepts or refuses a substream in this
95        /// list, or closes a substream in this list, the acceptance/refusal/closing is dismissed.
96        notifications_in_close_acknowledgments: VecDeque<established::SubstreamId>,
97
98        /// After a `NotificationsInOpenCancel` is emitted by the connection, an
99        /// entry is added to this list. If the coordinator accepts or refuses a substream in this
100        /// list, the acceptance/refusal is dismissed.
101        // TODO: this works only because SubstreamIds aren't reused
102        inbound_negotiated_cancel_acknowledgments:
103            hashbrown::HashSet<established::SubstreamId, fnv::FnvBuildHasher>,
104    },
105
106    /// Connection has finished its shutdown. A [`ConnectionToCoordinatorInner::ShutdownFinished`]
107    /// message has been sent and is waiting to be acknowledged.
108    ShutdownWaitingAck {
109        /// What has initiated the shutdown.
110        initiator: ShutdownInitiator,
111    },
112
113    /// Connection has finished its shutdown and its shutdown has been acknowledged. There is
114    /// nothing more to do except stop the connection task.
115    ShutdownAcked {
116        /// What has initiated the shutdown.
117        initiator: ShutdownInitiator,
118    },
119
120    /// Temporary state used to satisfy the borrow checker during state transitions.
121    Poisoned,
122}
123
124#[derive(Debug, Copy, Clone, PartialEq, Eq)]
125enum ShutdownInitiator {
126    /// The coordinator sent a [`CoordinatorToConnectionInner::StartShutdown`] message.
127    Coordinator,
128    /// [`SingleStreamConnectionTask::reset`] has been called.
129    Api,
130    /// The shutdown has been initiated due to a protocol error or because the connection has been
131    /// shut down cleanly by the remote.
132    Remote,
133}
134
135impl<TNow> SingleStreamConnectionTask<TNow>
136where
137    TNow: Clone + Add<Duration, Output = TNow> + Sub<TNow, Output = Duration> + Ord,
138{
139    // Note that the parameters of this function are a bit rough and undocumented, as this is
140    // a function only called from the parent module.
141    pub(super) fn new(config: Config<TNow>) -> Self {
142        SingleStreamConnectionTask {
143            connection: SingleStreamConnectionTaskInner::Handshake {
144                handshake: config.handshake,
145                randomness_seed: config.randomness_seed,
146                timeout: config.handshake_timeout,
147                max_inbound_substreams: config.max_inbound_substreams,
148                substreams_capacity: config.substreams_capacity,
149                max_protocol_name_len: config.max_protocol_name_len,
150                ping_protocol: config.ping_protocol,
151            },
152            pending_messages: VecDeque::with_capacity({
153                // We never buffer more than a few messages.
154                4
155            }),
156        }
157    }
158
159    /// Pulls a message to send back to the coordinator.
160    ///
161    /// This function takes ownership of `self` and optionally yields it back. If the first
162    /// option contains `None`, then no more message will be generated and the
163    /// [`SingleStreamConnectionTask`] has vanished. This will happen after the connection has been
164    /// shut down or reset.
165    /// It is possible for `self` to not be yielded back even if the [`ReadWrite`] that was last
166    /// passed to [`SingleStreamConnectionTask::read_write`] is still fully open, in which case the
167    /// API user should abruptly reset the connection, for example by sending a TCP RST flag. This
168    /// can happen for example if the connection seems unresponsive and that an attempt at closing
169    /// the connection in a clean way is futile.
170    ///
171    /// If any message is returned, it is the responsibility of the API user to send it to the
172    /// coordinator.
173    /// Do not attempt to buffer the message being returned, as it would work against the
174    /// back-pressure strategy used internally. As soon as a message is returned, it should be
175    /// delivered. If the coordinator is busy at the moment a message should be delivered, then
176    /// the entire thread of execution dedicated to this [`SingleStreamConnectionTask`] should be
177    /// paused until the coordinator is ready and the message delivered.
178    ///
179    /// Messages aren't generated spontaneously. In other words, you don't need to periodically
180    /// call this function just in case there's a new message. Messages are always generated after
181    /// either [`SingleStreamConnectionTask::read_write`] or [`SingleStreamConnectionTask::reset`]
182    /// has been called. Multiple messages can happen in a row.
183    ///
184    /// Because this function frees space in a buffer, calling
185    /// [`SingleStreamConnectionTask::read_write`] again after it has returned might read/write
186    /// more data and generate an event again. In other words, the API user should call
187    /// [`SingleStreamConnectionTask::read_write`] and
188    /// [`SingleStreamConnectionTask::pull_message_to_coordinator`] repeatedly in a loop until no
189    /// more message is generated.
190    pub fn pull_message_to_coordinator(
191        mut self,
192    ) -> (Option<Self>, Option<ConnectionToCoordinator>) {
193        // To be sure that there is no bug in the implementation, we make sure that the number of
194        // buffered messages doesn't go above a certain small limit.
195        debug_assert!(self.pending_messages.len() < 8);
196
197        let message = self
198            .pending_messages
199            .pop_front()
200            .map(|inner| ConnectionToCoordinator { inner });
201
202        // The `ShutdownAcked` state causes the task to exit.
203        let self_ret = if !matches!(
204            self.connection,
205            SingleStreamConnectionTaskInner::ShutdownAcked { .. }
206        ) {
207            Some(self)
208        } else {
209            None
210        };
211
212        (self_ret, message)
213    }
214
215    /// Injects a message that has been pulled from the coordinator.
216    ///
217    /// Calling this function might generate data to send to the connection. You should call
218    /// [`SingleStreamConnectionTask::read_write`] after this function has returned (unless you
219    /// have called [`SingleStreamConnectionTask::reset`] in the past).
220    pub fn inject_coordinator_message(&mut self, now: &TNow, message: CoordinatorToConnection) {
221        match (message.inner, &mut self.connection) {
222            (
223                CoordinatorToConnectionInner::AcceptInbound {
224                    substream_id,
225                    inbound_ty,
226                },
227                SingleStreamConnectionTaskInner::Established {
228                    established,
229                    inbound_negotiated_cancel_acknowledgments,
230                    ..
231                },
232            ) => {
233                if !inbound_negotiated_cancel_acknowledgments.remove(&substream_id) {
234                    established.accept_inbound(substream_id, inbound_ty, None);
235                } else {
236                    self.pending_messages.push_back(
237                        ConnectionToCoordinatorInner::InboundAcceptedCancel { id: substream_id },
238                    )
239                }
240            }
241            (
242                CoordinatorToConnectionInner::RejectInbound { substream_id },
243                SingleStreamConnectionTaskInner::Established {
244                    established,
245                    inbound_negotiated_cancel_acknowledgments,
246                    ..
247                },
248            ) => {
249                if !inbound_negotiated_cancel_acknowledgments.remove(&substream_id) {
250                    established.reject_inbound(substream_id);
251                }
252            }
253            (
254                CoordinatorToConnectionInner::SetMaxProtocolNameLen { new_max_length },
255                SingleStreamConnectionTaskInner::Handshake {
256                    max_protocol_name_len,
257                    ..
258                },
259            ) => {
260                *max_protocol_name_len = new_max_length;
261            }
262            (
263                CoordinatorToConnectionInner::SetMaxProtocolNameLen { new_max_length },
264                SingleStreamConnectionTaskInner::Established { established, .. },
265            ) => {
266                established.set_max_protocol_name_len(new_max_length);
267            }
268            (
269                CoordinatorToConnectionInner::StartRequest {
270                    protocol_name,
271                    request_data,
272                    timeout,
273                    max_response_size,
274                    substream_id,
275                },
276                SingleStreamConnectionTaskInner::Established {
277                    established,
278                    outbound_substreams_map,
279                    ..
280                },
281            ) => {
282                let inner_substream_id = established.add_request(
283                    protocol_name,
284                    request_data,
285                    now.clone() + timeout,
286                    max_response_size,
287                    Some(substream_id),
288                );
289
290                let _prev_value = outbound_substreams_map.insert(substream_id, inner_substream_id);
291                debug_assert!(_prev_value.is_none());
292            }
293            (
294                CoordinatorToConnectionInner::OpenOutNotifications {
295                    protocol_name,
296                    max_handshake_size,
297                    handshake,
298                    handshake_timeout,
299                    substream_id: outer_substream_id,
300                },
301                SingleStreamConnectionTaskInner::Established {
302                    established,
303                    outbound_substreams_map,
304                    ..
305                },
306            ) => {
307                let inner_substream_id = established.open_notifications_substream(
308                    protocol_name,
309                    handshake,
310                    max_handshake_size,
311                    now.clone() + handshake_timeout,
312                    Some(outer_substream_id),
313                );
314
315                let _prev_value =
316                    outbound_substreams_map.insert(outer_substream_id, inner_substream_id);
317                debug_assert!(_prev_value.is_none());
318            }
319            (
320                CoordinatorToConnectionInner::CloseOutNotifications { substream_id },
321                SingleStreamConnectionTaskInner::Established {
322                    established,
323                    outbound_substreams_map,
324                    ..
325                },
326            ) => {
327                // It is possible that the remote has closed the outbound notification substream
328                // while the `CloseOutNotifications` message was being delivered, or that the API
329                // user close the substream before the message about the substream being closed
330                // was delivered to the coordinator.
331                if let Some(inner_substream_id) = outbound_substreams_map.remove(&substream_id) {
332                    established.close_out_notifications_substream(inner_substream_id);
333                }
334            }
335            (
336                CoordinatorToConnectionInner::QueueNotification {
337                    substream_id,
338                    notification,
339                },
340                SingleStreamConnectionTaskInner::Established {
341                    established,
342                    outbound_substreams_map,
343                    ..
344                },
345            ) => {
346                // It is possible that the remote has closed the outbound notification substream
347                // while a `QueueNotification` message was being delivered, or that the API user
348                // queued a notification before the message about the substream being closed was
349                // delivered to the coordinator.
350                // If that happens, we intentionally silently discard the message, causing the
351                // notification to not be sent. This is consistent with the guarantees about
352                // notifications delivered that are documented in the public API.
353                if let Some(inner_substream_id) = outbound_substreams_map.get(&substream_id) {
354                    established.write_notification_unbounded(*inner_substream_id, notification);
355                }
356            }
357            (
358                CoordinatorToConnectionInner::AnswerRequest {
359                    substream_id,
360                    response,
361                },
362                SingleStreamConnectionTaskInner::Established { established, .. },
363            ) => match established.respond_in_request(substream_id, response) {
364                Ok(()) => {}
365                Err(established::RespondInRequestError::SubstreamClosed) => {
366                    // As documented, answering an obsolete request is simply ignored.
367                }
368            },
369            (
370                CoordinatorToConnectionInner::AcceptInNotifications {
371                    substream_id,
372                    handshake,
373                    max_notification_size,
374                },
375                SingleStreamConnectionTaskInner::Established {
376                    established,
377                    notifications_in_close_acknowledgments,
378                    ..
379                },
380            ) => {
381                if let Some(idx) = notifications_in_close_acknowledgments
382                    .iter()
383                    .position(|s| *s == substream_id)
384                {
385                    notifications_in_close_acknowledgments.remove(idx);
386                } else {
387                    established.accept_in_notifications_substream(
388                        substream_id,
389                        handshake,
390                        max_notification_size,
391                    );
392                }
393            }
394            (
395                CoordinatorToConnectionInner::RejectInNotifications { substream_id },
396                SingleStreamConnectionTaskInner::Established {
397                    established,
398                    notifications_in_close_acknowledgments,
399                    ..
400                },
401            ) => {
402                if let Some(idx) = notifications_in_close_acknowledgments
403                    .iter()
404                    .position(|s| *s == substream_id)
405                {
406                    notifications_in_close_acknowledgments.remove(idx);
407                } else {
408                    established.reject_in_notifications_substream(substream_id);
409                }
410            }
411            (
412                CoordinatorToConnectionInner::CloseInNotifications {
413                    substream_id,
414                    timeout,
415                },
416                SingleStreamConnectionTaskInner::Established {
417                    established,
418                    notifications_in_close_acknowledgments,
419                    ..
420                },
421            ) => {
422                if let Some(idx) = notifications_in_close_acknowledgments
423                    .iter()
424                    .position(|s| *s == substream_id)
425                {
426                    notifications_in_close_acknowledgments.remove(idx);
427                } else {
428                    established
429                        .close_in_notifications_substream(substream_id, now.clone() + timeout);
430                }
431            }
432            (
433                CoordinatorToConnectionInner::StartShutdown { .. },
434                SingleStreamConnectionTaskInner::Established { .. }
435                | SingleStreamConnectionTaskInner::Handshake { .. },
436            ) => {
437                // TODO: implement proper shutdown
438                self.pending_messages
439                    .push_back(ConnectionToCoordinatorInner::StartShutdown(None));
440                self.pending_messages
441                    .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
442                self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
443                    initiator: ShutdownInitiator::Coordinator,
444                };
445            }
446            (
447                CoordinatorToConnectionInner::AcceptInbound { .. }
448                | CoordinatorToConnectionInner::RejectInbound { .. }
449                | CoordinatorToConnectionInner::SetMaxProtocolNameLen { .. }
450                | CoordinatorToConnectionInner::AcceptInNotifications { .. }
451                | CoordinatorToConnectionInner::RejectInNotifications { .. }
452                | CoordinatorToConnectionInner::CloseInNotifications { .. }
453                | CoordinatorToConnectionInner::StartRequest { .. }
454                | CoordinatorToConnectionInner::AnswerRequest { .. }
455                | CoordinatorToConnectionInner::OpenOutNotifications { .. }
456                | CoordinatorToConnectionInner::CloseOutNotifications { .. }
457                | CoordinatorToConnectionInner::QueueNotification { .. },
458                SingleStreamConnectionTaskInner::Handshake { .. }
459                | SingleStreamConnectionTaskInner::ShutdownAcked { .. },
460            ) => unreachable!(),
461            (
462                CoordinatorToConnectionInner::AcceptInbound { .. }
463                | CoordinatorToConnectionInner::RejectInbound { .. }
464                | CoordinatorToConnectionInner::SetMaxProtocolNameLen { .. }
465                | CoordinatorToConnectionInner::AcceptInNotifications { .. }
466                | CoordinatorToConnectionInner::RejectInNotifications { .. }
467                | CoordinatorToConnectionInner::CloseInNotifications { .. }
468                | CoordinatorToConnectionInner::StartRequest { .. }
469                | CoordinatorToConnectionInner::AnswerRequest { .. }
470                | CoordinatorToConnectionInner::OpenOutNotifications { .. }
471                | CoordinatorToConnectionInner::CloseOutNotifications { .. }
472                | CoordinatorToConnectionInner::QueueNotification { .. },
473                SingleStreamConnectionTaskInner::ShutdownWaitingAck { .. },
474            )
475            | (
476                CoordinatorToConnectionInner::StartShutdown,
477                SingleStreamConnectionTaskInner::ShutdownWaitingAck {
478                    initiator: ShutdownInitiator::Api | ShutdownInitiator::Remote,
479                },
480            ) => {
481                // There might still be some messages coming from the coordinator after the
482                // connection task has sent a message indicating that it has shut down. This is
483                // due to the concurrent nature of the API and doesn't indicate a bug. These
484                // messages are simply ignored by the connection task.
485            }
486            (
487                CoordinatorToConnectionInner::ShutdownFinishedAck,
488                SingleStreamConnectionTaskInner::ShutdownWaitingAck { initiator },
489            ) => {
490                self.connection = SingleStreamConnectionTaskInner::ShutdownAcked {
491                    initiator: *initiator,
492                };
493            }
494            (
495                CoordinatorToConnectionInner::StartShutdown,
496                SingleStreamConnectionTaskInner::ShutdownWaitingAck {
497                    initiator: ShutdownInitiator::Coordinator,
498                    ..
499                }
500                | SingleStreamConnectionTaskInner::ShutdownAcked { .. },
501            ) => unreachable!(),
502            (CoordinatorToConnectionInner::ShutdownFinishedAck, _) => unreachable!(),
503            (_, SingleStreamConnectionTaskInner::Poisoned) => unreachable!(),
504        }
505    }
506
507    /// Sets the state of the connection to "reset".
508    ///
509    /// This should be called if the remote abruptly closes the connection, such as with a TCP/IP
510    /// RST flag.
511    ///
512    /// After this function has been called, it is illegal to call
513    /// [`SingleStreamConnectionTask::read_write`] or [`SingleStreamConnectionTask::reset`] again.
514    ///
515    /// Calling this function might have generated messages for the coordinator.
516    /// [`SingleStreamConnectionTask::pull_message_to_coordinator`] should be called afterwards in
517    /// order to process these messages.
518    ///
519    /// # Panic
520    ///
521    /// Panics if [`SingleStreamConnectionTask::reset`] has been called in the past.
522    ///
523    pub fn reset(&mut self) {
524        match self.connection {
525            SingleStreamConnectionTaskInner::ShutdownWaitingAck {
526                initiator: ShutdownInitiator::Api,
527            }
528            | SingleStreamConnectionTaskInner::ShutdownAcked {
529                initiator: ShutdownInitiator::Api,
530            } => {
531                // It is illegal to call `reset` a second time.
532                panic!()
533            }
534            SingleStreamConnectionTaskInner::ShutdownWaitingAck { ref mut initiator }
535            | SingleStreamConnectionTaskInner::ShutdownAcked { ref mut initiator } => {
536                // Mark the initiator as being the API in order to track proper API usage.
537                *initiator = ShutdownInitiator::Api;
538            }
539            _ => {
540                self.pending_messages
541                    .push_back(ConnectionToCoordinatorInner::StartShutdown(Some(
542                        ShutdownCause::RemoteReset,
543                    )));
544                self.pending_messages
545                    .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
546                self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
547                    initiator: ShutdownInitiator::Api,
548                };
549            }
550        }
551    }
552
553    /// Returns `true` if [`SingleStreamConnectionTask::reset`] has been called in the past.
554    pub fn is_reset_called(&self) -> bool {
555        matches!(
556            self.connection,
557            SingleStreamConnectionTaskInner::ShutdownWaitingAck {
558                initiator: ShutdownInitiator::Api,
559            } | SingleStreamConnectionTaskInner::ShutdownAcked {
560                initiator: ShutdownInitiator::Api,
561            }
562        )
563    }
564
565    /// Reads data coming from the connection, updates the internal state machine, and writes data
566    /// destined to the connection through the [`ReadWrite`].
567    ///
568    /// Calling this function might have generated messages for the coordinator.
569    /// [`SingleStreamConnectionTask::pull_message_to_coordinator`] should be called afterwards in
570    /// order to process these messages.
571    ///
572    /// # Panic
573    ///
574    /// Panics if [`SingleStreamConnectionTask::reset`] has been called in the past.
575    ///
576    pub fn read_write(&mut self, read_write: &mut ReadWrite<TNow>) {
577        // There is already at least one pending message. We back-pressure the connection by not
578        // performing any reading or writing, as this might generate more messages and open the
579        // door for a DoS attack by the remote. As documented, the API user is supposed to pull
580        // messages after this function has returned, meaning that they will drain the messages.
581        if !self.pending_messages.is_empty() {
582            return;
583        }
584
585        match mem::replace(
586            &mut self.connection,
587            SingleStreamConnectionTaskInner::Poisoned,
588        ) {
589            SingleStreamConnectionTaskInner::Established {
590                established,
591                mut outbound_substreams_map,
592                mut notifications_in_close_acknowledgments,
593                mut inbound_negotiated_cancel_acknowledgments,
594            } => match established.read_write(read_write) {
595                Ok((connection, event)) => {
596                    if read_write.is_dead() && read_write.wake_up_after.is_none() {
597                        self.pending_messages.push_back(
598                            ConnectionToCoordinatorInner::StartShutdown(Some(
599                                ShutdownCause::CleanShutdown,
600                            )),
601                        );
602                        self.pending_messages
603                            .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
604                        self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
605                            initiator: ShutdownInitiator::Remote,
606                        };
607                        return;
608                    }
609
610                    match event {
611                        Some(established::Event::NewOutboundSubstreamsForbidden) => {
612                            // TODO: handle properly
613                            self.pending_messages.push_back(
614                                ConnectionToCoordinatorInner::StartShutdown(Some(
615                                    ShutdownCause::CleanShutdown,
616                                )),
617                            );
618                            self.pending_messages
619                                .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
620                            self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
621                                initiator: ShutdownInitiator::Remote,
622                            };
623                            return;
624                        }
625                        Some(established::Event::InboundError(err)) => {
626                            self.pending_messages
627                                .push_back(ConnectionToCoordinatorInner::InboundError(err));
628                        }
629                        Some(established::Event::InboundNegotiated { id, protocol_name }) => {
630                            self.pending_messages.push_back(
631                                ConnectionToCoordinatorInner::InboundNegotiated {
632                                    id,
633                                    protocol_name,
634                                },
635                            );
636                        }
637                        Some(established::Event::InboundNegotiatedCancel { id, .. }) => {
638                            inbound_negotiated_cancel_acknowledgments.insert(id);
639                        }
640                        Some(established::Event::InboundAcceptedCancel { id, .. }) => {
641                            self.pending_messages.push_back(
642                                ConnectionToCoordinatorInner::InboundAcceptedCancel { id },
643                            );
644                        }
645                        Some(established::Event::RequestIn { id, request, .. }) => {
646                            self.pending_messages
647                                .push_back(ConnectionToCoordinatorInner::RequestIn { id, request });
648                        }
649                        Some(established::Event::Response {
650                            response,
651                            user_data,
652                            ..
653                        }) => {
654                            let Some(outer_substream_id) = user_data else {
655                                panic!()
656                            };
657                            outbound_substreams_map.remove(&outer_substream_id).unwrap();
658                            self.pending_messages.push_back(
659                                ConnectionToCoordinatorInner::Response {
660                                    response,
661                                    id: outer_substream_id,
662                                },
663                            );
664                        }
665                        Some(established::Event::NotificationsInOpen { id, handshake, .. }) => {
666                            self.pending_messages.push_back(
667                                ConnectionToCoordinatorInner::NotificationsInOpen { id, handshake },
668                            );
669                        }
670                        Some(established::Event::NotificationsInOpenCancel { id, .. }) => {
671                            notifications_in_close_acknowledgments.push_back(id);
672                            self.pending_messages.push_back(
673                                ConnectionToCoordinatorInner::NotificationsInOpenCancel { id },
674                            );
675                        }
676                        Some(established::Event::NotificationIn { id, notification }) => {
677                            self.pending_messages.push_back(
678                                ConnectionToCoordinatorInner::NotificationIn { id, notification },
679                            );
680                        }
681                        Some(established::Event::NotificationsInClose { id, outcome, .. }) => {
682                            // TODO: only do this if not sent by API user
683                            notifications_in_close_acknowledgments.push_back(id);
684                            self.pending_messages.push_back(
685                                ConnectionToCoordinatorInner::NotificationsInClose { id, outcome },
686                            );
687                        }
688                        Some(established::Event::NotificationsOutResult { id, result }) => {
689                            let (outer_substream_id, result) = match result {
690                                Ok(r) => {
691                                    let Some(outer_substream_id) = connection[id] else {
692                                        panic!()
693                                    };
694                                    (outer_substream_id, Ok(r))
695                                }
696                                Err((err, ud)) => {
697                                    let Some(outer_substream_id) = ud else {
698                                        panic!()
699                                    };
700                                    outbound_substreams_map.remove(&outer_substream_id);
701                                    (outer_substream_id, Err(NotificationsOutErr::Substream(err)))
702                                }
703                            };
704
705                            self.pending_messages.push_back(
706                                ConnectionToCoordinatorInner::NotificationsOutResult {
707                                    id: outer_substream_id,
708                                    result,
709                                },
710                            );
711                        }
712                        Some(established::Event::NotificationsOutCloseDemanded { id }) => {
713                            let Some(outer_substream_id) = connection[id] else {
714                                panic!()
715                            };
716                            self.pending_messages.push_back(
717                                ConnectionToCoordinatorInner::NotificationsOutCloseDemanded {
718                                    id: outer_substream_id,
719                                },
720                            );
721                        }
722                        Some(established::Event::NotificationsOutReset { user_data, .. }) => {
723                            let Some(outer_substream_id) = user_data else {
724                                panic!()
725                            };
726                            outbound_substreams_map.remove(&outer_substream_id);
727                            self.pending_messages.push_back(
728                                ConnectionToCoordinatorInner::NotificationsOutReset {
729                                    id: outer_substream_id,
730                                },
731                            );
732                        }
733                        Some(established::Event::PingOutSuccess { ping_time }) => {
734                            self.pending_messages.push_back(
735                                ConnectionToCoordinatorInner::PingOutSuccess { ping_time },
736                            );
737                        }
738                        Some(established::Event::PingOutFailed) => {
739                            self.pending_messages
740                                .push_back(ConnectionToCoordinatorInner::PingOutFailed);
741                        }
742                        None => {}
743                    }
744
745                    self.connection = SingleStreamConnectionTaskInner::Established {
746                        established: connection,
747                        outbound_substreams_map,
748                        notifications_in_close_acknowledgments,
749                        inbound_negotiated_cancel_acknowledgments,
750                    };
751                }
752                Err(err) => {
753                    self.pending_messages
754                        .push_back(ConnectionToCoordinatorInner::StartShutdown(Some(
755                            ShutdownCause::ProtocolError(err),
756                        )));
757                    self.pending_messages
758                        .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
759                    self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
760                        initiator: ShutdownInitiator::Remote,
761                    };
762                }
763            },
764
765            SingleStreamConnectionTaskInner::Handshake {
766                mut handshake,
767                randomness_seed,
768                timeout,
769                max_inbound_substreams,
770                substreams_capacity,
771                max_protocol_name_len,
772                ping_protocol,
773            } => {
774                // Check that the handshake isn't taking too long.
775                //
776                // Note that we check this condition before looking into the incoming data,
777                // and it is possible for the buffers to contain the data that leads to the
778                // handshake being finished. If that is the case, however, it is impossible to
779                // know whether this data arrived before or after the timeout.
780                // Whether to put this check before or after reading the buffer is a choice
781                // between having false negatives or having false positives for the timeout.
782                // We are more strict than necessary by having the check before, but this
783                // guarantees that no horrendously slow connections can accidentally make their
784                // way through.
785                if timeout < read_write.now {
786                    self.pending_messages
787                        .push_back(ConnectionToCoordinatorInner::StartShutdown(Some(
788                            ShutdownCause::HandshakeTimeout,
789                        )));
790                    self.pending_messages
791                        .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
792                    self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
793                        initiator: ShutdownInitiator::Remote,
794                    };
795                    return;
796                }
797
798                loop {
799                    let (read_before, written_before) =
800                        (read_write.read_bytes, read_write.write_bytes_queued);
801
802                    let result = match handshake.read_write(read_write) {
803                        Ok(rw) => rw,
804                        Err(err) => {
805                            self.pending_messages.push_back(
806                                ConnectionToCoordinatorInner::StartShutdown(Some(
807                                    ShutdownCause::HandshakeError(err),
808                                )),
809                            );
810                            self.pending_messages
811                                .push_back(ConnectionToCoordinatorInner::ShutdownFinished);
812                            self.connection = SingleStreamConnectionTaskInner::ShutdownWaitingAck {
813                                initiator: ShutdownInitiator::Remote,
814                            };
815                            return;
816                        }
817                    };
818
819                    match result {
820                        single_stream_handshake::Handshake::Healthy(updated_handshake)
821                            if (read_before, written_before)
822                                == (read_write.read_bytes, read_write.write_bytes_queued) =>
823                        {
824                            // `read_write()` should be called again as soon as possible
825                            // after `timeout`.
826                            read_write.wake_up_after(&timeout);
827                            self.connection = SingleStreamConnectionTaskInner::Handshake {
828                                handshake: updated_handshake,
829                                randomness_seed,
830                                timeout,
831                                max_inbound_substreams,
832                                substreams_capacity,
833                                max_protocol_name_len,
834                                ping_protocol,
835                            };
836                            break;
837                        }
838                        single_stream_handshake::Handshake::Healthy(updated_handshake) => {
839                            handshake = updated_handshake;
840                        }
841                        single_stream_handshake::Handshake::Success {
842                            remote_peer_id,
843                            connection,
844                        } => {
845                            self.pending_messages.push_back(
846                                ConnectionToCoordinatorInner::HandshakeFinished(remote_peer_id),
847                            );
848
849                            self.connection = SingleStreamConnectionTaskInner::Established {
850                                established: connection.into_connection(established::Config {
851                                    max_inbound_substreams,
852                                    substreams_capacity,
853                                    max_protocol_name_len,
854                                    randomness_seed,
855                                    ping_protocol: ping_protocol.to_string(), // TODO: cloning :-/
856                                    ping_interval: Duration::from_secs(20),   // TODO: hardcoded
857                                    ping_timeout: Duration::from_secs(10),    // TODO: hardcoded
858                                    first_out_ping: read_write.now.clone() + Duration::from_secs(2), // TODO: hardcoded
859                                }),
860                                outbound_substreams_map:
861                                    hashbrown::HashMap::with_capacity_and_hasher(
862                                        0,
863                                        Default::default(),
864                                    ), // TODO: capacity?
865                                notifications_in_close_acknowledgments: VecDeque::with_capacity(4),
866                                inbound_negotiated_cancel_acknowledgments:
867                                    hashbrown::HashSet::with_capacity_and_hasher(
868                                        2,
869                                        Default::default(),
870                                    ),
871                            };
872                            break;
873                        }
874                    }
875                }
876            }
877
878            c @ (SingleStreamConnectionTaskInner::ShutdownWaitingAck {
879                initiator: ShutdownInitiator::Coordinator | ShutdownInitiator::Remote,
880            }
881            | SingleStreamConnectionTaskInner::ShutdownAcked {
882                initiator: ShutdownInitiator::Coordinator | ShutdownInitiator::Remote,
883            }) => {
884                // The user might legitimately call this function after the connection has
885                // already shut down. In that case, just do nothing.
886                self.connection = c;
887
888                // This might have been done already during the shutdown process, but we do it
889                // again just in case.
890                read_write.close_write();
891            }
892
893            SingleStreamConnectionTaskInner::ShutdownWaitingAck {
894                initiator: ShutdownInitiator::Api,
895            }
896            | SingleStreamConnectionTaskInner::ShutdownAcked {
897                initiator: ShutdownInitiator::Api,
898            } => {
899                // As documented, it is illegal to call this function after calling `reset()`.
900                panic!()
901            }
902
903            SingleStreamConnectionTaskInner::Poisoned => unreachable!(),
904        }
905    }
