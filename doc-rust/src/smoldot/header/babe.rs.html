24/// A consensus log item for BABE.
25#[derive(Debug, Clone, PartialEq, Eq)]
26pub enum BabeConsensusLogRef<'a> {
27    /// The epoch has changed. This provides information about the _next_
28    /// epoch - information about the _current_ epoch (i.e. the one we've just
29    /// entered) should already be available earlier in the chain.
30    NextEpochData(BabeNextEpochRef<'a>),
31    /// Disable the authority with given index.
32    OnDisabled(u32),
33    /// The epoch has changed, and the epoch after the current one will
34    /// enact different epoch configurations.
35    NextConfigData(BabeNextConfig),
36}
37
38impl<'a> BabeConsensusLogRef<'a> {
39    /// Decodes a [`BabeConsensusLogRef`] from a slice of bytes.
40    pub fn from_slice(slice: &'a [u8]) -> Result<Self, Error> {
41        Ok(match slice.first() {
42            Some(1) => {
43                BabeConsensusLogRef::NextEpochData(BabeNextEpochRef::from_slice(&slice[1..])?)
44            }
45            Some(2) => {
46                let n = u32::from_le_bytes(
47                    <[u8; 4]>::try_from(&slice[1..]).map_err(|_| Error::DigestItemDecodeError)?,
48                );
49                BabeConsensusLogRef::OnDisabled(n)
50            }
51            Some(3) => {
52                // The Babe configuration info starts with a version number that is always `1`
53                // at the moment.
54                if slice.len() < 2 || slice[1] != 1 {
55                    return Err(Error::BadBabeNextConfigVersion);
56                }
57                BabeConsensusLogRef::NextConfigData(BabeNextConfig::from_slice(&slice[2..])?)
58            }
59            Some(_) => return Err(Error::BadBabeConsensusRefType),
60            None => return Err(Error::TooShort),
61        })
62    }
63
64    /// Returns an iterator to list of buffers which, when concatenated, produces the SCALE
65    /// encoding of that object.
66    pub fn scale_encoding(&self) -> impl Iterator<Item = impl AsRef<[u8]> + Clone> + Clone {
67        let index = iter::once(match self {
68            BabeConsensusLogRef::NextEpochData(_) => [1],
69            BabeConsensusLogRef::OnDisabled(_) => [2],
70            BabeConsensusLogRef::NextConfigData(_) => [3],
71        });
72
73        let body = match self {
74            BabeConsensusLogRef::NextEpochData(digest) => either::Left(either::Left(
75                digest.scale_encoding().map(either::Left).map(either::Left),
76            )),
77            BabeConsensusLogRef::OnDisabled(digest) => either::Left(either::Right(iter::once(
78                either::Left(either::Right(digest.to_le_bytes())),
79            ))),
80            BabeConsensusLogRef::NextConfigData(digest) => either::Right(
81                iter::once(either::Right(either::Left([1]))).chain(
82                    digest
83                        .scale_encoding()
84                        .map(either::Right)
85                        .map(either::Right),
86                ),
87            ),
88        };
89
90        index.map(either::Left).chain(body.map(either::Right))
91    }
92}

94impl<'a> From<&'a BabeConsensusLog> for BabeConsensusLogRef<'a> {
95    fn from(a: &'a BabeConsensusLog) -> Self {
96        match a {
97            BabeConsensusLog::NextEpochData(v) => BabeConsensusLogRef::NextEpochData(v.into()),
98            BabeConsensusLog::OnDisabled(v) => BabeConsensusLogRef::OnDisabled(*v),
99            BabeConsensusLog::NextConfigData(v) => BabeConsensusLogRef::NextConfigData(*v),
100        }
101    }
102}

128/// Information about the next epoch. This is broadcast in the first block
129/// of the epoch.
130#[derive(Debug, Clone, PartialEq, Eq)]
131pub struct BabeNextEpochRef<'a> {
132    /// The authorities.
133    pub authorities: BabeAuthoritiesIter<'a>,
134
135    /// The value of randomness to use for the slot-assignment.
136    pub randomness: &'a [u8; 32],
137}
138
139impl<'a> BabeNextEpochRef<'a> {
140    /// Decodes a [`BabePreDigestRef`] from a slice of bytes.
141    pub fn from_slice(slice: &'a [u8]) -> Result<Self, Error> {
142        let (slice, authorities_len) =
143            match crate::util::nom_scale_compact_usize::<nom::error::Error<&[u8]>>(slice) {
144                Ok(s) => s,
145                Err(_) => return Err(Error::TooShort),
146            };
147
148        if slice.len() != authorities_len * 40 + 32 {
149            return Err(Error::TooShort);
150        }
151
152        Ok(BabeNextEpochRef {
153            authorities: BabeAuthoritiesIter(BabeAuthoritiesIterInner::Raw(
154                slice[0..authorities_len * 40].chunks(40),
155            )),
156            randomness: <&[u8; 32]>::try_from(&slice[authorities_len * 40..]).unwrap(),
157        })
158    }
159
160    /// Returns an iterator to list of buffers which, when concatenated, produces the SCALE
161    /// encoding of that object.
162    pub fn scale_encoding(
163        &self,
164    ) -> impl Iterator<Item = impl AsRef<[u8]> + Clone + use<'a>> + Clone + use<'a> {
165        let header = util::encode_scale_compact_usize(self.authorities.len());
166        iter::once(either::Left(header))
167            .chain(
168                self.authorities
169                    .clone()
170                    .flat_map(|a| a.scale_encoding())
171                    .map(|buf| either::Right(either::Left(buf))),
172            )
173            .chain(iter::once(either::Right(either::Right(
174                &self.randomness[..],
175            ))))
176    }
177}

181        BabeNextEpochRef {
182            authorities: BabeAuthoritiesIter(BabeAuthoritiesIterInner::List(a.authorities.iter())),
183            randomness: &a.randomness,
184        }
185    }

212#[derive(Clone)]
213enum BabeAuthoritiesIterInner<'a> {
214    List(slice::Iter<'a, BabeAuthority>),
215    Raw(slice::Chunks<'a, u8>),
216}

218impl<'a> BabeAuthoritiesIter<'a> {
219    /// Builds a new [`BabeAuthoritiesIter`] iterating over the given slice.
220    pub fn from_slice(slice: &'a [BabeAuthority]) -> Self {
221        Self(BabeAuthoritiesIterInner::List(slice.iter()))
222    }

225impl<'a> Iterator for BabeAuthoritiesIter<'a> {
226    type Item = BabeAuthorityRef<'a>;
227
228    fn next(&mut self) -> Option<Self::Item> {
229        match &mut self.0 {
230            BabeAuthoritiesIterInner::List(l) => l.next().map(Into::into),
231            BabeAuthoritiesIterInner::Raw(l) => {
232                let item = l.next()?;
233                assert_eq!(item.len(), 40);
234                Some(BabeAuthorityRef {
235                    public_key: <&[u8; 32]>::try_from(&item[0..32]).unwrap(),
236                    weight: u64::from_le_bytes(<[u8; 8]>::try_from(&item[32..40]).unwrap()),
237                })
238            }
239        }
240    }
241
242    fn size_hint(&self) -> (usize, Option<usize>) {
243        match &self.0 {
244            BabeAuthoritiesIterInner::List(l) => l.size_hint(),
245            BabeAuthoritiesIterInner::Raw(l) => l.size_hint(),
246        }
247    }
248}

255        let mut b = other.clone();
256        loop {
257            match (a.next(), b.next()) {
258                (Some(a), Some(b)) if a == b => {}
259                (None, None) => return true,
260                _ => return false,
261            }
262        }
263    }

325/// Information about the next epoch config, if changed. This is broadcast in the first
326/// block of the epoch, and applies using the same rules as `NextEpochDescriptor`.
327#[derive(Debug, Copy, Clone, PartialEq, Eq)]
328pub struct BabeNextConfig {
329    /// Value of `c` in `BabeEpochConfiguration`.
330    pub c: (u64, u64),
331    /// Value of `allowed_slots` in `BabeEpochConfiguration`.
332    pub allowed_slots: BabeAllowedSlots,
333}
334
335impl BabeNextConfig {
336    /// Decodes a [`BabeNextConfig`] from a slice of bytes.
337    pub fn from_slice(slice: &[u8]) -> Result<Self, Error> {
338        if slice.len() < 16 {
339            return Err(Error::TooShort);
340        }
341
342        let c0 = u64::from_le_bytes(<[u8; 8]>::try_from(&slice[..8]).unwrap());
343        let c1 = u64::from_le_bytes(<[u8; 8]>::try_from(&slice[8..16]).unwrap());
344        let allowed_slots = BabeAllowedSlots::from_slice(&slice[16..])?;
345
346        Ok(BabeNextConfig {
347            c: (c0, c1),
348            allowed_slots,
349        })
350    }
351
352    /// Returns an iterator to list of buffers which, when concatenated, produces the SCALE
353    /// encoding of that object.
354    pub fn scale_encoding(&self) -> impl Iterator<Item = impl AsRef<[u8]> + Clone> + Clone {
355        iter::once(either::Left(self.c.0.to_le_bytes()))
356            .chain(iter::once(either::Left(self.c.1.to_le_bytes())))
357            .chain(self.allowed_slots.scale_encoding().map(either::Right))
358    }
359}

394/// A BABE pre-runtime digest. This contains all data required to validate a
395/// block and for the BABE runtime module. Slots can be assigned to a primary
396/// (VRF based) and to a secondary (slot number based).
397#[derive(Clone, Debug, PartialEq, Eq)]
398pub enum BabePreDigestRef<'a> {
399    /// A primary VRF-based slot assignment.
400    Primary(BabePrimaryPreDigestRef<'a>),
401    /// A secondary deterministic slot assignment.
402    SecondaryPlain(BabeSecondaryPlainPreDigest),
403    /// A secondary deterministic slot assignment with VRF outputs.
404    SecondaryVRF(BabeSecondaryVRFPreDigestRef<'a>),
405}
406
407impl<'a> BabePreDigestRef<'a> {
408    /// Decodes a [`BabePreDigestRef`] from a slice of bytes.
409    pub fn from_slice(slice: &'a [u8]) -> Result<Self, Error> {
410        Ok(match slice.first() {
411            Some(1) => BabePreDigestRef::Primary(BabePrimaryPreDigestRef::from_slice(&slice[1..])?),
412            Some(2) => BabePreDigestRef::SecondaryPlain(BabeSecondaryPlainPreDigest::from_slice(
413                &slice[1..],
414            )?),
415            Some(3) => BabePreDigestRef::SecondaryVRF(BabeSecondaryVRFPreDigestRef::from_slice(
416                &slice[1..],
417            )?),
418            Some(_) => return Err(Error::BadBabePreDigestRefType),
419            None => return Err(Error::TooShort),
420        })
421    }
422
423    /// Returns `true` for [`BabePreDigestRef::Primary`].
424    pub fn is_primary(&self) -> bool {
425        matches!(self, BabePreDigestRef::Primary(_))
426    }
427
428    /// Returns the slot number stored in the header.
429    pub fn slot_number(&self) -> u64 {
430        match self {
431            BabePreDigestRef::Primary(digest) => digest.slot_number,
432            BabePreDigestRef::SecondaryPlain(digest) => digest.slot_number,
433            BabePreDigestRef::SecondaryVRF(digest) => digest.slot_number,
434        }
435    }
436
437    /// Returns an iterator to list of buffers which, when concatenated, produces the SCALE
438    /// encoding of that object.
439    pub fn scale_encoding(&self) -> impl Iterator<Item = impl AsRef<[u8]> + Clone> + Clone {
440        let index = iter::once(match self {
441            BabePreDigestRef::Primary(_) => [1],
442            BabePreDigestRef::SecondaryPlain(_) => [2],
443            BabePreDigestRef::SecondaryVRF(_) => [3],
444        });
445
446        let body = match self {
447            BabePreDigestRef::Primary(digest) => either::Left(either::Left(
448                digest
449                    .scale_encoding()
450                    .map(|buf| either::Left(either::Left(buf))),
451            )),
452            BabePreDigestRef::SecondaryPlain(digest) => either::Left(either::Right(
453                digest
454                    .scale_encoding()
455                    .map(|buf| either::Left(either::Right(buf))),
456            )),
457            BabePreDigestRef::SecondaryVRF(digest) => {
458                either::Right(digest.scale_encoding().map(either::Right))
459            }
460        };
461
462        index.map(either::Left).chain(body.map(either::Right))
463    }
464}

499/// Raw BABE primary slot assignment pre-digest.
500#[derive(Debug, Clone)]
501pub struct BabePrimaryPreDigestRef<'a> {
502    /// Authority index
503    pub authority_index: u32,
504    /// Slot number
505    pub slot_number: u64,
506    /// VRF output
507    pub vrf_output: &'a [u8; 32],
508    /// VRF proof
509    pub vrf_proof: &'a [u8; 64],
510}
511
512impl<'a> BabePrimaryPreDigestRef<'a> {
513    /// Decodes a [`BabePrimaryPreDigestRef`] from a slice of bytes.
514    pub fn from_slice(slice: &'a [u8]) -> Result<Self, Error> {
515        if slice.len() != 4 + 8 + 32 + 64 {
516            return Err(Error::TooShort);
517        }
518
519        Ok(BabePrimaryPreDigestRef {
520            authority_index: u32::from_le_bytes(<[u8; 4]>::try_from(&slice[0..4]).unwrap()),
521            slot_number: u64::from_le_bytes(<[u8; 8]>::try_from(&slice[4..12]).unwrap()),
522            vrf_output: TryFrom::try_from(&slice[12..44]).unwrap(),
523            vrf_proof: TryFrom::try_from(&slice[44..108]).unwrap(),
524        })
525    }
526
527    /// Returns an iterator to list of buffers which, when concatenated, produces the SCALE
528    /// encoding of that object.
529    pub fn scale_encoding(
530        &self,
531    ) -> impl Iterator<Item = impl AsRef<[u8]> + Clone + use<'a>> + Clone + use<'a> {
532        let header = iter::once(either::Left(self.authority_index.to_le_bytes()))
533            .chain(iter::once(either::Right(self.slot_number.to_le_bytes())))
534            .map(either::Left);
535
536        header
537            .chain(iter::once(either::Right(&self.vrf_output[..])))
538            .chain(iter::once(either::Right(&self.vrf_proof[..])))
539    }
540}
