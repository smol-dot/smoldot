25impl<T> NonFinalizedTree<T> {
26    /// Returns a list of blocks (by their height and hash) that need to be finalized before any
27    /// of their descendants can be finalized.
28    ///
29    /// In other words, blocks in the [`NonFinalizedTree`] can be immediately finalized by call
30    /// to [`NonFinalizedTree::verify_justification`] or
31    /// [`NonFinalizedTree::verify_grandpa_commit_message`], unless they descend from any of the
32    /// blocks returned by this function, in which case that block must be finalized beforehand.
33    pub fn finality_checkpoints(&self) -> impl Iterator<Item = (u64, &[u8; 32])> {
34        // Note that the code below assumes that GrandPa is the only finality algorithm currently
35        // supported.
36        debug_assert!(
37            self.blocks_trigger_gp_change.is_empty()
38                || !matches!(self.finality, Finality::Outsourced)
39        );
40
41        self.blocks_trigger_gp_change
42            .range((
43                ops::Bound::Excluded((
44                    Some(self.finalized_block_number),
45                    fork_tree::NodeIndex::MAX,
46                )),
47                ops::Bound::Unbounded,
48            ))
49            .map(|(_prev_auth_change_trigger_number, block_index)| {
50                debug_assert!(
51                    _prev_auth_change_trigger_number
52                        .map_or(false, |n| n > self.finalized_block_number)
53                );
54                let block = self
55                    .blocks
56                    .get(*block_index)
57                    .unwrap_or_else(|| unreachable!());
58                (block.number, &block.hash)
59            })
60    }

62    /// Verifies the given justification.
63    ///
64    /// The verification is performed in the context of the chain. In particular, the
65    /// verification will fail if the target block isn't already in the chain.
66    ///
67    /// If the verification succeeds, a [`FinalityApply`] object will be returned which can
68    /// be used to apply the finalization.
69    ///
70    /// A randomness seed must be provided and will be used during the verification. Note that the
71    /// verification is nonetheless deterministic.
72    // TODO: expand the documentation about how blocks with authorities changes have to be finalized before any further block can be finalized
73    pub fn verify_justification(
74        &'_ mut self,
75        consensus_engine_id: [u8; 4],
76        scale_encoded_justification: &[u8],
77        randomness_seed: [u8; 32],
78    ) -> Result<FinalityApply<'_, T>, JustificationVerifyError> {
79        match (&self.finality, &consensus_engine_id) {
80            (Finality::Grandpa { .. }, b"FRNK") => {
81                // Turn justification into a strongly-typed struct.
82                let decoded = decode::decode_grandpa_justification(
83                    scale_encoded_justification,
84                    self.block_number_bytes,
85                )
86                .map_err(JustificationVerifyError::InvalidJustification)?;
87
88                // Delegate the first step to the other function.
89                let (block_index, authorities_set_id, authorities_list) = self
90                    .verify_grandpa_finality_inner(decoded.target_hash, decoded.target_number)
91                    .map_err(JustificationVerifyError::FinalityVerify)?;
92
93                verify::verify_justification(verify::JustificationVerifyConfig {
94                    justification: scale_encoded_justification,
95                    block_number_bytes: self.block_number_bytes,
96                    authorities_set_id,
97                    authorities_list,
98                    randomness_seed,
99                })
100                .map_err(JustificationVerifyError::VerificationFailed)?;
101
102                // Justification has been successfully verified!
103                Ok(FinalityApply {
104                    chain: self,
105                    to_finalize: block_index,
106                })
107            }
108            _ => Err(JustificationVerifyError::JustificationEngineMismatch),
109        }
110    }

112    /// Verifies the given Grandpa commit message.
113    ///
114    /// The verification is performed in the context of the chain. In particular, the
115    /// verification will fail if the target block isn't already in the chain or if one of the
116    /// voted blocks is unknown locally.
117    ///
118    /// If the verification succeeds, a [`FinalityApply`] object will be returned which can
119    /// be used to apply the finalization.
120    ///
121    /// A randomness seed must be provided and will be used during the verification. Note that the
122    /// verification is nonetheless deterministic.
123    pub fn verify_grandpa_commit_message(
124        &'_ mut self,
125        scale_encoded_commit: &[u8],
126        randomness_seed: [u8; 32],
127    ) -> Result<FinalityApply<'_, T>, CommitVerifyError> {
128        // The code below would panic if the chain doesn't use Grandpa.
129        if !matches!(self.finality, Finality::Grandpa { .. }) {
130            return Err(CommitVerifyError::NotGrandpa);
131        }
132
133        let decoded_commit =
134            decode::decode_grandpa_commit(scale_encoded_commit, self.block_number_bytes)
135                .map_err(|_| CommitVerifyError::InvalidCommit)?;
136
137        // Delegate the first step to the other function.
138        let (block_index, expected_authorities_set_id, authorities_list) = self
139            .verify_grandpa_finality_inner(decoded_commit.target_hash, decoded_commit.target_number)
140            .map_err(CommitVerifyError::FinalityVerify)?;
141
142        let mut verification = verify::verify_commit(verify::CommitVerifyConfig {
143            commit: scale_encoded_commit,
144            block_number_bytes: self.block_number_bytes,
145            expected_authorities_set_id,
146            num_authorities: u32::try_from(authorities_list.clone().count()).unwrap(),
147            randomness_seed,
148        });
149
150        loop {
151            match verification {
152                verify::CommitVerify::Finished(Ok(())) => {
153                    drop(authorities_list);
154                    return Ok(FinalityApply {
155                        chain: self,
156                        to_finalize: block_index,
157                    });
158                }
159                verify::CommitVerify::FinishedUnknown => {
160                    return Err(CommitVerifyError::NotEnoughKnownBlocks {
161                        target_block_number: decoded_commit.target_number,
162                    });
163                }
164                verify::CommitVerify::Finished(Err(error)) => {
165                    return Err(CommitVerifyError::VerificationFailed(error));
166                }
167                verify::CommitVerify::IsAuthority(is_authority) => {
168                    let to_find = is_authority.authority_public_key();
169                    let result = authorities_list.clone().any(|a| a == to_find);
170                    verification = is_authority.resume(result);
171                }
172                verify::CommitVerify::IsParent(is_parent) => {
173                    // Find in the list of non-finalized blocks the target of the check.
174                    match self.blocks_by_hash.get(is_parent.block_hash()) {
175                        Some(idx) => {
176                            let result = self.blocks.is_ancestor(block_index, *idx);
177                            verification = is_parent.resume(Some(result));
178                        }
179                        None => {
180                            verification = is_parent.resume(None);
181                        }
182                    };
183                }
184            }
185        }
186    }

188    /// Sets the latest known finalized block. Trying to verify a block that isn't a descendant of
189    /// that block will fail.
190    ///
191    /// The block must have been passed to [`NonFinalizedTree::verify_header`].
192    ///
193    /// Returns an iterator containing the now-finalized blocks and the pruned blocks in "reverse
194    /// hierarchical order". Each block is yielded by the iterator before its parent.
195    ///
196    /// > **Note**: This function returns blocks in this order, because any other ordering would
197    /// >           incur a performance cost. While returning blocks in hierarchical order would
198    /// >           often be more convenient, the overhead of doing so is moved to the user.
199    ///
200    /// The pruning is completely performed, even if the iterator is dropped eagerly.
201    ///
202    /// If necessary, the current best block will be updated to be a descendant of the
203    /// newly-finalized block.
204    pub fn set_finalized_block(
205        &'_ mut self,
206        block_hash: &[u8; 32],
207    ) -> Result<SetFinalizedBlockIter<'_, T>, SetFinalizedError> {
208        let block_index = match self.blocks_by_hash.get(block_hash) {
209            Some(idx) => *idx,
210            None => return Err(SetFinalizedError::UnknownBlock),
211        };
212
213        Ok(self.set_finalized_block_inner(block_index))
214    }
215
216    /// Common function for verifying GrandPa-finality-related messages.
217    ///
218    /// Returns the index of the possibly finalized block, the expected authorities set id, and
219    /// an iterator to the list of authorities.
220    ///
221    /// # Panic
222    ///
223    /// Panics if the finality algorithm of the chain isn't Grandpa.
224    ///
225    fn verify_grandpa_finality_inner(
226        &self,
227        target_hash: &[u8; 32],
228        target_number: u64,
229    ) -> Result<
230        (
231            fork_tree::NodeIndex,
232            u64,
233            impl Iterator<Item = &[u8]> + Clone,
234        ),
235        FinalityVerifyError,
236    > {
237        match &self.finality {
238            Finality::Outsourced => panic!(),
239            Finality::Grandpa {
240                after_finalized_block_authorities_set_id,
241                finalized_scheduled_change,
242                finalized_triggered_authorities,
243            } => {
244                match target_number.cmp(&self.finalized_block_number) {
245                    cmp::Ordering::Equal if *target_hash == self.finalized_block_hash => {
246                        return Err(FinalityVerifyError::EqualToFinalized);
247                    }
248                    cmp::Ordering::Equal => {
249                        return Err(FinalityVerifyError::EqualFinalizedHeightButInequalHash);
250                    }
251                    cmp::Ordering::Less => return Err(FinalityVerifyError::BelowFinalized),
252                    _ => {}
253                }
254
255                // Find in the list of non-finalized blocks the one targeted by the justification.
256                let block_index = match self.blocks_by_hash.get(target_hash) {
257                    Some(idx) => *idx,
258                    None => {
259                        return Err(FinalityVerifyError::UnknownTargetBlock {
260                            block_number: target_number,
261                            block_hash: *target_hash,
262                        });
263                    }
264                };
265
266                // If any block between the latest finalized one and the target block triggers any
267                // GrandPa authorities change, then we need to finalize that triggering block
268                // before finalizing the one targeted by the justification.
269                if let BlockFinality::Grandpa {
270                    ref prev_auth_change_trigger_number,
271                    ..
272                } = self.blocks.get(block_index).unwrap().finality
273                {
274                    if let Some(prev_auth_change_trigger_number) = prev_auth_change_trigger_number {
275                        if *prev_auth_change_trigger_number > self.finalized_block_number {
276                            return Err(FinalityVerifyError::TooFarAhead {
277                                justification_block_number: target_number,
278                                justification_block_hash: *target_hash,
279                                block_to_finalize_number: *prev_auth_change_trigger_number,
280                            });
281                        }
282                    }
283                } else {
284                    unreachable!()
285                }
286
287                // Find which authorities are supposed to finalize the target block.
288                let authorities_list = finalized_scheduled_change
289                    .as_ref()
290                    .filter(|(trigger_height, _)| *trigger_height < target_number)
291                    .map_or(finalized_triggered_authorities, |(_, list)| list);
292
293                // As per above check, we know that the authorities of the target block are either
294                // the same as the ones of the latest finalized block, or the ones contained in
295                // the header of the latest finalized block.
296
297                // First verification step complete.
298                Ok((
299                    block_index,
300                    *after_finalized_block_authorities_set_id,
301                    authorities_list.iter().map(|a| &a.public_key[..]),
302                ))
303            }
304        }
305    }

307    /// Implementation of [`NonFinalizedTree::set_finalized_block`].
308    ///
309    /// # Panic
310    ///
311    /// Panics if `block_index_to_finalize` isn't a valid node in the tree.
312    ///
313    fn set_finalized_block_inner(
314        &'_ mut self,
315        block_index_to_finalize: fork_tree::NodeIndex,
316    ) -> SetFinalizedBlockIter<'_, T> {
317        let new_finalized_block = self.blocks.get_mut(block_index_to_finalize).unwrap();
318
319        // Update `self.finality`.
320        match (&mut self.finality, &new_finalized_block.finality) {
321            (Finality::Outsourced, BlockFinality::Outsourced) => {}
322            (
323                Finality::Grandpa {
324                    after_finalized_block_authorities_set_id,
325                    finalized_scheduled_change,
326                    finalized_triggered_authorities,
327                },
328                BlockFinality::Grandpa {
329                    after_block_authorities_set_id,
330                    triggered_authorities,
331                    scheduled_change,
332                    ..
333                },
334            ) => {
335                // Some sanity checks.
336                debug_assert!(
337                    *after_finalized_block_authorities_set_id <= *after_block_authorities_set_id
338                );
339                debug_assert!(
340                    scheduled_change
341                        .as_ref()
342                        .map_or(true, |(n, _)| *n > new_finalized_block.number)
343                );
344
345                *after_finalized_block_authorities_set_id = *after_block_authorities_set_id;
346                *finalized_triggered_authorities = triggered_authorities.clone();
347                *finalized_scheduled_change = scheduled_change.clone();
348            }
349
350            // Mismatch between chain finality algorithm and block finality algorithm. Should never
351            // happen.
352            _ => unreachable!(),
353        }
354
355        // If the best block isn't a descendant of the block being finalized, then the best
356        // block will change to a different block.
357        // TODO: this is `O(n)`, does the user really need to know ahead of time whether the best block is updated?
358        let updates_best_block = {
359            let current_best: Option<fork_tree::NodeIndex> = self
360                .blocks_by_best_score
361                .last_key_value()
362                .map(|(_, idx)| *idx);
363            Some(block_index_to_finalize) == current_best
364                || current_best.map_or(true, |current_best| {
365                    !self
366                        .blocks
367                        .is_ancestor(block_index_to_finalize, current_best)
368                })
369        };
370
371        let new_finalized_block = self.blocks.get_mut(block_index_to_finalize).unwrap();
372
373        // Update `self.finalized_consensus`.
374        match (
375            &mut self.finalized_consensus,
376            &new_finalized_block.consensus,
377        ) {
378            (
379                FinalizedConsensus::Aura {
380                    authorities_list, ..
381                },
382                BlockConsensus::Aura {
383                    authorities_list: new_list,
384                },
385            ) => {
386                *authorities_list = new_list.clone();
387            }
388            (
389                FinalizedConsensus::Babe {
390                    block_epoch_information,
391                    next_epoch_transition,
392                    ..
393                },
394                BlockConsensus::Babe {
395                    current_epoch,
396                    next_epoch,
397                },
398            ) => {
399                *block_epoch_information = current_epoch.clone();
400                *next_epoch_transition = next_epoch.clone();
401            }
402            // Any mismatch of consensus engines between the chain and the newly-finalized block
403            // should have been detected when the block got added to the chain.
404            _ => unreachable!(),
405        }
406
407        // Update `self.finalized_block_header`, `self.finalized_block_hash`,
408        // `self.finalized_block_number`, and `self.finalized_best_score`.
409        mem::swap(
410            &mut self.finalized_block_header,
411            &mut new_finalized_block.header,
412        );
413        self.finalized_block_hash = new_finalized_block.hash;
414        self.finalized_block_number = new_finalized_block.number;
415        self.finalized_best_score = new_finalized_block.best_score;
416
417        debug_assert_eq!(self.blocks.len(), self.blocks_by_hash.len());
418        debug_assert_eq!(self.blocks.len(), self.blocks_by_best_score.len());
419        debug_assert!(self.blocks.len() >= self.blocks_trigger_gp_change.len());
420        SetFinalizedBlockIter {
421            iter: self.blocks.prune_ancestors(block_index_to_finalize),
422            blocks_by_hash: &mut self.blocks_by_hash,
423            blocks_by_best_score: &mut self.blocks_by_best_score,
424            blocks_trigger_gp_change: &mut self.blocks_trigger_gp_change,
425            updates_best_block,
426        }
427    }
428}

430/// Returned by [`NonFinalizedTree::verify_justification`] and
431/// [`NonFinalizedTree::verify_grandpa_commit_message`] on success.
432///
433/// As long as [`FinalityApply::apply`] isn't called, the underlying [`NonFinalizedTree`]
434/// isn't modified.
435#[must_use]
436pub struct FinalityApply<'c, T> {
437    chain: &'c mut NonFinalizedTree<T>,
438    to_finalize: fork_tree::NodeIndex,
439}
440
441impl<'c, T> FinalityApply<'c, T> {
442    /// Applies the justification, finalizing the given block.
443    ///
444    /// This function, including its return type, behaves in the same way as
445    /// [`NonFinalizedTree::set_finalized_block`].
446    pub fn apply(self) -> SetFinalizedBlockIter<'c, T> {
447        self.chain.set_finalized_block_inner(self.to_finalize)
448    }
449
450    /// Returns the user data of the block about to be justified.
451    pub fn block_user_data(&mut self) -> &mut T {
452        &mut self
453            .chain
454            .blocks
455            .get_mut(self.to_finalize)
456            .unwrap()
457            .user_data
458    }
459
460    /// Returns true if the block to be finalized is the current best block.
461    pub fn is_current_best_block(&self) -> bool {
462        Some(self.to_finalize)
463            == self
464                .chain
465                .blocks_by_best_score
466                .last_key_value()
467                .map(|(_, idx)| *idx)
468    }
469}

472    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
473        f.debug_tuple("FinalityApply").finish()
474    }
475}

477/// Error that can happen when verifying a justification.
478#[derive(Debug, derive_more::Display, derive_more::Error)]
479pub enum JustificationVerifyError {
480    /// Type of the justification doesn't match the finality mechanism used by the chain.
481    ///
482    /// > **Note**: If the chain's finality mechanism doesn't use justifications, this error is
483    /// >           always returned.
484    JustificationEngineMismatch,
485    /// Error while decoding the justification.
486    #[display("Error while decoding the justification: {_0}")]
487    InvalidJustification(decode::JustificationDecodeError),
488    /// The justification verification has failed. The justification is invalid and should be
489    /// thrown away.
490    #[display("{_0}")]
491    VerificationFailed(verify::JustificationVerifyError),
492    /// Error while verifying the finality in the context of the chain.
493    #[display("{_0}")]
494    FinalityVerify(FinalityVerifyError),
495}

497/// Error that can happen when verifying a Grandpa commit.
498#[derive(Debug, derive_more::Display, derive_more::Error)]
499pub enum CommitVerifyError {
500    /// Chain doesn't use the GrandPa algorithm.
501    NotGrandpa,
502    /// Error while decoding the commit.
503    InvalidCommit,
504    /// Error while verifying the finality in the context of the chain.
505    #[display("{_0}")]
506    FinalityVerify(FinalityVerifyError),
507    /// Not enough blocks are known by the tree to verify this commit.
508    ///
509    /// This doesn't mean that the commit is bad, but that it can't be verified without adding
510    /// more blocks to the tree.
511    #[display("Not enough blocks are known to verify this commit")]
512    NotEnoughKnownBlocks {
513        /// Block number that the commit targets.
514        target_block_number: u64,
515    },
516    /// The commit verification has failed. The commit is invalid and should be thrown away.
517    #[display("{_0}")]
518    VerificationFailed(verify::CommitVerifyError),
519}

521/// Error that can happen when verifying a proof of finality.
522#[derive(Debug, derive_more::Display, derive_more::Error)]
523pub enum FinalityVerifyError {
524    /// The target block height and hash are the same as the block that is already finalized.
525    /// While the proof couldn't be verified, nothing could be gained from actually verifying it.
526    EqualToFinalized,
527    /// The target block height is the same as the finalized block, but its hash is different.
528    /// This means that the proof can't possibly be correct.
529    EqualFinalizedHeightButInequalHash,
530    /// The target block height is strictly inferior to the finalized block height.
531    BelowFinalized,
532    /// Finality proof targets a block that isn't in the chain.
533    #[display("Justification targets a block (#{block_number}) that isn't in the chain.")]
534    UnknownTargetBlock {
535        /// Number of the block that isn't in the chain.
536        block_number: u64,
537        /// Hash of the block that isn't in the chain.
538        block_hash: [u8; 32],
539    },
540    /// There exists a block in-between the latest finalized block and the block targeted by the
541    /// justification that must first be finalized.
542    #[display(
543        "There exists a block in-between the latest finalized block and the block \
544        targeted by the justification that must first be finalized"
545    )]
546    TooFarAhead {
547        /// Number of the block contained in the justification.
548        justification_block_number: u64,
549        /// Hash of the block contained in the justification.
550        justification_block_hash: [u8; 32],
551        /// Number of the block to finalize first.
552        block_to_finalize_number: u64,
553    },
554}

556/// Iterator producing the newly-finalized blocks removed from the state when the finalized block
557/// is updated.
558pub struct SetFinalizedBlockIter<'a, T> {
559    iter: fork_tree::PruneAncestorsIter<'a, Block<T>>,
560    blocks_by_hash: &'a mut HashMap<[u8; 32], fork_tree::NodeIndex, fnv::FnvBuildHasher>,
561    blocks_by_best_score: &'a mut BTreeMap<BestScore, fork_tree::NodeIndex>,
562    blocks_trigger_gp_change: &'a mut BTreeSet<(Option<u64>, fork_tree::NodeIndex)>,
563    updates_best_block: bool,
564}
565
566impl<'a, T> SetFinalizedBlockIter<'a, T> {
567    /// Returns true if the finalization process modifies the best block of the chain.
568    pub fn updates_best_block(&self) -> bool {
569        self.updates_best_block
570    }
571}
572
573impl<'a, T> Iterator for SetFinalizedBlockIter<'a, T> {
574    type Item = RemovedBlock<T>;
575
576    fn next(&mut self) -> Option<Self::Item> {
577        let pruned = self.iter.next()?;
578        let _removed = self.blocks_by_hash.remove(&pruned.user_data.hash);
579        debug_assert_eq!(_removed, Some(pruned.index));
580        let _removed = self
581            .blocks_by_best_score
582            .remove(&pruned.user_data.best_score);
583        debug_assert_eq!(_removed, Some(pruned.index));
584        if let BlockFinality::Grandpa {
585            prev_auth_change_trigger_number,
586            triggers_change: true,
587            ..
588        } = pruned.user_data.finality
589        {
590            let _removed = self
591                .blocks_trigger_gp_change
592                .remove(&(prev_auth_change_trigger_number, pruned.index));
593            debug_assert!(_removed);
594        }
595
596        Some(RemovedBlock {
597            block_hash: pruned.user_data.hash,
598            block_number: pruned.user_data.number,
599            scale_encoded_header: pruned.user_data.header,
600            user_data: pruned.user_data.user_data,
601            ty: if pruned.is_prune_target_ancestor {
602                RemovedBlockType::Finalized
603            } else {
604                RemovedBlockType::Pruned
605            },
606        })
607    }
608
609    fn size_hint(&self) -> (usize, Option<usize>) {
610        self.iter.size_hint()
611    }
612}

614impl<'a, T> Drop for SetFinalizedBlockIter<'a, T> {
615    fn drop(&mut self) {
616        // Make sure the iteration goes to the end.
617        for _ in self {}
618    }
619}

621/// Error that can happen when setting the finalized block.
622#[derive(Debug, derive_more::Display, derive_more::Error)]
623pub enum SetFinalizedError {
624    /// Block must have been passed to [`NonFinalizedTree::insert_verified_header`] in the past.
625    UnknownBlock,
626}

628/// Block removed from the [`NonFinalizedTree`] by a [`SetFinalizedBlockIter`].
629#[derive(Debug, Clone, PartialEq, Eq, Hash)]
630pub struct RemovedBlock<T> {
631    /// Hash of the block.
632    pub block_hash: [u8; 32],
633    /// Height of the block.
634    pub block_number: u64,
635    /// User data that was associated with that block in the [`NonFinalizedTree`].
636    pub user_data: T,
637    /// Reason why the block was removed.
638    pub ty: RemovedBlockType,
639    /// SCALE-encoded header of the block.
640    pub scale_encoded_header: Vec<u8>,
641}
642
643/// Reason why a block was removed from the [`NonFinalizedTree`] by a [`SetFinalizedBlockIter`].
644#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
645pub enum RemovedBlockType {
646    /// Block is now part of the finalized chain.
647    Finalized,
648    /// Block is not a descendant of the new finalized block.
649    Pruned,
650}
