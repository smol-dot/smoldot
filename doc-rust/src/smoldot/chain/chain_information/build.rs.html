18//! Build the chain information of a chain given its runtime.
19//!
20//! This module contains the [`ChainInformationBuild`] struct, a state machine that drives the
21//! process of building the chain information of a certain finalized point of a chain.

34/// Configuration to provide to [`ChainInformationBuild::new`].
35pub struct Config {
36    /// Header of the finalized block, whose chain information is to retrieve.
37    ///
38    /// Stored within the chain information at the end.
39    pub finalized_block_header: ConfigFinalizedBlockHeader,
40
41    /// Runtime of the finalized block. Must be built using the Wasm code found at the `:code` key
42    /// of the block storage.
43    pub runtime: host::HostVmPrototype,
44
45    /// Number of bytes of the block number encoded in the block header.
46    pub block_number_bytes: usize,
47}
48
49/// See [`Config::finalized_block_header`].
50pub enum ConfigFinalizedBlockHeader {
51    /// The block is the genesis block of the chain.
52    Genesis {
53        /// Hash of the root of the state trie of the genesis.
54        state_trie_root_hash: [u8; 32],
55    },
56    /// The block can any block, genesis block of the chain or not.
57    Any {
58        /// Header of the block.
59        scale_encoded_header: Vec<u8>,
60        /// Can be used to pass information about the finality of the chain, if already known.
61        known_finality: Option<chain_information::ChainInformationFinality>,
62    },
63}
64
65/// Current state of the operation.
66#[must_use]
67pub enum ChainInformationBuild {
68    /// Fetching the chain information is over.
69    Finished {
70        /// The result of the computation.
71        ///
72        /// If successful, the chain information is guaranteed to be valid.
73        result: Result<chain_information::ValidChainInformation, Error>,
74        /// Value of [`Config::runtime`] passed back.
75        virtual_machine: host::HostVmPrototype,
76    },
77
78    /// Still in progress.
79    InProgress(InProgress),
80}
81
82/// Chain information building is still in progress.
83#[must_use]
84pub enum InProgress {
85    /// Loading a storage value is required in order to continue.
86    StorageGet(StorageGet),
87    /// Obtaining the Merkle value of the closest descendant of a trie node is required in order
88    /// to continue.
89    ClosestDescendantMerkleValue(ClosestDescendantMerkleValue),
90    /// Fetching the key that follows a given one is required in order to continue.
91    NextKey(NextKey),
92}

94/// Problem encountered during the chain building process.
95#[derive(Debug, derive_more::Display, derive_more::Error)]
96pub enum Error {
97    /// Error while starting the Wasm virtual machine.
98    #[display("While calling {call:?}: {error}")]
99    WasmStart {
100        call: RuntimeCall,
101        #[error(source)]
102        error: host::StartErr,
103    },
104    /// Error while running the Wasm virtual machine.
105    #[display("While calling {call:?}: {error}")]
106    WasmVm {
107        call: RuntimeCall,
108        #[error(source)]
109        error: runtime_call::ErrorDetail,
110    },
111    /// Runtime has called an offchain worker host function.
112    OffchainWorkerHostFunction,
113    /// Failed to decode the output of the `AuraApi_slot_duration` runtime call.
114    AuraSlotDurationOutputDecode,
115    /// Failed to decode the output of the `AuraApi_authorities` runtime call.
116    AuraAuthoritiesOutputDecode,
117    /// Failed to decode the output of the `BabeApi_current_epoch` runtime call.
118    BabeCurrentEpochOutputDecode,
119    /// Failed to decode the output of the `BabeApi_next_epoch` runtime call.
120    BabeNextEpochOutputDecode,
121    /// Failed to decode the output of the `BabeApi_configuration` runtime call.
122    BabeConfigurationOutputDecode,
123    /// The version of `GrandaApi` is too old to be able to build the chain information.
124    GrandpaApiTooOld,
125    /// Failed to decode the output of the `GrandpaApi_authorities` runtime call.
126    GrandpaAuthoritiesOutputDecode,
127    /// Failed to decode the output of the `GrandpaApi_current_set_id` runtime call.
128    GrandpaCurrentSetIdOutputDecode,
129    /// The combination of the information retrieved from the runtime doesn't make sense together.
130    #[display("{_0}")]
131    InvalidChainInformation(chain_information::ValidityError),
132    /// Multiple consensus algorithms have been detected.
133    MultipleConsensusAlgorithms,
134}

136/// Function call to perform or being performed.

138pub enum RuntimeCall {
139    AuraApiSlotDuration,
140    AuraApiAuthorities,
141    BabeApiCurrentEpoch,
142    BabeApiNextEpoch,
143    BabeApiConfiguration,
144    GrandpaApiAuthorities,
145    GrandpaApiCurrentSetId,
146}

148impl RuntimeCall {
149    /// Name of the runtime function corresponding to this call.
150    pub fn function_name(&self) -> &'static str {
151        match self {
152            RuntimeCall::AuraApiSlotDuration => "AuraApi_slot_duration",
153            RuntimeCall::AuraApiAuthorities => "AuraApi_authorities",
154            RuntimeCall::BabeApiCurrentEpoch => "BabeApi_current_epoch",
155            RuntimeCall::BabeApiNextEpoch => "BabeApi_next_epoch",
156            RuntimeCall::BabeApiConfiguration => "BabeApi_configuration",
157            RuntimeCall::GrandpaApiAuthorities => "GrandpaApi_grandpa_authorities",
158            RuntimeCall::GrandpaApiCurrentSetId => "GrandpaApi_current_set_id",
159        }
160    }
161
162    /// Returns the list of parameters to pass when making the call.
163    ///
164    /// The actual parameters are obtained by putting together all the returned buffers together.
165    pub fn parameter_vectored(&self) -> impl Iterator<Item = impl AsRef<[u8]> + Clone> + Clone {
166        iter::empty::<Vec<u8>>()
167    }
168
169    /// Returns the list of parameters to pass when making the call.
170    ///
171    /// This function is a convenience around [`RuntimeCall::parameter_vectored`].
172    pub fn parameter_vectored_vec(&self) -> Vec<u8> {
173        Vec::new()
174    }
175}

179        fmt::Debug::fmt(&self.function_name(), f)
180    }

183impl ChainInformationBuild {
184    /// Starts a new chain information build process.
185    ///
186    /// # Panic
187    ///
188    /// Panics if a [`ConfigFinalizedBlockHeader::Any`] is provided, and the header can't be
189    /// decoded.
190    ///
191    pub fn new(config: Config) -> Self {
192        let [aura_version, babe_version, grandpa_version] = config
193            .runtime
194            .runtime_version()
195            .decode()
196            .apis
197            .find_versions(["AuraApi", "BabeApi", "GrandpaApi"]);
198        let runtime_has_aura = aura_version.map_or(false, |version_number| version_number == 1);
199        let runtime_babeapi_is_v1 = babe_version.and_then(|version_number| match version_number {
200            1 => Some(true),
201            2 => Some(false),
202            _ => None,
203        });
204        let runtime_grandpa_supports_currentsetid =
205            grandpa_version.and_then(|version_number| match version_number {
206                // Version 1 is from 2019 and isn't used by any chain in production, so we don't
207                // care about it.
208                2 => Some(false),
209                3 => Some(true),
210                _ => None,
211            });
212
213        let inner = ChainInformationBuildInner {
214            finalized_block_header: config.finalized_block_header,
215            block_number_bytes: config.block_number_bytes,
216            call_in_progress: None,
217            virtual_machine: Some(config.runtime),
218            runtime_has_aura,
219            runtime_babeapi_is_v1,
220            runtime_grandpa_supports_currentsetid,
221            aura_autorities_call_output: None,
222            aura_slot_duration_call_output: None,
223            babe_current_epoch_call_output: None,
224            babe_next_epoch_call_output: None,
225            babe_configuration_call_output: None,
226            grandpa_autorities_call_output: None,
227            grandpa_current_set_id_call_output: None,
228        };
229
230        ChainInformationBuild::start_next_call(inner)
231    }
232}

234impl InProgress {
235    /// Returns the list of runtime calls that will be performed. Always includes the value
236    /// returned by [`InProgress::call_in_progress`].
237    ///
238    /// This list never changes, except for the fact that it gets shorter over time.
239    pub fn remaining_calls(&self) -> impl Iterator<Item = RuntimeCall> {
240        let inner = match self {
241            InProgress::StorageGet(StorageGet(_, shared)) => shared,
242            InProgress::ClosestDescendantMerkleValue(ClosestDescendantMerkleValue(_, shared)) => {
243                shared
244            }
245            InProgress::NextKey(NextKey(_, shared)) => shared,
246        };
247
248        ChainInformationBuild::necessary_calls(inner)
249    }
250
251    /// Returns the runtime call currently being made.
252    pub fn call_in_progress(&self) -> RuntimeCall {
253        let inner = match self {
254            InProgress::StorageGet(StorageGet(_, shared)) => shared,
255            InProgress::ClosestDescendantMerkleValue(ClosestDescendantMerkleValue(_, shared)) => {
256                shared
257            }
258            InProgress::NextKey(NextKey(_, shared)) => shared,
259        };
260
261        inner.call_in_progress.unwrap()
262    }
263}

389impl ChainInformationBuild {
390    fn necessary_calls(
391        inner: &ChainInformationBuildInner,
392    ) -> impl Iterator<Item = RuntimeCall> + use<> {
393        let aura_api_authorities =
394            if inner.runtime_has_aura && inner.aura_autorities_call_output.is_none() {
395                Some(RuntimeCall::AuraApiAuthorities)
396            } else {
397                None
398            };
399
400        let aura_slot_duration =
401            if inner.runtime_has_aura && inner.aura_slot_duration_call_output.is_none() {
402                Some(RuntimeCall::AuraApiSlotDuration)
403            } else {
404                None
405            };
406
407        let babe_current_epoch = if matches!(
408            inner.finalized_block_header,
409            ConfigFinalizedBlockHeader::Any {
410                ref scale_encoded_header,
411                ..
412            } if header::decode(scale_encoded_header, inner.block_number_bytes).unwrap().number != 0
413        ) && inner.runtime_babeapi_is_v1.is_some()
414            && inner.babe_current_epoch_call_output.is_none()
415        {
416            Some(RuntimeCall::BabeApiCurrentEpoch)
417        } else {
418            None
419        };
420
421        let babe_next_epoch = if matches!(
422            inner.finalized_block_header,
423            ConfigFinalizedBlockHeader::Any {
424                ref scale_encoded_header,
425                ..
426            } if header::decode(scale_encoded_header, inner.block_number_bytes).unwrap().number != 0
427        ) && inner.runtime_babeapi_is_v1.is_some()
428            && inner.babe_next_epoch_call_output.is_none()
429        {
430            Some(RuntimeCall::BabeApiNextEpoch)
431        } else {
432            None
433        };
434
435        let babe_configuration = if inner.runtime_babeapi_is_v1.is_some()
436            && inner.babe_configuration_call_output.is_none()
437        {
438            Some(RuntimeCall::BabeApiConfiguration)
439        } else {
440            None
441        };
442
443        let grandpa_authorities = if !matches!(
444            inner.finalized_block_header,
445            ConfigFinalizedBlockHeader::Any {
446                known_finality: Some(chain_information::ChainInformationFinality::Grandpa { .. }),
447                ..
448            },
449        ) && inner.runtime_grandpa_supports_currentsetid.is_some()
450            && inner.grandpa_autorities_call_output.is_none()
451        {
452            Some(RuntimeCall::GrandpaApiAuthorities)
453        } else {
454            None
455        };
456
457        // The grandpa set ID doesn't need to be retrieved if finality was provided by the user,
458        // but also doesn't need to be retrieved for the genesis block because we know it's
459        // always 0.
460        let grandpa_current_set_id = if matches!(
461            inner.finalized_block_header,
462            ConfigFinalizedBlockHeader::Any {
463                ref scale_encoded_header,
464                known_finality: None,
465                ..
466            } if header::decode(scale_encoded_header, inner.block_number_bytes).unwrap().number != 0,
467        ) && inner.runtime_grandpa_supports_currentsetid
468            == Some(true)
469            && inner.grandpa_current_set_id_call_output.is_none()
470        {
471            Some(RuntimeCall::GrandpaApiCurrentSetId)
472        } else {
473            None
474        };
475
476        [
477            aura_api_authorities,
478            aura_slot_duration,
479            babe_current_epoch,
480            babe_next_epoch,
481            babe_configuration,
482            grandpa_authorities,
483            grandpa_current_set_id,
484        ]
485        .into_iter()
486        .flatten()
487    }
488
489    fn start_next_call(mut inner: ChainInformationBuildInner) -> Self {
490        debug_assert!(inner.call_in_progress.is_none());
491        debug_assert!(inner.virtual_machine.is_some());
492
493        if let Some(call) = ChainInformationBuild::necessary_calls(&inner).next() {
494            let vm_start_result = runtime_call::run(runtime_call::Config {
495                function_to_call: call.function_name(),
496                parameter: call.parameter_vectored(),
497                virtual_machine: inner.virtual_machine.take().unwrap(),
498                max_log_level: 0,
499                storage_proof_size_behavior:
500                    runtime_call::StorageProofSizeBehavior::proof_recording_disabled(),
501                storage_main_trie_changes: Default::default(),
502                calculate_trie_changes: false,
503            });
504
505            let vm = match vm_start_result {
506                Ok(vm) => vm,
507                Err((error, virtual_machine)) => {
508                    return ChainInformationBuild::Finished {
509                        result: Err(Error::WasmStart { call, error }),
510                        virtual_machine,
511                    };
512                }
513            };
514
515            inner.call_in_progress = Some(call);
516            ChainInformationBuild::from_call_in_progress(vm, inner)
517        } else {
518            // If the logic of this module is correct, all the information that we need has been
519            // retrieved at this point.
520
521            let consensus = match (
522                inner.runtime_has_aura,
523                inner.runtime_babeapi_is_v1,
524                &inner.finalized_block_header,
525            ) {
526                (true, Some(_), _) => {
527                    return ChainInformationBuild::Finished {
528                        result: Err(Error::MultipleConsensusAlgorithms),
529                        virtual_machine: inner.virtual_machine.take().unwrap(),
530                    };
531                }
532                (false, None, _) => chain_information::ChainInformationConsensus::Unknown,
533                (
534                    false,
535                    Some(_),
536                    ConfigFinalizedBlockHeader::Any {
537                        scale_encoded_header,
538                        ..
539                    },
540                ) if header::decode(scale_encoded_header, inner.block_number_bytes)
541                    .unwrap()
542                    .number
543                    != 0 =>
544                {
545                    chain_information::ChainInformationConsensus::Babe {
546                        finalized_block_epoch_information: Some(Box::new(
547                            inner.babe_current_epoch_call_output.take().unwrap(),
548                        )),
549                        finalized_next_epoch_transition: Box::new(
550                            inner.babe_next_epoch_call_output.take().unwrap(),
551                        ),
552                        slots_per_epoch: inner
553                            .babe_configuration_call_output
554                            .take()
555                            .unwrap()
556                            .slots_per_epoch,
557                    }
558                }
559                (false, Some(_), _) => {
560                    let config = inner.babe_configuration_call_output.take().unwrap();
561                    chain_information::ChainInformationConsensus::Babe {
562                        slots_per_epoch: config.slots_per_epoch,
563                        finalized_block_epoch_information: None,
564                        finalized_next_epoch_transition: Box::new(
565                            chain_information::BabeEpochInformation {
566                                epoch_index: 0,
567                                start_slot_number: None,
568                                authorities: config.epoch0_information.authorities,
569                                randomness: config.epoch0_information.randomness,
570                                c: config.epoch0_configuration.c,
571                                allowed_slots: config.epoch0_configuration.allowed_slots,
572                            },
573                        ),
574                    }
575                }
576                (true, None, _) => chain_information::ChainInformationConsensus::Aura {
577                    finalized_authorities_list: inner.aura_autorities_call_output.take().unwrap(),
578                    slot_duration: inner.aura_slot_duration_call_output.take().unwrap(),
579                },
580            };
581
582            // Build the finalized block header, and extract the information about finality if it
583            // was already provided by the API user.
584            let (finalized_block_header, known_finality) = match inner.finalized_block_header {
585                ConfigFinalizedBlockHeader::Genesis {
586                    state_trie_root_hash,
587                } => {
588                    let header = header::HeaderRef {
589                        parent_hash: &[0; 32],
590                        number: 0,
591                        state_root: &state_trie_root_hash,
592                        extrinsics_root: &trie::EMPTY_BLAKE2_TRIE_MERKLE_VALUE,
593                        digest: header::DigestRef::empty(),
594                    }
595                    .scale_encoding_vec(inner.block_number_bytes);
596
597                    (header, None)
598                }
599                ConfigFinalizedBlockHeader::Any {
600                    scale_encoded_header: header,
601                    known_finality,
602                } => (header, known_finality),
603            };
604
605            // Build the finality information if not known yet.
606            let finality = if let Some(known_finality) = known_finality {
607                known_finality
608            } else if inner.runtime_grandpa_supports_currentsetid.is_some() {
609                chain_information::ChainInformationFinality::Grandpa {
610                    after_finalized_block_authorities_set_id: if header::decode(
611                        &finalized_block_header,
612                        inner.block_number_bytes,
613                    )
614                    .unwrap()
615                    .number
616                        == 0
617                    {
618                        0
619                    } else {
620                        // If the GrandPa runtime API version is too old, it is not possible to
621                        // determine the current set ID.
622                        let Some(grandpa_current_set_id_call_output) =
623                            inner.grandpa_current_set_id_call_output.take()
624                        else {
625                            debug_assert_eq!(
626                                inner.runtime_grandpa_supports_currentsetid,
627                                Some(false)
628                            );
629                            return ChainInformationBuild::Finished {
630                                result: Err(Error::GrandpaApiTooOld),
631                                virtual_machine: inner.virtual_machine.take().unwrap(),
632                            };
633                        };
634
635                        grandpa_current_set_id_call_output
636                    },
637                    // TODO: The runtime doesn't give us a way to know the current scheduled change. At the moment the runtime it never schedules changes with a delay of more than 0. So in practice this `None` is correct, but it relies on implementation details
638                    finalized_scheduled_change: None,
639                    finalized_triggered_authorities: inner
640                        .grandpa_autorities_call_output
641                        .take()
642                        .unwrap(),
643                }
644            } else {
645                chain_information::ChainInformationFinality::Outsourced
646            };
647
648            // Build a `ChainInformation` using the parameters found in the runtime.
649            // It is possible, however, that the runtime produces parameters that aren't
650            // coherent. For example the runtime could give "current" and "next" Babe
651            // epochs that don't follow each other.
652            let chain_information = match chain_information::ValidChainInformation::try_from(
653                chain_information::ChainInformation {
654                    finalized_block_header: Box::new(
655                        header::decode(&finalized_block_header, inner.block_number_bytes)
656                            .unwrap()
657                            .into(),
658                    ),
659                    finality,
660                    consensus,
661                },
662            ) {
663                Ok(ci) => ci,
664                Err(err) => {
665                    return ChainInformationBuild::Finished {
666                        result: Err(Error::InvalidChainInformation(err)),
667                        virtual_machine: inner.virtual_machine.take().unwrap(),
668                    };
669                }
670            };
671
672            ChainInformationBuild::Finished {
673                result: Ok(chain_information),
674                virtual_machine: inner.virtual_machine.take().unwrap(),
675            }
676        }
677    }

679    fn from_call_in_progress(
680        mut call: runtime_call::RuntimeCall,
681        mut inner: ChainInformationBuildInner,
682    ) -> Self {
683        loop {
684            debug_assert!(inner.call_in_progress.is_some());
685
686            match call {
687                runtime_call::RuntimeCall::Finished(Ok(success)) => {
688                    inner.virtual_machine = Some(match inner.call_in_progress.take() {
689                        None => unreachable!(),
690                        Some(RuntimeCall::AuraApiSlotDuration) => {
691                            let result = decode_aura_slot_duration_output(
692                                success.virtual_machine.value().as_ref(),
693                            );
694                            let virtual_machine = success.virtual_machine.into_prototype();
695                            match result {
696                                Ok(output) => inner.aura_slot_duration_call_output = Some(output),
697                                Err(err) => {
698                                    return ChainInformationBuild::Finished {
699                                        result: Err(err),
700                                        virtual_machine,
701                                    };
702                                }
703                            }
704                            virtual_machine
705                        }
706                        Some(RuntimeCall::AuraApiAuthorities) => {
707                            let result = decode_aura_authorities_output(
708                                success.virtual_machine.value().as_ref(),
709                            );
710                            let virtual_machine = success.virtual_machine.into_prototype();
711                            match result {
712                                Ok(output) => inner.aura_autorities_call_output = Some(output),
713                                Err(err) => {
714                                    return ChainInformationBuild::Finished {
715                                        result: Err(err),
716                                        virtual_machine,
717                                    };
718                                }
719                            }
720                            virtual_machine
721                        }
722                        Some(RuntimeCall::BabeApiCurrentEpoch) => {
723                            let result = decode_babe_epoch_output(
724                                success.virtual_machine.value().as_ref(),
725                                false,
726                            );
727                            let virtual_machine = success.virtual_machine.into_prototype();
728                            match result {
729                                Ok(output) => inner.babe_current_epoch_call_output = Some(output),
730                                Err(err) => {
731                                    return ChainInformationBuild::Finished {
732                                        result: Err(err),
733                                        virtual_machine,
734                                    };
735                                }
736                            }
737                            virtual_machine
738                        }
739                        Some(RuntimeCall::BabeApiNextEpoch) => {
740                            let result = decode_babe_epoch_output(
741                                success.virtual_machine.value().as_ref(),
742                                true,
743                            );
744                            let virtual_machine = success.virtual_machine.into_prototype();
745                            match result {
746                                Ok(output) => inner.babe_next_epoch_call_output = Some(output),
747                                Err(err) => {
748                                    return ChainInformationBuild::Finished {
749                                        result: Err(err),
750                                        virtual_machine,
751                                    };
752                                }
753                            }
754                            virtual_machine
755                        }
756                        Some(RuntimeCall::BabeApiConfiguration) => {
757                            let result = decode_babe_configuration_output(
758                                success.virtual_machine.value().as_ref(),
759                                inner.runtime_babeapi_is_v1.unwrap(),
760                            );
761                            let virtual_machine = success.virtual_machine.into_prototype();
762                            match result {
763                                Ok(output) => inner.babe_configuration_call_output = Some(output),
764                                Err(err) => {
765                                    return ChainInformationBuild::Finished {
766                                        result: Err(err),
767                                        virtual_machine,
768                                    };
769                                }
770                            }
771                            virtual_machine
772                        }
773                        Some(RuntimeCall::GrandpaApiAuthorities) => {
774                            let result = decode_grandpa_authorities_output(
775                                success.virtual_machine.value().as_ref(),
776                            );
777                            let virtual_machine = success.virtual_machine.into_prototype();
778                            match result {
779                                Ok(output) => inner.grandpa_autorities_call_output = Some(output),
780                                Err(err) => {
781                                    return ChainInformationBuild::Finished {
782                                        result: Err(err),
783                                        virtual_machine,
784                                    };
785                                }
786                            }
787                            virtual_machine
788                        }
789                        Some(RuntimeCall::GrandpaApiCurrentSetId) => {
790                            let result = decode_grandpa_current_set_id_output(
791                                success.virtual_machine.value().as_ref(),
792                            );
793                            let virtual_machine = success.virtual_machine.into_prototype();
794                            match result {
795                                Ok(output) => {
796                                    inner.grandpa_current_set_id_call_output = Some(output)
797                                }
798                                Err(err) => {
799                                    return ChainInformationBuild::Finished {
800                                        result: Err(err),
801                                        virtual_machine,
802                                    };
803                                }
804                            }
805                            virtual_machine
806                        }
807                    });
808
809                    break ChainInformationBuild::start_next_call(inner);
810                }
811                runtime_call::RuntimeCall::Finished(Err(err)) => {
812                    break ChainInformationBuild::Finished {
813                        result: Err(Error::WasmVm {
814                            call: inner.call_in_progress.unwrap(),
815                            error: err.detail,
816                        }),
817                        virtual_machine: err.prototype,
818                    };
819                }
820                runtime_call::RuntimeCall::StorageGet(call) => {
821                    break ChainInformationBuild::InProgress(InProgress::StorageGet(StorageGet(
822                        call, inner,
823                    )));
824                }
825                runtime_call::RuntimeCall::NextKey(call) => {
826                    break ChainInformationBuild::InProgress(InProgress::NextKey(NextKey(
827                        call, inner,
828                    )));
829                }
830                runtime_call::RuntimeCall::ClosestDescendantMerkleValue(call) => {
831                    break ChainInformationBuild::InProgress(
832                        InProgress::ClosestDescendantMerkleValue(ClosestDescendantMerkleValue(
833                            call, inner,
834                        )),
835                    );
836                }
837                runtime_call::RuntimeCall::SignatureVerification(sig) => {
838                    call = sig.verify_and_resume();
839                }
840                runtime_call::RuntimeCall::OffchainStorageSet(req) => {
841                    // Do nothing.
842                    call = req.resume();
843                }
844                runtime_call::RuntimeCall::Offchain(req) => {
845                    let virtual_machine = runtime_call::RuntimeCall::Offchain(req).into_prototype();
846                    break ChainInformationBuild::Finished {
847                        result: Err(Error::OffchainWorkerHostFunction),
848                        virtual_machine,
849                    };
850                }
851                runtime_call::RuntimeCall::LogEmit(req) => {
852                    // Generated logs are ignored.
853                    call = req.resume();
854                }
855            }
856        }
857    }
858}

860/// Struct shared by all the variants of the [`ChainInformationBuild`] enum. Contains the actual
861/// progress of the building.
862struct ChainInformationBuildInner {
863    /// See [`Config::finalized_block_header`].
864    finalized_block_header: ConfigFinalizedBlockHeader,
865    /// See [`Config::block_number_bytes`].
866    block_number_bytes: usize,
867
868    /// Which call is currently in progress, if any.
869    call_in_progress: Option<RuntimeCall>,
870    /// Runtime to use to start the calls.
871    ///
872    /// [`ChainInformationBuildInner::call_in_progress`] and
873    /// [`ChainInformationBuildInner::virtual_machine`] are never `Some` at the same time. However,
874    /// using an enum wouldn't make the code cleaner because we need to be able to extract the
875    /// values temporarily.
876    virtual_machine: Option<host::HostVmPrototype>,
877
878    /// If `true`, the runtime supports `AuraApi` functions.
879    runtime_has_aura: bool,
880    /// If `Some`, the runtime supports `BabeApi` functions. If `true`, the version is 1 (the old
881    /// version). If `false`, the version is 2.
882    runtime_babeapi_is_v1: Option<bool>,
883    /// If `Some`, the runtime supports `GrandpaApi` functions. If `true`, the API supports the
884    /// `GrandpaApi_current_set_id` runtime call.
885    runtime_grandpa_supports_currentsetid: Option<bool>,
886
887    /// Output of the call to `AuraApi_slot_duration`, if it was already made.
888    aura_slot_duration_call_output: Option<NonZero<u64>>,
889    /// Output of the call to `AuraApi_authorities`, if it was already made.
890    aura_autorities_call_output: Option<Vec<header::AuraAuthority>>,
891    /// Output of the call to `BabeApi_current_epoch`, if it was already made.
892    babe_current_epoch_call_output: Option<chain_information::BabeEpochInformation>,
893    /// Output of the call to `BabeApi_next_epoch`, if it was already made.
894    babe_next_epoch_call_output: Option<chain_information::BabeEpochInformation>,
895    /// Output of the call to `BabeApi_configuration`, if it was already made.
896    babe_configuration_call_output: Option<BabeGenesisConfiguration>,
897    /// Output of the call to `GrandpaApi_grandpa_authorities`, if it was already made.
898    grandpa_autorities_call_output: Option<Vec<header::GrandpaAuthority>>,
899    /// Output of the call to `GrandpaApi_current_set_id`, if it was already made.

901}

928/// Decodes the output of a call to `BabeApi_configuration`.
929fn decode_babe_configuration_output(
930    bytes: &[u8],
931    is_babe_api_v1: bool,
932) -> Result<BabeGenesisConfiguration, Error> {
933    let result: nom::IResult<_, _> = nom::Parser::parse(
934        &mut nom::combinator::all_consuming(nom::combinator::complete(nom::combinator::map(
935            (
936                nom::number::streaming::le_u64,
937                nom::combinator::map_opt(nom::number::streaming::le_u64, NonZero::<u64>::new),
938                nom::number::streaming::le_u64,
939                nom::number::streaming::le_u64,
940                nom::combinator::flat_map(crate::util::nom_scale_compact_usize, |num_elems| {
941                    nom::multi::many_m_n(
942                        num_elems,
943                        num_elems,
944                        nom::combinator::map(
945                            (
946                                nom::bytes::streaming::take(32u32),
947                                nom::number::streaming::le_u64,
948                            ),
949                            move |(public_key, weight)| header::BabeAuthority {
950                                public_key: <[u8; 32]>::try_from(public_key).unwrap(),
951                                weight,
952                            },
953                        ),
954                    )
955                }),
956                nom::combinator::map(nom::bytes::streaming::take(32u32), |b| {
957                    <[u8; 32]>::try_from(b).unwrap()
958                }),
959                |b| {
960                    if is_babe_api_v1 {
961                        nom::Parser::parse(
962                            &mut nom::branch::alt((
963                                nom::combinator::map(nom::bytes::streaming::tag(&[0][..]), |_| {
964                                    header::BabeAllowedSlots::PrimarySlots
965                                }),
966                                nom::combinator::map(nom::bytes::streaming::tag(&[1][..]), |_| {
967                                    header::BabeAllowedSlots::PrimaryAndSecondaryPlainSlots
968                                }),
969                            )),
970                            b,
971                        )
972                    } else {
973                        nom::Parser::parse(
974                            &mut nom::branch::alt((
975                                nom::combinator::map(nom::bytes::streaming::tag(&[0][..]), |_| {
976                                    header::BabeAllowedSlots::PrimarySlots
977                                }),
978                                nom::combinator::map(nom::bytes::streaming::tag(&[1][..]), |_| {
979                                    header::BabeAllowedSlots::PrimaryAndSecondaryPlainSlots
980                                }),
981                                nom::combinator::map(nom::bytes::streaming::tag(&[2][..]), |_| {
982                                    header::BabeAllowedSlots::PrimaryAndSecondaryVrfSlots
983                                }),
984                            )),
985                            b,
986                        )
987                    }
988                },
989            ),
990            |(_slot_duration, slots_per_epoch, c0, c1, authorities, randomness, allowed_slots)| {
991                // Note that the slot duration is unused as it is not modifiable anyway.
992                BabeGenesisConfiguration {
993                    slots_per_epoch,
994                    epoch0_configuration: header::BabeNextConfig {
995                        c: (c0, c1),
996                        allowed_slots,
997                    },
998                    epoch0_information: header::BabeNextEpoch {
999                        randomness,
1000                        authorities,
1001                    },
1002                }
1003            },
1004        ))),
1005        bytes,
1006    );
1007
1008    match result {
1009        Ok((_, out)) => Ok(out),
1010        Err(nom::Err::Error(_) | nom::Err::Failure(_)) => Err(Error::BabeConfigurationOutputDecode),
1011        Err(_) => unreachable!(),
1012    }
1013}

1015/// Decodes the output of a call to `BabeApi_current_epoch` (`is_next_epoch` is `false`) or
1016/// `BabeApi_next_epoch` (`is_next_epoch` is `true`).
1017fn decode_babe_epoch_output(
1018    scale_encoded: &[u8],
1019    is_next_epoch: bool,
1020) -> Result<chain_information::BabeEpochInformation, Error> {
1021    let mut combinator = nom::combinator::all_consuming(nom::combinator::map(
1022        (
1023            nom::number::streaming::le_u64,
1024            nom::number::streaming::le_u64,
1025            nom::number::streaming::le_u64,
1026            nom::combinator::flat_map(crate::util::nom_scale_compact_usize, |num_elems| {
1027                nom::multi::many_m_n(
1028                    num_elems,
1029                    num_elems,
1030                    nom::combinator::map(
1031                        (
1032                            nom::bytes::streaming::take(32u32),
1033                            nom::number::streaming::le_u64,
1034                        ),
1035                        move |(public_key, weight)| header::BabeAuthority {
1036                            public_key: <[u8; 32]>::try_from(public_key).unwrap(),
1037                            weight,
1038                        },
1039                    ),
1040                )
1041            }),
1042            nom::combinator::map(nom::bytes::streaming::take(32u32), |b| {
1043                <[u8; 32]>::try_from(b).unwrap()
1044            }),
1045            nom::number::streaming::le_u64,
1046            nom::number::streaming::le_u64,
1047            |b| {
1048                header::BabeAllowedSlots::from_slice(b)
1049                    .map(|v| (&[][..], v))
1050                    .map_err(|_| {
1051                        nom::Err::Error(nom::error::make_error(b, nom::error::ErrorKind::Verify))
1052                    })
1053            },
1054        ),
1055        |(
1056            epoch_index,
1057            start_slot_number,
1058            _duration,
1059            authorities,
1060            randomness,
1061            c0,
1062            c1,
1063            allowed_slots,
1064        )| {
1065            chain_information::BabeEpochInformation {
1066                epoch_index,
1067                // Smoldot requires `start_slot_number` to be `None` in the context of next
1068                // epoch #0, because its start slot number can't be known. The runtime function,
1069                // however, as it doesn't have a way to represent `None`, instead returns an
1070                // unspecified value (typically `0`).
1071                start_slot_number: if !is_next_epoch || epoch_index != 0 {
1072                    Some(start_slot_number)
1073                } else {
1074                    None
1075                },
1076                authorities,
1077                randomness,
1078                c: (c0, c1),
1079                allowed_slots,
1080            }
1081        },
1082    ));
1083
1084    let result: Result<_, nom::Err<nom::error::Error<&[u8]>>> =
1085        nom::Parser::parse(&mut combinator, scale_encoded);
1086    match result {
1087        Ok((_, info)) => Ok(info),
1088        Err(_) => Err(if is_next_epoch {
1089            Error::BabeNextEpochOutputDecode
1090        } else {
1091            Error::BabeCurrentEpochOutputDecode
1092        }),
1093    }
1094}

1096/// Decodes the output of a call to `GrandpaApi_grandpa_authorities`, or the content of the
1097/// `:grandpa_authorities` storage item.
1098fn decode_grandpa_authorities_output(
1099    scale_encoded: &[u8],
1100) -> Result<Vec<header::GrandpaAuthority>, Error> {
1101    let result: nom::IResult<_, _> = nom::Parser::parse(
1102        &mut nom::combinator::all_consuming(nom::combinator::complete(nom::combinator::flat_map(
1103            crate::util::nom_scale_compact_usize,
1104            |num_elems| {
1105                nom::multi::fold_many_m_n(
1106                    num_elems,
1107                    num_elems,
1108                    (
1109                        nom::bytes::streaming::take(32u32),
1110                        nom::combinator::map_opt(
1111                            nom::number::streaming::le_u64,
1112                            NonZero::<u64>::new,
1113                        ),
1114                    ),
1115                    move || Vec::with_capacity(num_elems),
1116                    |mut acc, (public_key, weight)| {
1117                        acc.push(header::GrandpaAuthority {
1118                            public_key: <[u8; 32]>::try_from(public_key).unwrap(),
1119                            weight,
1120                        });
1121                        acc
1122                    },
1123                )
1124            },
1125        ))),
1126        scale_encoded,
1127    );
1128
1129    match result {
1130        Ok((_, out)) => Ok(out),
1131        Err(nom::Err::Error(_) | nom::Err::Failure(_)) => {
1132            Err(Error::GrandpaAuthoritiesOutputDecode)
1133        }
1134        Err(_) => unreachable!(),
1135    }
1136}
