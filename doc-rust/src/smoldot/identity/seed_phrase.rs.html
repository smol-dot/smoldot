30/// Decodes a human-readable private key (a.k.a. a seed phrase) using the Sr25519 curve.
31///
32/// > **Note**: The key is returned within a `Box` in order to guarantee that no trace of the
33/// >           secret key is accidentally left in memory due to automatic copies of stack data.
34pub fn decode_sr25519_private_key(phrase: &str) -> Result<Box<[u8; 64]>, ParsePrivateKeyError> {
35    let parsed = parse_private_key(phrase)?;
36
37    // Note: `from_bytes` can only panic if the slice is of the wrong length, which we know can
38    // never happen.
39    let mini_key =
40        zeroize::Zeroizing::new(schnorrkel::MiniSecretKey::from_bytes(&*parsed.seed).unwrap());
41
42    let mut secret_key = mini_key
43        .expand_to_keypair(schnorrkel::ExpansionMode::Ed25519)
44        .secret
45        .clone();
46
47    for junction in parsed.path {
48        secret_key = match junction {
49            DeriveJunction::Soft(_) => todo!(), // TODO: return error
50            DeriveJunction::Hard(cc) => secret_key
51                .hard_derive_mini_secret_key(Some(schnorrkel::derive::ChainCode(cc)), b"")
52                .0
53                .expand(schnorrkel::ExpansionMode::Ed25519),
54        };
55    }
56
57    // TODO: unclear if the zeroizing works, we probably need some tweaks in the schnorrkel library
58    let bytes = zeroize::Zeroizing::new(secret_key.to_bytes());
59    let mut out = Box::new([0; 64]);
60    out.copy_from_slice(bytes.as_ref());
61    Ok(out)
62}

64/// Decodes a human-readable private key (a.k.a. a seed phrase) using the Ed25519 curve.
65///
66/// > **Note**: The key is returned within a `Box` in order to guarantee that no trace of the
67/// >           secret key is accidentally left in memory due to automatic copies of stack data.
68pub fn decode_ed25519_private_key(phrase: &str) -> Result<Box<[u8; 32]>, ParsePrivateKeyError> {
69    let parsed = parse_private_key(phrase)?;
70
71    let mut secret_key = parsed.seed;
72    for junction in parsed.path {
73        secret_key = match junction {
74            DeriveJunction::Soft(_) => todo!(), // TODO: return error
75            DeriveJunction::Hard(cc) => {
76                let mut hash = blake2_rfc::blake2b::Blake2b::new(32);
77                hash.update(crate::util::encode_scale_compact_usize(11).as_ref()); // Length of `"Ed25519HDKD"`
78                hash.update(b"Ed25519HDKD");
79                hash.update(&*secret_key);
80                hash.update(&cc);
81
82                let mut out = Box::new([0; 32]);
83                out.copy_from_slice(hash.finalize().as_ref());
84                // TODO: `hash` should be zero'ed on drop :-/
85                out
86            }
87        };
88    }
89
90    Ok(secret_key)
91}

93/// Turns a human-readable private key (a.k.a. a seed phrase) into a seed and a derivation path.
94pub fn parse_private_key(phrase: &str) -> Result<ParsedPrivateKey, ParsePrivateKeyError> {
95    let parse_result: Result<_, nom::Err<nom::error::Error<&str>>> = nom::Parser::parse(
96        &mut nom::combinator::all_consuming((
97            // Either BIP39 words or some hexadecimal
98            nom::branch::alt((
99                // Hexadecimal. Wrapped in `either::Left`
100                nom::combinator::complete(nom::combinator::map(
101                    nom::combinator::map_opt(
102                        nom::sequence::preceded(
103                            nom::bytes::streaming::tag("0x"),
104                            nom::character::complete::hex_digit0,
105                        ),
106                        |hex| {
107                            let mut out = Box::new([0; 32]);
108                            hex::decode_to_slice(hex, &mut *out).ok()?;
109                            Some(out)
110                        },
111                    ),
112                    either::Left,
113                )),
114                // BIP39. Wrapped in `either::Right`
115                nom::combinator::complete(nom::combinator::map(
116                    nom::bytes::complete::take_till(|c| c == '/'),
117                    either::Right,
118                )),
119            )),
120            // Derivation path
121            nom::multi::many0(nom::branch::alt((
122                // Soft
123                nom::combinator::complete(nom::combinator::map(
124                    nom::sequence::preceded(
125                        nom::bytes::streaming::tag("/"),
126                        nom::bytes::complete::take_till1(|c| c == '/'),
127                    ),
128                    |code| DeriveJunction::from_components(false, code),
129                )),
130                // Hard
131                nom::combinator::complete(nom::combinator::map(
132                    nom::sequence::preceded(
133                        nom::bytes::streaming::tag("//"),
134                        nom::bytes::complete::take_till1(|c| c == '/'),
135                    ),
136                    |code| DeriveJunction::from_components(true, code),
137                )),
138            ))),
139            // Optional password
140            nom::combinator::opt(nom::combinator::complete(nom::sequence::preceded(
141                nom::bytes::streaming::tag("///"),
142                |s| Ok(("", s)), // Take the rest of the input after the `///`
143            ))),
144        )),
145        phrase,
146    );
147
148    match parse_result {
149        Ok((_, (either::Left(seed), path, _password))) => {
150            // Hexadecimal seed
151            // TODO: what if there's a password? do we just ignore it?
152            Ok(ParsedPrivateKey { seed, path })
153        }
154        Ok((_, (either::Right(phrase), path, password))) => {
155            // BIP39 words
156            let phrase = if phrase.is_empty() {
157                DEFAULT_SEED_PHRASE
158            } else {
159                phrase
160            };
161
162            Ok(ParsedPrivateKey {
163                seed: bip39_to_seed(phrase, password.unwrap_or(""))
164                    .map_err(ParsePrivateKeyError::Bip39Decode)?,
165                path,
166            })
167        }
168        Err(_) => Err(ParsePrivateKeyError::InvalidFormat),
169    }
170}

172/// Successful outcome of [`parse_private_key`].
173pub struct ParsedPrivateKey {
174    /// Base seed phrase. Must be derived through [`ParsedPrivateKey::path`] to obtain the final
175    /// result.
176    ///
177    /// > **Note**: The key is embedded within a `Box` in order to guarantee that no trace of the
178    /// >           secret key is accidentally left in memory due to automatic copies of stack
179    /// >           data.
180    pub seed: Box<[u8; 32]>,
181
182    /// Derivation path found in the secret phrase.
183    pub path: Vec<DeriveJunction>,
184}
185
186/// Error in [`parse_private_key`].
187#[derive(Debug, derive_more::Display, derive_more::Error)]
188pub enum ParsePrivateKeyError {
189    /// Couldn't parse the string in any meaningful way.
190    InvalidFormat,
191    /// Failed to decode the provided BIP39 seed phrase.
192    Bip39Decode(Bip39ToSeedError),
193}
194
195#[derive(Debug, Copy, Clone, PartialEq, Eq)]
196pub enum DeriveJunction {
197    Soft([u8; 32]),
198    Hard([u8; 32]),
199}
200
201impl DeriveJunction {
202    fn from_components(hard: bool, code: &str) -> DeriveJunction {
203        // The algorithm here is the same as in Substrate, but way more readable.
204        let mut chain_code = [0; 32];
205        if let Ok(n) = str::parse::<u64>(code) {
206            chain_code[..8].copy_from_slice(&n.to_le_bytes());
207        } else {
208            // A SCALE-compact-encoded length prefix is added in front of the path.
209            let code = code.as_bytes();
210            let code_len_prefix = crate::util::encode_scale_compact_usize(code.len());
211            let code_len_prefix = code_len_prefix.as_ref();
212
213            if code_len_prefix.len() + code.len() > 32 {
214                let mut hash = blake2_rfc::blake2b::Blake2b::new(32);
215                hash.update(code_len_prefix);
216                hash.update(code);
217                chain_code.copy_from_slice(hash.finalize().as_bytes());
218            } else {
219                chain_code[..code_len_prefix.len()].copy_from_slice(code_len_prefix);
220                chain_code[code_len_prefix.len()..][..code.len()].copy_from_slice(code);
221            }
222        }
223
224        if hard {
225            DeriveJunction::Hard(chain_code)
226        } else {
227            DeriveJunction::Soft(chain_code)
228        }
229    }
230}

232/// Turns a BIP39 seed phrase into a 32 bytes cryptographic seed.
233///
234/// > **Note**: The key is returned within a `Box` in order to guarantee that no trace of the
235/// >           secret key is accidentally left in memory due to automatic copies of stack data.
236pub fn bip39_to_seed(phrase: &str, password: &str) -> Result<Box<[u8; 32]>, Bip39ToSeedError> {
237    let parsed = bip39::Mnemonic::parse_in_normalized(bip39::Language::English, phrase)
238        .map_err(|err| Bip39ToSeedError::WrongMnemonic(Bip39DecodeError(err)))?;
239
240    // Note that the `bip39` library implementation that turns the mnemonic to a seed isn't
241    // conformant to the BIP39 specification. Instead, we do it manually.
242
243    // `to_entropy_array()` returns the entropy as an array where only the first `entropy_len`
244    // bytes are meaningful. `entropy_len` depends on the number of words provided.
245    let (entropy, entropy_len) = parsed.to_entropy_array();
246
247    // These rules are part of the seed phrase format "specification" and have been copy-pasted
248    // from the Substrate code base.
249    if !(16..=32).contains(&entropy_len) || entropy_len % 4 != 0 {
250        return Err(Bip39ToSeedError::BadWordsCount);
251    }
252
253    let mut salt = zeroize::Zeroizing::new(String::with_capacity(8 + password.len()));
254    salt.push_str("mnemonic");
255    salt.push_str(password);
256
257    // This function returns an error only in case of wrong buffer length, making it safe to
258    // unwrap.
259    let mut seed_too_long = Box::new([0u8; 64]);
260    pbkdf2::pbkdf2::<hmac::Hmac<sha2::Sha512>>(
261        &entropy[..entropy_len],
262        salt.as_bytes(),
263        2048,
264        &mut *seed_too_long,
265    )
266    .unwrap();
267
268    // The seed is truncated to 32 bytes.
269    let mut seed = Box::new([0u8; 32]);
270    seed.copy_from_slice(&seed_too_long[..32]);
271    seed_too_long.zeroize();
272
273    Ok(seed)
274}

276/// Failed to decode BIP39 mnemonic phrase.
277#[derive(Debug, derive_more::Display, derive_more::Error)]
278pub enum Bip39ToSeedError {
279    /// Invalid BIP39 mnemonic phrase.
280    WrongMnemonic(Bip39DecodeError),
281    /// Number of mnemonic phrase words isn't supported by the SS58 format.
282    BadWordsCount,
283}
