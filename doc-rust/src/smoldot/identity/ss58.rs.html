21/// Decoded version of an SS58 address.
22#[derive(Debug, Copy, Clone, PartialEq, Eq)]
23pub struct Decoded<P> {
24    /// Identifier indicating which chain is concerned.
25    ///
26    /// The mapping between chains and this prefix can be found in this central registry:
27    /// <https://github.com/paritytech/ss58-registry/blob/main/ss58-registry.json>.
28    pub chain_prefix: ChainPrefix,
29
30    /// Public key of the account.
31    pub public_key: P,
32}
33
34/// Identifier indicating which chain is concerned.
35///
36/// The mapping between chains and this prefix can be found in this central registry:
37/// <https://github.com/paritytech/ss58-registry/blob/main/ss58-registry.json>.
38///
39/// This prefix is a 14 bits unsigned integer.
40//
41// Implementation note: the `u16` is guaranteed to be only up to 14 bits long. The upper two bits
42// are always 0.
43#[derive(Copy, Clone, PartialEq, Eq)]
44pub struct ChainPrefix(u16);
45
46impl fmt::Debug for ChainPrefix {
47    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
48        fmt::Debug::fmt(&self.0, f)
49    }
50}
51
52impl TryFrom<u16> for ChainPrefix {
53    type Error = PrefixTooLargeError;
54
55    fn try_from(prefix: u16) -> Result<Self, Self::Error> {
56        if (prefix >> 14) == 0 {
57            Ok(ChainPrefix(prefix))
58        } else {
59            Err(PrefixTooLargeError())
60        }
61    }
62}
63
64/// Integer is too large to be a valid prefix
65#[derive(Debug, Clone, derive_more::Display, derive_more::Error)]
66pub struct PrefixTooLargeError();
67
68impl From<u8> for ChainPrefix {
69    fn from(prefix: u8) -> ChainPrefix {
70        ChainPrefix(u16::from(prefix))
71    }
72}
73
74impl From<ChainPrefix> for u16 {
75    fn from(prefix: ChainPrefix) -> u16 {
76        prefix.0
77    }
78}

80/// Turns a decoded SS58 address into a string.
81pub fn encode(decoded: Decoded<impl AsRef<[u8]>>) -> String {
82    let prefix = decoded.chain_prefix.0;
83    let public_key = decoded.public_key.as_ref();
84
85    let mut bytes = Vec::with_capacity(2 + public_key.len() + 2);
86
87    if prefix < 64 {
88        bytes.push(prefix as u8);
89    } else {
90        // This encoding is plain weird.
91        bytes.push(((prefix & 0b0000_0000_1111_1100) as u8) >> 2 | 0b01000000);
92        bytes.push(((prefix >> 8) as u8) | ((prefix & 0b0000_0000_0000_0011) as u8) << 6);
93    }
94
95    bytes.extend_from_slice(public_key);
96
97    let checksum = calculate_checksum(&bytes);
98    bytes.extend_from_slice(&checksum);
99
100    bs58::encode(&bytes).into_string()
101}

103/// Decodes an SS58 address from a string.
104pub fn decode(encoded: &str) -> Result<Decoded<impl AsRef<[u8]>>, DecodeError> {
105    let mut bytes = bs58::decode(encoded)
106        .into_vec()
107        .map_err(|err| DecodeError::InvalidBs58(Bs58DecodeError(err)))?;
108
109    if bytes.len() < 4 {
110        return Err(DecodeError::TooShort);
111    }
112
113    // Verify the checksum.
114    let expected_checksum = calculate_checksum(&bytes[..bytes.len() - 2]);
115    if expected_checksum[..] != bytes[bytes.len() - 2..] {
116        return Err(DecodeError::InvalidChecksum);
117    }
118    bytes.truncate(bytes.len() - 2);
119
120    // Grab and remove the prefix.
121    let (prefix_len, chain_prefix) = if bytes[0] < 64 {
122        (1, ChainPrefix(u16::from(bytes[0])))
123    } else if bytes[0] < 128 {
124        let prefix = u16::from_be_bytes([bytes[1] & 0b00111111, (bytes[0] << 2) | (bytes[1] >> 6)]);
125        (2, ChainPrefix(prefix))
126    } else {
127        return Err(DecodeError::InvalidPrefix);
128    };
129
130    // Rather than remove the prefix from the beginning of `bytes`, we adjust the `AsRef`
131    // implementation to skip the prefix.
132    let public_key = {
133        struct Adjust(Vec<u8>, usize);
134        impl AsRef<[u8]> for Adjust {
135            fn as_ref(&self) -> &[u8] {
136                &self.0[self.1..]
137            }
138        }
139        Adjust(bytes, prefix_len)
140    };
141
142    Ok(Decoded {
143        chain_prefix,
144        public_key,
145    })
146}

148/// Error while decoding an SS58 address.
149#[derive(Debug, Clone, derive_more::Display, derive_more::Error)]
150pub enum DecodeError {
151    /// SS58 is too short to possibly be valid.
152    TooShort,
153    /// Invalid SS58 prefix encoding.
154    InvalidPrefix,
155    /// Invalid BS58 format.
156    #[display("{_0}")]
157    InvalidBs58(Bs58DecodeError),
158    /// Calculated checksum doesn't match the one provided.
159    InvalidChecksum,
160}

167fn calculate_checksum(data: &[u8]) -> [u8; 2] {
168    let mut hasher = blake2_rfc::blake2b::Blake2b::new(64);
169    hasher.update(b"SS58PRE");
170    hasher.update(data);
171
172    let hash = hasher.finalize();
173    *<&[u8; 2]>::try_from(&hash.as_bytes()[..2]).unwrap_or_else(|_| unreachable!())
174}
