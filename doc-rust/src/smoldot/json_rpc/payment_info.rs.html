18use super::methods;

20/// Produces the input to pass to the `TransactionPaymentApi_query_info` runtime call.
21pub fn payment_info_parameters(extrinsic: &[u8]) -> impl Iterator<Item = impl AsRef<[u8]>> + Clone {
22    [
23        either::Left(extrinsic),
24        either::Right(u32::try_from(extrinsic.len()).unwrap().to_le_bytes()),
25    ]
26    .into_iter()
27}
28
29/// Name of the runtime function to call in order to obtain the payment fees.
30pub const PAYMENT_FEES_FUNCTION_NAME: &str = "TransactionPaymentApi_query_info";
31
32/// Attempt to decode the output of the runtime call.
33///
34/// Must be passed the version of the `TransactionPaymentApi` API, according to the runtime
35/// specification.
36pub fn decode_payment_info(
37    scale_encoded: &[u8],
38    api_version: u32,
39) -> Result<methods::RuntimeDispatchInfo, DecodeError> {
40    let is_api_v2 = match api_version {
41        1 => false,
42        2 => true,
43        _ => return Err(DecodeError::UnknownRuntimeVersion),
44    };
45
46    match nom::Parser::parse(
47        &mut nom::combinator::all_consuming(nom_decode_payment_info::<nom::error::Error<&[u8]>>(
48            is_api_v2,
49        )),
50        scale_encoded,
51    ) {
52        Ok((_, info)) => Ok(info),
53        Err(_) => Err(DecodeError::ParseError),
54    }
55}
56
57/// Potential error when decoding payment information runtime output.
58#[derive(Debug, derive_more::Display, derive_more::Error)]
59pub enum DecodeError {
60    /// Failed to parse the return value of `TransactionPaymentApi_query_info`.
61    ParseError,
62    /// The `TransactionPaymentApi` API uses a version that smoldot doesn't support.
63    UnknownRuntimeVersion,
64}
65
66fn nom_decode_payment_info<'a, E: nom::error::ParseError<&'a [u8]>>(
67    is_api_v2: bool,
68) -> impl nom::Parser<&'a [u8], Output = methods::RuntimeDispatchInfo, Error = E> {
69    nom::combinator::map(
70        (
71            move |bytes| {
72                if is_api_v2 {
73                    nom::number::streaming::le_u64(bytes)
74                } else {
75                    nom::Parser::parse(
76                        &mut nom::combinator::map(
77                            (
78                                crate::util::nom_scale_compact_u64,
79                                crate::util::nom_scale_compact_u64,
80                            ),
81                            |(ref_time, _proof_size)| ref_time,
82                        ),
83                        bytes,
84                    )
85                }
86            },
87            nom::combinator::map_opt(nom::number::streaming::u8, |n| match n {
88                0 => Some(methods::DispatchClass::Normal),
89                1 => Some(methods::DispatchClass::Operational),
90                2 => Some(methods::DispatchClass::Mandatory),
91                _ => None,
92            }),
93            |bytes| {
94                // The exact format here is the SCALE encoding of the type `Balance`.
95                // Normally, determining the actual type of `Balance` would require parsing the
96                // metadata provided by the runtime. However, this is a pretty difficult to
97                // implement and CPU-heavy. Instead, given that there is no other field after
98                // the balance, we simply parse all the remaining bytes.
99                // Because the SCALE encoding of a number is the number in little endian format,
100                // we decode the bytes in little endian format in a way that works no matter the
101                // number of bytes.
102                // If a field was to be added after the balance, this code would need to be
103                // modified.
104                let mut num = 0u128;
105                let mut shift = 0u32;
106                for byte in <[u8]>::iter(bytes) {
107                    let shifted =
108                        u128::from(*byte)
109                            .checked_mul(1 << shift)
110                            .ok_or(nom::Err::Error(nom::error::make_error(
111                                bytes,
112                                nom::error::ErrorKind::Digit,
113                            )))?;
114                    num =
115                        num.checked_add(shifted)
116                            .ok_or(nom::Err::Error(nom::error::make_error(
117                                bytes,
118                                nom::error::ErrorKind::Digit,
119                            )))?;
120                    shift =
121                        shift
122                            .checked_add(16)
123                            .ok_or(nom::Err::Error(nom::error::make_error(
124                                bytes,
125                                nom::error::ErrorKind::Digit,
126                            )))?;
127                }
128
129                Ok((&[][..], num))
130            },
131        ),
132        |(weight, class, partial_fee)| methods::RuntimeDispatchInfo {
133            weight,
134            class,
135            partial_fee,
136        },
137    )
138}
