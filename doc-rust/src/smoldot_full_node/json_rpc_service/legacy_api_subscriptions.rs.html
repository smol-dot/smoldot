29/// Helper that provides the blocks of a `chain_subscribeAllHeads` subscription.
30pub struct SubscribeAllHeads {
31    consensus_service: Arc<consensus_service::ConsensusService>,
32
33    /// Active subscription to the consensus service blocks. `None` if not subscribed yet or if
34    /// the subscription has stopped.
35    subscription: Option<SubscribeAllHeadsSubscription>,
36}
37
38struct SubscribeAllHeadsSubscription {
39    subscription_id: consensus_service::SubscriptionId,
40    new_blocks: Pin<Box<async_channel::Receiver<consensus_service::Notification>>>,
41    blocks_to_unpin: Vec<[u8; 32]>,
42}
43
44impl SubscribeAllHeads {
45    /// Builds a new [`SubscribeAllHeads`].
46    pub fn new(consensus_service: Arc<consensus_service::ConsensusService>) -> Self {
47        SubscribeAllHeads {
48            consensus_service,
49            subscription: None,
50        }
51    }
52
53    /// Returns the SCALE-encoded header of the next block to provide as part of the subscription.
54    pub async fn next_scale_encoded_header(&mut self) -> Vec<u8> {
55        loop {
56            let subscription = match &mut self.subscription {
57                Some(s) => s,
58                None => {
59                    let subscribe_all = self
60                        .consensus_service
61                        .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
62                        .await;
63
64                    let blocks_to_unpin = iter::once(subscribe_all.finalized_block_hash)
65                        .chain(
66                            subscribe_all
67                                .non_finalized_blocks_ancestry_order
68                                .into_iter()
69                                .map(|b| b.block_hash),
70                        )
71                        .collect();
72
73                    self.subscription.insert(SubscribeAllHeadsSubscription {
74                        subscription_id: subscribe_all.id,
75                        new_blocks: Box::pin(subscribe_all.new_blocks),
76                        blocks_to_unpin,
77                    })
78                }
79            };
80
81            while let Some(block_to_unpin) = subscription.blocks_to_unpin.last() {
82                self.consensus_service
83                    .unpin_block(subscription.subscription_id, *block_to_unpin)
84                    .await;
85                let _ = subscription.blocks_to_unpin.pop();
86            }
87
88            loop {
89                match subscription.new_blocks.next().await {
90                    None => {
91                        self.subscription = None;
92                        break;
93                    }
94                    Some(consensus_service::Notification::Block { block, .. }) => {
95                        subscription.blocks_to_unpin.push(block.block_hash);
96                        return block.scale_encoded_header;
97                    }
98                    Some(consensus_service::Notification::Finalized { .. }) => {
99                        // Ignore event.
100                    }
101                }
102            }
103        }
104    }
105}

107/// Helper that provides the blocks of a `chain_subscribeFinalizedHeads` subscription.
108pub struct SubscribeFinalizedHeads {
109    consensus_service: Arc<consensus_service::ConsensusService>,
110
111    /// Active subscription to the consensus service blocks. `None` if not subscribed yet or if
112    /// the subscription has stopped.
113    subscription: Option<SubscribeFinalizedHeadsSubscription>,
114}
115
116struct SubscribeFinalizedHeadsSubscription {
117    subscription_id: consensus_service::SubscriptionId,
118    new_blocks: Pin<Box<async_channel::Receiver<consensus_service::Notification>>>,
119    pinned_blocks: HashMap<[u8; 32], Vec<u8>>,
120    blocks_to_unpin: Vec<[u8; 32]>,
121}
122
123impl SubscribeFinalizedHeads {
124    /// Builds a new [`SubscribeFinalizedHeads`].
125    pub fn new(consensus_service: Arc<consensus_service::ConsensusService>) -> Self {
126        SubscribeFinalizedHeads {
127            consensus_service,
128            subscription: None,
129        }
130    }
131
132    /// Returns the SCALE-encoded header of the next block to provide as part of the subscription.
133    pub async fn next_scale_encoded_header(&mut self) -> Vec<u8> {
134        loop {
135            let subscription = match &mut self.subscription {
136                Some(s) => s,
137                None => {
138                    let subscribe_all = self
139                        .consensus_service
140                        .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
141                        .await;
142
143                    let mut pinned_blocks = HashMap::with_capacity(
144                        subscribe_all.non_finalized_blocks_ancestry_order.len() + 1 + 8,
145                    );
146                    for block in subscribe_all.non_finalized_blocks_ancestry_order {
147                        pinned_blocks.insert(block.block_hash, block.scale_encoded_header);
148                    }
149
150                    let mut blocks_to_unpin = Vec::with_capacity(8);
151                    blocks_to_unpin.push(subscribe_all.finalized_block_hash);
152
153                    self.subscription = Some(SubscribeFinalizedHeadsSubscription {
154                        subscription_id: subscribe_all.id,
155                        new_blocks: Box::pin(subscribe_all.new_blocks),
156                        pinned_blocks,
157                        blocks_to_unpin,
158                    });
159
160                    return subscribe_all.finalized_block_scale_encoded_header;
161                }
162            };
163
164            while let Some(block_to_unpin) = subscription.blocks_to_unpin.last() {
165                self.consensus_service
166                    .unpin_block(subscription.subscription_id, *block_to_unpin)
167                    .await;
168                let _ = subscription.blocks_to_unpin.pop();
169            }
170
171            loop {
172                match subscription.new_blocks.next().await {
173                    None => {
174                        self.subscription = None;
175                        break;
176                    }
177                    Some(consensus_service::Notification::Block { block, .. }) => {
178                        subscription
179                            .pinned_blocks
180                            .insert(block.block_hash, block.scale_encoded_header);
181                    }
182                    Some(consensus_service::Notification::Finalized {
183                        finalized_blocks_newest_to_oldest,
184                        pruned_blocks_hashes,
185                        ..
186                    }) => {
187                        debug_assert!(!finalized_blocks_newest_to_oldest.is_empty());
188                        let finalized_block_hash =
189                            *finalized_blocks_newest_to_oldest.first().unwrap();
190                        subscription.blocks_to_unpin.push(finalized_block_hash);
191                        let finalized_block_header = subscription
192                            .pinned_blocks
193                            .remove(&finalized_block_hash)
194                            .unwrap();
195
196                        for block in pruned_blocks_hashes {
197                            subscription.blocks_to_unpin.push(block);
198                            let _was_in = subscription.pinned_blocks.remove(&block);
199                            debug_assert!(_was_in.is_some());
200                        }
201
202                        for block in finalized_blocks_newest_to_oldest.into_iter().skip(1) {
203                            subscription.blocks_to_unpin.push(block);
204                            let _was_in = subscription.pinned_blocks.remove(&block);
205                            debug_assert!(_was_in.is_some());
206                        }
207
208                        return finalized_block_header;
209                    }
210                }
211            }
212        }
213    }
214}

216/// Helper that provides the blocks of a `chain_subscribeNewHeads` subscription.
217pub struct SubscribeNewHeads {
218    consensus_service: Arc<consensus_service::ConsensusService>,
219
220    /// Active subscription to the consensus service blocks. `None` if not subscribed yet or if
221    /// the subscription has stopped.
222    subscription: Option<SubscribeNewHeadsSubscription>,
223}
224
225struct SubscribeNewHeadsSubscription {
226    subscription_id: consensus_service::SubscriptionId,
227    new_blocks: Pin<Box<async_channel::Receiver<consensus_service::Notification>>>,
228    pinned_blocks: HashMap<[u8; 32], Vec<u8>>,
229    blocks_to_unpin: Vec<[u8; 32]>,
230    current_best_block_hash: [u8; 32],
231}
232
233impl SubscribeNewHeads {
234    /// Builds a new [`SubscribeNewHeads`].
235    pub fn new(consensus_service: Arc<consensus_service::ConsensusService>) -> Self {
236        SubscribeNewHeads {
237            consensus_service,
238            subscription: None,
239        }
240    }
241
242    /// Returns the SCALE-encoded header of the next block to provide as part of the subscription.
243    pub async fn next_scale_encoded_header(&mut self) -> &Vec<u8> {
244        // Note: this function is convoluted with many unwraps due to a difficult fight with the
245        // Rust borrow checker.
246
247        loop {
248            if self.subscription.is_none() {
249                let subscribe_all = self
250                    .consensus_service
251                    .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
252                    .await;
253
254                let mut pinned_blocks = HashMap::with_capacity(
255                    subscribe_all.non_finalized_blocks_ancestry_order.len() + 1 + 8,
256                );
257
258                let mut current_best_block_hash = subscribe_all.finalized_block_hash;
259
260                pinned_blocks.insert(
261                    subscribe_all.finalized_block_hash,
262                    subscribe_all.finalized_block_scale_encoded_header,
263                );
264
265                for block in subscribe_all.non_finalized_blocks_ancestry_order {
266                    pinned_blocks.insert(block.block_hash, block.scale_encoded_header);
267                    if block.is_new_best {
268                        current_best_block_hash = block.block_hash;
269                    }
270                }
271
272                let subscription = self.subscription.insert(SubscribeNewHeadsSubscription {
273                    subscription_id: subscribe_all.id,
274                    new_blocks: Box::pin(subscribe_all.new_blocks),
275                    pinned_blocks,
276                    blocks_to_unpin: Vec::with_capacity(8),
277                    current_best_block_hash,
278                });
279
280                return subscription
281                    .pinned_blocks
282                    .get(&subscription.current_best_block_hash)
283                    .unwrap();
284            }
285
286            {
287                let subscription = self.subscription.as_mut().unwrap();
288                while let Some(block_to_unpin) = subscription.blocks_to_unpin.last() {
289                    self.consensus_service
290                        .unpin_block(subscription.subscription_id, *block_to_unpin)
291                        .await;
292                    let _ = subscription.blocks_to_unpin.pop();
293                }
294            }
295
296            loop {
297                let notification = self.subscription.as_mut().unwrap().new_blocks.next().await;
298                let Some(notification) = notification else {
299                    self.subscription = None;
300                    break;
301                };
302
303                match notification {
304                    consensus_service::Notification::Block { block, .. } => {
305                        let _previous_value = self
306                            .subscription
307                            .as_mut()
308                            .unwrap()
309                            .pinned_blocks
310                            .insert(block.block_hash, block.scale_encoded_header);
311                        debug_assert!(_previous_value.is_none());
312
313                        if block.is_new_best {
314                            self.subscription.as_mut().unwrap().current_best_block_hash =
315                                block.block_hash;
316                            return self
317                                .subscription
318                                .as_mut()
319                                .unwrap()
320                                .pinned_blocks
321                                .get(&block.block_hash)
322                                .unwrap();
323                        }
324                    }
325                    consensus_service::Notification::Finalized {
326                        pruned_blocks_hashes,
327                        finalized_blocks_newest_to_oldest,
328                        best_block_hash,
329                    } => {
330                        for hash in pruned_blocks_hashes {
331                            self.subscription
332                                .as_mut()
333                                .unwrap()
334                                .blocks_to_unpin
335                                .push(hash);
336                            let _was_in = self
337                                .subscription
338                                .as_mut()
339                                .unwrap()
340                                .pinned_blocks
341                                .remove(&hash);
342                            debug_assert!(_was_in.is_some());
343                        }
344
345                        for hash in finalized_blocks_newest_to_oldest.iter().skip(1) {
346                            self.subscription
347                                .as_mut()
348                                .unwrap()
349                                .blocks_to_unpin
350                                .push(*hash);
351                            let _was_in = self
352                                .subscription
353                                .as_mut()
354                                .unwrap()
355                                .pinned_blocks
356                                .remove(hash);
357                            debug_assert!(_was_in.is_some());
358                        }
359
360                        if best_block_hash
361                            != self.subscription.as_mut().unwrap().current_best_block_hash
362                        {
363                            self.subscription.as_mut().unwrap().current_best_block_hash =
364                                best_block_hash;
365                            return self
366                                .subscription
367                                .as_mut()
368                                .unwrap()
369                                .pinned_blocks
370                                .get(&best_block_hash)
371                                .unwrap();
372                        }
373                    }
374                }
375            }
376        }
377    }
378}

380/// Helper that provides the blocks of a `state_subscribeRuntimeVersion` subscription.
381pub struct SubscribeRuntimeVersion {
382    consensus_service: Arc<consensus_service::ConsensusService>,
383
384    /// Active subscription to the consensus service blocks. `None` if not subscribed yet or if
385    /// the subscription has stopped.
386    subscription: Option<SubscribeRuntimeVersionSubscription>,
387}
388
389struct SubscribeRuntimeVersionSubscription {
390    subscription_id: consensus_service::SubscriptionId,
391    new_blocks: Pin<Box<async_channel::Receiver<consensus_service::Notification>>>,
392    pinned_blocks: HashMap<[u8; 32], Arc<HostVmPrototype>>,
393    blocks_to_unpin: Vec<[u8; 32]>,
394    current_best_block_hash: [u8; 32],
395    current_best_block_runtime: Arc<HostVmPrototype>,
396}
397
398impl SubscribeRuntimeVersion {
399    /// Builds a new [`SubscribeRuntimeVersion`].
400    pub fn new(consensus_service: Arc<consensus_service::ConsensusService>) -> Self {
401        SubscribeRuntimeVersion {
402            consensus_service,
403            subscription: None,
404        }
405    }
406
407    /// Returns the next runtime version to provide as part of the subscription.
408    pub async fn next_runtime_version(&mut self) -> &CoreVersion {
409        // Note: this function is convoluted with many unwraps due to a difficult fight with the
410        // Rust borrow checker.
411
412        loop {
413            if self.subscription.is_none() {
414                let subscribe_all = self
415                    .consensus_service
416                    .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
417                    .await;
418
419                let mut pinned_blocks = HashMap::with_capacity(
420                    subscribe_all.non_finalized_blocks_ancestry_order.len() + 1 + 8,
421                );
422
423                let mut current_best_block_hash = subscribe_all.finalized_block_hash;
424                let mut current_best_block_runtime = subscribe_all.finalized_block_runtime.clone();
425
426                pinned_blocks.insert(
427                    subscribe_all.finalized_block_hash,
428                    subscribe_all.finalized_block_runtime,
429                );
430
431                for block in subscribe_all.non_finalized_blocks_ancestry_order {
432                    let runtime = block
433                        .runtime_update
434                        .unwrap_or_else(|| pinned_blocks.get(&block.parent_hash).unwrap().clone());
435                    if block.is_new_best {
436                        current_best_block_hash = block.block_hash;
437                        current_best_block_runtime = runtime.clone();
438                    }
439                    pinned_blocks.insert(block.block_hash, runtime);
440                }
441
442                let subscription = self
443                    .subscription
444                    .insert(SubscribeRuntimeVersionSubscription {
445                        subscription_id: subscribe_all.id,
446                        new_blocks: Box::pin(subscribe_all.new_blocks),
447                        pinned_blocks,
448                        blocks_to_unpin: Vec::with_capacity(8),
449                        current_best_block_hash,
450                        current_best_block_runtime: current_best_block_runtime.clone(),
451                    });
452
453                return subscription.current_best_block_runtime.runtime_version();
454            }
455
456            {
457                let subscription = self.subscription.as_mut().unwrap();
458                while let Some(block_to_unpin) = subscription.blocks_to_unpin.last() {
459                    self.consensus_service
460                        .unpin_block(subscription.subscription_id, *block_to_unpin)
461                        .await;
462                    let _ = subscription.blocks_to_unpin.pop();
463                }
464            }
465
466            loop {
467                let notification = self.subscription.as_mut().unwrap().new_blocks.next().await;
468                let Some(notification) = notification else {
469                    self.subscription = None;
470                    break;
471                };
472
473                match notification {
474                    consensus_service::Notification::Block { block, .. } => {
475                        let runtime = block.runtime_update.unwrap_or_else(|| {
476                            self.subscription
477                                .as_ref()
478                                .unwrap()
479                                .pinned_blocks
480                                .get(&block.parent_hash)
481                                .unwrap()
482                                .clone()
483                        });
484
485                        let _previous_value = self
486                            .subscription
487                            .as_mut()
488                            .unwrap()
489                            .pinned_blocks
490                            .insert(block.block_hash, runtime.clone());
491                        debug_assert!(_previous_value.is_none());
492
493                        if block.is_new_best {
494                            self.subscription.as_mut().unwrap().current_best_block_hash =
495                                block.block_hash;
496                            if !Arc::ptr_eq(
497                                &self
498                                    .subscription
499                                    .as_mut()
500                                    .unwrap()
501                                    .current_best_block_runtime,
502                                &runtime,
503                            ) {
504                                self.subscription
505                                    .as_mut()
506                                    .unwrap()
507                                    .current_best_block_runtime = runtime;
508                                return self
509                                    .subscription
510                                    .as_mut()
511                                    .unwrap()
512                                    .current_best_block_runtime
513                                    .runtime_version();
514                            }
515                        }
516                    }
517                    consensus_service::Notification::Finalized {
518                        pruned_blocks_hashes,
519                        finalized_blocks_newest_to_oldest,
520                        best_block_hash,
521                    } => {
522                        for hash in pruned_blocks_hashes {
523                            self.subscription
524                                .as_mut()
525                                .unwrap()
526                                .blocks_to_unpin
527                                .push(hash);
528                            let _was_in = self
529                                .subscription
530                                .as_mut()
531                                .unwrap()
532                                .pinned_blocks
533                                .remove(&hash);
534                            debug_assert!(_was_in.is_some());
535                        }
536
537                        for hash in finalized_blocks_newest_to_oldest.iter().skip(1) {
538                            self.subscription
539                                .as_mut()
540                                .unwrap()
541                                .blocks_to_unpin
542                                .push(*hash);
543                            let _was_in = self
544                                .subscription
545                                .as_mut()
546                                .unwrap()
547                                .pinned_blocks
548                                .remove(hash);
549                            debug_assert!(_was_in.is_some());
550                        }
551
552                        if best_block_hash
553                            != self.subscription.as_mut().unwrap().current_best_block_hash
554                        {
555                            self.subscription.as_mut().unwrap().current_best_block_hash =
556                                best_block_hash;
557                            let new_best_runtime = self
558                                .subscription
559                                .as_ref()
560                                .unwrap()
561                                .pinned_blocks
562                                .get(&best_block_hash)
563                                .unwrap()
564                                .clone();
565                            if !Arc::ptr_eq(
566                                &self
567                                    .subscription
568                                    .as_mut()
569                                    .unwrap()
570                                    .current_best_block_runtime,
571                                &new_best_runtime,
572                            ) {
573                                self.subscription
574                                    .as_mut()
575                                    .unwrap()
576                                    .current_best_block_runtime = new_best_runtime;
577                                return self
578                                    .subscription
579                                    .as_mut()
580                                    .unwrap()
581                                    .current_best_block_runtime
582                                    .runtime_version();
583                            }
584                        }
585                    }
586                }
587            }
588        }
589    }
590}

592/// Helper that provides the blocks of a `state_subscribeStorage` subscription.
593///
594/// Note that various corner cases are weirdly handled, due to `state_subscribeStorage` not being
595/// properly defined anyway.
596pub struct SubscribeStorage {
597    /// Consensus service that was passed to [`SubscribeStorage::new`].
598    consensus_service: Arc<consensus_service::ConsensusService>,
599    /// Database that was passed to [`SubscribeStorage::new`].
600    database: Arc<database_thread::DatabaseThread>,
601    /// List of keys that was passed to [`SubscribeStorage::new`].
602    keys: Vec<Vec<u8>>,
603
604    /// Active subscription to the consensus service blocks. `None` if not subscribed yet or if
605    /// the subscription has stopped.
606    subscription: Option<SubscribeStorageSubscription>,
607}
608
609struct SubscribeStorageSubscription {
610    /// Next changes report currently being prepared.
611    new_report_preparation: Vec<(Vec<u8>, Option<Vec<u8>>)>,
612    /// List of keys that remain to be included
613    /// in [`SubscribeStorageSubscription::new_report_preparation`].
614    new_report_remaining_keys: hashbrown::HashSet<Vec<u8>, fnv::FnvBuildHasher>,
615
616    /// Identifier of the subscription towards the consensus service.
617    subscription_id: consensus_service::SubscriptionId,
618    /// Channel connected to the consensus service where notifications are received.
619    new_blocks: Pin<Box<async_channel::Receiver<consensus_service::Notification>>>,
620    /// List of block hashes that are still pinned but are not necessary anymore and should be
621    /// unpinned.
622    blocks_to_unpin: Vec<[u8; 32]>,
623
624    /// Tree of all pinned blocks. Doesn't include the current finalized block.
625    pinned_blocks: fork_tree::ForkTree<[u8; 32]>,
626    /// Content of [`SubscribeStorageSubscription::pinned_blocks`], indexed by block hashes.
627    pinned_blocks_by_hash: hashbrown::HashMap<[u8; 32], fork_tree::NodeIndex>,
628    /// Contains all the storage changes related to the keys found by [`SubscribeStorage::keys`]
629    /// (or all keys if subscribing to all keys) made in the pinned blocks found
630    /// in [`SubscribeStorageSubscription::pinned_blocks`].
631    ///
632    /// Because the storage changes of blocks that were already present at the time when the
633    /// subscription starts are unknown, they are also not in this list. This leads to corner
634    /// cases where some changes aren't provided, but we don't really care
635    /// as `state_subscribeStorage` is not properly defined anyway.
636    pinned_blocks_storage_changes: BTreeSet<(fork_tree::NodeIndex, Vec<u8>)>,
637    /// Hash of the current finalized block. Not found
638    /// in [`SubscribeStorageSubscription::pinned_blocks`].
639    current_finalized_block_hash: [u8; 32],
640    /// Index of the current  best block within [`SubscribeStorageSubscription::pinned_blocks`],
641    /// or `None` if the best block is equal to the finalized block.
642    current_best_block_index: Option<fork_tree::NodeIndex>,
643}
644
645impl SubscribeStorage {
646    /// Builds a new [`SubscribeStorage`].
647    ///
648    /// If the list of keys is empty, then all storage changes are reported, in accordance to the
649    /// behavior of `state_subscribeStorage`.
650    pub fn new(
651        consensus_service: Arc<consensus_service::ConsensusService>,
652        database: Arc<database_thread::DatabaseThread>,
653        subscribed_keys: Vec<Vec<u8>>,
654    ) -> Self {
655        SubscribeStorage {
656            consensus_service,
657            database,
658            keys: subscribed_keys,
659            subscription: None,
660        }
661    }
662
663    /// Returns the next storage change notification.
664    pub async fn next_storage_update(
665        &mut self,
666    ) -> ([u8; 32], impl Iterator<Item = (Vec<u8>, Option<Vec<u8>>)>) {
667        'main_subscription: loop {
668            // Get the active consensus service subscription, or subscribe if necessary.
669            let subscription = match &mut self.subscription {
670                Some(s) => s,
671                subscription @ None => {
672                    let subscribe_all = self
673                        .consensus_service
674                        .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
675                        .await;
676
677                    let mut pinned_blocks_by_hash = HashMap::with_capacity(
678                        subscribe_all.non_finalized_blocks_ancestry_order.len() + 1 + 8,
679                    );
680                    let mut pinned_blocks =
681                        fork_tree::ForkTree::with_capacity(pinned_blocks_by_hash.capacity());
682
683                    let mut current_best_block_index = None;
684
685                    for block in subscribe_all.non_finalized_blocks_ancestry_order {
686                        let node_index = pinned_blocks.insert(
687                            if block.parent_hash != subscribe_all.finalized_block_hash {
688                                Some(*pinned_blocks_by_hash.get(&block.parent_hash).unwrap())
689                            } else {
690                                None
691                            },
692                            block.block_hash,
693                        );
694                        pinned_blocks_by_hash.insert(block.block_hash, node_index);
695                        if block.is_new_best {
696                            current_best_block_index = Some(node_index);
697                        }
698                    }
699
700                    subscription.insert(SubscribeStorageSubscription {
701                        new_report_preparation: Vec::with_capacity(self.keys.len()),
702                        // We put all the keys in `new_report_remaining_keys`, as we must indicate
703                        // the initial values of all the keys.
704                        // If the list of keys is empty (meaning that the API user wants to
705                        // subscribe to all keys), this will intentionally not lead to any report,
706                        // as said report would be huge.
707                        // It is unclear how `state_subscribeStorage` is supposed to behave in
708                        // that situation.
709                        //
710                        // Also note that this initial report will happen after a re-subscription,
711                        // in order to not miss a storage change.
712                        new_report_remaining_keys: self.keys.iter().cloned().collect(),
713                        subscription_id: subscribe_all.id,
714                        new_blocks: Box::pin(subscribe_all.new_blocks),
715                        pinned_blocks,
716                        pinned_blocks_by_hash,
717                        pinned_blocks_storage_changes: BTreeSet::new(),
718                        blocks_to_unpin: Vec::with_capacity(8),
719                        current_finalized_block_hash: subscribe_all.finalized_block_hash,
720                        current_best_block_index,
721                    })
722                }
723            };
724
725            // Unpin the blocks that must be unpinned.
726            while let Some(block_to_unpin) = subscription.blocks_to_unpin.last() {
727                self.consensus_service
728                    .unpin_block(subscription.subscription_id, *block_to_unpin)
729                    .await;
730                let _ = subscription.blocks_to_unpin.pop();
731            }
732
733            // Continue to fill the next storage changes report.
734            while let Some(key) = subscription.new_report_remaining_keys.iter().next() {
735                let best_block_hash = subscription
736                    .current_best_block_index
737                    .map_or(subscription.current_finalized_block_hash, |idx| {
738                        *subscription.pinned_blocks.get(idx).unwrap()
739                    });
740
741                let key = key.clone();
742
743                let (key, result) = self
744                    .database
745                    .with_database(move |database| {
746                        let result = database.block_storage_get(
747                            &best_block_hash,
748                            iter::empty::<iter::Empty<_>>(),
749                            trie::bytes_to_nibbles(key.iter().copied()).map(u8::from),
750                        );
751                        (key, result)
752                    })
753                    .await;
754
755                subscription.new_report_remaining_keys.remove(&key);
756
757                match result {
758                    Ok(value) => subscription
759                        .new_report_preparation
760                        .push((key, value.map(|(v, _)| v))),
761                    Err(database_thread::StorageAccessError::UnknownBlock)
762                    | Err(database_thread::StorageAccessError::IncompleteStorage) => {
763                        self.subscription = None;
764                        continue 'main_subscription;
765                    }
766                    Err(database_thread::StorageAccessError::Corrupted(_)) => {
767                        // Database corruption errors are ignored.
768                        continue;
769                    }
770                }
771            }
772
773            // Send the storage changes report if it is complete.
774            debug_assert!(subscription.new_report_remaining_keys.is_empty());
775            if !subscription.new_report_preparation.is_empty() {
776                let best_block_hash = subscription
777                    .current_best_block_index
778                    .map_or(subscription.current_finalized_block_hash, |idx| {
779                        *subscription.pinned_blocks.get(idx).unwrap()
780                    });
781                return (
782                    best_block_hash,
783                    mem::replace(
784                        &mut subscription.new_report_preparation,
785                        Vec::with_capacity(self.keys.len()),
786                    )
787                    .into_iter(),
788                );
789            }
790
791            // Process the next incoming consensus service notification.
792            let notification = subscription.new_blocks.next().await;
793            let Some(mut notification) = notification else {
794                self.subscription = None;
795                continue;
796            };
797
798            // If the notification is about a new block, insert said new block in the state
799            // machine.
800            if let consensus_service::Notification::Block {
801                block,
802                storage_changes,
803            } = &mut notification
804            {
805                let node_index = subscription.pinned_blocks.insert(
806                    if block.parent_hash != subscription.current_finalized_block_hash {
807                        Some(
808                            *subscription
809                                .pinned_blocks_by_hash
810                                .get(&block.parent_hash)
811                                .unwrap(),
812                        )
813                    } else {
814                        None
815                    },
816                    block.block_hash,
817                );
818
819                subscription
820                    .pinned_blocks_by_hash
821                    .insert(block.block_hash, node_index);
822
823                if !self.keys.is_empty() {
824                    for key in &self.keys {
825                        if storage_changes.main_trie_diff_get(key).is_some() {
826                            subscription
827                                .pinned_blocks_storage_changes
828                                .insert((node_index, key.clone()));
829                        }
830                    }
831                } else {
832                    for (changed_key, _) in
833                        storage_changes.main_trie_storage_changes_iter_unordered()
834                    {
835                        subscription
836                            .pinned_blocks_storage_changes
837                            .insert((node_index, changed_key.to_owned()));
838                    }
839                }
840            }
841
842            // If the notification changes the best block, find the keys that have changed and
843            // put them in `new_report_remaining_keys`.
844            if let consensus_service::Notification::Block {
845                block:
846                    consensus_service::BlockNotification {
847                        block_hash: best_block_hash,
848                        is_new_best: true,
849                        ..
850                    },
851                ..
852            }
853            | consensus_service::Notification::Finalized {
854                best_block_hash, ..
855            } = &notification
856            {
857                let new_best_block_node_index = *subscription
858                    .pinned_blocks_by_hash
859                    .get(best_block_hash)
860                    .unwrap();
861
862                let ascend_descend_iter = match subscription.current_best_block_index {
863                    Some(prev_best_idx) => {
864                        let (a, d) = subscription
865                            .pinned_blocks
866                            .ascend_and_descend(prev_best_idx, new_best_block_node_index);
867                        either::Left(a.chain(d))
868                    }
869                    None => either::Right(
870                        subscription
871                            .pinned_blocks
872                            .root_to_node_path(new_best_block_node_index),
873                    ),
874                };
875
876                for block in ascend_descend_iter {
877                    let storage_changes = subscription.pinned_blocks_storage_changes.range((
878                        ops::Bound::Included((block, Vec::new())),
879                        if let Some(block_plus_one) = block.inc() {
880                            ops::Bound::Excluded((block_plus_one, Vec::new()))
881                        } else {
882                            ops::Bound::Unbounded
883                        },
884                    ));
885
886                    for (_, key) in storage_changes {
887                        subscription.new_report_remaining_keys.insert(key.clone());
888                    }
889                }
890
891                subscription.current_best_block_index = Some(new_best_block_node_index);
892            }
893
894            // Remove from the state machine the blocks that have been finalized.
895            if let consensus_service::Notification::Finalized {
896                finalized_blocks_newest_to_oldest,
897                ..
898            } = notification
899            {
900                subscription.current_finalized_block_hash =
901                    *finalized_blocks_newest_to_oldest.first().unwrap();
902
903                for pruned_block in subscription.pinned_blocks.prune_ancestors(
904                    *subscription
905                        .pinned_blocks_by_hash
906                        .get(&subscription.current_finalized_block_hash)
907                        .unwrap(),
908                ) {
909                    let _was_in = subscription
910                        .pinned_blocks_by_hash
911                        .remove(&pruned_block.user_data);
912                    debug_assert_eq!(_was_in, Some(pruned_block.index));
913
914                    let mut after_split_off_point = subscription
915                        .pinned_blocks_storage_changes
916                        .split_off(&(pruned_block.index, Vec::new()));
917                    if let Some(index_plus_one) = pruned_block.index.inc() {
918                        let mut after_changes =
919                            after_split_off_point.split_off(&(index_plus_one, Vec::new()));
920                        subscription
921                            .pinned_blocks_storage_changes
922                            .append(&mut after_changes);
923                    }
924
925                    subscription.blocks_to_unpin.push(pruned_block.user_data);
926                }
927            }
928        }
929    }
930}
