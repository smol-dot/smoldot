18//! Jaeger integration.
19//!
20//! See <https://www.jaegertracing.io/> for an introduction.
21//!
22//! The easiest way to try Jaeger is:
23//!
24//! - Start a docker container with the all-in-one docker image (see below).
25//! - Run [`JaegerService`] with [`Config::jaeger_agent`] set to `127.0.0.1:6831`.
26//! - Open your browser and navigate to <http://localhost:16686> to access the UI.
27//!
28//! The all-in-one docker image can be started with:
29//!
30//! ```not_rust
31//! docker run -d --name jaeger -e COLLECTOR_ZIPKIN_HTTP_PORT=9411 -p 5775:5775/udp -p 6831:6831/udp -p 6832:6832/udp -p 5778:5778 -p 16686:16686 -p 14268:14268 -p 14250:14250 -p 9411:9411 jaegertracing/all-in-one:1
32//! ```
33//!

41/// Configuration for a [`JaegerService`].
42pub struct Config<'a> {
43    /// Closure that spawns background tasks.
44    pub tasks_executor: &'a mut dyn FnMut(Pin<Box<dyn Future<Output = ()> + Send>>),
45
46    /// Service name to report to the Jaeger agent.
47    pub service_name: String,
48
49    /// Address of the Jaeger agent to send traces to. Uses UDP.
50    ///
51    /// If this is `None`, the service will still be created but do nothing.
52    pub jaeger_agent: Option<SocketAddr>,
53}
54
55pub struct JaegerService {
56    traces_in: Arc<mick_jaeger::TracesIn>,
57
58    /// Notified when the service is destroyed.
59    shutdown_notify: event_listener::Event,
60}
61
62impl JaegerService {
63    pub async fn new(config: Config<'_>) -> Result<Arc<Self>, io::Error> {
64        let (traces_in, mut traces_out) = mick_jaeger::init(mick_jaeger::Config {
65            service_name: config.service_name,
66        });
67
68        let shutdown_notify = event_listener::Event::new();
69
70        if let Some(jaeger_agent) = config.jaeger_agent {
71            let udp_socket = UdpSocket::bind("0.0.0.0:0").await?;
72            let mut on_shutdown = shutdown_notify.listen();
73
74            // Spawn a background task that pulls span information and sends them on the network.
75            (config.tasks_executor)(Box::pin(async move {
76                loop {
77                    let Some(buf) = future::or(
78                        async {
79                            (&mut on_shutdown).await;
80                            None
81                        },
82                        async { Some(traces_out.next().await) },
83                    )
84                    .await
85                    else {
86                        break;
87                    };
88
89                    // UDP sending errors happen only either if the API is misused (in which case
90                    // panicking is desirable) or in case of missing priviledge, in which case a
91                    // panic is preferable in order to inform the user.
92                    udp_socket.send_to(&buf, jaeger_agent).await.unwrap();
93                }
94            }));
95        }
96
97        Ok(Arc::new(JaegerService {
98            traces_in,
99            shutdown_notify,
100        }))
101    }

103    pub fn block_announce_receive_span(
104        &self,
105        local_peer_id: &PeerId,
106        remote_peer_id: &PeerId,
107        block_number: u64,
108        block_hash: &[u8; 32],
109    ) -> mick_jaeger::Span {
110        let mut span =
111            self.net_connection_span(local_peer_id, remote_peer_id, "block-announce-received");
112        if let Ok(block_number) = i64::try_from(block_number) {
113            span.add_int_tag("number", block_number);
114        }
115        span.add_string_tag("hash", &hex::encode(block_hash));
116        span
117    }
118
119    pub fn block_announce_process_span(&self, block_hash: &[u8; 32]) -> mick_jaeger::Span {
120        self.block_span(block_hash, "block-announce-process")
121    }

140    // TODO: better return type
141    pub fn incoming_block_request_span(
142        &self,
143        local_peer_id: &PeerId,
144        remote_peer_id: &PeerId,
145        num_requested_blocks: u32,
146        block_hash: Option<&[u8; 32]>,
147    ) -> [Option<mick_jaeger::Span>; 2] {
148        let mut span1 =
149            self.net_connection_span(local_peer_id, remote_peer_id, "incoming-blocks-request");
150        span1.add_int_tag("num-blocks", num_requested_blocks.into());
151
152        let span2 = if let Some(block_hash) = block_hash {
153            let mut span = self.block_span(block_hash, "incoming-blocks-request");
154            let hex = hex::encode(block_hash);
155            span.add_string_tag("hash", &hex);
156            span1.add_string_tag("hash", &hex);
157            Some(span)
158        } else {
159            None
160        };
161
162        [Some(span1), span2]
163    }

190    /// Creates a new `Span` that refers to an event about a given block.
191    ///
192    /// This function is private so that only the code in the `jaeger_service` module decides
193    /// which names and labels to apply to spans. This makes it possible to easily ensure some
194    /// consistency in these names and labels.
195    fn block_span(
196        &self,
197        block_hash: &[u8; 32],
198        operation_name: impl Into<String>,
199    ) -> mick_jaeger::Span {
200        let trace_id = NonZero::<u128>::new(u128::from_be_bytes(
201            <[u8; 16]>::try_from(&block_hash[16..]).unwrap(),
202        ))
203        .unwrap_or_else(|| NonZero::<u128>::new(1u128).unwrap());
204        self.traces_in.span(trace_id, operation_name)
205    }
206
207    /// Creates a new `Span` that refers to a specific network connection between two nodes.
208    ///
209    /// This function is private so that only the code in the `jaeger_service` module decides
210    /// which names and labels to apply to spans. This makes it possible to easily ensure some
211    /// consistency in these names and labels.
212    fn net_connection_span(
213        &self,
214        local_peer_id: &PeerId,
215        remote_peer_id: &PeerId,
216        operation_name: impl Into<String>,
217    ) -> mick_jaeger::Span {
218        let local_peer_id = local_peer_id.as_bytes();
219        let remote_peer_id = remote_peer_id.as_bytes();
220
221        let mut buf = [0; 16];
222        if local_peer_id < remote_peer_id {
223            buf[..8].copy_from_slice(&local_peer_id[local_peer_id.len() - 8..]);
224            buf[8..].copy_from_slice(&remote_peer_id[remote_peer_id.len() - 8..]);
225        } else {
226            buf[..8].copy_from_slice(&remote_peer_id[remote_peer_id.len() - 8..]);
227            buf[8..].copy_from_slice(&local_peer_id[local_peer_id.len() - 8..]);
228        };
229
230        let trace_id = NonZero::<u128>::new(u128::from_be_bytes(buf)).unwrap();
231        self.traces_in.span(trace_id, operation_name)
232    }
233}

235impl Drop for JaegerService {
236    fn drop(&mut self) {
237        self.shutdown_notify.notify(usize::MAX);
238    }
