18//! Background network service.
19//!
20//! The [`NetworkService`] manages background tasks dedicated to connecting to other nodes.
21//! Importantly, its design is oriented towards the particular use case of the full node.
22//!
23//! The [`NetworkService`] spawns one background task (using the [`Config::tasks_executor`]) for
24//! each active TCP socket, plus one for each TCP listening socket. Messages are exchanged between
25//! the service and these background tasks.

355    /// Initializes the network service with the given configuration.
356    pub async fn new(
357        config: Config,
358    ) -> Result<
359        (
360            Arc<Self>,
361            Vec<ChainId>,
362            Vec<Pin<Box<dyn Stream<Item = Event> + Send>>>,
363        ),
364        InitError,
365    > {
366        let (event_senders, event_receivers): (Vec<_>, Vec<_>) = (0..config.num_events_receivers)
367            .map(|_| channel::bounded(16))
368            .unzip();
369
370        let mut network = service::ChainNetwork::new(service::Config {
371            chains_capacity: config.chains.len(),
372            connections_capacity: 100, // TODO: ?
373            handshake_timeout: Duration::from_secs(8),
374            randomness_seed: rand::random(),
375        });
376
377        let mut peering_strategy =
378            basic_peering_strategy::BasicPeeringStrategy::new(basic_peering_strategy::Config {
379                randomness_seed: rand::random(),
380                peers_capacity: 200, // TODO: ?
381                chains_capacity: config.chains.len(),
382            });
383
384        let mut chain_names =
385            hashbrown::HashMap::with_capacity_and_hasher(config.chains.len(), Default::default());
386
387        for chain in config.chains {
388            let chain_id = network
389                .add_chain(service::ChainConfig {
390                    fork_id: chain.fork_id.clone(),
391                    block_number_bytes: chain.block_number_bytes,
392                    best_hash: chain.best_block.1,
393                    best_number: chain.best_block.0,
394                    genesis_hash: chain.genesis_block_hash,
395                    role: codec::Role::Full,
396                    grandpa_protocol_config: chain.grandpa_protocol_finalized_block_height.map(
397                        // TODO: dummy values
398                        |commit_finalized_height| service::GrandpaState {
399                            commit_finalized_height,
400                            round_number: 1,
401                            set_id: 0,
402                        },
403                    ),
404                    allow_inbound_block_requests: true,
405                    user_data: Chain {
406                        log_name: chain.log_name.clone(),
407                        database: chain.database,
408                        max_in_peers: chain.max_in_peers,
409                        max_slots: chain.max_slots,
410                    },
411                })
412                .unwrap(); // TODO: don't unwrap?
413
414            for (peer_id, addr) in chain.bootstrap_nodes {
415                // Note that we must call this function before `insert_address`, as documented
416                // in `basic_peering_strategy`.
417                peering_strategy.insert_chain_peer(chain_id, peer_id.clone(), usize::MAX);
418                peering_strategy.insert_address(&peer_id, addr.into_bytes(), usize::MAX);
419            }
420
421            chain_names.insert(chain_id, chain.log_name);
422        }
423
424        let (to_background_tx, to_background_rx) = channel::bounded(16);
425        let (from_connections_tx, from_connections_rx) = channel::bounded(64);
426
427        let local_peer_id =
428            peer_id::PublicKey::Ed25519(*config.noise_key.libp2p_public_ed25519_key())
429                .into_peer_id();
430
431        // For each listening address in the configuration, create a background task dedicated to
432        // listening on that address.
433        let mut incoming_connections = SelectAll::new();
434        for listen_address in config.listen_addresses {
435            // Try to parse the requested address and create the corresponding listening socket.
436            let tcp_listener: smol::net::TcpListener = {
437                let addr = {
438                    let mut iter = listen_address.iter();
439                    let proto1 = iter.next();
440                    let proto2 = iter.next();
441                    let proto3 = iter.next();
442                    match (proto1, proto2, proto3) {
443                        (Some(Protocol::Ip4(ip)), Some(Protocol::Tcp(port)), None) => {
444                            Some(SocketAddr::from((ip, port)))
445                        }
446                        (Some(Protocol::Ip6(ip)), Some(Protocol::Tcp(port)), None) => {
447                            Some(SocketAddr::from((ip, port)))
448                        }
449                        _ => None,
450                    }
451                };
452
453                if let Some(addr) = addr {
454                    match smol::net::TcpListener::bind(addr).await {
455                        Ok(l) => l,
456                        Err(err) => {
457                            return Err(InitError::ListenerIo(listen_address, err));
458                        }
459                    }
460                } else {
461                    // TODO: support WebSocket server
462                    return Err(InitError::BadListenMultiaddr {
463                        address: listen_address,
464                    });
465                }
466            };
467
468            // Add a task dedicated to this listener.
469            let log_callback = config.log_callback.clone();
470            incoming_connections.push(Box::pin(stream::unfold(tcp_listener, move |tcp_listener| {
471                let log_callback = log_callback.clone();
472                async move {
473                    loop {
474                        match tcp_listener.accept().await {
475                            Ok((socket, socket_addr)) => {
476                                break Some(((socket, socket_addr), tcp_listener));
477                            }
478                            Err(error) => {
479                                // Errors here can happen if the accept failed, for example
480                                // if no file descriptor is available.
481                                // A wait is added in order to avoid having a busy-loop
482                                // failing to accept connections.
483                                log_callback.log(
484                                    LogLevel::Warn,
485                                    format!("tcp-accept-error; error={}", error),
486                                );
487                                smol::Timer::after(Duration::from_secs(2)).await;
488                            }
489                        }
490                    }
491                }
492            })) as Pin<Box<_>>);
493        }
494
495        // Initialize the inner network service.
496        run(Inner {
497            local_peer_id: local_peer_id.clone(),
498            identify_agent_version: config.identify_agent_version,
499            event_senders: either::Left(event_senders),
500            event_pending_send: None,
501            num_pending_out_attempts: 0,
502            to_background_rx: Box::pin(to_background_rx),
503            from_connections_rx: Box::pin(from_connections_rx),
504            from_connections_tx,
505            tasks_executor: config.tasks_executor,
506            log_callback: config.log_callback,
507            network,
508            noise_key: config.noise_key,
509            peering_strategy,
510            blocks_requests: hashbrown::HashMap::with_capacity_and_hasher(
511                50, // TODO: ?
512                Default::default(),
513            ),
514            warp_sync_requests: hashbrown::HashMap::with_capacity_and_hasher(
515                2, // TODO: ?
516                Default::default(),
517            ),
518            storage_requests: hashbrown::HashMap::with_capacity_and_hasher(
519                5, // TODO: ?
520                Default::default(),
521            ),
522            call_proof_requests: hashbrown::HashMap::with_capacity_and_hasher(
523                5, // TODO: ?
524                Default::default(),
525            ),
526            jaeger_service: config.jaeger_service.clone(),
527            next_discovery: smol::Timer::after(Duration::from_secs(1)),
528            next_discovery_period: Duration::from_secs(1),
529            incoming_connections,
530        });
531
532        // Build the final network service.
533        let network_service = Arc::new(NetworkService {
534            local_peer_id,
535            chain_names,
536            _jaeger_service: config.jaeger_service,
537            to_background_tx: Mutex::new(to_background_tx),
538        });
539
540        // Adjust the receivers to keep the `network_service` alive.
541        // TODO: no, hacky
542        let receivers = event_receivers
543            .into_iter()
544            .map(|rx| {
545                let mut network_service = Some(network_service.clone());
546                rx.chain(smol::stream::poll_fn(move |_| {
547                    drop(network_service.take());
548                    Poll::Ready(None)
549                }))
550                .boxed()
551            })
552            .collect();
553
554        let chain_ids = network_service.chain_names.keys().cloned().collect();
555        Ok((network_service, chain_ids, receivers))
556    }

648            .await
649            .send(ToBackground::ForegroundDisconnectAndBan {
650                peer_id,

652                severity,
653                reason,

839fn run(mut inner: Inner) {
840    // This function is a small hack because I didn't find a better way to store the executor
841    // within `Inner` while at the same time spawning the `Inner` using said executor.
842    let mut actual_executor = mem::replace(&mut inner.tasks_executor, Box::new(|_| unreachable!()));
843    let (tx, rx) = oneshot::channel();
844    actual_executor(Box::pin(async move {
845        let actual_executor = rx.await.unwrap();
846        inner.tasks_executor = actual_executor;
847        background_task(inner).await;
848    }));
849    tx.send(actual_executor).unwrap_or_else(|_| panic!());
850}
851
852async fn background_task(mut inner: Inner) {
853    loop {
854        enum WakeUpReason {
855            IncomingConnection {
856                socket: TcpStream,
857                socket_addr: SocketAddr,
858            },
859            NetworkEvent(service::Event<channel::Sender<service::CoordinatorToConnection>>),
860            Message(ToBackground),
861            ForegroundClosed,
862            FromConnectionTask {
863                connection_id: service::ConnectionId,
864                // TODO: this Option is weird
865                message: Option<service::ConnectionToCoordinator>,
866            },
867            EventSendersReady,
868            CanAssignSlot(PeerId, ChainId),
869            CanStartConnect(PeerId),
870            CanOpenGossip(PeerId, ChainId),
871            StartKademliaDiscoveries,
872            MessageToConnection {
873                connection_id: service::ConnectionId,
874                message: service::CoordinatorToConnection,
875            },
876        }
877
878        let wake_up_reason = async {
879            inner
880                .to_background_rx
881                .next()
882                .await
883                .map_or(WakeUpReason::ForegroundClosed, WakeUpReason::Message)
884        }
885        .or({
886            let event_senders_ready = matches!(inner.event_senders, either::Left(_));
887            let event_pending_send = &inner.event_pending_send;
888            let network = &mut inner.network;
889            let peering_strategy = &mut inner.peering_strategy;
890            let num_pending_out_attempts = &inner.num_pending_out_attempts;
891            async move {
892                if let Some(event) = (event_senders_ready && event_pending_send.is_none())
893                    .then(|| network.next_event())
894                    .flatten()
895                {
896                    WakeUpReason::NetworkEvent(event)
897                } else if let Some((connection_id, message)) = network.pull_message_to_connection()
898                {
899                    WakeUpReason::MessageToConnection {
900                        connection_id,
901                        message,
902                    }
903                } else if let Some((peer_id, chain_id)) = network
904                    .connected_unopened_gossip_desired()
905                    .next()
906                    .map(|(peer_id, chain_id, _)| (peer_id.clone(), chain_id))
907                {
908                    WakeUpReason::CanOpenGossip(peer_id, chain_id)
909                } else if let Some(peer_id) = (*num_pending_out_attempts < 16)
910                    .then(|| network.unconnected_desired().next().cloned())
911                    .flatten()
912                {
913                    WakeUpReason::CanStartConnect(peer_id)
914                } else {
915                    'search: loop {
916                        let mut earlier_unban = None;
917
918                        for chain_id in network.chains().collect::<Vec<_>>() {
919                            if network.gossip_desired_num(
920                                chain_id,
921                                service::GossipKind::ConsensusTransactions,
922                            ) >= network[chain_id].max_slots
923                            {
924                                continue;
925                            }
926
927                            match peering_strategy.pick_assignable_peer(&chain_id, &Instant::now())
928                            {
929                                basic_peering_strategy::AssignablePeer::Assignable(peer_id) => {
930                                    break 'search WakeUpReason::CanAssignSlot(
931                                        peer_id.clone(),
932                                        chain_id,
933                                    );
934                                }
935                                basic_peering_strategy::AssignablePeer::AllPeersBanned {
936                                    next_unban,
937                                } => {
938                                    if earlier_unban.as_ref().map_or(true, |b| b > next_unban) {
939                                        earlier_unban = Some(next_unban.clone());
940                                    }
941                                }
942                                basic_peering_strategy::AssignablePeer::NoPeer => continue,
943                            }
944                        }
945
946                        if let Some(earlier_unban) = earlier_unban {
947                            smol::Timer::at(earlier_unban).await;
948                        } else {
949                            future::pending::<()>().await;
950                        }
951                    }
952                }
953            }
954        })
955        .or(async {
956            if let either::Right(sending) = &mut inner.event_senders {
957                let event_senders = sending.await;
958                inner.event_senders = either::Left(event_senders);
959                WakeUpReason::EventSendersReady
960            } else if inner.event_pending_send.is_some() {
961                WakeUpReason::EventSendersReady
962            } else {
963                future::pending().await
964            }
965        })
966        .or(async {
967            (&mut inner.next_discovery).await;
968            inner.next_discovery = smol::Timer::after(inner.next_discovery_period);
969            inner.next_discovery_period =
970                cmp::min(inner.next_discovery_period * 2, Duration::from_secs(120));
971            WakeUpReason::StartKademliaDiscoveries
972        })
973        .or(async {
974            let (connection_id, message) = inner.from_connections_rx.next().await.unwrap();
975            WakeUpReason::FromConnectionTask {
976                connection_id,
977                message,
978            }
979        })
980        .or(async {
981            let Some((socket, socket_addr)) = inner.incoming_connections.next().await else {
982                future::pending().await
983            };
984            WakeUpReason::IncomingConnection {
985                socket,
986                socket_addr,
987            }
988        })
989        .await;
990
991        match wake_up_reason {
992            WakeUpReason::MessageToConnection {
993                connection_id,
994                message,
995            } => {
996                // Note that it is critical for the sending to not take too long here, in order to
997                // not block the process of the network service.
998                // In particular, if sending the message to the connection is blocked due to
999                // sending a message on the connection-to-coordinator channel, this will result
1000                // in a deadlock.
1001                // For this reason, the connection task is always ready to immediately accept a
1002                // message on the coordinator-to-connection channel.
1003                inner.network[connection_id].send(message).await.unwrap();
1004            }
1005
1006            WakeUpReason::FromConnectionTask {
1007                connection_id,
1008                message,
1009            } => {
1010                if let Some(message) = message {
1011                    inner
1012                        .network
1013                        .inject_connection_message(connection_id, message);
1014                }
1015            }
1016
1017            WakeUpReason::IncomingConnection {
1018                socket,
1019                socket_addr,
1020            } => {
1021                // The Nagle algorithm, implemented in the kernel, consists in buffering the
1022                // data to be sent out and waiting a bit before actually sending it out, in
1023                // order to potentially merge multiple writes in a row into one packet. In
1024                // the implementation below, it is guaranteed that the buffer in `WithBuffers`
1025                // is filled with as much data as possible before the operating system gets
1026                // involved. As such, we disable the Nagle algorithm, in order to avoid adding
1027                // an artificial delay to all sends.
1028                let _ = socket.set_nodelay(true);
1029
1030                let multiaddr = [
1031                    match socket_addr.ip() {
1032                        IpAddr::V4(ip) => Protocol::<&[u8]>::Ip4(ip.octets()),
1033                        IpAddr::V6(ip) => Protocol::Ip6(ip.octets()),
1034                    },
1035                    Protocol::Tcp(socket_addr.port()),
1036                ]
1037                .into_iter()
1038                .collect::<Multiaddr>();
1039
1040                inner.log_callback.log(
1041                    LogLevel::Debug,
1042                    format!("incoming-connection; multiaddr={}", multiaddr),
1043                );
1044
1045                let (tx, rx) = channel::bounded(16); // TODO: ?!
1046
1047                let (connection_id, connection_task) = inner.network.add_single_stream_connection(
1048                    Instant::now(),
1049                    service::SingleStreamHandshakeKind::MultistreamSelectNoiseYamux {
1050                        is_initiator: false,
1051                        noise_key: &inner.noise_key,
1052                    },
1053                    multiaddr.clone().into_bytes(),
1054                    None,
1055                    tx,
1056                );
1057
1058                (inner.tasks_executor)(Box::pin(tasks::connection_task(
1059                    inner.log_callback.clone(),
1060                    multiaddr.to_string(),
1061                    async move { Ok(socket) },
1062                    connection_id,
1063                    connection_task,
1064                    rx,
1065                    inner.from_connections_tx.clone(),
1066                )));
1067            }
1068
1069            WakeUpReason::StartKademliaDiscoveries => {
1070                for chain_id in inner.network.chains().collect::<Vec<_>>() {
1071                    let random_peer_id =
1072                        PeerId::from_public_key(&peer_id::PublicKey::Ed25519(rand::random()));
1073
1074                    // TODO: select target closest to the random peer instead
1075                    let target = inner
1076                        .network
1077                        .gossip_connected_peers(
1078                            chain_id,
1079                            service::GossipKind::ConsensusTransactions,
1080                        )
1081                        .next()
1082                        .cloned();
1083
1084                    if let Some(target) = target {
1085                        match inner.network.start_kademlia_find_node_request(
1086                            &target,
1087                            chain_id,
1088                            &random_peer_id,
1089                            Duration::from_secs(20),
1090                        ) {
1091                            Ok(_) => {}
1092                            Err(service::StartRequestError::NoConnection) => unreachable!(),
1093                        };
1094                    } else {
1095                        // TODO: log message
1096                    }
1097                }
1098            }
1099
1100            WakeUpReason::ForegroundClosed => {
1101                // TODO: do a clean shutdown of all the connections
1102                return;
1103            }
1104
1105            WakeUpReason::Message(ToBackground::ForegroundDisconnectAndBan {
1106                peer_id,
1107                chain_id,
1108                severity,
1109                reason,
1110            }) => {
1111                // Note that peer doesn't necessarily have an out slot.
1112                inner.peering_strategy.unassign_slot_and_ban(
1113                    &chain_id,
1114                    &peer_id,
1115                    Instant::now()
1116                        + Duration::from_secs(match severity {
1117                            BanSeverity::Low => 10,
1118                            BanSeverity::High => 40,
1119                        }),
1120                );
1121                if inner.network.gossip_remove_desired(
1122                    chain_id,
1123                    &peer_id,
1124                    service::GossipKind::ConsensusTransactions,
1125                ) {
1126                    inner.log_callback.log(
1127                        LogLevel::Debug,
1128                        format!(
1129                            "slot-unassigned; peer_id={}; chain={}; reason=user-ban; user-reason={}",
1130                            peer_id, inner.network[chain_id].log_name, reason
1131                        ),
1132                    );
1133                }
1134
1135                if inner.network.gossip_is_connected(
1136                    chain_id,
1137                    &peer_id,
1138                    service::GossipKind::ConsensusTransactions,
1139                ) {
1140                    let _close_result = inner.network.gossip_close(
1141                        chain_id,
1142                        &peer_id,
1143                        service::GossipKind::ConsensusTransactions,
1144                    );
1145                    debug_assert!(_close_result.is_ok());
1146
1147                    inner.log_callback.log(
1148                        LogLevel::Debug,
1149                        format!(
1150                            "chain-disconnected; peer_id={}; chain={}",
1151                            peer_id, inner.network[chain_id].log_name
1152                        ),
1153                    );
1154
1155                    debug_assert!(inner.event_pending_send.is_none());
1156                    inner.event_pending_send = Some(Event::Disconnected { chain_id, peer_id });
1157                }
1158            }
1159
1160            WakeUpReason::Message(ToBackground::ForegroundAnnounceBlock {
1161                target,
1162                chain_id,
1163                scale_encoded_header,
1164                is_best,
1165                result_tx,
1166            }) => {
1167                let _ = result_tx.send(inner.network.gossip_send_block_announce(
1168                    &target,
1169                    chain_id,
1170                    &scale_encoded_header,
1171                    is_best,
1172                ));
1173            }
1174            WakeUpReason::Message(ToBackground::ForegroundSetLocalBestBlock {
1175                chain_id,
1176                best_hash,
1177                best_number,
1178            }) => {
1179                inner
1180                    .network
1181                    .set_chain_local_best_block(chain_id, best_hash, best_number);
1182            }
1183            WakeUpReason::Message(ToBackground::ForegroundBlocksRequest {
1184                target,
1185                chain_id,
1186                config,
1187                result_tx,
1188            }) => {
1189                inner.log_callback.log(
1190                    LogLevel::Debug,
1191                    format!(
1192                        "blocks-request-start; peer_id={}; chain={}; start={}; desired_count={}; direction={}",
1193                        target,
1194                        inner.network[chain_id].log_name,
1195                        match &config.start {
1196                            codec::BlocksRequestConfigStart::Hash(h) => either::Left(HashDisplay(h)),
1197                            codec::BlocksRequestConfigStart::Number(n) => either::Right(n),
1198                        },
1199                        config.desired_count,
1200                        match config.direction {
1201                            codec::BlocksRequestDirection::Ascending => "ascending",
1202                            codec::BlocksRequestDirection::Descending => "descending",
1203                        },
1204                    ),
1205                );
1206
1207                match inner.network.start_blocks_request(
1208                    &target,
1209                    chain_id,
1210                    config,
1211                    Duration::from_secs(12),
1212                ) {
1213                    Ok(request_id) => {
1214                        // TODO: somehow cancel the request if the `rx` is dropped?
1215                        inner.blocks_requests.insert(request_id, result_tx);
1216                    }
1217                    Err(service::StartRequestError::NoConnection) => {
1218                        inner.log_callback.log(
1219                            LogLevel::Debug,
1220                            format!(
1221                                "blocks-request-ended; peer_id={}; chain={}; outcome=failure; error=no-connection",

1223                                inner.network[chain_id].log_name,
1224                            ),
1225                        );
1226                        let _ = result_tx.send(Err(BlocksRequestError::NoConnection));

1228                }
1229            }

1231                target,
1232                chain_id,
1233                begin_hash,
1234                result_tx,
1235            }) => {
1236                inner.log_callback.log(
1237                    LogLevel::Debug,
1238                    format!(

2266    database: &database_thread::DatabaseThread,

2268    config: codec::BlocksRequestConfig,
2269) -> Result<Vec<codec::BlockData>, full_sqlite::CorruptedError> {
2270    database
2271        .with_database(move |database| {
2272            let num_blocks = cmp::min(
2273                usize::try_from(config.desired_count.get()).unwrap_or(usize::MAX),
2274                128,
2275            );
2276
2277            let mut output = Vec::with_capacity(num_blocks);
2278            let mut next_block = config.start;
2279
2280            loop {

2290                            Some(h) => h,
2291                            None => break,
2292                        }

2300

2306                            codec::BlocksRequestConfigStart::Number(decoded.number + 1)
