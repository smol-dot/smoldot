18//! As explained in the documentation of smoldot, the database uses synchronous I/O operations.
19//! For this reason, it is undesirable to access it from an asynchronous context.
20
21use futures_channel::oneshot;
22use smol::{channel, lock::Mutex, stream::StreamExt as _};
23use smoldot::database::full_sqlite::SqliteFullDatabase;
24use std::{pin::pin, thread};
25
26pub use smoldot::database::full_sqlite::StorageAccessError;
27
28/// Handle to the thread were the database accesses are performed.
29///
30/// Destroying this object stops the thread.
31///
32/// Use the `From` trait implementation to build a [`DatabaseThread`].
33pub struct DatabaseThread {
34    sender: Mutex<channel::Sender<Exec>>,
35}
36
37type Exec = Box<dyn FnOnce(&SqliteFullDatabase) + Send>;
38
39impl DatabaseThread {
40    /// Sends a closure to the database thread, executes it, then returns the value that the
41    /// closure returned.
42    pub async fn with_database<T: Send + 'static>(
43        &self,
44        closure: impl FnOnce(&SqliteFullDatabase) -> T + Send + 'static,
45    ) -> T {
46        let (tx, rx) = oneshot::channel();
47        self.sender
48            .lock()
49            .await
50            .send(Box::new(move |db| {
51                let _ = tx.send(closure(db));
52            }))
53            .await
54            .unwrap();
55        rx.await.unwrap()
56    }
57
58    /// Similar to [`DatabaseThread::with_database`], but without any return value. This function
59    /// is slightly more optimized for this use case.
60    pub async fn with_database_detached(
61        &self,
62        closure: impl FnOnce(&SqliteFullDatabase) + Send + 'static,
63    ) {
64        self.sender
65            .lock()
66            .await
67            .send(Box::new(move |db| {
68                closure(db);
69            }))
70            .await
71            .unwrap();
72    }
73}
74
75impl From<SqliteFullDatabase> for DatabaseThread {
76    fn from(db: SqliteFullDatabase) -> DatabaseThread {
77        let (sender, rx) = channel::bounded::<Box<dyn FnOnce(&SqliteFullDatabase) + Send>>(256);
78
79        thread::Builder::new()
80            .name("sqlite-database".into())
81            .spawn(move || {
82                // When the `DatabaseThread` is dropped, the sender will close, `rx.next()`
83                // will return `None`, and the closure here will finish, ending the thread.
84                let mut rx = pin!(rx);
85                while let Some(closure) = smol::block_on(rx.next()) {
86                    closure(&db)
87                }
88            })
89            .unwrap();
90
91        DatabaseThread {
92            sender: Mutex::new(sender),
93        }
94    }
95}
