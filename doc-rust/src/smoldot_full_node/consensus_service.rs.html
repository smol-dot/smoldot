91    /// Receiver for events coming from the network, as returned by
92    /// [`network_service::NetworkService::new`].

94
95    /// Service to use to report traces.
96    pub jaeger_service: Arc<jaeger_service::JaegerService>,
97
98    /// A node has the authorization to author a block during a slot.
99    ///
100    /// In order for the network to perform well, a block should be authored and propagated
101    /// throughout the peer-to-peer network before the end of the slot. In order for this to
102    /// happen, the block creation process itself should end a few seconds before the end of the
103    /// slot. This threshold after which the block creation should end is determined by this value.
104    ///
105    /// The moment in the slot when the authoring ends is determined by
106    /// `slot_duration * slot_duration_author_ratio / u16::MAX`.
107    /// For example, passing `u16::MAX` means that the entire slot is used. Passing
108    /// `u16::MAX / 2` means that half of the slot is used.
109    ///
110    /// A typical value is `43691_u16`, representing 2/3 of a slot.
111    ///
112    /// Note that this value doesn't determine the moment when creating the block has ended, but
113    /// the moment when creating the block should start its final phase.
114    pub slot_duration_author_ratio: u16,
115}

160/// Potential error when calling [`ConsensusService::new`].
161#[derive(Debug, derive_more::Display, derive_more::Error)]
162pub enum InitError {
163    /// Database is corrupted.
164    DatabaseCorruption(full_sqlite::CorruptedError),
165    /// Error parsing the header of a block in the database.
166    InvalidHeader(header::Error),
167    /// `:code` key is missing from the finalized block storage.
168    FinalizedCodeMissing,
169    /// Error parsing the `:heappages` of the finalized block.
170    FinalizedHeapPagesInvalid(executor::InvalidHeapPagesError),
171    /// Error initializing the runtime of the finalized block.
172    FinalizedRuntimeInit(executor::host::NewErr),
173}

176    /// Initializes the [`ConsensusService`] with the given configuration.
177    pub async fn new(mut config: Config) -> Result<Arc<Self>, InitError> {
178        // Perform the initial access to the database to load a bunch of information.
179        let (
180            finalized_block_number,
181            finalized_heap_pages,
182            finalized_code,
183            best_block_hash,
184            best_block_number,
185            finalized_chain_information,
186        ) = config
187            .database
188            .with_database({
189                let block_number_bytes = config.block_number_bytes;
190                move |database| {
191                    // If the previous run of the full node crashed, the database will contain
192                    // blocks that are no longer useful in any way. We purge them all here.
193                    database
194                        .purge_finality_orphans()
195                        .map_err(InitError::DatabaseCorruption)?;
196
197                    let finalized_block_hash = database
198                        .finalized_block_hash()
199                        .map_err(InitError::DatabaseCorruption)?;
200                    let finalized_block_number = header::decode(
201                        &database
202                            .block_scale_encoded_header(&finalized_block_hash)
203                            .map_err(InitError::DatabaseCorruption)?
204                            .unwrap(), // A panic here would indicate a bug in the database code.
205                        block_number_bytes,
206                    )
207                    .map_err(InitError::InvalidHeader)?
208                    .number;
209                    let best_block_hash = database.best_block_hash().unwrap();
210                    let best_block_number = header::decode(
211                        &database
212                            .block_scale_encoded_header(&best_block_hash)
213                            .map_err(InitError::DatabaseCorruption)?
214                            .unwrap(), // A panic here would indicate a bug in the database code.
215                        block_number_bytes,
216                    )
217                    .map_err(InitError::InvalidHeader)?
218                    .number;
219                    let finalized_chain_information =
220                        match database.to_chain_information(&finalized_block_hash) {
221                            Ok(info) => info,
222                            Err(full_sqlite::StorageAccessError::Corrupted(err)) => {
223                                return Err(InitError::DatabaseCorruption(err));
224                            }
225                            Err(full_sqlite::StorageAccessError::IncompleteStorage)
226                            | Err(full_sqlite::StorageAccessError::UnknownBlock) => unreachable!(),
227                        };
228                    let finalized_code = match database.block_storage_get(
229                        &finalized_block_hash,
230                        iter::empty::<iter::Empty<_>>(),
231                        trie::bytes_to_nibbles(b":code".iter().copied()).map(u8::from),
232                    ) {
233                        Ok(Some((code, _))) => code,
234                        Ok(None) => return Err(InitError::FinalizedCodeMissing),
235                        Err(full_sqlite::StorageAccessError::Corrupted(err)) => {
236                            return Err(InitError::DatabaseCorruption(err));
237                        }
238                        Err(full_sqlite::StorageAccessError::IncompleteStorage)
239                        | Err(full_sqlite::StorageAccessError::UnknownBlock) => unreachable!(),
240                    };
241                    let finalized_heap_pages = match database.block_storage_get(
242                        &finalized_block_hash,
243                        iter::empty::<iter::Empty<_>>(),
244                        trie::bytes_to_nibbles(b":heappages".iter().copied()).map(u8::from),
245                    ) {
246                        Ok(Some((hp, _))) => Some(hp),
247                        Ok(None) => None,
248                        Err(full_sqlite::StorageAccessError::Corrupted(err)) => {
249                            return Err(InitError::DatabaseCorruption(err));
250                        }
251                        Err(full_sqlite::StorageAccessError::IncompleteStorage)
252                        | Err(full_sqlite::StorageAccessError::UnknownBlock) => unreachable!(),
253                    };
254                    Ok((
255                        finalized_block_number,
256                        finalized_heap_pages,
257                        finalized_code,
258                        best_block_hash,
259                        best_block_number,
260                        finalized_chain_information,
261                    ))
262                }
263            })
264            .await?;
265
266        // The Kusama chain contains a fork hardcoded in the official Polkadot client.
267        // See <https://github.com/paritytech/polkadot/blob/93f45f996a3d5592a57eba02f91f2fc2bc5a07cf/node/service/src/grandpa_support.rs#L111-L216>
268        // Because we don't want to support this in smoldot, a warning is printed instead if we
269        // recognize Kusama.
270        // See also <https://github.com/paritytech/smoldot/issues/1866>.
271        if config.genesis_block_hash
272            == [
273                176, 168, 212, 147, 40, 92, 45, 247, 50, 144, 223, 183, 230, 31, 135, 15, 23, 180,
274                24, 1, 25, 122, 20, 156, 169, 54, 84, 73, 158, 163, 218, 254,
275            ]
276            && finalized_block_number <= 1500988
277        {
278            config.log_callback.log(
279                LogLevel::Warn,
280                "The Kusama chain is known to be borked at block #1491596. The official Polkadot \
281                client works around this issue by hardcoding a fork in its source code. Smoldot \
282                does not support this hardcoded fork and will thus fail to sync past this block."
283                    .to_string(),
284            );
285        }
286
287        let mut sync = all::AllSync::new(all::Config {
288            chain_information: finalized_chain_information,
289            block_number_bytes: config.block_number_bytes,
290            allow_unknown_consensus_engines: false,
291            sources_capacity: 32,
292            blocks_capacity: {
293                // This is the maximum number of blocks between two consecutive justifications.
294                1024
295            },
296            max_disjoint_headers: 1024,
297            max_requests_per_block: NonZero::<u32>::new(3).unwrap(),
298            download_ahead_blocks: {
299                // Assuming a verification speed of 1k blocks/sec and a 99th download time
300                // percentile of two second, the number of blocks to download ahead of time
301                // in order to not block is 2000.
302                // In practice, however, the verification speed and download speed depend on
303                // the chain and the machine of the user.
304                NonZero::<u32>::new(2000).unwrap()
305            },
306            download_bodies: true,
307            // We ask for all the chain-information-related storage proofs and call proofs to be
308            // downloaded during the warp syncing in order to guarantee that the necessary
309            // information will be found in the database at the next reload.
310            download_all_chain_information_storage_proofs: true,
311            code_trie_node_hint: None,
312        });
313
314        let finalized_runtime = {
315            // Builds the runtime of the finalized block.
316            // Assumed to always be valid, otherwise the block wouldn't have been
317            // saved in the database, hence the large number of unwraps here.
318            let heap_pages = executor::storage_heap_pages_to_value(finalized_heap_pages.as_deref())
319                .map_err(InitError::FinalizedHeapPagesInvalid)?;
320            executor::host::HostVmPrototype::new(executor::host::Config {
321                module: finalized_code,
322                heap_pages,
323                exec_hint: executor::vm::ExecHint::ValidateAndCompile, // TODO: probably should be decided by the optimisticsync
324                allow_unresolved_imports: false,
325            })
326            .map_err(InitError::FinalizedRuntimeInit)?
327        };
328
329        let block_author_sync_source = sync
330            .prepare_add_source(best_block_number, best_block_hash)
331            .add_source(None, NonFinalizedBlock::NotVerified);
332
333        let (to_background_tx, to_background_rx) = mpsc::channel(4);
334
335        let background_sync = SyncBackground {
336            sync,
337            block_author_sync_source,
338            block_authoring: None,
339            authored_block: None,
340            slot_duration_author_ratio: config.slot_duration_author_ratio,
341            keystore: config.keystore,
342            finalized_runtime: Arc::new(finalized_runtime),
343            network_service: config.network_service.0,
344            network_chain_id: config.network_service.1,
345            network_local_chain_update_needed: true,
346            pending_block_announce: None,
347            to_background_rx,
348            blocks_notifications: Vec::with_capacity(8),
349            pending_notification: None,
350            from_network_service: config.network_events_receiver,
351            database: config.database,
352            database_catch_up_download: DatabaseCatchUpDownload::NoDownloadInProgress,
353            database_catch_up_download_block_verification:
354                DatabaseCatchUpDownloadBlockVerification::None,
355            peers_source_id_map: Default::default(),
356            sub_tasks: FuturesUnordered::new(),
357            log_callback: config.log_callback,
358            jaeger_service: config.jaeger_service,
359        };
360
361        (config.tasks_executor)(Box::pin(background_sync.run()));
362
363        Ok(Arc::new(ConsensusService {
364            block_number_bytes: config.block_number_bytes,
365            to_background_tx: Mutex::new(to_background_tx),
366        }))
367    }

380        let _ = self

389    /// Subscribes to the state of the chain: the current state and the new blocks.
390    ///
391    /// Only up to `buffer_size` notifications are buffered in the channel. If the channel is full
392    /// when a new notification is attempted to be pushed, the channel gets closed.
393    ///
394    /// A maximum number of finalized or non-canonical (i.e. not part of the finalized chain)
395    /// pinned blocks must be passed, indicating the maximum number of blocks that are finalized
396    /// or non-canonical that the consensus service will pin at the same time for this
397    /// subscription. If this maximum is reached, the channel will get closed. In situations
398    /// where the subscriber is guaranteed to always properly unpin blocks, a value of
399    /// `usize::MAX` can be passed in order to ignore this maximum.
400    ///
401    /// All the blocks being reported are guaranteed to be present in the database associated to
402    /// this [`ConsensusService`].
403    ///
404    /// See [`SubscribeAll`] for information about the return value.
405    ///
406    /// While this function is asynchronous, it is guaranteed to finish relatively quickly. Only
407    /// CPU operations are performed.
408    pub async fn subscribe_all(
409        &self,
410        buffer_size: usize,
411        max_finalized_pinned_blocks: NonZero<usize>,
412    ) -> SubscribeAll {
413        let (result_tx, result_rx) = oneshot::channel();
414        let _ = self
415            .to_background_tx
416            .lock()
417            .await
418            .send(ToBackground::SubscribeAll {
419                buffer_size,
420                _max_finalized_pinned_blocks: max_finalized_pinned_blocks,
421                result_tx,
422            })
423            .await;
424        result_rx.await.unwrap()
425    }
426
427    /// Unpins a block that was reported as part of a subscription.
428    ///
429    /// Has no effect if the [`SubscriptionId`] is not or no longer valid (as the consensus service
430    /// can kill any subscription at any moment).
431    ///
432    /// # Panic
433    ///
434    /// Panics if the block hash has not been reported or has already been unpinned.
435    ///
436    pub async fn unpin_block(&self, subscription_id: SubscriptionId, block_hash: [u8; 32]) {
437        let (result_tx, result_rx) = oneshot::channel();
438        let _ = self
439            .to_background_tx
440            .lock()
441            .await
442            .send(ToBackground::Unpin {
443                _subscription_id: subscription_id,
444                _block_hash: block_hash,
445                result_tx,
446            })
447            .await;
448        result_rx.await.unwrap()
449    }
450
451    /// Returns `true` if the syncing is currently downloading blocks at a high rate in order to
452    /// catch up with the head of the chain.
453    ///
454    /// > **Note**: This function is used to implement the `system_health` JSON-RPC function and
455    /// >           is basically a hack that shouldn't be relied upon.
456    pub async fn is_major_syncing_hint(&self) -> bool {
457        let (result_tx, result_rx) = oneshot::channel();
458        let _ = self
459            .to_background_tx
460            .lock()
461            .await
462            .send(ToBackground::IsMajorSyncingHint { result_tx })
463            .await;
464        result_rx.await.unwrap()
465    }
466}

499/// Notification about a new block or a new finalized block.
500///
501/// See [`ConsensusService::subscribe_all`].
502#[derive(Debug, Clone)]
503pub enum Notification {
504    /// A non-finalized block has been finalized.
505    Finalized {
506        /// BLAKE2 hash of the blocks that have been finalized, in deceasing block number. In
507        /// other words, each block in this list is the parent of the previous one. The first block
508        /// in this list is the new finalized block.
509        ///
510        /// A block with this hash is guaranteed to have earlier been reported in a
511        /// [`BlockNotification`], either in [`SubscribeAll::non_finalized_blocks_ancestry_order`]
512        /// or in a [`Notification::Block`].
513        finalized_blocks_newest_to_oldest: Vec<[u8; 32]>,
514
515        /// Hash of the best block after the finalization.
516        ///
517        /// If the newly-finalized block is an ancestor of the current best block, then this field
518        /// contains the hash of this current best block. Otherwise, the best block is now
519        /// the non-finalized block with the given hash.
520        ///
521        /// A block with this hash is guaranteed to have earlier been reported in a
522        /// [`BlockNotification`], either in [`SubscribeAll::non_finalized_blocks_ancestry_order`]
523        /// or in a [`Notification::Block`].
524        best_block_hash: [u8; 32],
525
526        /// List of BLAKE2 hashes of blocks that are no longer part of the canonical chain. In
527        /// unspecified order.
528        pruned_blocks_hashes: Vec<[u8; 32]>,
529    },
530
531    /// A new block has been added to the list of unfinalized blocks.
532    Block {
533        /// Information about the block.
534        block: BlockNotification,
535
536        /// Changes to the storage that the block has performed.
537        ///
538        /// Note that this field is only available when a new block is available.
539        storage_changes: Arc<runtime_call::StorageChanges>,
540    },
541}

543/// Notification about a new block.
544///
545/// See [`ConsensusService::subscribe_all`].
546#[derive(Debug, Clone)]
547pub struct BlockNotification {
548    /// True if this block is considered as the best block of the chain.
549    pub is_new_best: bool,
550
551    /// SCALE-encoded header of the block.
552    pub scale_encoded_header: Vec<u8>,
553
554    /// Hash of the block, to provide to [`ConsensusService::unpin_block`].
555    pub block_hash: [u8; 32],
556
557    /// If the block has a different runtime compared to its parent, contains the new runtime.
558    /// Contains `None` if the runtime of the block is the same as its parent's.
559    pub runtime_update: Option<Arc<executor::host::HostVmPrototype>>,
560
561    /// BLAKE2 hash of the header of the parent of this block.
562    ///
563    ///
564    /// A block with this hash is guaranteed to have earlier been reported in a
565    /// [`BlockNotification`], either in [`SubscribeAll::non_finalized_blocks_ancestry_order`] or
566    /// in a [`Notification::Block`].
567    ///
568    /// > **Note**: The header of a block contains the hash of its parent. When it comes to
569    /// >           consensus algorithms such as Babe or Aura, the syncing code verifies that this
570    /// >           hash, stored in the header, actually corresponds to a valid block. However,
571    /// >           when it comes to parachain consensus, no such verification is performed.
572    /// >           Contrary to the hash stored in the header, the value of this field is
573    /// >           guaranteed to refer to a block that is known by the syncing service. This
574    /// >           allows a subscriber of the state of the chain to precisely track the hierarchy
575    /// >           of blocks, without risking to run into a problem in case of a block with an
576    /// >           invalid header.
577    pub parent_hash: [u8; 32],
578}

580struct SyncBackground {
581    /// State machine containing the list of all the peers, all the non-finalized blocks, and all
582    /// the network requests in progress.
583    ///
584    /// Each peer holds a struct containing either information about a networking peer, or `None`
585    /// if this is the "special source" representing the local block authoring. Only one source
586    /// must contain `None` and its id must be [`SyncBackground::block_author_sync_source`].
587    ///
588    /// Each block holds its runtime if it has been verified.
589    ///
590    /// Some of the sources can represent networking peers that have already been disconnected. If
591    /// that is the case, no new request is started against these sources but existing requests
592    /// are allowed to finish.
593    /// This "trick" is necessary in order to not cancel requests that have already been started
594    /// against a peer when it disconnects and that might already have a response.
595    ///
596    /// Each on-going request has a corresponding background task in [`SyncBackground::sub_tasks`].
597    sync: all::AllSync<(), Option<NetworkSourceInfo>, NonFinalizedBlock>,
598
599    /// Source within the [`SyncBackground::sync`] to use to import locally-authored blocks.
600    block_author_sync_source: all::SourceId,
601
602    /// State of the authoring. If `None`, the builder should be (re)created. If `Some`, also
603    /// contains the list of public keys that were loaded from the keystore when creating the
604    /// builder.
605    ///
606    /// The difference between a value of `None` and a value of `Some(Builder::Idle)` is that
607    /// `None` indicates that we should try to author a block as soon as possible, while `Idle`
608    /// means that we shouldn't try again until some event occurs (at which point this field is

2429        // TODO: print logs
2430        let new_block_hash = header::hash_from_scale_encoded_header(&new_block_header);
2431        self.log_callback.log(
2432            LogLevel::Info,
2433            format!(
2434                "block-generated; hash={}; body_len={}",
2435                HashDisplay(&new_block_hash),
2436                new_block_body.len()
2437            ),
2438        );
2439        let _jaeger_span = self
2440            .jaeger_service
2441            .block_authorship_span(&new_block_hash, block_author_jaeger_start_time);
2442
2443        // Print a warning if generating the block has taken more time than expected.
2444        // This can happen because the node is completely overloaded, is running on a slow machine,
2445        // or if the runtime code being executed contains a very heavy operation.
2446        // In any case, there is not much that a node operator can do except try increase the
2447        // performance of their machine.
2448        match authoring_end.elapsed() {
2449            Ok(now_minus_end) if now_minus_end < Duration::from_millis(500) => {}
2450            _ => {
2451                self.log_callback.log(
2452                    LogLevel::Warn,
2453                    format!(
2454                        "block-generation-too-long; hash={}",
2455                        HashDisplay(&new_block_hash)
2456                    ),
2457                );
2458            }
2459        }
2460
2461        // Switch the block authoring to a state where we won't try to generate a new block again
2462        // until something new happens.
2463        // TODO: nothing prevents the node from generating two blocks at the same height at the moment
2464        self.block_authoring = Some((author::build::Builder::Idle, Vec::new()));
2465
2466        // The next step is to import the block in `self.sync`. This is done by pretending that
2467        // the local node is a source of block similar to networking peers.
2468        let all::BlockAnnounceOutcome::Unknown(block_insert) = self.sync.block_announce(
2469            self.block_author_sync_source,
2470            new_block_header.clone(),
2471            true, // Since the new block is a child of the current best block, it always becomes the new best.
2472        ) else {
2473            unreachable!();
2474        };
2475        block_insert.insert_and_update_source(NonFinalizedBlock::NotVerified);
2476
2477        debug_assert!(self.authored_block.is_none());
2478        self.authored_block = Some((
2479            parent_number + 1,
2480            new_block_hash,
2481            new_block_header,
2482            new_block_body,
2483        ));
2484    }

2949                            ),
2950                        );
2951                        self.sync = sync_out;
2952                        (self, true)
2953                    }
2954                }
2955            }

2960/// Executes the given block. On success, inserts it and its storage into the database.
2961// TODO: use a config struct for the parameters?
2962pub async fn execute_block_and_insert(
2963    database: &database_thread::DatabaseThread,
2964    mut parent_runtime: host::HostVmPrototype,
2965    parent_block_hash: &[u8; 32],
2966    block_header: &[u8],
2967    block_number_bytes: usize,
2968    block_body: impl ExactSizeIterator<Item = impl AsRef<[u8]> + Clone> + Clone,
2969    now_from_unix_epoch: Duration,
2970    is_new_best: bool,
2971) -> Result<ExecuteBlockSuccess, ExecuteBlockError> {
2972    let mut database_accesses_duration = Duration::new(0, 0);
2973    let mut runtime_build_duration = Duration::new(0, 0);
2974
2975    let mut storage_changes = runtime_call::StorageChanges::empty();
2976    let mut state_trie_version = runtime_call::TrieEntryVersion::V0; // TODO: shouldn't have to be initialized
2977    for (call_function, call_parameter) in [
2978        (
2979            body_only::CHECK_INHERENTS_FUNCTION_NAME,
2980            body_only::check_inherents_parameter(
2981                block_header,
2982                block_number_bytes,
2983                block_body.clone(),
2984                now_from_unix_epoch,
2985            )
2986            .unwrap()
2987            .fold(Vec::new(), |mut a, b| {
2988                a.extend_from_slice(b.as_ref());
2989                a
2990            }),
2991        ),
2992        (
2993            body_only::EXECUTE_BLOCK_FUNCTION_NAME,
2994            body_only::execute_block_parameter(
2995                block_header,
2996                block_number_bytes,
2997                block_body.clone(),
2998            )
2999            .unwrap()
3000            .fold(Vec::new(), |mut a, b| {
3001                a.extend_from_slice(b.as_ref());
3002                a
3003            }),
3004        ),
3005    ] {
3006        match runtime_call(
3007            database,
3008            parent_block_hash,
3009            parent_runtime,
3010            call_function,
3011            &call_parameter,
3012            runtime_call::StorageProofSizeBehavior::Unimplemented,
3013            storage_changes,
3014        )
3015        .await
3016        {
3017            Ok(success) => {
3018                // TODO: a bit crappy to compare function names, however I got insanely weird borrowck errors if putting a fn() in the for loop above
3019                let error = if call_function == body_only::CHECK_INHERENTS_FUNCTION_NAME {
3020                    body_only::check_check_inherents_output(&success.output)
3021                        .map_err(ExecuteBlockInvalidBlockError::CheckInherentsOutputError)
3022                        .err()
3023                } else {
3024                    body_only::check_execute_block_output(&success.output)
3025                        .map_err(ExecuteBlockInvalidBlockError::ExecuteBlockOutputError)
3026                        .err()
3027                };
3028                if let Some(error) = error {
3029                    return Err(ExecuteBlockError::InvalidBlock(error));
3030                }
3031
3032                parent_runtime = success.runtime;
3033                storage_changes = success.storage_changes;
3034                state_trie_version = success.state_trie_version;
3035                database_accesses_duration += success.database_accesses_duration;
3036            }
3037            Err(RuntimeCallError::RuntimeStartError(error)) => {
3038                return Err(ExecuteBlockError::VerificationFailure(
3039                    ExecuteBlockVerificationFailureError::RuntimeStartError(error),
3040                ));
3041            }
3042            Err(RuntimeCallError::RuntimeExecutionError(error)) => {
3043                return Err(ExecuteBlockError::InvalidBlock(
3044                    ExecuteBlockInvalidBlockError::RuntimeExecutionError(error),
3045                ));
3046            }

3188                        };

3191                        let references_merkle_value = key
3192                            .iter()

3210                                },
3211                                runtime_call::TrieChangeStorageValue::Modified {
3212                                    new_value: None,
3213                                } => full_sqlite::InsertTrieNodeStorageValue::NoValue,
3214                                runtime_call::TrieChangeStorageValue::Unmodified => {
3215                                    // TODO: overhead, and no child trie support
3216                                    if let Some((value_in_parent, _)) = database
3217                                        .block_storage_get(
3218                                            &parent_block_hash,
3219                                            iter::empty::<iter::Empty<_>>(),
3220                                            key.iter().map(|n| u8::from(*n)),
3221                                        )
3222                                        .unwrap()
3223                                    {
3224                                        full_sqlite::InsertTrieNodeStorageValue::Value {
3225                                            value: Cow::Owned(value_in_parent),
3226                                            references_merkle_value,
3227                                        }
3228                                    } else {
3229                                        full_sqlite::InsertTrieNodeStorageValue::NoValue
3230                                    }
3231                                }
3232                            },
3233                            partial_key_nibbles: partial_key
3234                                .iter()
3235                                .map(|n| u8::from(*n))

3237                                .into(),

3239                    })
3240                    .collect::<Vec<_>>();

3243                    .insert_trie_nodes(trie_nodes.into_iter(), u8::from(state_trie_version))

3246        })
3247        .await;

3252        new_runtime,
3253        storage_changes,
3254        database_accesses_duration,
3255        runtime_build_duration,
3256    })

3356        function_to_call,
3357        parameter: iter::once(&parameter),

3405                            key.iter().copied(),
3406                        )

3413                            .map_err(|_| RuntimeCallError::DatabaseInvalidStateTrieVersion)?,
