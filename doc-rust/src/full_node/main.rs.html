30fn main() {
31    smol::block_on(async_main())
32}

34async fn async_main() {
35    match <cli::CliOptions as clap::Parser>::parse().command {
36        cli::CliOptionsCommand::Run(r) => run(*r).await,
37        cli::CliOptionsCommand::Blake264BitsHash(opt) => {
38            let hash = blake2_rfc::blake2b::blake2b(8, &[], opt.payload.as_bytes());
39            println!("0x{}", hex::encode(hash));
40        }
41        cli::CliOptionsCommand::Blake2256BitsHash(opt) => {
42            let content = fs::read(opt.file).expect("Failed to read file content");
43            let hash = blake2_rfc::blake2b::blake2b(32, &[], &content);
44            println!("0x{}", hex::encode(hash));
45        }
46    }
47}

49async fn run(cli_options: cli::CliOptionsRun) {
50    // Determine the actual CLI output by replacing `Auto` with the actual value.
51    let cli_output = if let cli::Output::Auto = cli_options.output {
52        if io::IsTerminal::is_terminal(&io::stderr()) && cli_options.log_level.is_none() {
53            cli::Output::Informant
54        } else {
55            cli::Output::Logs
56        }
57    } else {
58        cli_options.output
59    };
60    debug_assert!(!matches!(cli_output, cli::Output::Auto));
61
62    // Setup the logging system of the binary.
63    let log_callback: Arc<dyn smoldot_full_node::LogCallback + Send + Sync> = match cli_output {
64        cli::Output::None => Arc::new(|_level, _message| {}),
65        cli::Output::Informant | cli::Output::Logs => {
66            let color_choice = cli_options.color.clone();
67            let log_level = cli_options.log_level.clone().unwrap_or(
68                if matches!(cli_output, cli::Output::Informant) {
69                    cli::LogLevel::Info
70                } else {
71                    cli::LogLevel::Debug
72                },
73            );
74
75            Arc::new(move |level, message| {
76                match (&level, &log_level) {
77                    (_, cli::LogLevel::Off) => return,
78                    (
79                        smoldot_full_node::LogLevel::Warn
80                        | smoldot_full_node::LogLevel::Info
81                        | smoldot_full_node::LogLevel::Debug
82                        | smoldot_full_node::LogLevel::Trace,
83                        cli::LogLevel::Error,
84                    ) => return,
85                    (
86                        smoldot_full_node::LogLevel::Info
87                        | smoldot_full_node::LogLevel::Debug
88                        | smoldot_full_node::LogLevel::Trace,
89                        cli::LogLevel::Warn,
90                    ) => return,
91                    (
92                        smoldot_full_node::LogLevel::Debug | smoldot_full_node::LogLevel::Trace,
93                        cli::LogLevel::Info,
94                    ) => return,
95                    (smoldot_full_node::LogLevel::Trace, cli::LogLevel::Debug) => return,
96                    _ => {}
97                }
98
99                let when = humantime::format_rfc3339_millis(SystemTime::now());
100
101                let level_str = match (level, &color_choice) {
102                    (smoldot_full_node::LogLevel::Trace, cli::ColorChoice::Never) => "trace",
103                    (smoldot_full_node::LogLevel::Trace, cli::ColorChoice::Always) => {
104                        "\x1b[36mtrace\x1b[0m"
105                    }
106                    (smoldot_full_node::LogLevel::Debug, cli::ColorChoice::Never) => "debug",
107                    (smoldot_full_node::LogLevel::Debug, cli::ColorChoice::Always) => {
108                        "\x1b[34mdebug\x1b[0m"
109                    }
110                    (smoldot_full_node::LogLevel::Info, cli::ColorChoice::Never) => "info",
111                    (smoldot_full_node::LogLevel::Info, cli::ColorChoice::Always) => {
112                        "\x1b[32minfo\x1b[0m"
113                    }
114                    (smoldot_full_node::LogLevel::Warn, cli::ColorChoice::Never) => "warn",
115                    (smoldot_full_node::LogLevel::Warn, cli::ColorChoice::Always) => {
116                        "\x1b[33;1mwarn\x1b[0m"
117                    }
118                    (smoldot_full_node::LogLevel::Error, cli::ColorChoice::Never) => "error",
119                    (smoldot_full_node::LogLevel::Error, cli::ColorChoice::Always) => {
120                        "\x1b[31;1merror\x1b[0m"
121                    }
122                };
123
124                eprintln!("[{}] [{}] {}", when, level_str, message);
125            }) as Arc<dyn smoldot_full_node::LogCallback + Send + Sync>
126        }
127        cli::Output::LogsJson => {
128            let log_level = cli_options
129                .log_level
130                .clone()
131                .unwrap_or(cli::LogLevel::Debug);
132            Arc::new(move |level, message| {
133                match (&level, &log_level) {
134                    (_, cli::LogLevel::Off) => return,
135                    (
136                        smoldot_full_node::LogLevel::Warn
137                        | smoldot_full_node::LogLevel::Info
138                        | smoldot_full_node::LogLevel::Debug
139                        | smoldot_full_node::LogLevel::Trace,
140                        cli::LogLevel::Error,
141                    ) => return,
142                    (
143                        smoldot_full_node::LogLevel::Info
144                        | smoldot_full_node::LogLevel::Debug
145                        | smoldot_full_node::LogLevel::Trace,
146                        cli::LogLevel::Warn,
147                    ) => return,
148                    (
149                        smoldot_full_node::LogLevel::Debug | smoldot_full_node::LogLevel::Trace,
150                        cli::LogLevel::Info,
151                    ) => return,
152                    (smoldot_full_node::LogLevel::Trace, cli::LogLevel::Debug) => return,
153                    _ => {}
154                }
155
156                #[derive(serde::Serialize)]
157                struct Record {
158                    timestamp: u128,
159                    level: &'static str,
160                    message: String,
161                }
162
163                let mut lock = std::io::stderr().lock();
164                if serde_json::to_writer(
165                    &mut lock,
166                    &Record {
167                        timestamp: SystemTime::now()
168                            .duration_since(UNIX_EPOCH)
169                            .map(|d| d.as_millis())
170                            .unwrap_or(0),
171                        level: match level {
172                            smoldot_full_node::LogLevel::Trace => "trace",
173                            smoldot_full_node::LogLevel::Debug => "debug",
174                            smoldot_full_node::LogLevel::Info => "info",
175                            smoldot_full_node::LogLevel::Warn => "warn",
176                            smoldot_full_node::LogLevel::Error => "error",
177                        },
178                        message,
179                    },
180                )
181                .is_ok()
182                {
183                    let _ = io::Write::write_all(&mut lock, b"\n");
184                }
185            })
186        }
187        cli::Output::Auto => unreachable!(), // Handled above.
188    };
189
190    let chain_spec =
191        fs::read(&cli_options.path_to_chain_spec).expect("Failed to read chain specification");
192    let parsed_chain_spec = {
193        smoldot::chain_spec::ChainSpec::from_json_bytes(&chain_spec)
194            .expect("Failed to decode chain specification")
195    };
196
197    // Directory where we will store everything on the disk, such as the database, secret keys,
198    // etc.
199    let base_storage_directory = if cli_options.tmp {
200        None
201    } else if let Some(base) = directories::ProjectDirs::from("io", "smoldot", "smoldot") {
202        Some(base.data_dir().to_owned())
203    } else {
204        log_callback.log(
205            smoldot_full_node::LogLevel::Warn,
206            "Failed to fetch $HOME directory. Falling back to storing everything in memory, \
207                meaning that everything will be lost when the node stops. If this is intended, \
208                please make this explicit by passing the `--tmp` flag instead."
209                .to_string(),
210        );
211        None
212    };
213
214    // Create the directory if necessary.
215    if let Some(base_storage_directory) = base_storage_directory.as_ref() {
216        fs::create_dir_all(base_storage_directory.join(parsed_chain_spec.id())).unwrap();
217    }
218    // Directory supposed to contain the database.
219    let sqlite_database_path = base_storage_directory
220        .as_ref()
221        .map(|d| d.join(parsed_chain_spec.id()).join("database"));
222    // Directory supposed to contain the keystore.
223    let keystore_path = base_storage_directory
224        .as_ref()
225        .map(|path| path.join(parsed_chain_spec.id()).join("keys"));
226
227    // Build the relay chain information if relevant.
228    let (relay_chain, relay_chain_name) =
229        if let Some((relay_chain_name, _parachain_id)) = parsed_chain_spec.relay_chain() {
230            let spec_json = {
231                let relay_chain_path = cli_options
232                    .path_to_chain_spec
233                    .parent()
234                    .unwrap()
235                    .join(format!("{relay_chain_name}.json"));
236                fs::read(&relay_chain_path).expect("Failed to read relay chain specification")
237            };
238
239            let parsed_relay_spec = smoldot::chain_spec::ChainSpec::from_json_bytes(&spec_json)
240                .expect("Failed to decode relay chain chain specs");
241
242            // Make sure we're not accidentally opening the same chain twice, otherwise weird
243            // interactions will happen.
244            assert_ne!(parsed_relay_spec.id(), parsed_chain_spec.id());
245
246            // Create the directory if necessary.
247            if let Some(base_storage_directory) = base_storage_directory.as_ref() {
248                fs::create_dir_all(base_storage_directory.join(parsed_relay_spec.id())).unwrap();
249            }
250
251            let cfg = smoldot_full_node::ChainConfig {
252                chain_spec: spec_json.into(),
253                additional_bootnodes: Vec::new(),
254                keystore_memory: Vec::new(),
255                sqlite_database_path: base_storage_directory.as_ref().map(|d| {
256                    d.join(parsed_relay_spec.id())
257                        .join("database")
258                        .join("database.sqlite")
259                }),
260                sqlite_cache_size: cli_options.relay_chain_database_cache_size.0,
261                keystore_path: base_storage_directory
262                    .as_ref()
263                    .map(|path| path.join(parsed_relay_spec.id()).join("keys")),
264                json_rpc_listen: None,
265            };
266
267            (Some(cfg), Some(relay_chain_name.to_owned()))
268        } else {
269            (None, None)
270        };
271
272    // Determine which networking key to use.
273    //
274    // This is either passed as a CLI option, loaded from disk, or generated randomly.
275    // TODO: move this code to `/lib/src/identity`?
276    let libp2p_key = if let Some(node_key) = cli_options.libp2p_key {
277        node_key
278    } else if let Some(dir) = base_storage_directory.as_ref() {
279        let path = dir.join("libp2p_ed25519_secret_key.secret");
280        let libp2p_key = if path.exists() {
281            let file_content = zeroize::Zeroizing::new(
282                fs::read_to_string(&path).expect("failed to read libp2p secret key file content"),
283            );
284            let mut hex_decoded = Box::new([0u8; 32]);
285            hex::decode_to_slice(file_content, &mut *hex_decoded)
286                .expect("invalid libp2p secret key file content");
287            hex_decoded
288        } else {
289            let mut actual_key = Box::new([0u8; 32]);
290            rand::Fill::try_fill(&mut *actual_key, &mut rand::thread_rng()).unwrap();
291            let mut hex_encoded = Box::new([0; 64]);
292            hex::encode_to_slice(*actual_key, &mut *hex_encoded).unwrap();
293            fs::write(&path, *hex_encoded).expect("failed to write libp2p secret key file");
294            zeroize::Zeroize::zeroize(&mut *hex_encoded);
295            actual_key
296        };
297        // On Unix platforms, set the permission as 0o400 (only reading and by owner is permitted).
298        // TODO: do something equivalent on Windows
299        #[cfg(unix)]
300        let _ = fs::set_permissions(&path, std::os::unix::fs::PermissionsExt::from_mode(0o400));
301        libp2p_key
302    } else {
303        let mut key = Box::new([0u8; 32]);
304        rand::Fill::try_fill(&mut *key, &mut rand::thread_rng()).unwrap();
305        key
306    };
307
308    // Create an executor where tasks are going to be spawned onto.
309    let executor = Arc::new(smol::Executor::new());
310    for n in 0..thread::available_parallelism()
311        .map(|n| n.get() - 1)
312        .unwrap_or(3)
313    {
314        let executor = executor.clone();
315
316        let spawn_result = thread::Builder::new()
317            .name(format!("tasks-pool-{}", n))
318            .spawn(move || smol::block_on(executor.run(smol::future::pending::<()>())));
319
320        // Ignore a failure to spawn a thread, as we're going to run tasks on the current thread
321        // later down this function.
322        if let Err(err) = spawn_result {
323            log_callback.log(
324                smoldot_full_node::LogLevel::Warn,
325                format!("tasks-pool-thread-spawn-failure; err={err}"),
326            );
327        }
328    }
329
330    // Print some general information.
331    log_callback.log(
332        smoldot_full_node::LogLevel::Info,
333        "smoldot full node".to_string(),
334    );
335    log_callback.log(
336        smoldot_full_node::LogLevel::Info,
337        "Copyright (C) 2019-2022  Parity Technologies (UK) Ltd.".to_string(),
338    );
339    log_callback.log(
340        smoldot_full_node::LogLevel::Info,
341        "Copyright (C) 2023  Pierre Krieger.".to_string(),
342    );
343    log_callback.log(
344        smoldot_full_node::LogLevel::Info,
345        "This program comes with ABSOLUTELY NO WARRANTY.".to_string(),
346    );
347    log_callback.log(
348        smoldot_full_node::LogLevel::Info,
349        "This is free software, and you are welcome to redistribute it under certain conditions."
350            .to_string(),
351    );
352
353    // This warning message should be removed if/when the full node becomes mature.
354    log_callback.log(
355        smoldot_full_node::LogLevel::Warn,
356        "Please note that this full node is experimental. It is not feature complete and is \
357        known to panic often. Please report any panic you might encounter to \
358        <https://github.com/smol-dot/smoldot/issues>."
359            .to_string(),
360    );
361
362    let client_init_result = smoldot_full_node::start(smoldot_full_node::Config {
363        chain: smoldot_full_node::ChainConfig {
364            chain_spec: chain_spec.into(),
365            additional_bootnodes: cli_options
366                .additional_bootnode
367                .iter()
368                .map(|cli::Bootnode { address, peer_id }| (peer_id.clone(), address.clone()))
369                .collect(),
370            keystore_memory: cli_options.keystore_memory,
371            sqlite_database_path,
372            sqlite_cache_size: cli_options.database_cache_size.0,
373            keystore_path,
374            json_rpc_listen: if let Some(address) = cli_options.json_rpc_address.0 {
375                Some(smoldot_full_node::JsonRpcListenConfig {
376                    address,
377                    max_json_rpc_clients: cli_options.json_rpc_max_clients,
378                })
379            } else {
380                None
381            },
382        },
383        relay_chain,
384        libp2p_key,
385        listen_addresses: cli_options.listen_addr,
386        tasks_executor: {
387            let executor = executor.clone();
388            Arc::new(move |task| executor.spawn(task).detach())
389        },
390        log_callback: log_callback.clone(),
391        jaeger_agent: cli_options.jaeger,
392    })
393    .await;
394
395    let client = match client_init_result {
396        Ok(c) => c,
397        Err(err) => {
398            log_callback.log(
399                smoldot_full_node::LogLevel::Error,
400                format!("Failed to initialize client: {}", err),
401            );
402            panic!("Failed to initialize client: {}", err);
403        }
404    };
405
406    if let Some(addr) = client.json_rpc_server_addr() {
407        log_callback.log(
408            smoldot_full_node::LogLevel::Info,
409            format!(
410                "JSON-RPC server listening on {addr}. Visit \
411                <https://ipfs.io/ipns/dotapps.io/?rpc=ws%3A%2F%2F{addr}> in order to \
412                interact with the node."
413            ),
414        );
415    }
416
417    // Starting from here, a SIGINT (or equivalent) handler is set up. If the user does Ctrl+C,
418    // an event will be triggered on `ctrlc_detected`.
419    // This should be performed after all the expensive initialization is done, as otherwise these
420    // expensive initializations aren't interrupted by Ctrl+C, which could be frustrating for the
421    // user.
422    let ctrlc_detected = {
423        let event = event_listener::Event::new();
424        let listen = event.listen();
425        if let Err(err) = ctrlc::set_handler(move || {
426            event.notify(usize::MAX);
427        }) {
428            // It is not critical to fail to setup the Ctrl-C handler.
429            log_callback.log(
430                smoldot_full_node::LogLevel::Warn,
431                format!("ctrlc-handler-setup-fail; err={err}"),
432            );
433        }
434        listen
435    };
436
437    // Spawn a task that prints the informant at a regular interval.
438    // The interval is fast enough that the informant should be visible roughly at any time,
439    // even if the terminal is filled with logs.
440    // Note that this task also holds the smoldot `client` alive, and thus we spawn it even if
441    // the informant is disabled.
442    let main_task = executor.spawn({
443        let show_informant = matches!(cli_output, cli::Output::Informant);
444        let informant_colors = match cli_options.color {
445            cli::ColorChoice::Always => true,
446            cli::ColorChoice::Never => false,
447        };
448
449        async move {
450            let mut informant_timer = if show_informant {
451                smol::Timer::after(Duration::new(0, 0))
452            } else {
453                smol::Timer::never()
454            };
455
456            loop {
457                informant_timer =
458                    smol::Timer::at(informant_timer.await + Duration::from_millis(100));
459
460                // We end the informant line with a `\r` so that it overwrites itself
461                // every time. If any other line gets printed, it will overwrite the
462                // informant, and the informant will then print itself below, which is
463                // a fine behaviour.
464                let sync_state = client.sync_state().await;
465                eprint!(
466                    "{}\r",
467                    smoldot::informant::InformantLine {
468                        enable_colors: informant_colors,
469                        chain_name: parsed_chain_spec.name(),
470                        relay_chain: client.relay_chain_sync_state().await.map(
471                            |relay_sync_state| smoldot::informant::RelayChain {
472                                chain_name: relay_chain_name.as_ref().unwrap(),
473                                best_number: relay_sync_state.best_block_number,
474                            }
475                        ),
476                        max_line_width: terminal_size::terminal_size()
477                            .map_or(80, |(w, _)| w.0.into()),
478                        num_peers: client.num_peers().await,
479                        num_network_connections: client.num_network_connections().await,
480                        best_number: sync_state.best_block_number,
481                        finalized_number: sync_state.finalized_block_number,
482                        best_hash: &sync_state.best_block_hash,
483                        finalized_hash: &sync_state.finalized_block_hash,
484                        network_known_best: client.network_known_best().await,
485                    }
486                );
487            }
488        }
489    });
490
491    // Now run all the tasks that have been spawned.
492    executor.run(ctrlc_detected).await;
493
494    // Add a new line after the informant so that the user's shell doesn't
495    // overwrite it.
496    if matches!(cli_output, cli::Output::Informant) {
497        eprintln!();
498    }
499
500    // After `ctrlc_detected` has triggered, we destroy `main_task`, which cancels it and destroys
501    // the smoldot client.
502    drop::<smol::Task<_>>(main_task);
503
504    // TODO: consider running the executor until all tasks shut down gracefully; unfortunately this currently hangs
505}
