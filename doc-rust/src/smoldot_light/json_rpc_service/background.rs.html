224/// State of the subscription towards the runtime service.
225/// See [`Background::runtime_service_subscription`].
226enum RuntimeServiceSubscription<TPlat: PlatformRef> {
227    /// Subscription is active.
228    Active {
229        /// Object representing the subscription.
230        subscription: runtime_service::Subscription<TPlat>,
231
232        /// Hash of the current best block. Guaranteed to be in
233        /// [`RuntimeServiceSubscription::Active::pinned_blocks`].
234        current_best_block: [u8; 32],
235
236        /// If `Some`, the new heads and runtime version subscriptions haven't been updated about
237        /// the new current best block yet. Contains the previous best block that the
238        /// subscriptions are aware of. The previous best block is guaranteed to be in
239        /// [`RuntimeServiceSubscription::Active::pinned_blocks`].
240        new_heads_and_runtime_subscriptions_stale: Option<Option<[u8; 32]>>,
241
242        /// Hash of the current finalized block. Guaranteed to be in
243        /// [`RuntimeServiceSubscription::Active::pinned_blocks`].
244        current_finalized_block: [u8; 32],
245
246        /// If `true`, the finalized heads subscriptions haven't been updated about the new
247        /// current finalized block yet.
248        finalized_heads_subscriptions_stale: bool,
249
250        /// When the runtime service reports a new block, it is kept pinned and inserted in this
251        /// list.
252        ///
253        /// Blocks are removed from this container and unpinned when they leave
254        /// [`RuntimeServiceSubscription::Active::finalized_and_pruned_lru`].
255        ///
256        /// JSON-RPC clients are more likely to ask for information about recent blocks and
257        /// perform calls on them, hence a cache of recent blocks.
258        pinned_blocks: hashbrown::HashMap<[u8; 32], RecentBlock, fnv::FnvBuildHasher>,
259
260        /// When a block is finalized or pruned, it is inserted into this LRU cache. The least
261        /// recently used blocks are removed and unpinned.
262        finalized_and_pruned_lru: lru::LruCache<[u8; 32], (), fnv::FnvBuildHasher>,
263    },
264
265    /// Waiting for the runtime service to start the subscription. Can potentially take a long
266    /// time.
267    Pending(Pin<Box<dyn Future<Output = runtime_service::SubscribeAll<TPlat>> + Send>>),
268
269    /// Subscription not requested yet. Should transition to
270    /// [`RuntimeServiceSubscription::Pending`] as soon as possible.
271    NotCreated,
272}

486pub(super) async fn run<TPlat: PlatformRef>(
487    log_target: String,
488    config: Config<TPlat>,
489    requests_rx: async_channel::Receiver<String>,
490    responses_tx: async_channel::Sender<String>,
491) {
492    let mut me = Background {
493        log_target,
494        chain_name: config.chain_name,
495        chain_ty: config.chain_ty,
496        chain_is_live: config.chain_is_live,
497        chain_properties_json: config.chain_properties_json,
498        system_name: config.system_name.clone(),
499        system_version: config.system_version.clone(),
500        randomness: ChaCha20Rng::from_seed({
501            let mut seed = [0; 32];
502            config.platform.fill_random_bytes(&mut seed);
503            seed
504        }),
505        next_garbage_collection: Box::pin(config.platform.sleep(Duration::new(0, 0))),
506        network_service: config.network_service.clone(),
507        sync_service: config.sync_service.clone(),
508        runtime_service: config.runtime_service.clone(),
509        transactions_service: config.transactions_service.clone(),
510        background_tasks: stream::FuturesUnordered::new(),
511        runtime_service_subscription: RuntimeServiceSubscription::NotCreated,
512        all_heads_subscriptions: hashbrown::HashSet::with_capacity_and_hasher(
513            2,
514            Default::default(),
515        ),
516        new_heads_subscriptions: hashbrown::HashSet::with_capacity_and_hasher(
517            2,
518            Default::default(),
519        ),
520        finalized_heads_subscriptions: hashbrown::HashSet::with_capacity_and_hasher(
521            2,
522            Default::default(),
523        ),
524        runtime_version_subscriptions: hashbrown::HashSet::with_capacity_and_hasher(
525            2,
526            Default::default(),
527        ),
528        transactions_subscriptions: hashbrown::HashMap::with_capacity_and_hasher(
529            2,
530            Default::default(),
531        ),
532        chain_head_follow_subscriptions: hashbrown::HashMap::with_hasher(Default::default()),
533        legacy_api_storage_subscriptions: BTreeSet::new(),
534        legacy_api_storage_subscriptions_by_key: BTreeSet::new(),
535        legacy_api_stale_storage_subscriptions: hashbrown::HashSet::with_capacity_and_hasher(
536            0,
537            Default::default(),
538        ),
539        legacy_api_storage_query_in_progress: false,
540        requests_rx: Box::pin(requests_rx),
541        responses_tx,
542        multistage_requests_to_advance: VecDeque::new(),
543        block_headers_cache: lru::LruCache::with_hasher(
544            NonZero::<usize>::new(32).unwrap_or_else(|| unreachable!()),
545            Default::default(),
546        ),
547        best_block_hash_pending: Vec::new(),
548        pending_get_finalized_head: Vec::new(),
549        block_headers_pending: hashbrown::HashMap::with_capacity_and_hasher(0, Default::default()),
550        block_runtimes_cache: lru::LruCache::with_hasher(
551            NonZero::<usize>::new(32).unwrap_or_else(|| unreachable!()),
552            Default::default(),
553        ),
554        block_runtimes_pending: hashbrown::HashMap::with_capacity_and_hasher(0, Default::default()),
555        state_get_keys_paged_cache: lru::LruCache::with_hasher(
556            NonZero::<usize>::new(2).unwrap(),
557            util::SipHasherBuild::new({
558                let mut seed = [0; 16];
559                config.platform.fill_random_bytes(&mut seed);
560                seed
561            }),
562        ),
563        genesis_block_hash: config.genesis_block_hash,
564        printed_legacy_json_rpc_warning: false,
565        platform: config.platform,
566    };
567
568    loop {
569        // Yield at every loop in order to provide better tasks granularity.
570        futures_lite::future::yield_now().await;
571
572        enum WakeUpReason<'a, TPlat: PlatformRef> {
573            ForegroundDead,
574            GarbageCollection,
575            IncomingJsonRpcRequest(String),
576            AdvanceMultiStageRequest {
577                request_id_json: String,
578                stage: MultiStageRequestStage,
579                request_ty: MultiStageRequestTy,
580            },
581            Event(Event<TPlat>),
582            RuntimeServiceSubscriptionReady(runtime_service::SubscribeAll<TPlat>),
583            RuntimeServiceSubscriptionNotification {
584                notification: runtime_service::Notification,
585                subscription: &'a mut runtime_service::Subscription<TPlat>,
586                pinned_blocks:
587                    &'a mut hashbrown::HashMap<[u8; 32], RecentBlock, fnv::FnvBuildHasher>,
588                finalized_and_pruned_lru: &'a mut lru::LruCache<[u8; 32], (), fnv::FnvBuildHasher>,
589                current_best_block: &'a mut [u8; 32],
590                new_heads_and_runtime_subscriptions_stale: &'a mut Option<Option<[u8; 32]>>,
591                current_finalized_block: &'a mut [u8; 32],
592                finalized_heads_subscriptions_stale: &'a mut bool,
593            },
594            RuntimeServiceSubscriptionDead,
595            StartStorageSubscriptionsUpdates,
596            NotifyFinalizedHeads,
597            NotifyNewHeadsRuntimeSubscriptions(Option<[u8; 32]>),
598        }
599
600        // Wait until there is something to do.

5453                                result = Err(());

5455                        } else {

5457                        }
5458                    } else {
5459                        result = Err(());

5470                                    .body
5471                                    .unwrap()

5503                } = &mut me.runtime_service_subscription

5598                                        result: Err(error),

5635                >::with_capacity_and_hasher(

5657                }

5660                // relevant.

5664                            .remove(&subscription_id);
5665                    }
5666                    let _ = me
5667                        .responses_tx
5668                        .send(
5669                            methods::ServerToClient::state_storage {
5670                                subscription: Cow::Borrowed(&*subscription_id),
5671                                result: methods::StorageChangeSet {
5672                                    block: methods::HashHexString(block_hash),
5673                                    changes,
5674                                },
5675                            }
5676                            .to_json_request_object_parameters(None),
5677                        )
5678                        .await;
5679                }

5686                ..

5815                                .responses_tx
5816                                .send(

5831                // out the notifications.
5832                me.legacy_api_stale_storage_subscriptions.extend(
5833                    me.legacy_api_storage_subscriptions
5834                        .iter()
5835                        .map(|(s_id, _)| s_id.clone()),

5842fn convert_runtime_version_legacy(
5843    runtime_spec: &'_ smoldot::executor::CoreVersion,
5844) -> methods::RuntimeVersion<'_> {
5845    let runtime_spec = runtime_spec.decode();
5846    methods::RuntimeVersion {
5847        spec_name: runtime_spec.spec_name.into(),
5848        impl_name: runtime_spec.impl_name.into(),
5849        authoring_version: u64::from(runtime_spec.authoring_version),
5850        spec_version: u64::from(runtime_spec.spec_version),
5851        impl_version: u64::from(runtime_spec.impl_version),
5852        transaction_version: runtime_spec.transaction_version.map(u64::from),
5853        state_version: runtime_spec.state_version.map(u8::from).map(u64::from),
5854        apis: runtime_spec
5855            .apis
5856            .map(|api| (methods::HexString(api.name_hash.to_vec()), api.version))
5857            .collect(),
5858    }
5859}
