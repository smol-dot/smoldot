31/// Starts a sync service background task to synchronize a parachain.
32pub(super) async fn start_parachain<TPlat: PlatformRef>(
33    log_target: String,
34    platform: TPlat,
35    finalized_block_header: Vec<u8>,
36    block_number_bytes: usize,
37    relay_chain_sync: Arc<runtime_service::RuntimeService<TPlat>>,
38    parachain_id: u32,
39    from_foreground: Pin<Box<async_channel::Receiver<ToBackground>>>,
40    network_service: Arc<network_service::NetworkServiceChain<TPlat>>,
41) {
42    ParachainBackgroundTask {
43        log_target,
44        from_foreground,
45        block_number_bytes,
46        parachain_id,
47        from_network_service: None,
48        network_service,
49        obsolete_finalized_parahead: finalized_block_header,
50        sync_sources: HashMap::with_capacity_and_hasher(
51            0,
52            util::SipHasherBuild::new({
53                let mut seed = [0; 16];
54                platform.fill_random_bytes(&mut seed);
55                seed
56            }),
57        ),
58        subscription_state: ParachainBackgroundState::NotSubscribed {
59            all_subscriptions: Vec::new(),
60            subscribe_future: {
61                let relay_chain_sync = relay_chain_sync.clone();
62                Box::pin(async move {
63                    relay_chain_sync
64                        .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
65                        .await
66                })
67            },
68        },
69        relay_chain_sync,
70        platform,
71    }
72    .run()
73    .await;
74}

76/// Task that is running in the background.
77struct ParachainBackgroundTask<TPlat: PlatformRef> {
78    /// Target to use for all logs.
79    log_target: String,
80
81    /// Access to the platform's capabilities.
82    platform: TPlat,
83
84    /// Channel receiving message from the sync service frontend.
85    from_foreground: Pin<Box<async_channel::Receiver<ToBackground>>>,
86
87    /// Number of bytes to use to encode the parachain block numbers in headers.
88    block_number_bytes: usize,
89
90    /// Id of the parachain registered within the relay chain. Chosen by the user.
91    parachain_id: u32,
92
93    /// Networking service connected to the peer-to-peer network of the parachain.
94    network_service: Arc<network_service::NetworkServiceChain<TPlat>>,
95
96    /// Events coming from the networking service. `None` if not subscribed yet.
97    from_network_service: Option<Pin<Box<async_channel::Receiver<network_service::Event>>>>,
98
99    /// Runtime service of the relay chain.
100    relay_chain_sync: Arc<runtime_service::RuntimeService<TPlat>>,
101
102    /// Last-known finalized parachain header. Can be very old and obsolete.
103    /// Updated after we successfully fetch the parachain head of a relay chain finalized block,
104    /// and left untouched if the fetch fails.
105    /// Initialized to the parachain genesis block header.
106    obsolete_finalized_parahead: Vec<u8>,
107
108    /// List of parachain network sources.
109    ///
110    /// Values are their role, and self-reported best block when we connected to them. This best
111    /// block is never updated.
112    ///
113    /// > **Note**: In the past, smoldot used to track exactly which peer knows which block
114    /// >           based on block announces. This, however, caused issues due to the fact that
115    /// >           there's a disconnect between the parachain best block on the relay chain
116    /// >           and the parachain best block on the network. We currently simply assume that
117    /// >           all parachain nodes know about all parachain blocks from the relay chain.
118    sync_sources: HashMap<PeerId, (codec::Role, u64, [u8; 32]), util::SipHasherBuild>,
119
120    /// Extra fields that are set after the subscription to the runtime service events has
121    /// succeeded.
122    subscription_state: ParachainBackgroundState<TPlat>,
123}
124
125enum ParachainBackgroundState<TPlat: PlatformRef> {
126    /// Currently subscribing to the relay chain runtime service.
127    NotSubscribed {
128        /// List of senders that will get notified when the tree of blocks is modified.
129        ///
130        /// These subscriptions are pending and no notification should be sent to them until the
131        /// subscription to the relay chain runtime service is finished.
132        all_subscriptions: Vec<async_channel::Sender<super::Notification>>,
133
134        /// Future when the subscription has finished.
135        subscribe_future: future::BoxFuture<'static, runtime_service::SubscribeAll<TPlat>>,
136    },
137
138    /// Subscribed to the relay chain runtime service.
139    Subscribed(ParachainBackgroundTaskAfterSubscription<TPlat>),
140}
141
142struct ParachainBackgroundTaskAfterSubscription<TPlat: PlatformRef> {
143    /// List of senders that get notified when the tree of blocks is modified.
144    all_subscriptions: Vec<async_channel::Sender<super::Notification>>,
145
146    /// Stream of blocks of the relay chain this parachain is registered on.
147    /// The buffer size should be large enough so that, if the CPU is busy, it doesn't become full
148    /// before the execution of the sync service resumes.
149    /// The maximum number of pinned block is ignored, as this maximum is a way to avoid malicious
150    /// behaviors. This code is by definition not considered malicious.
151    relay_chain_subscribe_all: runtime_service::Subscription<TPlat>,
152
153    /// Hash of the best parachain that has been reported to the subscriptions.
154    /// `None` if and only if no finalized parachain head is known yet.
155    reported_best_parahead_hash: Option<[u8; 32]>,
156
157    /// Tree of relay chain blocks. Blocks are inserted when received from the relay chain
158    /// sync service. Once inside, their corresponding parachain head is fetched. Once the
159    /// parachain head is fetched, this parachain head is reported to our subscriptions.
160    ///
161    /// The root of the tree is a "virtual" block. It can be thought as the parent of the relay
162    /// chain finalized block, but is there even if the relay chain finalized block is block 0.
163    ///
164    /// All block in the tree has an associated parachain head behind an `Option`. This `Option`
165    /// always contains `Some`, except for the "virtual" root block for which it is `None`.
166    ///
167    /// If the output finalized block has a parachain head equal to `None`, it therefore means
168    /// that no finalized parachain head is known yet.
169    /// Note that, when it is the case, `SubscribeAll` messages from the frontend are still
170    /// answered with a single finalized block set to `obsolete_finalized_parahead`. Once a
171    /// finalized parachain head is known, it is important to reset all subscriptions.
172    ///
173    /// The set of blocks in this tree whose parachain block hasn't been fetched yet is the same
174    /// as the set of blocks that is maintained pinned on the runtime service. Blocks are unpinned
175    /// when their parachain head fetching succeeds or when they are removed from the tree.
176    async_tree: async_tree::AsyncTree<TPlat::Instant, [u8; 32], Option<Vec<u8>>>,
177
178    /// If `true`, [`ParachainBackgroundTaskAfterSubscription::async_tree`] might need to
179    /// be advanced.
180    must_process_sync_tree: bool,
181
182    /// List of in-progress parachain head fetching operations.
183    ///
184    /// The operations require some blocks to be pinned within the relay chain runtime service,
185    /// which is guaranteed by the fact that `relay_chain_subscribe_all.new_blocks` stays
186    /// alive for longer than this container, and by the fact that we unpin block after a
187    /// fetching operation has finished and that we never fetch twice for the same block.
188    in_progress_paraheads: stream::FuturesUnordered<
189        future::BoxFuture<'static, (async_tree::AsyncOpId, Result<Vec<u8>, ParaheadError>)>,
190    >,
191
192    /// Future that is ready when we need to start a new parachain head fetch operation.
193    next_start_parahead_fetch: Pin<Box<dyn Future<Output = ()> + Send>>,
194}
195
196impl<TPlat: PlatformRef> ParachainBackgroundTask<TPlat> {
197    async fn run(mut self) {
198        loop {
199            // Yield at every loop in order to provide better tasks granularity.
200            futures_lite::future::yield_now().await;
201
202            // Wait until something interesting happens.
203            enum WakeUpReason<TPlat: PlatformRef> {
204                ForegroundClosed,
205                ForegroundMessage(ToBackground),
206                NewSubscription(runtime_service::SubscribeAll<TPlat>),
207                StartParaheadFetch,
208                ParaheadFetchFinished {
209                    async_op_id: async_tree::AsyncOpId,
210                    parahead_result: Result<Vec<u8>, ParaheadError>,
211                },
212                Notification(runtime_service::Notification),
213                SubscriptionDead,
214                MustSubscribeNetworkEvents,
215                NetworkEvent(network_service::Event),
216                AdvanceSyncTree,
217            }
218
219            let wake_up_reason: WakeUpReason<_> = {
220                let (
221                    subscribe_future,
222                    next_start_parahead_fetch,
223                    relay_chain_subscribe_all,
224                    in_progress_paraheads,
225                    must_process_sync_tree,
226                    is_relaychain_subscribed,
227                ) = match &mut self.subscription_state {
228                    ParachainBackgroundState::NotSubscribed {
229                        subscribe_future, ..
230                    } => (Some(subscribe_future), None, None, None, None, false),
231                    ParachainBackgroundState::Subscribed(runtime_subscription) => (
232                        None,
233                        Some(&mut runtime_subscription.next_start_parahead_fetch),
234                        Some(&mut runtime_subscription.relay_chain_subscribe_all),
235                        Some(&mut runtime_subscription.in_progress_paraheads),
236                        Some(&mut runtime_subscription.must_process_sync_tree),
237                        true,
238                    ),
239                };
240
241                async {
242                    if let Some(subscribe_future) = subscribe_future {
243                        WakeUpReason::NewSubscription(subscribe_future.await)
244                    } else {
245                        future::pending().await
246                    }
247                }
248                .or(async {
249                    match self.from_foreground.next().await {
250                        Some(msg) => WakeUpReason::ForegroundMessage(msg),
251                        None => WakeUpReason::ForegroundClosed,
252                    }
253                })
254                .or(async {
255                    if let Some(relay_chain_subscribe_all) = relay_chain_subscribe_all {
256                        match relay_chain_subscribe_all.next().await {
257                            Some(notif) => WakeUpReason::Notification(notif),
258                            None => WakeUpReason::SubscriptionDead,
259                        }
260                    } else {
261                        future::pending().await
262                    }
263                })
264                .or(async {
265                    if is_relaychain_subscribed {
266                        if let Some(from_network_service) = self.from_network_service.as_mut() {
267                            match from_network_service.next().await {
268                                Some(ev) => WakeUpReason::NetworkEvent(ev),
269                                None => {
270                                    self.from_network_service = None;
271                                    WakeUpReason::MustSubscribeNetworkEvents
272                                }
273                            }
274                        } else {
275                            WakeUpReason::MustSubscribeNetworkEvents
276                        }
277                    } else {
278                        future::pending().await
279                    }
280                })
281                .or(async {
282                    if let Some(next_start_parahead_fetch) = next_start_parahead_fetch {
283                        next_start_parahead_fetch.as_mut().await;
284                        *next_start_parahead_fetch = Box::pin(future::pending());
285                        WakeUpReason::StartParaheadFetch
286                    } else {
287                        future::pending().await
288                    }
289                })
290                .or(async {
291                    if let Some(in_progress_paraheads) = in_progress_paraheads {
292                        if !in_progress_paraheads.is_empty() {
293                            let (async_op_id, parahead_result) =
294                                in_progress_paraheads.next().await.unwrap();
295                            WakeUpReason::ParaheadFetchFinished {
296                                async_op_id,
297                                parahead_result,
298                            }
299                        } else {
300                            future::pending().await
301                        }
302                    } else {
303                        future::pending().await
304                    }
305                })
306                .or(async {
307                    if let Some(must_process_sync_tree) = must_process_sync_tree {
308                        if *must_process_sync_tree {
309                            *must_process_sync_tree = false;
310                            WakeUpReason::AdvanceSyncTree
311                        } else {
312                            future::pending().await
313                        }
314                    } else {
315                        future::pending().await
316                    }
317                })
318                .await
319            };
320
321            match (wake_up_reason, &mut self.subscription_state) {
322                (WakeUpReason::ForegroundClosed, _) => {
323                    // Terminate the background task.
324                    return;
325                }
326
327                (WakeUpReason::NewSubscription(relay_chain_subscribe_all), _) => {
328                    // Subscription to the relay chain has finished.
329                    log!(
330                        &self.platform,
331                        Debug,
332                        &self.log_target,
333                        "relay-chain-new-subscription",
334                        finalized_hash = HashDisplay(&header::hash_from_scale_encoded_header(
335                            &relay_chain_subscribe_all.finalized_block_scale_encoded_header
336                        )),
337                        subscription_id = ?relay_chain_subscribe_all.new_blocks.id(),
338                    );
339                    log!(
340                        &self.platform,
341                        Debug,
342                        &self.log_target,
343                        "parahead-fetch-operations-cleared"
344                    );
345
346                    let async_tree = {
347                        let mut async_tree =
348                            async_tree::AsyncTree::<TPlat::Instant, [u8; 32], _>::new(
349                                async_tree::Config {
350                                    finalized_async_user_data: None,
351                                    retry_after_failed: Duration::from_secs(5),
352                                    blocks_capacity: 32,
353                                },
354                            );
355                        let finalized_hash = header::hash_from_scale_encoded_header(
356                            &relay_chain_subscribe_all.finalized_block_scale_encoded_header,
357                        );
358                        let finalized_index =
359                            async_tree.input_insert_block(finalized_hash, None, false, true);
360                        async_tree.input_finalize(finalized_index);
361                        for block in relay_chain_subscribe_all.non_finalized_blocks_ancestry_order {
362                            let hash =
363                                header::hash_from_scale_encoded_header(&block.scale_encoded_header);
364                            let parent = async_tree
365                                .input_output_iter_unordered()
366                                .find(|b| *b.user_data == block.parent_hash)
367                                .map(|b| b.id)
368                                .unwrap_or(finalized_index);
369                            async_tree.input_insert_block(
370                                hash,
371                                Some(parent),
372                                false,
373                                block.is_new_best,
374                            );
375                        }
376                        async_tree
377                    };
378
379                    self.subscription_state = ParachainBackgroundState::Subscribed(
380                        ParachainBackgroundTaskAfterSubscription {
381                            all_subscriptions: match &mut self.subscription_state {
382                                ParachainBackgroundState::NotSubscribed {
383                                    all_subscriptions,
384                                    ..
385                                } => mem::take(all_subscriptions),
386                                _ => unreachable!(),
387                            },
388                            relay_chain_subscribe_all: relay_chain_subscribe_all.new_blocks,
389                            reported_best_parahead_hash: None,
390                            async_tree,
391                            must_process_sync_tree: false,
392                            in_progress_paraheads: stream::FuturesUnordered::new(),
393                            next_start_parahead_fetch: Box::pin(future::ready(())),
394                        },
395                    );
396                }
397
398                (
399                    WakeUpReason::AdvanceSyncTree,
400                    ParachainBackgroundState::Subscribed(runtime_subscription),
401                ) => {
402                    if let Some(update) = runtime_subscription.async_tree.try_advance_output() {
403                        // Make sure to process any notification that comes after.
404                        runtime_subscription.must_process_sync_tree = true;
405
406                        match update {
407                            async_tree::OutputUpdate::Finalized {
408                                former_finalized_async_op_user_data: former_finalized_parahead,
409                                pruned_blocks,
410                                best_output_block_updated,
411                                ..
412                            } if *runtime_subscription
413                                .async_tree
414                                .output_finalized_async_user_data()
415                                != former_finalized_parahead =>
416                            {
417                                let new_finalized_parahead = runtime_subscription
418                                    .async_tree
419                                    .output_finalized_async_user_data();
420                                debug_assert!(new_finalized_parahead.is_some());
421
422                                // If this is the first time a finalized parahead is known, any
423                                // `SubscribeAll` message that has been answered beforehand was
424                                // answered in a dummy way with a potentially obsolete finalized
425                                // header.
426                                // For this reason, we reset all subscriptions to force all
427                                // subscribers to re-subscribe.
428                                if former_finalized_parahead.is_none() {
429                                    runtime_subscription.all_subscriptions.clear();
430                                }
431
432                                let hash = header::hash_from_scale_encoded_header(
433                                    new_finalized_parahead.as_ref().unwrap(),
434                                );
435
436                                self.obsolete_finalized_parahead =
437                                    new_finalized_parahead.clone().unwrap();
438
439                                // Must unpin the pruned blocks if they haven't already been unpinned.
440                                let mut pruned_blocks_hashes =
441                                    Vec::with_capacity(pruned_blocks.len());
442                                for (_, hash, pruned_block_parahead) in pruned_blocks {
443                                    if pruned_block_parahead.is_none() {
444                                        runtime_subscription
445                                            .relay_chain_subscribe_all
446                                            .unpin_block(hash)
447                                            .await;
448                                    }
449                                    pruned_blocks_hashes.push(hash);
450                                }
451
452                                log!(
453                                    &self.platform,
454                                    Debug,
455                                    &self.log_target,
456                                    "subscriptions-notify-parablock-finalized",
457                                    hash = HashDisplay(&hash)
458                                );
459
460                                let best_block_hash = runtime_subscription
461                                    .async_tree
462                                    .output_best_block_index()
463                                    .map(|(_, parahead)| {
464                                        header::hash_from_scale_encoded_header(
465                                            parahead.as_ref().unwrap(),
466                                        )
467                                    })
468                                    .unwrap_or(hash);
469                                runtime_subscription.reported_best_parahead_hash =
470                                    Some(best_block_hash);
471
472                                // Elements in `all_subscriptions` are removed one by one and
473                                // inserted back if the channel is still open.
474                                for index in (0..runtime_subscription.all_subscriptions.len()).rev()
475                                {
476                                    let sender =
477                                        runtime_subscription.all_subscriptions.swap_remove(index);
478                                    let notif = super::Notification::Finalized {
479                                        hash,
480                                        best_block_hash_if_changed: if best_output_block_updated {
481                                            Some(best_block_hash)
482                                        } else {
483                                            None
484                                        },
485                                        pruned_blocks: pruned_blocks_hashes.clone(),
486                                    };
487                                    if sender.try_send(notif).is_ok() {
488                                        runtime_subscription.all_subscriptions.push(sender);
489                                    }
490                                }
491                            }
492
493                            async_tree::OutputUpdate::Finalized { .. }
494                            | async_tree::OutputUpdate::BestBlockChanged { .. } => {
495                                // Do not report anything to subscriptions if no finalized parahead is
496                                // known yet.
497                                let finalized_parahead = match runtime_subscription
498                                    .async_tree
499                                    .output_finalized_async_user_data()
500                                {
501                                    Some(p) => p,
502                                    None => continue,
503                                };
504
505                                // Calculate hash of the parablock corresponding to the new best relay
506                                // chain block.
507                                let parahash = header::hash_from_scale_encoded_header(
508                                    runtime_subscription
509                                        .async_tree
510                                        .output_best_block_index()
511                                        .map(|(_, b)| b.as_ref().unwrap())
512                                        .unwrap_or(finalized_parahead),
513                                );
514
515                                if runtime_subscription.reported_best_parahead_hash.as_ref()
516                                    != Some(&parahash)
517                                {
518                                    runtime_subscription.reported_best_parahead_hash =
519                                        Some(parahash);
520
521                                    // The networking service needs to be kept up to date with what the local
522                                    // node considers as the best block.
523                                    if let Ok(header) =
524                                        header::decode(finalized_parahead, self.block_number_bytes)
525                                    {
526                                        self.network_service
527                                            .set_local_best_block(parahash, header.number)
528                                            .await;
529                                    }
530
531                                    log!(
532                                        &self.platform,
533                                        Debug,
534                                        &self.log_target,
535                                        "subscriptions-notify-best-block-changed",
536                                        hash = HashDisplay(&parahash)
537                                    );
538
539                                    // Elements in `all_subscriptions` are removed one by one and
540                                    // inserted back if the channel is still open.
541                                    for index in
542                                        (0..runtime_subscription.all_subscriptions.len()).rev()
543                                    {
544                                        let sender = runtime_subscription
545                                            .all_subscriptions
546                                            .swap_remove(index);
547                                        let notif = super::Notification::BestBlockChanged {
548                                            hash: parahash,
549                                        };
550                                        if sender.try_send(notif).is_ok() {
551                                            runtime_subscription.all_subscriptions.push(sender);
552                                        }
553                                    }
554                                }
555                            }
556
557                            async_tree::OutputUpdate::Block(block) => {
558                                // `block` borrows `async_tree`. We need to mutably access `async_tree`
559                                // below, so deconstruct `block` beforehand.
560                                let is_new_best = block.is_new_best;
561                                let block_index = block.index;
562                                let scale_encoded_header: Vec<u8> = runtime_subscription
563                                    .async_tree
564                                    .block_async_user_data(block.index)
565                                    .unwrap()
566                                    .clone()
567                                    .unwrap();
568                                let parahash =
569                                    header::hash_from_scale_encoded_header(&scale_encoded_header);
570
571                                // Do not report anything to subscriptions if no finalized parahead is
572                                // known yet.
573                                let finalized_parahead = match runtime_subscription
574                                    .async_tree
575                                    .output_finalized_async_user_data()
576                                {
577                                    Some(p) => p,
578                                    None => continue,
579                                };
580
581                                // Do not report the new block if it has already been reported in the
582                                // past. This covers situations where the parahead is identical to the
583                                // relay chain's parent's parahead, but also situations where multiple
584                                // sibling relay chain blocks have the same parahead.
585                                if *finalized_parahead == scale_encoded_header
586                                    || runtime_subscription
587                                        .async_tree
588                                        .input_output_iter_unordered()
589                                        .filter(|item| item.id != block_index)
590                                        .filter_map(|item| item.async_op_user_data)
591                                        .any(|item| item.as_ref() == Some(&scale_encoded_header))
592                                {
593                                    // While the parablock has already been reported, it is possible that
594                                    // it becomes the new best block while it wasn't before, in which
595                                    // case we should send a notification.
596                                    if is_new_best
597                                        && runtime_subscription.reported_best_parahead_hash.as_ref()
598                                            != Some(&parahash)
599                                    {
600                                        runtime_subscription.reported_best_parahead_hash =
601                                            Some(parahash);
602
603                                        // The networking service needs to be kept up to date with what the
604                                        // local node considers as the best block.
605                                        if let Ok(header) = header::decode(
606                                            finalized_parahead,
607                                            self.block_number_bytes,
608                                        ) {
609                                            self.network_service
610                                                .set_local_best_block(parahash, header.number)
611                                                .await;
612                                        }
613
614                                        log!(
615                                            &self.platform,
616                                            Debug,
617                                            &self.log_target,
618                                            "subscriptions-notify-best-block-changed",
619                                            hash = HashDisplay(&parahash)
620                                        );
621
622                                        // Elements in `all_subscriptions` are removed one by one and
623                                        // inserted back if the channel is still open.
624                                        for index in
625                                            (0..runtime_subscription.all_subscriptions.len()).rev()
626                                        {
627                                            let sender = runtime_subscription
628                                                .all_subscriptions
629                                                .swap_remove(index);
630                                            let notif = super::Notification::BestBlockChanged {
631                                                hash: parahash,
632                                            };
633                                            if sender.try_send(notif).is_ok() {
634                                                runtime_subscription.all_subscriptions.push(sender);
635                                            }
636                                        }
637                                    }
638
639                                    continue;
640                                }
641
642                                if is_new_best {
643                                    runtime_subscription.reported_best_parahead_hash =
644                                        Some(parahash);
645                                }
646
647                                let parent_hash = header::hash_from_scale_encoded_header(
648                                    runtime_subscription
649                                        .async_tree
650                                        .parent(block_index)
651                                        .map(|idx| {
652                                            runtime_subscription
653                                                .async_tree
654                                                .block_async_user_data(idx)
655                                                .unwrap()
656                                                .as_ref()
657                                                .unwrap()
658                                        })
659                                        .unwrap_or(finalized_parahead),
660                                );
661
662                                log!(
663                                    &self.platform,
664                                    Debug,
665                                    &self.log_target,
666                                    "subscriptions-notify-new-parablock",
667                                    hash = HashDisplay(&parahash),
668                                    parent_hash = HashDisplay(&parent_hash),
669                                    ?is_new_best
670                                );
671
672                                // Elements in `all_subscriptions` are removed one by one and
673                                // inserted back if the channel is still open.
674                                for index in (0..runtime_subscription.all_subscriptions.len()).rev()
675                                {
676                                    let sender =
677                                        runtime_subscription.all_subscriptions.swap_remove(index);
678                                    let notif =
679                                        super::Notification::Block(super::BlockNotification {
680                                            is_new_best,
681                                            parent_hash,
682                                            scale_encoded_header: scale_encoded_header.clone(),
683                                        });
684                                    if sender.try_send(notif).is_ok() {
685                                        runtime_subscription.all_subscriptions.push(sender);
686                                    }
687                                }
688                            }
689                        }
690                    }
691                }
692
693                (
694                    WakeUpReason::StartParaheadFetch,
695                    ParachainBackgroundState::Subscribed(runtime_subscription),
696                ) => {
697                    // Must start downloading a parahead.
698
699                    // Internal state check.
700                    debug_assert_eq!(
701                        runtime_subscription.reported_best_parahead_hash.is_some(),
702                        runtime_subscription
703                            .async_tree
704                            .output_finalized_async_user_data()
705                            .is_some()
706                    );
707
708                    // Limit the maximum number of simultaneous downloads.
709                    if runtime_subscription.in_progress_paraheads.len() >= 4 {
710                        continue;
711                    }
712
713                    match runtime_subscription
714                        .async_tree
715                        .next_necessary_async_op(&self.platform.now())
716                    {
717                        async_tree::NextNecessaryAsyncOp::NotReady { when: Some(when) } => {
718                            runtime_subscription.next_start_parahead_fetch =
719                                Box::pin(self.platform.sleep_until(when));
720                        }
721                        async_tree::NextNecessaryAsyncOp::NotReady { when: None } => {
722                            runtime_subscription.next_start_parahead_fetch =
723                                Box::pin(future::pending());
724                        }
725                        async_tree::NextNecessaryAsyncOp::Ready(op) => {
726                            log!(
727                                &self.platform,
728                                Debug,
729                                &self.log_target,
730                                "parahead-fetch-operation-started",
731                                relay_block_hash =
732                                    HashDisplay(&runtime_subscription.async_tree[op.block_index]),
733                            );
734
735                            runtime_subscription.in_progress_paraheads.push({
736                                let relay_chain_sync = self.relay_chain_sync.clone();
737                                let subscription_id =
738                                    runtime_subscription.relay_chain_subscribe_all.id();
739                                let block_hash = runtime_subscription.async_tree[op.block_index];
740                                let async_op_id = op.id;
741                                let parachain_id = self.parachain_id;
742                                Box::pin(async move {
743                                    (
744                                        async_op_id,
745                                        fetch_parahead(
746                                            &relay_chain_sync,
747                                            subscription_id,
748                                            parachain_id,
749                                            &block_hash,
750                                        )
751                                        .await,
752                                    )
753                                })
754                            });
755
756                            // There might be more downloads to start.
757                            runtime_subscription.next_start_parahead_fetch =
758                                Box::pin(future::ready(()));
759                        }
760                    }
761                }
762
763                (
764                    WakeUpReason::Notification(runtime_service::Notification::Finalized {
765                        hash,
766                        best_block_hash_if_changed,
767                        ..
768                    }),
769                    ParachainBackgroundState::Subscribed(runtime_subscription),
770                ) => {
771                    // Relay chain has a new finalized block.
772                    log!(
773                        &self.platform,
774                        Debug,
775                        &self.log_target,
776                        "relay-chain-block-finalized",
777                        hash = HashDisplay(&hash)
778                    );
779
780                    if let Some(best_block_hash_if_changed) = best_block_hash_if_changed {
781                        let best = runtime_subscription
782                            .async_tree
783                            .input_output_iter_unordered()
784                            .find(|b| *b.user_data == best_block_hash_if_changed)
785                            .unwrap()
786                            .id;
787                        runtime_subscription
788                            .async_tree
789                            .input_set_best_block(Some(best));
790                    }
791
792                    let finalized = runtime_subscription
793                        .async_tree
794                        .input_output_iter_unordered()
795                        .find(|b| *b.user_data == hash)
796                        .unwrap()
797                        .id;
798                    runtime_subscription.async_tree.input_finalize(finalized);
799                    runtime_subscription.must_process_sync_tree = true;
800                }
801
802                (
803                    WakeUpReason::Notification(runtime_service::Notification::Block(block)),
804                    ParachainBackgroundState::Subscribed(runtime_subscription),
805                ) => {
806                    // Relay chain has a new block.
807                    let hash = header::hash_from_scale_encoded_header(&block.scale_encoded_header);
808
809                    log!(
810                        &self.platform,
811                        Debug,
812                        &self.log_target,
813                        "relay-chain-new-block",
814                        hash = HashDisplay(&hash),
815                        parent_hash = HashDisplay(&block.parent_hash)
816                    );
817
818                    let parent = runtime_subscription
819                        .async_tree
820                        .input_output_iter_unordered()
821                        .find(|b| *b.user_data == block.parent_hash)
822                        .map(|b| b.id); // TODO: check if finalized
823                    runtime_subscription.async_tree.input_insert_block(
824                        hash,
825                        parent,
826                        false,
827                        block.is_new_best,
828                    );
829                    runtime_subscription.must_process_sync_tree = true;
830
831                    runtime_subscription.next_start_parahead_fetch = Box::pin(future::ready(()));
832                }
833
834                (
835                    WakeUpReason::Notification(runtime_service::Notification::BestBlockChanged {
836                        hash,
837                    }),
838                    ParachainBackgroundState::Subscribed(runtime_subscription),
839                ) => {
840                    // Relay chain has a new best block.
841                    log!(
842                        &self.platform,
843                        Debug,
844                        &self.log_target,
845                        "relay-chain-best-block-changed",
846                        hash = HashDisplay(&hash)
847                    );
848
849                    // If the block isn't found in `async_tree`, assume that it is equal to the
850                    // finalized block (that has left the tree already).
851                    let node_idx = runtime_subscription
852                        .async_tree
853                        .input_output_iter_unordered()
854                        .find(|b| *b.user_data == hash)
855                        .map(|b| b.id);
856                    runtime_subscription
857                        .async_tree
858                        .input_set_best_block(node_idx);
859
860                    runtime_subscription.must_process_sync_tree = true;
861                }
862
863                (WakeUpReason::SubscriptionDead, _) => {
864                    // Recreate the channel.
865                    log!(
866                        &self.platform,
867                        Debug,
868                        &self.log_target,
869                        "relay-chain-subscription-reset"
870                    );
871                    self.subscription_state = ParachainBackgroundState::NotSubscribed {
872                        all_subscriptions: Vec::new(),
873                        subscribe_future: {
874                            let relay_chain_sync = self.relay_chain_sync.clone();
875                            Box::pin(async move {
876                                relay_chain_sync
877                                    .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
878                                    .await
879                            })
880                        },
881                    };
882                }
883
884                (
885                    WakeUpReason::ParaheadFetchFinished {
886                        async_op_id,
887                        parahead_result: Ok(parahead),
888                    },
889                    ParachainBackgroundState::Subscribed(runtime_subscription),
890                ) => {
891                    // A parahead fetching operation is successful.
892                    log!(
893                        &self.platform,
894                        Debug,
895                        &self.log_target,
896                        "parahead-fetch-operation-success",
897                        parahead_hash = HashDisplay(
898                            blake2_rfc::blake2b::blake2b(32, b"", &parahead).as_bytes()
899                        ),
900                        relay_blocks = runtime_subscription
901                            .async_tree
902                            .async_op_blocks(async_op_id)
903                            .map(|b| HashDisplay(b))
904                            .join(",")
905                    );
906
907                    // Unpin the relay blocks whose parahead is now known.
908                    for block in runtime_subscription
909                        .async_tree
910                        .async_op_finished(async_op_id, Some(parahead))
911                    {
912                        let hash = &runtime_subscription.async_tree[block];
913                        runtime_subscription
914                            .relay_chain_subscribe_all
915                            .unpin_block(*hash)
916                            .await;
917                    }
918
919                    runtime_subscription.must_process_sync_tree = true;
920
921                    runtime_subscription.next_start_parahead_fetch = Box::pin(future::ready(()));
922                }
923
924                (
925                    WakeUpReason::ParaheadFetchFinished {
926                        parahead_result:
927                            Err(ParaheadError::PinRuntimeError(
928                                runtime_service::PinPinnedBlockRuntimeError::ObsoleteSubscription,
929                            )),
930                        ..
931                    },
932                    _,
933                ) => {
934                    // The relay chain runtime service has some kind of gap or issue and has
935                    // discarded the runtime.
936                    // Destroy the subscription and recreate the channels.
937                    log!(
938                        &self.platform,
939                        Debug,
940                        &self.log_target,
941                        "relay-chain-subscription-reset"
942                    );
943                    self.subscription_state = ParachainBackgroundState::NotSubscribed {
944                        all_subscriptions: Vec::new(),
945                        subscribe_future: {
946                            let relay_chain_sync = self.relay_chain_sync.clone();
947                            Box::pin(async move {
948                                relay_chain_sync
949                                    .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
950                                    .await
951                            })
952                        },
953                    };
954                }
955
956                (
957                    WakeUpReason::ParaheadFetchFinished {
958                        async_op_id,
959                        parahead_result: Err(error),
960                    },
961                    ParachainBackgroundState::Subscribed(runtime_subscription),
962                ) => {
963                    // Failed fetching a parahead.
964
965                    // Several relay chains initially didn't support parachains, and have later
966                    // been upgraded to support them. Similarly, the parachain might not have had a
967                    // core on the relay chain until recently. For these reasons, errors when the
968                    // relay chain is not near head of the chain are most likely normal and do
969                    // not warrant logging an error.
970                    // Note that `is_near_head_of_chain_heuristic` is normally not acceptable to
971                    // use due to being too vague, but since this is just about whether to print a
972                    // log message, it's completely fine.
973                    if self
974                        .relay_chain_sync
975                        .is_near_head_of_chain_heuristic()
976                        .await
977                        && !error.is_network_problem()
978                    {
979                        log!(
980                            &self.platform,
981                            Error,
982                            &self.log_target,
983                            format!(
984                                "Failed to fetch the parachain head from relay chain blocks {}: {}",
985                                runtime_subscription
986                                    .async_tree
987                                    .async_op_blocks(async_op_id)
988                                    .map(|b| HashDisplay(b))
989                                    .join(", "),
990                                error
991                            )
992                        );
993                    }
994
995                    log!(
996                        &self.platform,
997                        Debug,
998                        &self.log_target,
999                        "parahead-fetch-operation-error",
1000                        relay_blocks = runtime_subscription
1001                            .async_tree
1002                            .async_op_blocks(async_op_id)
1003                            .map(|b| HashDisplay(b))
1004                            .join(","),
1005                        ?error
1006                    );
1007
1008                    runtime_subscription
1009                        .async_tree
1010                        .async_op_failure(async_op_id, &self.platform.now());
1011
1012                    runtime_subscription.next_start_parahead_fetch = Box::pin(future::ready(()));
1013                }
1014
1015                (
1016                    WakeUpReason::ForegroundMessage(ToBackground::IsNearHeadOfChainHeuristic {
1017                        send_back,
1018                    }),
1019                    ParachainBackgroundState::Subscribed(sub),
1020                ) if sub.async_tree.output_finalized_async_user_data().is_some() => {
1021                    // Since there is a mapping between relay chain blocks and parachain blocks,
1022                    // whether a parachain is at the head of the chain is the same thing as whether
1023                    // its relay chain is at the head of the chain.
1024                    // Note that there is no ordering guarantee of any kind w.r.t. block
1025                    // subscriptions notifications.
1026                    let val = self
1027                        .relay_chain_sync
1028                        .is_near_head_of_chain_heuristic()
1029                        .await;
1030                    let _ = send_back.send(val);
1031                }
1032
1033                (
1034                    WakeUpReason::ForegroundMessage(ToBackground::IsNearHeadOfChainHeuristic {
1035                        send_back,
1036                    }),
1037                    _,
1038                ) => {
1039                    // If no finalized parahead is known yet, we might be very close to the head
1040                    // but also maybe very very far away. We lean on the cautious side and always
1041                    // return `false`.
1042                    let _ = send_back.send(false);
1043                }
1044
1045                (
1046                    WakeUpReason::ForegroundMessage(ToBackground::SubscribeAll {
1047                        send_back,
1048                        buffer_size,
1049                        ..
1050                    }),
1051                    ParachainBackgroundState::NotSubscribed {
1052                        all_subscriptions, ..
1053                    },
1054                ) => {
1055                    let (tx, new_blocks) = async_channel::bounded(buffer_size.saturating_sub(1));
1056
1057                    // No known finalized parahead.
1058                    let _ = send_back.send(super::SubscribeAll {
1059                        finalized_block_scale_encoded_header: self
1060                            .obsolete_finalized_parahead
1061                            .clone(),
1062                        finalized_block_runtime: None,
1063                        non_finalized_blocks_ancestry_order: Vec::new(),
1064                        new_blocks,
1065                    });
1066
1067                    all_subscriptions.push(tx);
1068                }
1069
1070                (
1071                    WakeUpReason::ForegroundMessage(ToBackground::SubscribeAll {
1072                        send_back,
1073                        buffer_size,
1074                        ..
1075                    }),
1076                    ParachainBackgroundState::Subscribed(runtime_subscription),
1077                ) => {
1078                    let (tx, new_blocks) = async_channel::bounded(buffer_size.saturating_sub(1));
1079
1080                    // There are two possibilities here: either we know of any recent finalized
1081                    // parahead, or we don't. In case where we don't know of any finalized parahead
1082                    // yet, we report a single obsolete finalized parahead, which is
1083                    // `obsolete_finalized_parahead`. The rest of this module makes sure that no
1084                    // other block is reported to subscriptions as long as this is the case, and
1085                    // that subscriptions are reset once the first known finalized parahead
1086                    // is known.
1087                    if let Some(finalized_parahead) = runtime_subscription
1088                        .async_tree
1089                        .output_finalized_async_user_data()
1090                    {
1091                        // Finalized parahead is known.
1092                        let finalized_parahash =
1093                            header::hash_from_scale_encoded_header(finalized_parahead);
1094                        let _ = send_back.send(super::SubscribeAll {
1095                            finalized_block_scale_encoded_header: finalized_parahead.clone(),
1096                            finalized_block_runtime: None,
1097                            non_finalized_blocks_ancestry_order: {
1098                                let mut list =
1099                                    Vec::<([u8; 32], super::BlockNotification)>::with_capacity(
1100                                        runtime_subscription
1101                                            .async_tree
1102                                            .num_input_non_finalized_blocks(),
1103                                    );
1104
1105                                for relay_block in runtime_subscription
1106                                    .async_tree
1107                                    .input_output_iter_ancestry_order()
1108                                {
1109                                    let parablock = match relay_block.async_op_user_data {
1110                                        Some(b) => b.as_ref().unwrap(),
1111                                        None => continue,
1112                                    };
1113
1114                                    let parablock_hash =
1115                                        header::hash_from_scale_encoded_header(parablock);
1116
1117                                    // TODO: O(n)
1118                                    if let Some((_, entry)) =
1119                                        list.iter_mut().find(|(h, _)| *h == parablock_hash)
1120                                    {
1121                                        // Block is already in the list. Don't add it a second time.
1122                                        if relay_block.is_output_best {
1123                                            entry.is_new_best = true;
1124                                        }
1125                                        continue;
1126                                    }
1127
1128                                    // Find the parent of the parablock. This is done by going through
1129                                    // the ancestors of the corresponding relay chain block (until and
1130                                    // including the finalized relay chain block) until we find one
1131                                    // whose parablock is different from the parablock in question.
1132                                    // If none is found, the parablock is the same as the finalized
1133                                    // parablock.
1134                                    let parent_hash = runtime_subscription
1135                                        .async_tree
1136                                        .ancestors(relay_block.id)
1137                                        .find_map(|idx| {
1138                                            let hash = header::hash_from_scale_encoded_header(
1139                                                runtime_subscription
1140                                                    .async_tree
1141                                                    .block_async_user_data(idx)
1142                                                    .unwrap()
1143                                                    .as_ref()
1144                                                    .unwrap(),
1145                                            );
1146                                            if hash != parablock_hash {
1147                                                Some(hash)
1148                                            } else {
1149                                                None
1150                                            }
1151                                        })
1152                                        .or_else(|| {
1153                                            if finalized_parahash != parablock_hash {
1154                                                Some(finalized_parahash)
1155                                            } else {
1156                                                None
1157                                            }
1158                                        });
1159
1160                                    // `parent_hash` is `None` if the parablock is
1161                                    // the same as the finalized parablock, in which case we
1162                                    // don't add it to the list.
1163                                    if let Some(parent_hash) = parent_hash {
1164                                        debug_assert!(
1165                                            list.iter().filter(|(h, _)| *h == parent_hash).count()
1166                                                == 1
1167                                                || parent_hash == finalized_parahash
1168                                        );
1169                                        list.push((
1170                                            parablock_hash,
1171                                            super::BlockNotification {
1172                                                is_new_best: relay_block.is_output_best,
1173                                                scale_encoded_header: parablock.clone(),
1174                                                parent_hash,
1175                                            },
1176                                        ));
1177                                    }
1178                                }
1179
1180                                list.into_iter().map(|(_, v)| v).collect()
1181                            },
1182                            new_blocks,
1183                        });
1184                    } else {
1185                        // No known finalized parahead.
1186                        let _ = send_back.send(super::SubscribeAll {
1187                            finalized_block_scale_encoded_header: self
1188                                .obsolete_finalized_parahead
1189                                .clone(),
1190                            finalized_block_runtime: None,
1191                            non_finalized_blocks_ancestry_order: Vec::new(),
1192                            new_blocks,
1193                        });
1194                    }
1195
1196                    runtime_subscription.all_subscriptions.push(tx);
1197                }
1198
1199                (
1200                    WakeUpReason::ForegroundMessage(ToBackground::PeersAssumedKnowBlock {
1201                        send_back,
1202                        ..
1203                    }),
1204                    _,
1205                ) => {
1206                    // Simply assume that all peers know about all blocks.
1207                    //
1208                    // We could in principle check whether the block is higher than the current
1209                    // finalized block, and if not if it is in the list of paraheads found in the
1210                    // relay chain. But because parachain blocks might not be decodable, we can't
1211                    // know their number, and thus we can't know if the requested block is a
1212                    // descendant of the finalized block.
1213                    // Assuming that all peers know all blocks is the only sane way of
1214                    // implementing this.
1215                    let _ = send_back.send(self.sync_sources.keys().cloned().collect());
1216                }
1217
1218                (WakeUpReason::ForegroundMessage(ToBackground::SyncingPeers { send_back }), _) => {
1219                    let _ = send_back.send(
1220                        self.sync_sources
1221                            .iter()
1222                            .map(|(peer_id, (role, best_height, best_hash))| {
1223                                //let (height, hash) = self.sync_sources.best_block(local_id);
1224                                (peer_id.clone(), *role, *best_height, *best_hash)
1225                            })
1226                            .collect(),
1227                    );
1228                }
1229
1230                (
1231                    WakeUpReason::ForegroundMessage(ToBackground::SerializeChainInformation {
1232                        send_back,
1233                    }),
1234                    _,
1235                ) => {
1236                    let _ = send_back.send(None);
1237                }
1238
1239                (WakeUpReason::MustSubscribeNetworkEvents, _) => {
1240                    debug_assert!(self.from_network_service.is_none());
1241                    self.sync_sources.clear();
1242                    self.from_network_service = Some(Box::pin(
1243                        // As documented, `subscribe().await` is expected to return quickly.
1244                        self.network_service.subscribe().await,
1245                    ));
1246                }
1247
1248                (
1249                    WakeUpReason::NetworkEvent(network_service::Event::Connected {
1250                        peer_id,
1251                        role,
1252                        best_block_number,
1253                        best_block_hash,
1254                    }),
1255                    _,
1256                ) => {
1257                    let _former_value = self
1258                        .sync_sources
1259                        .insert(peer_id, (role, best_block_number, best_block_hash));
1260                    debug_assert!(_former_value.is_none());
1261                }
1262
1263                (
1264                    WakeUpReason::NetworkEvent(network_service::Event::Disconnected { peer_id }),
1265                    _,
1266                ) => {
1267                    let _role = self.sync_sources.remove(&peer_id);
1268                    debug_assert!(_role.is_some());
1269                }
1270
1271                (
1272                    WakeUpReason::NetworkEvent(network_service::Event::BlockAnnounce {
1273                        peer_id: _peer_id,
1274                        ..
1275                    }),
1276                    _,
1277                ) => {
1278                    debug_assert!(self.sync_sources.contains_key(&_peer_id));
1279                }
1280
1281                (WakeUpReason::NetworkEvent(_), _) => {
1282                    // Uninteresting message.
1283                }
1284
1285                (
1286                    WakeUpReason::ParaheadFetchFinished { .. }
1287                    | WakeUpReason::AdvanceSyncTree
1288                    | WakeUpReason::Notification(_)
1289                    | WakeUpReason::StartParaheadFetch,
1290                    ParachainBackgroundState::NotSubscribed { .. },
1291                ) => {
1292                    // These paths are unreachable.
1293                    debug_assert!(false);
1294                }
1295            }
1296        }
1297    }
1298}
1299
1300async fn fetch_parahead<TPlat: PlatformRef>(
1301    relay_chain_sync: &Arc<runtime_service::RuntimeService<TPlat>>,
1302    subscription_id: runtime_service::SubscriptionId,
1303    parachain_id: u32,
1304    block_hash: &[u8; 32],
1305) -> Result<Vec<u8>, ParaheadError> {
1306    // Call `ParachainHost_persisted_validation_data` in order to know where the parachain is.
1307    let (pinned_runtime, block_state_trie_root, block_number) = relay_chain_sync
1308        .pin_pinned_block_runtime(subscription_id, *block_hash)
1309        .await
1310        .map_err(ParaheadError::PinRuntimeError)?;
1311    let success = relay_chain_sync
1312        .runtime_call(
1313            pinned_runtime,
1314            *block_hash,
1315            block_number,
1316            block_state_trie_root,
1317            para::PERSISTED_VALIDATION_FUNCTION_NAME.to_owned(),
1318            None, // TODO: /!\
1319            para::persisted_validation_data_parameters(
1320                parachain_id,
1321                para::OccupiedCoreAssumption::TimedOut,
1322            )
1323            .fold(Vec::new(), |mut a, b| {
1324                a.extend_from_slice(b.as_ref());
1325                a
1326            }),
1327            6,
1328            Duration::from_secs(10),
1329            NonZero::<u32>::new(2).unwrap(),
1330        )
1331        .await
1332        .map_err(ParaheadError::RuntimeCall)?;
1333
1334    // Try decode the result of the runtime call.
1335    // If this fails, it indicates an incompatibility between smoldot and the relay chain.
1336    match para::decode_persisted_validation_data_return_value(
1337        &success.output,
1338        relay_chain_sync.block_number_bytes(),
1339    ) {
1340        Ok(Some(pvd)) => Ok(pvd.parent_head.to_vec()),
1341        Ok(None) => Err(ParaheadError::NoCore),
1342        Err(error) => Err(ParaheadError::InvalidRuntimeOutput(error)),
1343    }
1344}
1345
1346/// Error that can happen when fetching the parachain head corresponding to a relay chain block.
1347#[derive(Debug, derive_more::Display, derive_more::Error)]
1348enum ParaheadError {
1349    /// Error while performing call request over the network.
1350    #[display("Error while performing call request over the network: {_0}")]
1351    RuntimeCall(runtime_service::RuntimeCallError),
1352    /// Error pinning the runtime of the block.
1353    PinRuntimeError(runtime_service::PinPinnedBlockRuntimeError),
1354    /// Parachain doesn't have a core in the relay chain.
1355    NoCore,
1356    /// Error while decoding the output of the call.
1357    ///
1358    /// This indicates some kind of incompatibility between smoldot and the relay chain.
1359    #[display("Error while decoding the output of the call: {_0}")]
1360    InvalidRuntimeOutput(para::Error),
1361}
1362
1363impl ParaheadError {
1364    /// Returns `true` if this is caused by networking issues, as opposed to a consensus-related
1365    /// issue.
1366    fn is_network_problem(&self) -> bool {
1367        match self {
1368            ParaheadError::RuntimeCall(runtime_service::RuntimeCallError::Inaccessible(_)) => true,
1369            ParaheadError::RuntimeCall(
1370                runtime_service::RuntimeCallError::Execution(_)
1371                | runtime_service::RuntimeCallError::Crash
1372                | runtime_service::RuntimeCallError::InvalidRuntime(_)
1373                | runtime_service::RuntimeCallError::ApiVersionRequirementUnfulfilled,
1374            ) => false,
1375            ParaheadError::PinRuntimeError(_) => false,
1376            ParaheadError::NoCore => false,
1377            ParaheadError::InvalidRuntimeOutput(_) => false,
1378        }
1379    }
1380}
