18//! Background transactions service.
19//!
20//! The role of the [`TransactionsService`] is to manage the transactions that the user wants to
21//! send out, and report about their status.
22//!
23//! The [`TransactionsService`] is most of the time idle. When the user wants to emit a
24//! transaction on the network, it gets reported to the service, which then tries to send it to
25//! the peers the node is currently connected to. Afterwards, the service will inspect the stream
26//! of best and finalized blocks to find out whether the transaction has been included or not.
27//!
28//! # How watching transactions works
29//!
30//! Calling [`TransactionsService::submit_transaction`] returns a channel receiver that will contain
31//! status updates about this transaction.
32//!
33//! In order to implement this, the [`TransactionsService`] will follow all the blocks that are
34//! verified locally by the [`runtime_service::RuntimeService`] (see
35//! [`runtime_service::RuntimeService::subscribe_all`]) and download from the network the body of
36//! all the blocks in the best chain.
37//!
38//! When a block body download fails, it is ignored, in the hopes that the block will not be part
39//! of the finalized chain. If the block body download of a finalized block fails, we enter "panic
40//! mode" (not an actual Rust panic, just a way to describe the logic) and all watched
41//! transactions are dropped.
42//!
43//! The same "panic mode" happens if there's an accidental gap in the chain, which will typically
44//! happen if the [`runtime_service::RuntimeService`] is overwhelmed.
45//!
46//! If the channel returned by [`TransactionsService::submit_transaction`] is full, it will
47//! automatically be closed so as to not block the transactions service if the receive is too slow
48//! to be processed.
49//!
50//! # About duplicate unsigned transactions
51//!
52//! The Substrate and Polkadot runtimes support nonce-less unsigned transactions. In other words,
53//! a user can submit the same transaction (the exact same bytes every time) as many time as they
54//! want.
55//!
56//! While the chain can accept the same transaction multiple times over time, a Substrate node
57//! will only allow submitting it *once at a time*. In other words, any given unsigned transaction
58//! will never be included more than once in any given block. If you try to submit an unsigned
59//! transaction while the same transaction is already pending, the Substrate node will ignore it
60//! or return an error.
61//!
62//! Contrary to Substrate, the smoldot Wasm client can be used by multiple UIs at the same time.
63//! When a UI submits an unsigned transaction, we don't want to do the same as Substrate and
64//! refuse it if it is already pending, as it would make it possible for a UI to determine
65//! whether another UI has already submitted this transaction, and thus allow communications
66//! between UIs. Instead, the smoldot Wasm client return another sender to the same already-pending
67//! transaction.
68//!

96    /// Name of the chain, for logging purposes.
97    ///
98    /// > **Note**: This name will be directly printed out. Any special character should already
99    /// >           have been filtered out from this name.
100    pub log_name: String,
101
102    /// Access to the platform's capabilities.
103    pub platform: TPlat,
104
105    /// Service responsible for synchronizing the chain.
106    pub sync_service: Arc<sync_service::SyncService<TPlat>>,
107
108    /// Service responsible for synchronizing the chain.
109    pub runtime_service: Arc<runtime_service::RuntimeService<TPlat>>,
110
111    /// Access to the network, and identifier of the chain to use to gossip transactions from the
112    /// point of view of the network service.
113    pub network_service: Arc<network_service::NetworkServiceChain<TPlat>>,
114
115    /// Maximum number of pending transactions allowed in the service.
116    ///
117    /// Any extra transaction will lead to [`DropReason::MaxPendingTransactionsReached`].
118    pub max_pending_transactions: NonZero<u32>,

120    /// Maximum number of block body downloads that can be performed in parallel.
121    ///
122    /// > **Note**: This is the maximum number of *blocks* whose body is being download, not the
123    /// >           number of block requests emitted on the network.
124    pub max_concurrent_downloads: NonZero<u32>,
125
126    /// Maximum number of transaction validations that can be performed in parallel.
127    pub max_concurrent_validations: NonZero<u32>,
128}

130/// See [the module-level documentation](..).
131pub struct TransactionsService<TPlat: PlatformRef> {
132    /// Sending messages to the background task.
133    to_background: async_lock::Mutex<async_channel::Sender<ToBackground>>,
134
135    /// Configuration of the background task. Used in order to restart it if necessary.
136    background_task_config: BackgroundTaskConfig<TPlat>,
137}
138
139impl<TPlat: PlatformRef> TransactionsService<TPlat> {
140    /// Builds a new service.
141    pub fn new(config: Config<TPlat>) -> Self {
142        let log_target = format!("tx-service-{}", config.log_name);
143        let (to_background, from_foreground) = async_channel::bounded(8);
144
145        let background_task_config = BackgroundTaskConfig {
146            log_target: log_target.clone(),
147            platform: config.platform.clone(),
148            sync_service: config.sync_service,
149            runtime_service: config.runtime_service,
150            network_service: config.network_service,
151            max_concurrent_downloads: usize::try_from(config.max_concurrent_downloads.get())
152                .unwrap_or(usize::MAX),
153            max_pending_transactions: usize::try_from(config.max_pending_transactions.get())
154                .unwrap_or(usize::MAX),
155            max_concurrent_validations: usize::try_from(config.max_concurrent_validations.get())
156                .unwrap_or(usize::MAX),
157        };
158
159        let task = Box::pin(background_task::<TPlat>(
160            background_task_config.clone(),
161            from_foreground,
162        ));
163
164        config.platform.spawn_task(log_target.clone().into(), {
165            let platform = config.platform.clone();
166            async move {
167                task.await;
168                log!(&platform, Debug, &log_target, "shutdown");
169            }
170        });
171
172        TransactionsService {
173            to_background: async_lock::Mutex::new(to_background),
174            background_task_config,
175        }
176    }
177
178    /// Adds a transaction to the service. The service will try to send it out as soon as
179    /// possible.
180    ///
181    /// Must pass as parameter the SCALE-encoded transaction.
182    ///
183    /// The return value of this method is an object receives updates on the state of the
184    /// transaction.
185    ///
186    /// If `detached` is `true`, then dropping the value returned does not cancel sending out
187    /// the transaction. If `detached` is `false`, then it does.
188    ///
189    /// If this exact same transaction has already been submitted before, the transaction isn't
190    /// added a second time. Instead, a second channel is created pointing to the already-existing
191    /// transaction.
192    pub async fn submit_and_watch_transaction(
193        &self,
194        transaction_bytes: Vec<u8>,
195        channel_size: usize,
196        detached: bool,
197    ) -> TransactionWatcher {
198        let (updates_report, rx) = async_channel::bounded(channel_size);
199
200        self.send_to_background(ToBackground::SubmitTransaction {
201            transaction_bytes,
202            updates_report: Some((updates_report, detached)),
203        })
204        .await;
205
206        TransactionWatcher {
207            rx,
208            has_yielded_drop_reason: false,
209            _dummy_keep_alive: self.to_background.lock().await.clone(),
210        }
211    }
212
213    /// Similar to [`TransactionsService::submit_and_watch_transaction`], but doesn't return any
214    /// channel.
215    pub async fn submit_transaction(&self, transaction_bytes: Vec<u8>) {
216        self.send_to_background(ToBackground::SubmitTransaction {
217            transaction_bytes,
218            updates_report: None,
219        })
220        .await;
221    }
222
223    async fn send_to_background(&self, message: ToBackground) {
224        let mut lock = self.to_background.lock().await;
225
226        if lock.is_closed() {
227            let log_target = self.background_task_config.log_target.clone();
228            let (tx, rx) = async_channel::bounded(8);
229            let platform = self.background_task_config.platform.clone();
230            let task = background_task(self.background_task_config.clone(), rx);
231            self.background_task_config.platform.spawn_task(
232                log_target.clone().into(),
233                async move {
234                    // Sleep for a bit in order to avoid potential infinite loops
235                    // of repeated crashing.
236                    platform.sleep(Duration::from_secs(2)).await;
237                    log!(&platform, Debug, &log_target, "restart");
238                    task.await;
239                    log!(&platform, Debug, &log_target, "shutdown");
240                },
241            );
242            *lock = tx;
243        }
244
245        // Note that the background task might have crashed already at this point, so errors can
246        // be expected.
247        let _ = lock.send(message).await;
248    }
249}

251/// Returned by [`TransactionsService::submit_and_watch_transaction`].
252#[pin_project::pin_project]
253pub struct TransactionWatcher {
254    /// Channel connected to the background task.
255    #[pin]
256    rx: async_channel::Receiver<TransactionStatus>,
257    /// `true` if a [`TransactionStatus::Dropped`] has already been yielded.
258    has_yielded_drop_reason: bool,
259    /// Dummy copy of [`TransactionsService::to_background`] that guarantees that the background
260    /// stays alive.
261    _dummy_keep_alive: async_channel::Sender<ToBackground>,
262}
263
264impl TransactionWatcher {
265    /// Returns the next status update of the transaction.
266    ///
267    /// The last event is always a [`TransactionStatus::Dropped`], and then `None` is yielded
268    /// repeatedly forever.
269    pub async fn next(self: pin::Pin<&mut Self>) -> Option<TransactionStatus> {
270        let mut this = self.project();
271        if *this.has_yielded_drop_reason {
272            debug_assert!(this.rx.is_closed() || this.rx.next().await.is_none());
273            return None;
274        }
275
276        match this.rx.next().await {
277            Some(update) => {
278                if matches!(update, TransactionStatus::Dropped(_)) {
279                    debug_assert!(!*this.has_yielded_drop_reason);
280                    *this.has_yielded_drop_reason = true;
281                }
282                Some(update)
283            }
284            None => {
285                *this.has_yielded_drop_reason = true;
286                Some(TransactionStatus::Dropped(DropReason::Crashed))
287            }
288        }
289    }
290}

292/// Update on the state of a transaction in the service.
293///
294/// > **Note**: Because this code isn't an *actual* transactions pool that leverages the runtime,
295/// >           some variants (e.g. `Invalid`) are missing compared to the ones that can be found
296/// >           in Substrate, as they can't possibly be generated by this implementation.
297/// >           Additionally, an equivalent to the `Ready` state in Substrate is missing as it
298/// >           is the default state.
299#[derive(Debug, Clone)]
300pub enum TransactionStatus {
301    /// Transaction has been broadcasted to the given peers.
302    Broadcast(Vec<PeerId>),
303
304    /// Transaction is now known to be valid. If it ever becomes invalid in the future, a
305    /// [`TransactionStatus::Dropped`] will be generated.
306    Validated,
307
308    /// The block in which a block is included has changed.
309    IncludedBlockUpdate {
310        /// If `Some`, the transaction is included in the block of the best chain with the given
311        /// hash and at the given index. If `None`, the transaction isn't present in the best
312        /// chain.
313        block_hash: Option<([u8; 32], u32)>,
314    },
315
316    /// Transaction has been removed from the pool.
317    ///
318    /// This is always the last message sent back by the channel reporting the status.
319    Dropped(DropReason),
320}
321
322/// See [`TransactionStatus::Dropped`].
323#[derive(Debug, Clone)]
324pub enum DropReason {
325    /// Transaction has been included in a finalized block.
326    ///
327    /// This is a success path.
328    Finalized { block_hash: [u8; 32], index: u32 },
329
330    /// Transaction has been dropped because there was a gap in the chain of blocks. It is
331    /// impossible to know.
332    GapInChain,
333
334    /// Transaction has been dropped because the maximum number of transactions in the pool has
335    /// been reached.
336    MaxPendingTransactionsReached,
337
338    /// Transaction has been dropped because it is invalid.
339    Invalid(validate::TransactionValidityError),
340
341    /// Transaction has been dropped because we have failed to validate it.
342    ValidateError(ValidateTransactionError),
343
344    /// Transaction service background task has crashed.
345    Crashed,
346}
347
348/// Failed to check the validity of a transaction.
349#[derive(Debug, derive_more::Display, derive_more::Error, Clone)]
350pub enum ValidateTransactionError {
351    /// The runtime of the requested block is invalid.
352    InvalidRuntime(runtime_service::RuntimeError),
353
354    /// The runtime doesn't implement the API required to validate transactions.
355    ApiVersionRequirementUnfulfilled,
356
357    /// Runtime service has crashed while the call was in progress.
358    Crash,
359
360    /// Error during the execution of the runtime.
361    ///
362    /// There is no point in trying to validate the transaction call again, as it would result
363    /// in the same error.
364    #[display("Error during the execution of the runtime: {_0}")]
365    Execution(runtime_service::RuntimeCallExecutionError),
366
367    /// Error trying to access the storage required for the runtime call.
368    ///
369    /// Because these errors are non-fatal, the operation is attempted multiple times, and as such
370    /// there can be multiple errors.
371    ///
372    /// Trying the same transaction again might succeed.
373    #[display("Error trying to access the storage required for the runtime call")]
374    // TODO: better display?
375    Inaccessible(#[error(not(source))] Vec<runtime_service::RuntimeCallInaccessibleError>),
376
377    /// Error while decoding the output of the runtime.
378    OutputDecodeError(validate::DecodeError),
379}

401/// Configuration for [`background_task`].

403struct BackgroundTaskConfig<TPlat: PlatformRef> {
404    log_target: String,
405    platform: TPlat,
406    sync_service: Arc<sync_service::SyncService<TPlat>>,
407    runtime_service: Arc<runtime_service::RuntimeService<TPlat>>,
408    network_service: Arc<network_service::NetworkServiceChain<TPlat>>,
409    max_concurrent_downloads: usize,
410    max_pending_transactions: usize,
411    max_concurrent_validations: usize,
412}

414/// Background task running in parallel of the front service.
415async fn background_task<TPlat: PlatformRef>(
416    config: BackgroundTaskConfig<TPlat>,
417    from_foreground: async_channel::Receiver<ToBackground>,
418) {
419    let transactions_capacity = cmp::min(8, config.max_pending_transactions);
420    let blocks_capacity = 32;
421    let mut from_foreground = pin::pin!(from_foreground);
422
423    let mut worker = Worker {
424        platform: config.platform,
425        sync_service: config.sync_service,
426        runtime_service: config.runtime_service,
427        network_service: config.network_service,
428        pending_transactions: light_pool::LightPool::new(light_pool::Config {
429            transactions_capacity,
430            blocks_capacity,
431            finalized_block_hash: [0; 32], // Dummy value. Pool is re-initialized below.
432        }),
433        block_downloads: FuturesUnordered::new(),
434        validations_in_progress: FuturesUnordered::new(),
435        next_reannounce: FuturesUnordered::new(),
436        max_concurrent_downloads: config.max_concurrent_downloads,
437        max_pending_transactions: config.max_pending_transactions,
438    };
439
440    // TODO: must periodically re-send transactions that aren't included in block yet
441
442    'channels_rebuild: loop {
443        // This loop is entered when it is necessary to rebuild the subscriptions with the runtime
444        // service. This happens when there is a gap in the blocks, either intentionally (e.g.
445        // after a Grandpa warp sync) or because the transactions service was too busy to process
446        // the new blocks.
447        let mut subscribe_all = {
448            let sub_future = async {
449                Some(
450                    // The buffer size should be large enough so that, if the CPU is busy, it
451                    // doesn't become full before the execution of the transactions service resumes.
452                    // The maximum number of pinned block is ignored, as this maximum is a way to
453                    // avoid malicious behaviors. This code is by definition not considered
454                    // malicious.
455                    worker
456                        .runtime_service
457                        .subscribe_all(32, NonZero::<usize>::new(usize::MAX).unwrap())
458                        .await,
459                )
460            };
461
462            // Because `runtime_service.subscribe_all()` might take a long time (potentially
463            // forever), we need to process messages coming from the foreground in parallel.
464            let from_foreground = &mut from_foreground;
465            let messages_process = async move {
466                loop {
467                    match from_foreground.next().await {
468                        Some(ToBackground::SubmitTransaction {
469                            updates_report: Some(updates_report),
470                            ..
471                        }) => {
472                            let _ = updates_report
473                                .0
474                                .send(TransactionStatus::Dropped(DropReason::GapInChain))
475                                .await;
476                        }
477                        Some(ToBackground::SubmitTransaction { .. }) => {}
478                        None => break None,
479                    }
480                }
481            };
482
483            match sub_future.or(messages_process).await {
484                Some(s) => s,
485                None => return,
486            }
487        };
488
489        let initial_finalized_block_hash = header::hash_from_scale_encoded_header(
490            &subscribe_all.finalized_block_scale_encoded_header,
491        );
492
493        // Drop all pending transactions of the pool.
494        for (_, pending) in worker.pending_transactions.transactions_iter_mut() {
495            // TODO: only do this if transaction hasn't been validated yet
496            pending.update_status(TransactionStatus::Dropped(DropReason::GapInChain));
497        }
498
499        // Reset the blocks tracking state machine.
500        let dropped_transactions = worker
501            .pending_transactions
502            .transactions_iter()
503            .map(|(tx_id, _)| {
504                HashDisplay(worker.pending_transactions.scale_encoding(tx_id).unwrap())
505            })
506            .join(",");
507        worker.pending_transactions = light_pool::LightPool::new(light_pool::Config {
508            transactions_capacity,
509            blocks_capacity,
510            finalized_block_hash: initial_finalized_block_hash,
511        });
512
513        for block in subscribe_all.non_finalized_blocks_ancestry_order {
514            let hash = header::hash_from_scale_encoded_header(&block.scale_encoded_header);
515            worker.pending_transactions.add_block(
516                hash,
517                &block.parent_hash,
518                Block {
519                    scale_encoded_header: block.scale_encoded_header,
520                    failed_downloads: 0,
521                    downloading: false,
522                },
523            );
524            if block.is_new_best {
525                worker.set_best_block(&config.log_target, &hash);
526            }
527        }
528
529        // Reset the other fields.
530        worker.block_downloads.clear();
531        worker.validations_in_progress.clear();
532        worker.next_reannounce.clear();
533
534        log!(
535            &worker.platform,
536            Debug,
537            &config.log_target,
538            "reset",
539            new_finalized = HashDisplay(&initial_finalized_block_hash),
540            subscription_id = ?subscribe_all.new_blocks.id(),
541            dropped_transactions
542        );
543
544        loop {
545            // If the finalized block moved in such a way that there would be blocks in the
546            // pool whose height is inferior to `latest_finalized - 32`, then jump to
547            // "catastrophic mode" and reset everything. This is to avoid the possibility of an
548            // unreasonable memory consumption.
549            if worker.pending_transactions.oldest_block_finality_lag() >= 32 {
550                continue 'channels_rebuild;
551            }
552
553            // Try to find transactions whose status update channels have all been closed.
554            while let Some(tx_id) = {
555                let id = worker
556                    .pending_transactions
557                    .transactions_iter()
558                    .find(|(_, tx)| {
559                        !tx.status_update.iter().any(|s| !s.is_closed()) && !tx.detached
560                    })
561                    .map(|(id, _)| id);
562                id
563            } {
564                worker.pending_transactions.remove_transaction(tx_id);
565            }
566
567            // Start the validation process of transactions that need to be validated.
568            while worker.validations_in_progress.len() < config.max_concurrent_validations {
569                // Find a transaction that needs to be validated.
570                //
571                // While this is an `O(n)` process, in practice we pick the first transaction not
572                // currently being validated, and only `max_concurrent_validations` transactions
573                // in the list don't match that criteria. Since `max_concurrent_validations`
574                // should be pretty low, this search should complete very quickly.
575                let to_start_validate = worker
576                    .pending_transactions
577                    .unvalidated_transactions()
578                    .find(|(_, tx)| tx.validation_in_progress.is_none())
579                    .map(|(tx_id, ..)| tx_id);
580                let to_start_validate = match to_start_validate {
581                    Some(tx_id) => tx_id,
582                    None => break,
583                };
584
585                // Create the `Future` of the validation process.
586                let validation_future = {
587                    // Find which block to validate the transaction against.
588                    let block_hash = *worker.pending_transactions.best_block_hash();
589
590                    // It is possible for the current best block to be equal to the finalized
591                    // block, in which case it will not be in the data structure and will already
592                    // be unpinned in the runtime service.
593                    // In that situation, we simply don't start any validation.
594                    // TODO: is this problem worth solving? ^
595                    let scale_encoded_header =
596                        match worker.pending_transactions.block_user_data(&block_hash) {
597                            Some(b) => b.scale_encoded_header.clone(),
598                            None => break,
599                        };
600
601                    // Make copies of everything in order to move the values into the future.
602                    let runtime_service = worker.runtime_service.clone();
603                    let platform = worker.platform.clone();
604                    let log_target = config.log_target.clone();
605                    let relay_chain_sync_subscription_id = subscribe_all.new_blocks.id();
606                    let scale_encoded_transaction = worker
607                        .pending_transactions
608                        .scale_encoding(to_start_validate)
609                        .unwrap()
610                        .to_owned();
611                    // TODO: race condition /!\ the block could be pruned and unpinned before this future starts executing
612                    async move {
613                        let result = validate_transaction(
614                            &platform,
615                            &log_target,
616                            &runtime_service,
617                            relay_chain_sync_subscription_id,
618                            block_hash,
619                            &scale_encoded_header,
620                            scale_encoded_transaction,
621                            validate::TransactionSource::External,
622                        )
623                        .await;
624                        (block_hash, result)
625                    }
626                };
627
628                // The future that will receive the validation result is stored in the
629                // `PendingTransaction`, while the future that executes the validation (and
630                // yields `()`) is stored in `validations_in_progress`.
631                let (result_tx, result_rx) = oneshot::channel();
632                worker
633                    .validations_in_progress
634                    .push(Box::pin(validation_future.map(move |result| {
635                        let _ = result_tx.send(result);
636                        to_start_validate
637                    })));
638                let tx = worker
639                    .pending_transactions
640                    .transaction_user_data_mut(to_start_validate)
641                    .unwrap();
642                debug_assert!(tx.validation_in_progress.is_none());
643                tx.validation_in_progress = Some(result_rx);
644            }
645
646            // Remove transactions that have been determined to be invalid.
647            loop {
648                // Note that we really would like to use a `while let` loop, but the Rust borrow
649                // checker doesn't permit it.
650                let (tx_id, _, error) = match worker
651                    .pending_transactions
652                    .invalid_transactions_finalized_block()
653                    .next()
654                {
655                    Some(v) => v,
656                    None => break,
657                };
658
659                // Clone the error because we need to unborrow `worker.pending_transactions`.
660                let error = error.clone();
661
662                let (tx_body, mut transaction) =
663                    worker.pending_transactions.remove_transaction(tx_id);
664
665                log!(
666                    &worker.platform,
667                    Debug,
668                    &config.log_target,
669                    "discarded",
670                    tx_hash = HashDisplay(&blake2_hash(&tx_body)),
671                    ?error
672                );
673
674                transaction.update_status(TransactionStatus::Dropped(match error {
675                    InvalidOrError::Invalid(err) => DropReason::Invalid(err),
676                    InvalidOrError::ValidateError(err) => DropReason::ValidateError(err),
677                }));
678            }
679
680            // Start block bodies downloads that need to be started.
681            while worker.block_downloads.len() < worker.max_concurrent_downloads {
682                // TODO: prioritize best chain?
683                let block_hash_number = worker
684                    .pending_transactions
685                    .missing_block_bodies()
686                    .find(|(_, block)| {
687                        // The transaction pool isn't aware of the fact that we're currently
688                        // downloading a block's body. Skip when that is the case.
689                        if block.downloading {
690                            return false;
691                        }
692
693                        // Don't try again block downloads that have failed before.
694                        if block.failed_downloads >= 1 {
695                            // TODO: try downloading again if finalized or best chain
696                            return false;
697                        }
698
699                        true
700                    })
701                    .map(|(hash, block)| {
702                        // TODO: unwrap?! should only insert valid blocks in the worker
703                        let decoded = header::decode(
704                            &block.scale_encoded_header,
705                            worker.sync_service.block_number_bytes(),
706                        )
707                        .unwrap();
708                        (*hash, decoded.number)
709                    });
710                let (block_hash, block_number) = match block_hash_number {
711                    Some(b) => b,
712                    None => break,
713                };
714
715                // Actual download start.
716                worker.block_downloads.push({
717                    let download_future = worker.sync_service.clone().block_query(
718                        block_number,
719                        block_hash,
720                        codec::BlocksRequestFields {
721                            body: true,
722                            header: true, // TODO: must be true in order to avoid an error being generated, fix this in sync service
723                            justifications: false,
724                        },
725                        3,
726                        Duration::from_secs(8),
727                        NonZero::<u32>::new(3).unwrap(),
728                    );
729
730                    Box::pin(async move {
731                        (
732                            block_hash,
733                            download_future.await.and_then(|b| b.body.ok_or(())),
734                        )
735                    })
736                });
737
738                worker
739                    .pending_transactions
740                    .block_user_data_mut(&block_hash)
741                    .unwrap()
742                    .downloading = true;
743
744                log!(
745                    &worker.platform,
746                    Debug,
747                    &config.log_target,
748                    "blocks-download-started",
749                    block = HashDisplay(&block_hash)
750                );
751            }
752
753            // Remove finalized blocks from the pool when possible.
754            for block in worker.pending_transactions.prune_finalized_with_body() {
755                log!(
756                    &worker.platform,
757                    Debug,
758                    &config.log_target,
759                    "finalized",
760                    block = HashDisplay(&block.block_hash),
761                    body_transactions = block
762                        .included_transactions
763                        .iter()
764                        .map(|tx| HashDisplay(&blake2_hash(&tx.scale_encoding)).to_string())
765                        .join(", ")
766                );
767
768                // All blocks in `pending_transactions` are pinned within the runtime service.
769                // Unpin them when they're removed.
770                subscribe_all.new_blocks.unpin_block(block.block_hash).await;
771
772                debug_assert!(!block.user_data.downloading);
773                for mut tx in block.included_transactions {
774                    // We assume that there's no more than 2<<32 transactions per block.
775                    let body_index = u32::try_from(tx.index_in_block).unwrap();
776                    tx.user_data
777                        .update_status(TransactionStatus::Dropped(DropReason::Finalized {
778                            block_hash: block.block_hash,
779                            index: body_index,
780                        }));
781                    // `tx` is no longer in the pool.
782                }
783            }
784
785            // Yield at every loop in order to provide better tasks granularity.
786            futures_lite::future::yield_now().await;
787
788            enum WakeUpReason {
789                Notification(Option<runtime_service::Notification>),
790                BlockDownloadFinished([u8; 32], Result<Vec<Vec<u8>>, ()>),
791                MustMaybeReannounce(light_pool::TransactionId),
792                MaybeValidated(light_pool::TransactionId),
793                ForegroundMessage(Option<ToBackground>),
794            }
795
796            let wake_up_reason: WakeUpReason = {
797                async { WakeUpReason::Notification(subscribe_all.new_blocks.next().await) }
798                    .or(async {
799                        if !worker.block_downloads.is_empty() {
800                            let (block_hash, result) =
801                                worker.block_downloads.select_next_some().await;
802                            WakeUpReason::BlockDownloadFinished(block_hash, result)
803                        } else {
804                            future::pending().await
805                        }
806                    })
807                    .or(async {
808                        if !worker.next_reannounce.is_empty() {
809                            WakeUpReason::MustMaybeReannounce(
810                                worker.next_reannounce.select_next_some().await,
811                            )
812                        } else {
813                            future::pending().await
814                        }
815                    })
816                    .or(async {
817                        if !worker.validations_in_progress.is_empty() {
818                            WakeUpReason::MaybeValidated(
819                                worker.validations_in_progress.select_next_some().await,
820                            )
821                        } else {
822                            future::pending().await
823                        }
824                    })
825                    .or(async { WakeUpReason::ForegroundMessage(from_foreground.next().await) })
826                    .await
827            };
828
829            match wake_up_reason {
830                WakeUpReason::Notification(Some(runtime_service::Notification::Block(
831                    new_block,
832                ))) => {
833                    let hash =
834                        header::hash_from_scale_encoded_header(&new_block.scale_encoded_header);
835                    worker.pending_transactions.add_block(
836                        header::hash_from_scale_encoded_header(&new_block.scale_encoded_header),
837                        &new_block.parent_hash,
838                        Block {
839                            scale_encoded_header: new_block.scale_encoded_header,
840                            failed_downloads: 0,
841                            downloading: false,
842                        },
843                    );
844                    if new_block.is_new_best {
845                        worker.set_best_block(&config.log_target, &hash);
846                    }
847                }
848                WakeUpReason::Notification(Some(runtime_service::Notification::Finalized {
849                    hash,
850                    best_block_hash_if_changed,
851                    ..
852                })) => {
853                    if let Some(best_block_hash_if_changed) = best_block_hash_if_changed {
854                        worker.set_best_block(&config.log_target, &best_block_hash_if_changed);
855                    }
856                    for pruned in worker.pending_transactions.set_finalized_block(&hash) {
857                        log!(
858                            &worker.platform,
859                            Debug,
860                            &config.log_target,
861                            "pruned-block-discard",
862                            block = HashDisplay(&pruned.0),
863                        );
864
865                        // All blocks in `pending_transactions` are pinned within the
866                        // runtime service. Unpin them when they're removed.
867                        subscribe_all.new_blocks.unpin_block(pruned.0).await;
868
869                        // Note that we could in principle interrupt any on-going
870                        // download of that block, but it is not worth the effort.
871                    }
872                }
873                WakeUpReason::Notification(Some(
874                    runtime_service::Notification::BestBlockChanged { hash },
875                )) => {
876                    worker.set_best_block(&config.log_target, &hash);
877                }
878                WakeUpReason::Notification(None) => continue 'channels_rebuild,
879
880                WakeUpReason::BlockDownloadFinished(block_hash, mut block_body) => {
881                    // A block body download has finished, successfully or not.
882                    let block = match worker.pending_transactions.block_user_data_mut(&block_hash) {
883                        Some(b) => b,
884                        None => {
885                            // It is possible that this block has been discarded because a sibling
886                            // or uncle has been finalized. This is a normal situation.
887                            continue;
888                        }
889                    };
890
891                    debug_assert!(block.downloading);
892                    block.downloading = false;
893
894                    // Make sure that the downloaded body is the one of this block, otherwise
895                    // we consider the download as failed.
896                    if let Ok(body) = &block_body {
897                        // TODO: unwrap the decoding?! is that correct?
898                        if header::extrinsics_root(body)
899                            != *header::decode(
900                                &block.scale_encoded_header,
901                                worker.sync_service.block_number_bytes(),
902                            )
903                            .unwrap()
904                            .extrinsics_root
905                        {
906                            block_body = Err(());
907                        }
908                    }
909
910                    if let Ok(block_body) = block_body {
911                        let block_body_size = block_body.len();
912                        let included_transactions = worker
913                            .pending_transactions
914                            .set_block_body(&block_hash, block_body.into_iter())
915                            .collect::<Vec<_>>();
916
917                        log!(
918                            &worker.platform,
919                            Debug,
920                            &config.log_target,
921                            "blocks-download-success",
922                            block = HashDisplay(&block_hash),
923                            included_transactions = included_transactions
924                                .iter()
925                                .map(|(id, _)| HashDisplay(&blake2_hash(
926                                    worker.pending_transactions.scale_encoding(*id).unwrap()
927                                ))
928                                .to_string())
929                                .join(", ")
930                        );
931
932                        for (tx_id, body_index) in included_transactions {
933                            debug_assert!(body_index < block_body_size);
934                            let tx = worker
935                                .pending_transactions
936                                .transaction_user_data_mut(tx_id)
937                                .unwrap();
938                            // We assume that there's no more than 2<<32 transactions per block.
939                            let body_index = u32::try_from(body_index).unwrap();
940                            tx.update_status(TransactionStatus::IncludedBlockUpdate {
941                                block_hash: Some((block_hash, body_index)),
942                            });
943                        }
944                    } else {
945                        block.failed_downloads = block.failed_downloads.saturating_add(1);
946                        log!(
947                            &worker.platform,
948                            Debug,
949                            &config.log_target,
950                            "blocks-download-failure",
951                            block = HashDisplay(&block_hash)
952                        );
953                    }
954                }
955
956                WakeUpReason::MustMaybeReannounce(maybe_reannounce_tx_id) => {
957                    // A transaction reannounce future has finished. This doesn't necessarily mean
958                    // that a validation actually needs to be reannounced. The provided
959                    // `maybe_reannounce_tx_id` is a hint as to which transaction might need to be
960                    // reannounced, but without a strong guarantee.
961
962                    // `continue` if transaction doesn't exist. False positive.
963                    if worker
964                        .pending_transactions
965                        .transaction_user_data(maybe_reannounce_tx_id)
966                        .is_none()
967                    {
968                        continue;
969                    }
970
971                    // Don't gossip the transaction if it hasn't been validated or is already
972                    // included.
973                    // TODO: if best block changes, we would need to reset all the re-announce period of all transactions, awkward!
974                    // TODO: also, if this is false, then the transaction might never be re-announced ever again
975                    if worker
976                        .pending_transactions
977                        .is_included_best_chain(maybe_reannounce_tx_id)
978                        || !worker
979                            .pending_transactions
980                            .is_valid_against_best_block(maybe_reannounce_tx_id)
981                    {
982                        continue;
983                    }
984
985                    let now = worker.platform.now();
986                    let tx = worker
987                        .pending_transactions
988                        .transaction_user_data_mut(maybe_reannounce_tx_id)
989                        .unwrap();
990                    if tx.when_reannounce > now {
991                        continue;
992                    }
993
994                    // TODO: only announce if propagate is true
995
996                    // Update transaction state for the next re-announce.
997                    tx.when_reannounce = now + Duration::from_secs(5);
998                    worker.next_reannounce.push({
999                        let platform = worker.platform.clone();
1000                        Box::pin(async move {
1001                            platform.sleep(Duration::from_secs(5)).await;
1002                            maybe_reannounce_tx_id
1003                        })
1004                    });
1005
1006                    // Perform the announce.
1007                    let peers_sent = worker
1008                        .network_service
1009                        .clone()
1010                        .announce_transaction(
1011                            worker
1012                                .pending_transactions
1013                                .scale_encoding(maybe_reannounce_tx_id)
1014                                .unwrap(),
1015                        )
1016                        .await;
1017                    log!(
1018                        &worker.platform,
1019                        Debug,
1020                        &config.log_target,
1021                        "announced-to-network",
1022                        transaction = HashDisplay(&blake2_hash(
1023                            worker
1024                                .pending_transactions
1025                                .scale_encoding(maybe_reannounce_tx_id)
1026                                .unwrap()
1027                        )),
1028                        peers = peers_sent.iter().join(", ")
1029                    );
1030
1031                    // TODO: is this correct? and what should we do if announcing the same transaction multiple times? is it cumulative? `Broadcast` isn't super well documented
1032                    if !peers_sent.is_empty() {
1033                        worker
1034                            .pending_transactions
1035                            .transaction_user_data_mut(maybe_reannounce_tx_id)
1036                            .unwrap()
1037                            .update_status(TransactionStatus::Broadcast(peers_sent));
1038                    }
1039                }
1040
1041                WakeUpReason::MaybeValidated(maybe_validated_tx_id) => {
1042                    // A transaction validation future has finished. This doesn't necessarily mean
1043                    // that a validation has actually finished. The provided
1044                    // `maybe_validated_tx_id` is a hint as to which transaction might have
1045                    // finished being validated, but without a strong guarantee.
1046
1047                    // Try extract the validation result of this transaction, or `continue` if it
1048                    // is a false positive.
1049                    let (block_hash, validation_result) = match worker
1050                        .pending_transactions
1051                        .transaction_user_data_mut(maybe_validated_tx_id)
1052                    {
1053                        None => continue, // Normal. `maybe_validated_tx_id` is just a hint.
1054                        Some(tx) => match tx
1055                            .validation_in_progress
1056                            .as_mut()
1057                            .and_then(|f| f.now_or_never())
1058                        {
1059                            None => continue,               // Normal. `maybe_validated_tx_id` is just a hint.
1060                            Some(Err(_)) => unreachable!(), // Validations are never interrupted.
1061                            Some(Ok(result)) => {
1062                                tx.validation_in_progress = None;
1063                                result
1064                            }
1065                        },
1066                    };
1067
1068                    let tx_hash = blake2_hash(
1069                        worker
1070                            .pending_transactions
1071                            .scale_encoding(maybe_validated_tx_id)
1072                            .unwrap(),
1073                    );
1074
1075                    // The validation is made using the runtime service, while the state
1076                    // of the chain is tracked using the sync service. As such, it is
1077                    // possible for the validation to have been performed against a block
1078                    // that has already been finalized and removed from the pool.
1079                    if !worker.pending_transactions.has_block(&block_hash) {
1080                        log!(
1081                            &worker.platform,
1082                            Debug,
1083                            &config.log_target,
1084                            "transaction-validation-obsolete-block",
1085                            transaction = HashDisplay(&tx_hash),
1086                            block = HashDisplay(&block_hash)
1087                        );
1088                        continue;
1089                    }
1090
1091                    let validation_result = match validation_result {
1092                        Ok(result) => {
1093                            log!(
1094                                &worker.platform,
1095                                Debug,
1096                                &config.log_target,
1097                                "transaction-validation-success",
1098                                transaction = HashDisplay(&tx_hash),
1099                                block = HashDisplay(&block_hash),
1100                                priority = result.priority,
1101                                longevity = result.longevity,
1102                                propagate = ?result.propagate,
1103                            );
1104
1105                            log!(
1106                                &worker.platform,
1107                                Info,
1108                                &config.log_target,
1109                                format!(
1110                                    "Successfully validated transaction {}",
1111                                    HashDisplay(&tx_hash)
1112                                )
1113                            );
1114
1115                            worker
1116                                .pending_transactions
1117                                .transaction_user_data_mut(maybe_validated_tx_id)
1118                                .unwrap_or_else(|| unreachable!())
1119                                .update_status(TransactionStatus::Validated);
1120
1121                            // Schedule this transaction for announcement.
1122                            worker
1123                                .next_reannounce
1124                                .push(Box::pin(async move { maybe_validated_tx_id }));
1125
1126                            Ok(result)
1127                        }
1128                        Err(ValidationError::ObsoleteSubscription) => {
1129                            // Runtime service subscription is obsolete. Throw away everything and
1130                            // rebuild it.
1131                            continue 'channels_rebuild;
1132                        }
1133                        Err(ValidationError::InvalidOrError(InvalidOrError::Invalid(error))) => {
1134                            log!(
1135                                &worker.platform,
1136                                Debug,
1137                                &config.log_target,
1138                                "transaction-validation-invalid-tx",
1139                                transaction = HashDisplay(&tx_hash),
1140                                block = HashDisplay(&block_hash),
1141                                ?error,
1142                            );
1143
1144                            log!(
1145                                &worker.platform,
1146                                Warn,
1147                                &config.log_target,
1148                                format!(
1149                                    "Transaction {} invalid against block {}: {}",
1150                                    HashDisplay(&tx_hash),
1151                                    HashDisplay(&block_hash),
1152                                    error,
1153                                )
1154                            );
1155
1156                            Err(InvalidOrError::Invalid(error))
1157                        }
1158                        Err(ValidationError::InvalidOrError(InvalidOrError::ValidateError(
1159                            error,
1160                        ))) => {
1161                            log!(
1162                                &worker.platform,
1163                                Debug,
1164                                &config.log_target,
1165                                "transaction-validation-error",
1166                                transaction = HashDisplay(&tx_hash),
1167                                block = HashDisplay(&block_hash),
1168                                ?error,
1169                            );
1170
1171                            log!(
1172                                &worker.platform,
1173                                Warn,
1174                                &config.log_target,
1175                                format!(
1176                                    "Failed to validate transaction {}: {}",
1177                                    HashDisplay(&tx_hash),
1178                                    error
1179                                )
1180                            );
1181
1182                            Err(InvalidOrError::ValidateError(error))
1183                        }
1184                    };
1185
1186                    // No matter whether the validation is successful, we store the result in
1187                    // the transactions pool. This will later be picked up by the code that removes
1188                    // invalid transactions from the pool.
1189                    // TODO: shouldn't mark a transaction as invalid if it failed due to network errors
1190                    worker.pending_transactions.set_validation_result(
1191                        maybe_validated_tx_id,
1192                        &block_hash,
1193                        validation_result,
1194                    );
1195                }
1196
1197                WakeUpReason::ForegroundMessage(None) => return,
1198
1199                WakeUpReason::ForegroundMessage(Some(ToBackground::SubmitTransaction {
1200                    transaction_bytes,
1201                    updates_report,
1202                })) => {
1203                    // Handle the situation where the same transaction has already been
1204                    // submitted in the pool before.
1205                    let existing_tx_id = worker
1206                        .pending_transactions
1207                        .find_transaction(&transaction_bytes)
1208                        .next();
1209                    if let Some(existing_tx_id) = existing_tx_id {
1210                        let existing_tx = worker
1211                            .pending_transactions
1212                            .transaction_user_data_mut(existing_tx_id)
1213                            .unwrap();
1214                        if let Some((channel, detached)) = updates_report {
1215                            existing_tx.add_status_update(channel);
1216                            if detached {
1217                                existing_tx.detached = true;
1218                            }
1219                        }
1220                        continue;
1221                    }
1222
1223                    // We intentionally limit the number of transactions in the pool,
1224                    // and immediately drop new transactions of this limit is reached.
1225                    if worker.pending_transactions.num_transactions()
1226                        >= worker.max_pending_transactions
1227                    {
1228                        if let Some((updates_report, _)) = updates_report {
1229                            let _ = updates_report.try_send(TransactionStatus::Dropped(
1230                                DropReason::MaxPendingTransactionsReached,
1231                            ));
1232                        }
1233                        continue;
1234                    }
1235
1236                    // Success path. Inserting in pool.
1237                    worker.pending_transactions.add_unvalidated(
1238                        transaction_bytes,
1239                        PendingTransaction {
1240                            when_reannounce: worker.platform.now(),
1241                            detached: match &updates_report {
1242                                Some((_, true)) | None => true,
1243                                Some((_, false)) => false,
1244                            },
1245                            status_update: {
1246                                let mut vec = Vec::with_capacity(1);
1247                                if let Some((channel, _)) = updates_report {
1248                                    vec.push(channel);
1249                                }
1250                                vec
1251                            },
1252                            latest_status: None,
1253                            validation_in_progress: None,
1254                        },
1255                    );
1256                }
1257            }
1258        }
1259    }
1260}

1262/// Background worker running in parallel of the front service.
1263struct Worker<TPlat: PlatformRef> {
1264    /// Access to the platform's capabilities.
1265    platform: TPlat,
1266
1267    // How to download the bodies of blocks and synchronize the chain.
1268    sync_service: Arc<sync_service::SyncService<TPlat>>,
1269
1270    /// How to validate transactions.
1271    runtime_service: Arc<runtime_service::RuntimeService<TPlat>>,
1272
1273    /// How to gossip transactions.
1274    network_service: Arc<network_service::NetworkServiceChain<TPlat>>,
1275
1276    /// List of pending transactions.
1277    ///
1278    /// Contains all transactions that were submitted with
1279    /// [`TransactionsService::submit_transaction`] and their channel to send back their status.
1280    ///
1281    /// All the entries in this map represent transactions that we're trying to include on the
1282    /// network. It is normal to find entries where the status report channels are closed, as they
1283    /// still represent transactions that we're trying to include but whose status isn't
1284    /// interesting the frontend.
1285    ///
1286    /// All the blocks within this data structure are also pinned within the runtime service. They
1287    /// must be unpinned when they leave the data structure.
1288    pending_transactions: light_pool::LightPool<PendingTransaction<TPlat>, Block, InvalidOrError>,
1289
1290    /// See [`Config::max_pending_transactions`].
1291    max_pending_transactions: usize,
1292
1293    /// List of ongoing block body downloads.
1294    /// The output of the future is a block hash and a block body.
1295    block_downloads:
1296        FuturesUnordered<future::BoxFuture<'static, ([u8; 32], Result<Vec<Vec<u8>>, ()>)>>,
1297
1298    /// List of transactions currently being validated.
1299    /// Returns the [`light_pool::TransactionId`] of the transaction that has finished being
1300    /// validated. The result can then be read from [`PendingTransaction::validation_in_progress`].
1301    /// Since transaction IDs can be reused, the returned ID might not correspond to a transaction
1302    /// or might correspond to the wrong transaction. This ID being returned is just a hint as to
1303    /// which transaction to check, and not an authoritative value.
1304    validations_in_progress:
1305        FuturesUnordered<future::BoxFuture<'static, light_pool::TransactionId>>,
1306
1307    /// List of transactions that need to be reannounced.
1308    /// Returns the [`light_pool::TransactionId]` of the transaction that needs to be re-announced.
1309    /// Since transaction IDs can be reused, the returned ID might not correspond to a transaction
1310    /// or might correspond to the wrong transaction. This ID being returned is just a hint as to
1311    /// which transaction to check, not an authoritative value, and
1312    /// [`PendingTransaction::when_reannounce`] should be checked.
1313    next_reannounce: FuturesUnordered<future::BoxFuture<'static, light_pool::TransactionId>>,
1314
1315    /// See [`Config::max_concurrent_downloads`]. Maximum number of elements in
1316    /// [`Worker::block_downloads`].
1317    max_concurrent_downloads: usize,
1318}
1319
1320impl<TPlat: PlatformRef> Worker<TPlat> {
1321    /// Update the best block. Must have been previously inserted with
1322    /// [`light_pool::LightPool::add_block`].
1323    fn set_best_block(&mut self, log_target: &str, new_best_block_hash: &[u8; 32]) {
1324        let updates = self
1325            .pending_transactions
1326            .set_best_block(new_best_block_hash);
1327
1328        // There might be entries in common between `retracted_transactions` and
1329        // `included_transactions`, in the case of a re-org where a transaction is part of both
1330        // the old and new best chain.
1331        // In that situation we need to first signal `Retracted`, then only `InBlock`.
1332        // Consequently, process `retracted_transactions` first.
1333
1334        log!(
1335            &self.platform,
1336            Debug,
1337            &log_target,
1338            "best-chain-update",
1339            new_best_block = HashDisplay(new_best_block_hash),
1340            included_transactions = updates
1341                .included_transactions
1342                .iter()
1343                .map(|(id, _, _)| HashDisplay(&blake2_hash(
1344                    self.pending_transactions.scale_encoding(*id).unwrap()
1345                ))
1346                .to_string())
1347                .join(", "),
1348            retracted_transactions = updates
1349                .retracted_transactions
1350                .iter()
1351                .map(|(id, _, _)| HashDisplay(&blake2_hash(
1352                    self.pending_transactions.scale_encoding(*id).unwrap()
1353                ))
1354                .to_string())
1355                .join(", ")
1356        );
1357
1358        for (tx_id, _, _) in updates.retracted_transactions {
1359            let tx = self
1360                .pending_transactions
1361                .transaction_user_data_mut(tx_id)
1362                .unwrap();
1363            tx.update_status(TransactionStatus::IncludedBlockUpdate { block_hash: None });
1364        }
1365
1366        for (tx_id, block_hash, block_body_index) in updates.included_transactions {
1367            let tx = self
1368                .pending_transactions
1369                .transaction_user_data_mut(tx_id)
1370                .unwrap();
1371            // We assume that there's no more than 2<<32 transactions per block.
1372            let block_body_index = u32::try_from(block_body_index).unwrap();
1373            tx.update_status(TransactionStatus::IncludedBlockUpdate {
1374                block_hash: Some((block_hash, block_body_index)),
1375            });
1376        }
1377    }
1378}

1381    /// Header of the block, in SCALE encoding. Necessary in order to be able to validate blocks.
1382    scale_encoded_header: Vec<u8>,
1383

1391struct PendingTransaction<TPlat: PlatformRef> {
1392    /// Earliest moment when to gossip the transaction on the network again.
1393    ///
1394    /// This should be interpreted as the moment before which to not reannounce, rather than the
1395    /// moment when to announce.
1396    ///
1397    /// In particular, this value might be long in the past, in case for example of a transaction
1398    /// that is not validated.
1399    when_reannounce: TPlat::Instant,
1400
1401    /// List of channels that should receive changes to the transaction status.
1402    status_update: Vec<async_channel::Sender<TransactionStatus>>,
1403
1404    /// If `false`, then dropping all the [`PendingTransaction::status_update`] channels will
1405    /// remove the transaction from the pool.
1406    detached: bool,
1407
1408    /// Latest known status of the transaction. Used when a new sender is added to
1409    /// [`PendingTransaction::status_update`].
1410    latest_status: Option<TransactionStatus>,
1411
1412    /// If `Some`, will receive the result of the validation of the transaction.
1413    validation_in_progress: Option<
1414        oneshot::Receiver<(
1415            [u8; 32],
1416            Result<validate::ValidTransaction, ValidationError>,
1417        )>,
1418    >,
1419}
1420
1421impl<TPlat: PlatformRef> PendingTransaction<TPlat> {
1422    fn add_status_update(&mut self, channel: async_channel::Sender<TransactionStatus>) {
1423        if let Some(latest_status) = &self.latest_status {
1424            if channel.try_send(latest_status.clone()).is_err() {
1425                return;
1426            }
1427        }
1428
1429        self.status_update.push(channel);
1430    }
1431
1432    fn update_status(&mut self, status: TransactionStatus) {
1433        for n in 0..self.status_update.len() {
1434            let channel = self.status_update.swap_remove(n);
1435            if channel.try_send(status.clone()).is_ok() {
1436                self.status_update.push(channel);
1437            }
1438        }
1439
1440        self.latest_status = Some(status);
1441    }
1442}

1444/// Actual transaction validation logic. Validates the transaction against the given block of the
1445/// [`runtime_service::RuntimeService`].
1446///
1447/// Returns the result of the validation, and the hash of the block it was validated against.
1448async fn validate_transaction<TPlat: PlatformRef>(
1449    platform: &TPlat,
1450    log_target: &str,
1451    relay_chain_sync: &Arc<runtime_service::RuntimeService<TPlat>>,
1452    relay_chain_sync_subscription_id: runtime_service::SubscriptionId,
1453    block_hash: [u8; 32],
1454    block_scale_encoded_header: &[u8],
1455    scale_encoded_transaction: impl AsRef<[u8]> + Clone,
1456    source: validate::TransactionSource,
1457) -> Result<validate::ValidTransaction, ValidationError> {
1458    // TODO: move somewhere else?
1459    log!(
1460        platform,
1461        Debug,
1462        &log_target,
1463        "transaction-validation-started",
1464        transaction = HashDisplay(&blake2_hash(scale_encoded_transaction.as_ref())),
1465        block = HashDisplay(&block_hash),
1466        block_height = header::decode(
1467            block_scale_encoded_header,
1468            relay_chain_sync.block_number_bytes()
1469        )
1470        .ok()
1471        .map(|h| format!("#{}", h.number))
1472        .unwrap_or_else(|| "unknown".to_owned())
1473    );
1474
1475    let (pinned_runtime, block_state_root_hash, block_number) = match relay_chain_sync
1476        .pin_pinned_block_runtime(relay_chain_sync_subscription_id, block_hash)
1477        .await
1478    {
1479        Ok(r) => r,
1480        Err(runtime_service::PinPinnedBlockRuntimeError::ObsoleteSubscription) => {
1481            return Err(ValidationError::ObsoleteSubscription);
1482        }
1483        Err(runtime_service::PinPinnedBlockRuntimeError::BlockNotPinned) => unreachable!(),
1484    };
1485
1486    let runtime_call_future = relay_chain_sync.runtime_call(
1487        pinned_runtime,
1488        block_hash,
1489        block_number,
1490        block_state_root_hash,
1491        validate::VALIDATION_FUNCTION_NAME.to_owned(),
1492        Some(("TaggedTransactionQueue".to_owned(), 3..=3)),
1493        validate::validate_transaction_runtime_parameters_v3(
1494            iter::once(scale_encoded_transaction.as_ref()),
1495            source,
1496            &block_hash,
1497        )
1498        .fold(Vec::new(), |mut a, b| {
1499            a.extend_from_slice(b.as_ref());
1500            a
1501        }),
1502        3,
1503        Duration::from_secs(8),
1504        NonZero::<u32>::new(1).unwrap(),
1505    );
1506
1507    let success = match runtime_call_future.await {
1508        Ok(output) => output,
1509        Err(runtime_service::RuntimeCallError::Execution(error)) => {
1510            return Err(ValidationError::InvalidOrError(
1511                InvalidOrError::ValidateError(ValidateTransactionError::Execution(error)),
1512            ));
1513        }
1514        Err(runtime_service::RuntimeCallError::Crash) => {
1515            return Err(ValidationError::InvalidOrError(
1516                InvalidOrError::ValidateError(ValidateTransactionError::Crash),
1517            ));
1518        }
1519        Err(runtime_service::RuntimeCallError::Inaccessible(errors)) => {
1520            return Err(ValidationError::InvalidOrError(
1521                InvalidOrError::ValidateError(ValidateTransactionError::Inaccessible(errors)),
1522            ));
1523        }
1524        Err(runtime_service::RuntimeCallError::InvalidRuntime(error)) => {
1525            return Err(ValidationError::InvalidOrError(
1526                InvalidOrError::ValidateError(ValidateTransactionError::InvalidRuntime(error)),
1527            ));
1528        }
1529        Err(runtime_service::RuntimeCallError::ApiVersionRequirementUnfulfilled) => {
1530            return Err(ValidationError::InvalidOrError(
1531                InvalidOrError::ValidateError(
1532                    ValidateTransactionError::ApiVersionRequirementUnfulfilled,
1533                ),
1534            ));
1535        }
1536    };
1537
1538    match validate::decode_validate_transaction_return_value(&success.output) {
1539        Ok(Ok(decoded)) => Ok(decoded),
1540        Ok(Err(err)) => Err(ValidationError::InvalidOrError(InvalidOrError::Invalid(
1541            err,
1542        ))),
1543        Err(err) => Err(ValidationError::InvalidOrError(
1544            InvalidOrError::ValidateError(ValidateTransactionError::OutputDecodeError(err)),
1545        )),
1546    }
1547}
